112

115  real inv_d12 = 1.0 / d12aux;
116 
117  real d12 = d12aux / sigma;

119  real inv_d12_a = 1.0 / (d12-getCutoff());
120 
121  real term1 = B * pow(d12, -p);
122  real term2 = pow(d12, -q);
123  real term12 = term1 - term2;
124 
125  real energy = epsilonA * term12 * exp ( inv_d12_a );
126 
127  real inv_d12_a2 = inv_d12_a*inv_d12_a;
128 
129  real term3 = ( p*term1 - q*term2 ) * sigmaInv_d12 / term12;
130 
131  real ffactor = energy * ( term3 + inv_d12_a2);
