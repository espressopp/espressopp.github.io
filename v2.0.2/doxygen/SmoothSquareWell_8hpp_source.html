98  real _computeEnergySqrRaw(real distSqr) const {
99  real r = sqrt(distSqr);
100  real energy = halfepsilon*(exp((sigma-r)/a)+tanh((r-rb)/a)-1);
101  return energy;

104  bool _computeForceRaw(Real3D& force, const Real3D& dist, real distSqr) const {
105  real r = sqrt(distSqr);
106  real ffactor = -halfepsilon/r/a*(-exp((sigma-r)/a)+pow(cosh((r-rb)/a), -2));
107  force = dist * ffactor;
108  return true;
109  }
