135  template < typename _Potential > inline void
136  VerletListAdressCGInteractionTemplate < _Potential >::
137  addForces() {
138  LOG4ESPP_INFO(theLogger, "add forces computed by the Verlet List");
139 
140  // We only compute the CG contribution of all forces in this template.
141 
142  // Pairs not inside the AdResS Zone (CG region)
143  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
144 
145  Particle &p1 = *it->first;
146  Particle &p2 = *it->second;
147  int type1 = p1.type();
148  int type2 = p2.type();
149 
150  const Potential &potential = getPotential(type1, type2);
151 
152  Real3D force(0.0, 0.0, 0.0);
153 
154  // CG forces
155  if(potential._computeForce(force, p1, p2)) {
156  p1.force() += force;
157  p2.force() -= force;
158  }
159  }
160 
161  // Compute forces (AT and VP) of Pairs inside AdResS zone
162  for (PairList::Iterator it(verletList->getAdrPairs()); it.isValid(); ++it) {
163 
164  // these are the two VP interacting
165  Particle &p1 = *it->first;
166  Particle &p2 = *it->second;
167 
168  // read weights
169  real w1 = p1.lambda();
170  real w2 = p2.lambda();
171  real w12 = w1 * w2; // AdResS
172 
173  // force between VP particles
174  int type1 = p1.type();
175  int type2 = p2.type();
176  const Potential &potential = getPotential(type1, type2);
177  Real3D forcevp(0.0, 0.0, 0.0);
178  if (w12 != 1) { // calculate VP force if both VP are outside AT region (CG-HY, HY-HY)
179  if(potential._computeForce(forcevp, p1, p2)) {
180  forcevp *= (1 - w12);
181  p1.force() += forcevp;
182  p2.force() -= forcevp;
183  }
184  }
185  }
186 
187  // Loop over CG particles and overwrite AT forces and velocity.
188  // This makes the AT particles move along with CG particles.
189 
190  // Note (Karsten): This is a different approach compared to H-AdResS. Here, we overwrite intra-atomistic
191  // rotations and vibrations in the CG zone. This leads to failures in the kinetic energy. However, in Force-AdResS there is no energy conservation anyway.
192  // Here we calculate CG forces/velocities and distribute them to AT particles. In contrast, in H-AdResS, we calculate AT forces from intra-molecular
193  // interactions and inter-molecular center-of-mass interactions and just update the positions of the center-of-mass CG particles.
194  std::set<Particle*> cgZone = verletList->getCGZone();
195  for (std::set<Particle*>::iterator it=cgZone.begin();
196  it != cgZone.end(); ++it) {
197 
198  Particle &vp = **it;
199 
200  FixedTupleListAdress::iterator it3;
201  it3 = fixedtupleList->find(&vp);
202 
203  if (it3 != fixedtupleList->end()) {
204 
205  std::vector<Particle*> atList1;
206  atList1 = it3->second;
207 
208  //Real3D vpfm = vp.force() / vp.getMass();
209  for (std::vector<Particle*>::iterator itv = atList1.begin();
210  itv != atList1.end(); ++itv) {
211  Particle &at = **itv;
212  at.velocity() = vp.velocity(); // Overwrite velocity - Note (Karsten): See comment above.
213  //at.force() += at.mass() * vpfm;
214  }
215 
216  }
217  else { // this should not happen
218  std::stringstream ss;
219  ss << " VP particle not found in tuples: " << vp.id() << "-" << vp.ghost();
220  throw std::runtime_error(ss.str());
221  }
222  }
223  }

225  template < typename _Potential > inline real
226  VerletListAdressCGInteractionTemplate < _Potential >::
227  computeEnergy() {
228 
229  std::set<Particle*> cgZone = verletList->getCGZone();
230  for (std::set<Particle*>::iterator it=cgZone.begin();
231  it != cgZone.end(); ++it) {
232 
233  Particle &vp = **it;
234  vp.lambda() = 0.0;
235  }
236 
237  std::set<Particle*> adrZone = verletList->getAdrZone();
238  for (std::set<Particle*>::iterator it=adrZone.begin();
239  it != adrZone.end(); ++it) {
240 
241  Particle &vp = **it;
242 
243  FixedTupleListAdress::iterator it3;
244  it3 = fixedtupleList->find(&vp);
245 
246  if (it3 != fixedtupleList->end()) {
247 
248  std::vector<Particle*> atList;
249  atList = it3->second;
250 
251  // compute center of mass
252  Real3D cmp(0.0, 0.0, 0.0); // center of mass position
253  Real3D cmv(0.0, 0.0, 0.0); // center of mass velocity
254  for (std::vector<Particle*>::iterator it2 = atList.begin();
255  it2 != atList.end(); ++it2) {
256  Particle &at = **it2;
257  cmp += at.mass() * at.position();
258  cmv += at.mass() * at.velocity();
259  }
260  cmp /= vp.getMass();
261  cmv /= vp.getMass();
262 
263  // update (overwrite) the position and velocity of the VP
264  vp.position() = cmp;
265  vp.velocity() = cmv;
266 
267  // calculate distance to nearest adress particle or center
268  std::vector<Real3D*>::iterator it2 = verletList->getAdrPositions().begin();
269  Real3D pa = **it2; // position of adress particle
270  Real3D d1;
271  verletList->getSystem()->bc->getMinimumImageVector(d1, vp.position(), pa);
272  // set min1sq before loop
273  real min1sq;
274  if (verletList->getAdrRegionType()) { // spherical adress region
275  min1sq = d1.sqr();
276  } else { //slab-type adress region
277  min1sq = d1[0]*d1[0];
278  }
279  ++it2;
280  for (; it2 != verletList->getAdrPositions().end(); ++it2) {
281  pa = **it2;
282  verletList->getSystem()->bc->getMinimumImageVector(d1, vp.position(), pa);
283  real distsq1;
284  if (verletList->getAdrRegionType()) { // spherical adress region
285  distsq1 = d1.sqr();
286  } else { //slab-type adress region
287  distsq1 = d1[0]*d1[0];
288  }
289  if (distsq1 < min1sq) min1sq = distsq1;
290  }
291 
292  // calculate weight and write it in the map
293  real w;
294  if (dex2 > min1sq) w = 1;
295  else if (dexdhy2 < min1sq) w = 0;
296  else {
297  w = cos(pidhy2 * (sqrt(min1sq) - dex));
298  w *= w;
299  // real argument = sqrt(min1sq) - dex;
300  // w = 1.0-(30.0/(pow(dhy, 5.0)))*(1.0/5.0*pow(argument, 5.0)-dhy/2.0*pow(argument, 4.0)+1.0/3.0*pow(argument, 3.0)*dhy*dhy);
301  }
302  vp.lambda() = w;
303  }
304  else { // this should not happen
305  std::stringstream ss;
306  ss << " VP particle not found in tuples: " << vp.id() << "-" << vp.ghost();
307  throw std::runtime_error(ss.str());
308  }
309  }
310 
311  LOG4ESPP_INFO(theLogger, "compute energy of the Verlet list pairs");
312 
313  real e = 0.0;
314 
315  for (PairList::Iterator it(verletList->getPairs());
316  it.isValid(); ++it) {
317  Particle &p1 = *it->first;
318  Particle &p2 = *it->second;
319  int type1 = p1.type();
320  int type2 = p2.type();
321  const Potential &potential = getPotential(type1, type2);
322  e += potential._computeEnergy(p1, p2);
323  }
324 
325  for (PairList::Iterator it(verletList->getAdrPairs());
326  it.isValid(); ++it) {
327  Particle &p1 = *it->first;
328  Particle &p2 = *it->second;
329  real w1 = p1.lambda();
330  real w2 = p2.lambda();
331  real w12 = w1 * w2;
332  int type1 = p1.type();
333  int type2 = p2.type();
334  const Potential &potential = getPotential(type1, type2);
335  e += (1.0-w12)*potential._computeEnergy(p1, p2);
336  }
337 
338  real esum;
339  boost::mpi::all_reduce(*getVerletList()->getSystem()->comm, e, esum, std::plus<real>());
340  return esum;
341  }
