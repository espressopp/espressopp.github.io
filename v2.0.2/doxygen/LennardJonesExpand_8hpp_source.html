79  real _computeEnergySqrRaw(real distSqr) const {
80  real frac2 = sigma*sigma / pow(sqrt(distSqr) - delta, 2);
81  real frac6 = frac2 * frac2 * frac2;
82  real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
83  return energy;
84  }
85 
86  bool _computeForceRaw(Real3D& force,
87  const Real3D& dist,
88  real distSqr) const {
89  real r = sqrt(distSqr);
90  real rshift = r - delta;
91  real rshiftsq = rshift * rshift;
92  real frac2 = sigma * sigma / rshiftsq;
93  real frac6 = frac2 * frac2 * frac2;
94  real ffactor = 4.0 * epsilon * frac6 * (12.0 * frac6 - 6.0) / rshift / r;
95  force = dist * ffactor;
96  return true;
97  }
