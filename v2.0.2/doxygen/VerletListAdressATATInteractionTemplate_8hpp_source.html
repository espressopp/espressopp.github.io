149  template < typename _Potential1, typename _Potential2 > inline void
150  VerletListAdressATATInteractionTemplate < _Potential1, _Potential2 >::
151  addForces() {
152  LOG4ESPP_INFO(theLogger, "add forces computed by the Verlet List");
153 
154  // We only compute the atomistic contribution of all forces in this template. Hence, we do not loop over the pairs in the CG region.
155 
156  // Compute atomistic forces of Pairs inside AdResS zone
157  for (PairList::Iterator it(verletList->getAdrPairs()); it.isValid(); ++it) {
158 
159  // these are the two VP interacting
160  Particle &p1 = *it->first;
161  Particle &p2 = *it->second;
162 
163  // read weights
164  real w1 = p1.lambda();
165  real w2 = p2.lambda();
166  real w12 = w1 * w2; // AdResS
167 
168  // force between AT particles
169  if (w12 != 0) { // calculate AT force if both VP are outside CG region (HY-HY, HY-AT, AT-AT)
170  FixedTupleListAdress::iterator it3;
171  FixedTupleListAdress::iterator it4;
172  it3 = fixedtupleList->find(&p1);
173  it4 = fixedtupleList->find(&p2);
174 
175  if (it3 != fixedtupleList->end() && it4 != fixedtupleList->end()) {
176 
177  std::vector<Particle*> atList1;
178  std::vector<Particle*> atList2;
179  atList1 = it3->second;
180  atList2 = it4->second;
181 
182  for (std::vector<Particle*>::iterator itv = atList1.begin();
183  itv != atList1.end(); ++itv) {
184 
185  Particle &p3 = **itv;
186 
187  for (std::vector<Particle*>::iterator itv2 = atList2.begin();
188  itv2 != atList2.end(); ++itv2) {
189 
190  Particle &p4 = **itv2;
191 
192  // AT forces
193  Real3D force(0.0, 0.0, 0.0);
194  const Potential1 &potential1 = getPotential1(p3.type(), p4.type());
195  if(potential1._computeForce(force, p3, p4)) {
196  force *= w12;
197  p3.force() += force;
198  p4.force() -= force;
199  }
200  const Potential2 &potential2 = getPotential2(p3.type(), p4.type());
201  if(potential2._computeForce(force, p3, p4)) {
202  force *= w12;
203  p3.force() += force;
204  p4.force() -= force;
205  }
206 
207  }
208 
209  }
210  }
211  else { // this should not happen
212  std::stringstream ss;
213  ss << "One of the VP particles not found in tuples: " << p1.id() << "-" << p1.ghost() << ", " << p2.id() << "-" << p2.ghost() << " (" << p1.position() << ") (" << p2.position();
214  throw std::runtime_error(ss.str());
215  }
216  }
217  }
218 
219  // Loop over CG particles and overwrite AT forces and velocity.
220  // This makes the AT particles move along with CG particles.
221 
222  // Note (Karsten): This is a different approach compared to H-AdResS. Here, we overwrite intra-atomistic
223  // rotations and vibrations in the CG zone. This leads to failures in the kinetic energy. However, in Force-AdResS there is no energy conservation anyway.
224  // Here we calculate CG forces/velocities and distribute them to AT particles. In contrast, in H-AdResS, we calculate AT forces from intra-molecular
225  // interactions and inter-molecular center-of-mass interactions and just update the positions of the center-of-mass CG particles.
226  std::set<Particle*> cgZone = verletList->getCGZone();
227  for (std::set<Particle*>::iterator it=cgZone.begin();
228  it != cgZone.end(); ++it) {
229 
230  Particle &vp = **it;
231 
232  FixedTupleListAdress::iterator it3;
233  it3 = fixedtupleList->find(&vp);
234 
235  if (it3 != fixedtupleList->end()) {
236 
237  std::vector<Particle*> atList1;
238  atList1 = it3->second;
239 
240  //Real3D vpfm = vp.force() / vp.getMass();
241  for (std::vector<Particle*>::iterator itv = atList1.begin();
242  itv != atList1.end(); ++itv) {
243  Particle &at = **itv;
244  at.velocity() = vp.velocity(); // Overwrite velocity - Note (Karsten): See comment above.
245  //at.force() += at.mass() * vpfm;
246  }
247 
248  }
249  else { // this should not happen
250  std::cout << " VP particle not found in tuples: " << vp.id() << "-" << vp.ghost();
251  exit(1);
252  return;
253  }
254  }

257  template < typename _Potential1, typename _Potential2 >
258  inline real
259  VerletListAdressATATInteractionTemplate < _Potential1, _Potential2 >::
260  computeEnergy() {
261 
262  std::set<Particle*> cgZone = verletList->getCGZone();
263  for (std::set<Particle*>::iterator it=cgZone.begin();
264  it != cgZone.end(); ++it) {
265 
266  Particle &vp = **it;
267  vp.lambda() = 0.0;
268  }
269 
270  std::set<Particle*> adrZone = verletList->getAdrZone();
271  for (std::set<Particle*>::iterator it=adrZone.begin();
272  it != adrZone.end(); ++it) {
273 
274  Particle &vp = **it;
275 
276  FixedTupleListAdress::iterator it3;
277  it3 = fixedtupleList->find(&vp);
278 
279  if (it3 != fixedtupleList->end()) {
280 
281  std::vector<Particle*> atList;
282  atList = it3->second;
283 
284  // compute center of mass
285  Real3D cmp(0.0, 0.0, 0.0); // center of mass position
286  Real3D cmv(0.0, 0.0, 0.0); // center of mass velocity
287  for (std::vector<Particle*>::iterator it2 = atList.begin();
288  it2 != atList.end(); ++it2) {
289  Particle &at = **it2;
290  cmp += at.mass() * at.position();
291  cmv += at.mass() * at.velocity();
292  }
293  cmp /= vp.getMass();
294  cmv /= vp.getMass();
295 
296  // update (overwrite) the position and velocity of the VP
297  vp.position() = cmp;
298  vp.velocity() = cmv;
299 
300  // calculate distance to nearest adress particle or center
301  std::vector<Real3D*>::iterator it2 = verletList->getAdrPositions().begin();
302  Real3D pa = **it2; // position of adress particle
303  Real3D d1;
304  verletList->getSystem()->bc->getMinimumImageVector(d1, vp.position(), pa);
305  // set min1sq before loop
306  real min1sq;
307  if (verletList->getAdrRegionType()) { // spherical adress region
308  min1sq = d1.sqr();
309  } else { //slab-type adress region
310  min1sq = d1[0]*d1[0];
311  }
312  ++it2;
313  for (; it2 != verletList->getAdrPositions().end(); ++it2) {
314  pa = **it2;
315  verletList->getSystem()->bc->getMinimumImageVector(d1, vp.position(), pa);
316  real distsq1;
317  if (verletList->getAdrRegionType()) { // spherical adress region
318  distsq1 = d1.sqr();
319  } else { //slab-type adress region
320  distsq1 = d1[0]*d1[0];
321  }
322  if (distsq1 < min1sq) min1sq = distsq1;
323  }
324 
325  // calculate weight and write it in the map
326  real w;
327  if (dex2 > min1sq) w = 1;
328  else if (dexdhy2 < min1sq) w = 0;
329  else {
330  w = cos(pidhy2 * (sqrt(min1sq) - dex));
331  w *= w;
332  // real argument = sqrt(min1sq) - dex;
333  // w = 1.0-(30.0/(pow(dhy, 5.0)))*(1.0/5.0*pow(argument, 5.0)-dhy/2.0*pow(argument, 4.0)+1.0/3.0*pow(argument, 3.0)*dhy*dhy);
334  }
335  vp.lambda() = w;
336  }
337  else { // this should not happen
338  std::cout << " VP particle " << vp.id() << "-" << vp.ghost() << " not found in tuples ";
339  std::cout << " (" << vp.position() << ")\n";
340  exit(1);
341  }
342  }
343 
344  LOG4ESPP_INFO(theLogger, "compute energy of the Verlet list pairs");
345 
346  real e = 0.0;
347 
348  for (PairList::Iterator it(verletList->getAdrPairs());
349  it.isValid(); ++it) {
350  Particle &p1 = *it->first;
351  Particle &p2 = *it->second;
352  real w1 = p1.lambda();
353  real w2 = p2.lambda();
354  real w12 = w1 * w2;
355 
356  FixedTupleListAdress::iterator it3;
357  FixedTupleListAdress::iterator it4;
358  it3 = fixedtupleList->find(&p1);
359  it4 = fixedtupleList->find(&p2);
360 
361  if (it3 != fixedtupleList->end() && it4 != fixedtupleList->end()) {
362  std::vector<Particle*> atList1;
363  std::vector<Particle*> atList2;
364  atList1 = it3->second;
365  atList2 = it4->second;
366 
367  for (std::vector<Particle*>::iterator itv = atList1.begin();
368  itv != atList1.end(); ++itv) {
369 
370  Particle &p3 = **itv;
371  for (std::vector<Particle*>::iterator itv2 = atList2.begin();
372  itv2 != atList2.end(); ++itv2) {
373  Particle &p4 = **itv2;
374 
375  // AT energies
376  const Potential1 &potential1 = getPotential1(p3.type(), p4.type());
377  const Potential2 &potential2 = getPotential2(p3.type(), p4.type());
378  e += w12*(potential1._computeEnergy(p3, p4) + potential2._computeEnergy(p3, p4));
379  }
380  }
381  }
382  }
383  real esum;
384  boost::mpi::all_reduce(*getVerletList()->getSystem()->comm, e, esum, std::plus<real>());
385  return esum;
386  }
