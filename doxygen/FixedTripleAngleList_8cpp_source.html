00170     std::vector< longint > toSendInt;
00171     std::vector< real > toSendReal;
00172     // loop over the particle list
00173     for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
00174       longint pid = pit->id();
00175       
00176       // find all triples that involve this particle
00177       int n = triplesAngles.count(pid);
00178 
00179       if (n > 0) {
00180         std::pair<TriplesAngles::const_iterator,
00181         TriplesAngles::const_iterator> equalRange = triplesAngles.equal_range(pid);
00182 
00183         // first write the pid of this particle
00184         // then the number of partners (n)
00185         // and then the pids of the partners
00186         toSendInt.reserve(toSendInt.size()+2*n+1);
00187         toSendReal.reserve(toSendReal.size()+n);
00188         toSendInt.push_back(pid);
00189         toSendInt.push_back(n);
00190         for(TriplesAngles::const_iterator it = equalRange.first; it!=equalRange.second; ++it) {
00191           toSendInt.push_back(it->second.first.first);
00192           toSendInt.push_back(it->second.first.second);
00193           toSendReal.push_back(it->second.second);
00194         }
00195 
00196         // delete all of these triples from the global list
00197         triplesAngles.erase(pid);
00198       }
00199     }
00200     // send the list
00201     buf.write(toSendInt);
00202     buf.write(toSendReal);
00203     LOG4ESPP_INFO(theLogger, "prepared fixed triple list before send particles");
