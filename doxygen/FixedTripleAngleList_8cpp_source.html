60  bool FixedTripleAngleList::
61  add(longint pid1, longint pid2, longint pid3) {
62  bool returnVal = true;
63  System& system = storage->getSystemRef();
64  esutil::Error err(system.comm);
65 
66  // ADD THE LOCAL TRIPLET
67  Particle *p1 = storage->lookupLocalParticle(pid1);
68  Particle *p2 = storage->lookupRealParticle(pid2);
69  Particle *p3 = storage->lookupLocalParticle(pid3);
70 
71  // middle particle is the reference particle and must exist here
72  if (!p2){
73  // particle does not exists here (some other CPU must have it)
74  returnVal = false;
75  }
76  else{
77  if (!p1) {
78  std::stringstream msg;
79  msg << "triple particle p1 " << pid1 << " does not exists here and cannot be added";
80  msg << " triplet: " << pid1 << "-" << pid2 << "-" << pid3;
81  err.setException( msg.str() );
82  }
83 
84  if (!p3) {
85  std::stringstream msg;
86  msg << "triple particle p3 " << pid3 << " does not exists here and cannot be added";
87  msg << " triplet: " << pid1 << "-" << pid2 << "-" << pid3;
88  err.setException( msg.str() );
89  }
90  }
91  err.checkException();
92 
93  if(returnVal){
94  // add the triple locally
95  this->add(p1, p2, p3);
96  //printf("me = %d: pid1 %d, pid2 %d, pid3 %d\n", mpiWorld->rank(), pid1, pid2, pid3);
97 
98  Real3D pos1 = p1->position();
99  Real3D pos2 = p2->position();
100  Real3D pos3 = p3->position();
101 
102  Real3D r12(0.0,0.0,0.0);
103  system.bc->getMinimumImageVectorBox(r12, pos2, pos1);
104  Real3D r32(0.0,0.0,0.0);
105  system.bc->getMinimumImageVectorBox(r32, pos2, pos3);
106 
107  real cosVal = r12*r32 / (r12.abs() * r32.abs());
108 
109  if(cosVal>1) cosVal=1;
110  if(cosVal<-1) cosVal=-1;
111 
112  triplesAngles.insert(std::make_pair(pid2,
113  std::make_pair(std::make_pair(pid1, pid3), acos(cosVal) )));
114  }
115  LOG4ESPP_INFO(theLogger, "added fixed triple to global triple list");
116  return returnVal;
117  }

169  void FixedTripleAngleList::beforeSendParticles(ParticleList& pl, OutBuffer& buf) {
170  std::vector< longint > toSendInt;
171  std::vector< real > toSendReal;
172  // loop over the particle list
173  for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
174  longint pid = pit->id();
175 
176  // find all triples that involve this particle
177  int n = triplesAngles.count(pid);
178 
179  if (n > 0) {
180  std::pair<TriplesAngles::const_iterator,
181  TriplesAngles::const_iterator> equalRange = triplesAngles.equal_range(pid);
182 
183  // first write the pid of this particle
184  // then the number of partners (n)
185  // and then the pids of the partners
186  toSendInt.reserve(toSendInt.size()+2*n+1);
187  toSendReal.reserve(toSendReal.size()+n);
188  toSendInt.push_back(pid);
189  toSendInt.push_back(n);
190  for(TriplesAngles::const_iterator it = equalRange.first; it!=equalRange.second; ++it) {
191  toSendInt.push_back(it->second.first.first);
192  toSendInt.push_back(it->second.first.second);
193  toSendReal.push_back(it->second.second);
194  }
195 
196  // delete all of these triples from the global list
197  triplesAngles.erase(pid);
198  }
199  }
200  // send the list
201  buf.write(toSendInt);
202  buf.write(toSendReal);
203  LOG4ESPP_INFO(theLogger, "prepared fixed triple list before send particles");
204  }

206  void FixedTripleAngleList::afterRecvParticles(ParticleList &pl, InBuffer& buf) {
207 
208  std::vector< longint > receivedInt;
209  std::vector< real > receivedReal;
210  int n;
211  longint pid1, pid2, pid3;
212  real angleVal;
213  TriplesAngles::iterator it = triplesAngles.begin();
214  // receive the triple list
215  buf.read(receivedInt);
216  buf.read(receivedReal);
217  int size = receivedInt.size(); int i = 0; int j = 0;
218  while (i < size) {
219  // unpack the list
220  pid2 = receivedInt[i++];
221  n = receivedInt[i++];
222  for (; n > 0; --n) {
223  pid1 = receivedInt[i++];
224  pid3 = receivedInt[i++];
225  angleVal = receivedReal[j++];
226 
227  // add the triple
228  it = triplesAngles.insert(it, std::make_pair(pid2,
229  std::make_pair(std::make_pair(pid1, pid3), angleVal)));
230  }
231  }
232  if (i != size) {

234  }

238  void FixedTripleAngleList::onParticlesChanged() {
239  System& system = storage->getSystemRef();
240  esutil::Error err(system.comm);
241 
242  // (re-)generate the local triple list from the global list
243  //printf("FixedTripleAngleList: rebuild local triple list from global\n");
244  this->clear();
245  longint lastpid2 = -1;
246  Particle *p1;
247  Particle *p2;
248  Particle *p3;
249  for (TriplesAngles::const_iterator it = triplesAngles.begin(); it!=triplesAngles.end(); ++it) {
250  //printf("lookup global triple %d %d %d\n", it->first, it->second.first, it->second.second);
251  if (it->first != lastpid2) {
252  p2 = storage->lookupRealParticle(it->first);
253  if (p2 == NULL) {
254  std::stringstream msg;
255  msg << "triple particle p2 " << it->first << " does not exists here";
256  err.setException( msg.str() );
257  }
258  lastpid2 = it->first;
259  }
260  p1 = storage->lookupLocalParticle(it->second.first.first);
261  if (p1 == NULL) {
262  std::stringstream msg;
263  msg << "triple particle p1 " << it->second.first.first << " does not exists here";
264  err.setException( msg.str() );
265  }
266  p3 = storage->lookupLocalParticle(it->second.first.second);
267  if (p3 == NULL) {
268  std::stringstream msg;
269  msg<< "triple particle p3 " << it->second.first.second << " does not exists here";
270  err.setException( msg.str() );
271  }
272  this->add(p1, p2, p3);
273  }
274  err.checkException();
275  LOG4ESPP_INFO(theLogger, "regenerated local fixed triple list from global list");
