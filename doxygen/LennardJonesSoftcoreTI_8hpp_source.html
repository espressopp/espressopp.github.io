60  void preset() {
61  complLambdaTI = 1.0 - lambdaTI;
62 
63  sigmaSC_A6 = pow(sigmaSC_A,6);
64  sigmaSC_B6 = pow(sigmaSC_B,6);
65  lambdaTI_powerSC = pow(lambdaTI,powerSC);
66  lambdaTI_powerSCm1 = pow(lambdaTI,powerSC-1.0);
67  compllambdaTI_powerSC = pow(1.0-lambdaTI,powerSC);
68  compllambdaTI_powerSCm1 = pow(1.0-lambdaTI,powerSC-1.0);
69  alpha_sigmaA6_lambdaP = alphaSC * sigmaSC_A6 * lambdaTI_powerSC;
70  alpha_sigmaB6_compllambdaP = alphaSC * sigmaSC_B6 * compllambdaTI_powerSC;
71  powerSC_alphaSC_inv6 = powerSC*alphaSC/6.0;
72 
73  real sig2 = sigmaSC_A * sigmaSC_A;
74  real sig6 = sig2 * sig2 * sig2;
75  ff1A = 48.0 * epsilonA * sig6 * sig6;
76  ff2A = 24.0 * epsilonA * sig6;
77 
78  sig2 = sigmaSC_B * sigmaSC_B;
79  sig6 = sig2 * sig2 * sig2;
80  ff1B = 48.0 * epsilonB * sig6 * sig6;
81  ff2B = 24.0 * epsilonB * sig6;
82  }

118  bool checkTIpair(size_t pid1, size_t pid2) const {
119  if (annihilate) {
120  //if at least one of pid1 and pid2 in pidsTI (OR)
121  if ((pidsTI.find(pid1) != pidsTI.end()) or (pidsTI.find(pid2) != pidsTI.end())) {
122  return true;
123  } else {
124  return false;
125  }
126  } else { //decouple
127  //if pid1 or pid2 in pidsTI, but not both (XOR)
128  if ((pidsTI.find(pid1) != pidsTI.end()) != (pidsTI.find(pid2) != pidsTI.end())) {
129  return true;
130  } else {
131  return false;
132  }
133  }

136  real _computeEnergy(const Particle& p1, const Particle& p2) const {
137  Real3D dist = p1.position() - p2.position();
138  real distSqr = dist.sqr();
139  if (distSqr>cutoffSqr) return 0.0;
140 
141  real frac2, frac6;
142  real energyA, energyB;
143  real rA, rB, rA5, rB5, rA2, rB2;
144  real r6, r5;
145 
146  if (checkTIpair(p1.id(),p2.id())) { //use softcore and lambdaTI
147  r6 = distSqr*distSqr*distSqr;
148 
149  rA = pow(alpha_sigmaA6_lambdaP + r6,1.0/6.0);
150  rA2 = rA*rA;
151  frac2 = sigmaSC_A*sigmaSC_A / rA2;
152  frac6 = frac2 * frac2 * frac2;
153  energyA = 4.0 * epsilonA * (frac6 * frac6 - frac6);
154 
155  rB = pow(alpha_sigmaB6_compllambdaP + r6,1.0/6.0);
156  rB2 = rB*rB;
157  frac2 = sigmaSC_B*sigmaSC_B / rB2;
158  frac6 = frac2 * frac2 * frac2;
159  energyB = 4.0 * epsilonB * (frac6 * frac6 - frac6);
160 
161  real energy = complLambdaTI*energyA + lambdaTI*energyB;
162  return energy;
163  } else {
164  //hardcore, epsilon and sigma used are from state A
165  frac2 = sigmaSC_A*sigmaSC_A / distSqr;
166  frac6 = frac2 * frac2 * frac2;
167  real energy = 4.0 * epsilonA * (frac6 * frac6 - frac6);
168  return energy;
169  }

172  bool _computeForce(Real3D& force, const Particle &p1,
173  const Particle &p2) const {
174  Real3D dist = p1.position() - p2.position();
175  real distSqr = dist.sqr();
176  if (distSqr>cutoffSqr) return true;
177 
178  real frac2, frac6;
179 
180  if (checkTIpair(p1.id(),p2.id())) {
181 
182  real forceA, forceB; //this is dV/dr not dV/dr * 1/r
183  real rA, rB, rA5, rB5, rA2, rB2;
184  real r6, r5;
185 
186  r6 = distSqr*distSqr*distSqr;
187  r5 = distSqr*distSqr*sqrt(distSqr);
188 
189  rA = pow(alpha_sigmaA6_lambdaP + r6,1.0/6.0);
190  rA2 = rA*rA;
191  rA5 = rA2*rA2*rA;
192  frac2 = 1.0 / rA2;
193  frac6 = frac2 * frac2 * frac2;
194  forceA = frac6 * (ff1A * frac6 - ff2A) * frac2 * rA;
195 
196  rB = pow(alpha_sigmaB6_compllambdaP + r6,1.0/6.0);
197  rB2 = rB*rB;
198  rB5 = rB2*rB2*rB;
199  frac2 = 1.0 / rB2;
200  frac6 = frac2 * frac2 * frac2;
201  forceB = frac6 * (ff1B * frac6 - ff2B) * frac2 * rB;
202 
203  real ffactor = complLambdaTI*r5/rA5*forceA + lambdaTI*r5/rB5*forceB;
204  //ffactor = dV/dr * 1/r; force = dV/dr
205  ffactor /= sqrt(distSqr);
206  force = dist * ffactor ;
207  return true;
208  } else {
209  real frac2 = 1.0 / distSqr;
210  real frac6 = frac2 * frac2 * frac2;
211  real ffactor = frac6 * (ff1A * frac6 - ff2A) * frac2; //dV/dr * 1/r
212  force = dist * ffactor;
213  return true;
214  }
215  }
