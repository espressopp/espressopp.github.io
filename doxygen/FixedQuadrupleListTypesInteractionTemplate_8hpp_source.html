109 inline void
110 FixedQuadrupleListTypesInteractionTemplate<_DihedralPotential>::
111 addForces() {
112  LOG4ESPP_INFO(theLogger, "add forces computed by FixedQuadrupleList");
113 
114  const bc::BC &bc = *getSystemRef().bc; // boundary conditions
115 
116  for (FixedQuadrupleList::QuadrupleList::Iterator it(*fixedquadrupleList); it.isValid(); ++it) {
117  Particle &p1 = *it->first;
118  Particle &p2 = *it->second;
119  Particle &p3 = *it->third;
120  Particle &p4 = *it->fourth;
121 
122  longint type1 = p1.type();
123  longint type2 = p2.type();
124  longint type3 = p3.type();
125  longint type4 = p4.type();
126 
127  const Potential &potential = getPotential(type1, type2, type3, type4);
128 
129  Real3D dist21, dist32, dist43;
130 
131  bc.getMinimumImageVectorBox(dist21, p2.position(), p1.position());
132  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
133  bc.getMinimumImageVectorBox(dist43, p4.position(), p3.position());
134 
135  Real3D force1, force2, force3, force4; // result forces
136 
137  potential.computeForce(force1, force2, force3, force4,
138  dist21, dist32, dist43);
139  p1.force() += force1;
140  p2.force() += force2;
141  p3.force() += force3;
142  p4.force() += force4;
143  }
144 }
