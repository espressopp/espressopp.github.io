106  template < typename _Potential > inline void
107  FixedPairDistListInteractionTemplate < _Potential >::addForces() {
108  LOG4ESPP_INFO(theLogger, "add forces computed by the FixedPair List");
109  const bc::BC& bc = *getSystemRef().bc;
110  for (FixedPairDistList::PairList::Iterator it(*fixedPairDistList); it.isValid(); ++it) {
111  Particle &p1 = *it->first;
112  Particle &p2 = *it->second;
113  Real3D r21(0.0,0.0,0.0);
114  bc.getMinimumImageVectorBox(r21, p1.position(), p2.position());
115  Real3D force;
116 
117  real currentDist = fixedPairDistList->getDist(p1.getId(), p2.getId());
118 
119  if(potential->_computeForce(force, r21, currentDist)) {
120  p1.force() += force;
121  p2.force() -= force;
122  }
123  }
124  }

263  template < typename _Potential > inline void
264  FixedPairDistListInteractionTemplate < _Potential >::
265  computeVirialTensor(Tensor *w, int n){
266  LOG4ESPP_INFO(theLogger, "compute the virial tensor for the FixedPair List");
267 
268  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
269  Real3D Li = bc.getBoxL();
270  Tensor *wlocal = new Tensor[n];
271  for(int i=0; i<n; i++) wlocal[i] = Tensor(0.0);
272  for (FixedPairDistList::PairList::Iterator it(*fixedPairDistList);
273  it.isValid(); ++it) {
274  const Particle &p1 = *it->first;
275  const Particle &p2 = *it->second;
276  Real3D p1pos = p1.position();
277  Real3D p2pos = p2.position();
278 
279  int position1 = (int)( n * p1pos[2]/Li[2]);
280  int position2 = (int)( n * p1pos[2]/Li[2]);
281 
282  int maxpos = std::max(position1, position2);
283  int minpos = std::min(position1, position2);
284 
285  Real3D r21(0.0,0.0,0.0);
286  bc.getMinimumImageVectorBox(r21, p1pos, p2pos);
287  real currentDist = fixedPairDistList->getDist(p1.getId(), p2.getId());
288 
289  Real3D force;
290  Tensor ww;
291  if(potential->_computeForce(force, r21, currentDist)) {
292  ww = Tensor(r21, force);
293  }
294 
295  int i = minpos + 1;
296  while(i<=maxpos){
297  wlocal[i] += ww;
298  i++;
299  }
300  }
301 
302  // reduce over all CPUs
303  Tensor *wsum = new Tensor[n];
304  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, n, (double*)&wsum, std::plus<double>());
305 
306  for(int j=0; j<n; j++){
307  w[j] += wsum[j];
308  }
309 
310  delete [] wsum;
311  delete [] wlocal;
312  }
