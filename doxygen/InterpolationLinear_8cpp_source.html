96  // read file, store values in r, e, f, and return number of read lines (called by readRaw())
97  int InterpolationLinear::readFile(const char* file, bool dummy) {
98 
99  char line[MAXLINE];
100  real r, e, f;
101  FILE *fp = fopen(file, "r");
102  if (fp == NULL) {
103  LOG4ESPP_ERROR(theLogger, "could not open file " << file);
104  return 0;
105  }
106 
107  int N = 0;
108 
109  while (1) {
110  if (fgets(line, MAXLINE, fp) == NULL) break;
111  int k = sscanf(line, "%lg %lg %lg", &r, &e, &f);
112  if (k < 3) continue; // do not accept this line
113  if (!dummy) {
114  radius[N] = r;
115  energy[N] = e;
116  force[N] = f;
117  }
118  N++; // increment counter for table entries
119  }
120  fclose(fp);
121 
122  if (dummy) {
123  LOG4ESPP_INFO(theLogger, "found " << N << " table entries file " << file);
124  } else {
125  LOG4ESPP_INFO(theLogger, "read " << N << " table entries in file " << file);
126  }
127 
128  return N;
129  }// readFile
