92  Real3D dist = p1.position() - p2.position();
93  real abs_dist = dist.abs();
94  real sqr_dist = dist.sqr();

96  real forceFactor = prefactor * p1.q() * p2.q() * ( factor * exp( - alpha2 * sqr_dist ) + erfc(alpha*abs_dist) / abs_dist ) / sqr_dist;
