48  real auxCoef; // This is temporary solution for empty potential. This problem
49  // appears when there is a potential for, for example, 00 and
50  // 01, but no potential for 11. The loop in interaction
51  // template will call potential(1, 1) anyway if these particles are

75  void preset() {
76  sqrcutoff = 1.5 * 1.5;
77  pot_border = pow(2.0, 1.0/6.0);
78  sqr_pot_border = pot_border * pot_border;
79  alpha = M_PIl / (2.25 - sqr_pot_border);
80  beta = M_PIl - sqr_pot_border*alpha;
81 
82  one_phi = 1.0 - phi;
83  half_phi = 0.5 * phi;
84  phi_alpha = phi * alpha;
85  }

90  preset();
91  }
92  real getPhi() const { return phi; }

94  real _computeEnergySqrRaw(real distSqr) const {
95  real energy;
96  if(distSqr<=sqr_pot_border){
97  real frac2 = auxCoef / distSqr;
98  real frac6 = frac2 * frac2 * frac2;
99  energy = 4.0 * (frac6 * frac6 - frac6) + one_phi;
100  }
101  else{
102  energy = half_phi * cos(alpha*distSqr+beta) - half_phi;
103  }
104 
105  return energy;
106  }
107 
108  bool _computeForceRaw(Real3D& force,
109  const Real3D& r21,
110  real distSqr) const {
111  real ffactor;
112  if(distSqr<=sqr_pot_border){
113  real frac2 = auxCoef / distSqr;
114  real frac6 = frac2 * frac2 * frac2;
115  ffactor = frac6 * ( 48.0 * frac6 - 24.0 ) * frac2;
116  }
117  else{
118  ffactor = phi_alpha * sin( alpha * distSqr + beta );
119  }
120  force = r21 * ffactor;
121  return true;
122  }
