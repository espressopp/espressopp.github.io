78  void preset() {
79  real sig2 = sigma * sigma;
80  real sig6 = sig2 * sig2 * sig2;
81  r_min = pow(2.0, 1.0/6.0);
82  sqr_r_min = r_min * r_min;
83  sqr_cutoff = getCutoff() * getCutoff();
84  alpha = M_PIl / (sqr_cutoff - sqr_r_min);
85  beta = M_PIl - sqr_r_min * alpha;
86  gamma = -0.5;
87 
88  ff1 = 48.0 * epsilon * sig6 * sig6;
89  ff2 = 24.0 * epsilon * sig6;
90 
91  one_phi = (1.0 - phi) * epsilon;
92  half_phi = 0.5 * phi * epsilon;
93  gamma_phi = gamma * phi * epsilon;
94  alpha_phi = alpha * phi * epsilon;
95  }

112  real _computeEnergySqrRaw(real distSqr) const {
113  real energy;
114  if(distSqr<=sqr_r_min){
115  real frac2 = sigma*sigma*auxCoef / distSqr;
116  real frac6 = frac2 * frac2 * frac2;
117  energy = 4.0 * epsilon * (frac6 * frac6 - frac6) + one_phi;
118  }
119  else{
120  energy = half_phi * cos(alpha * distSqr + beta) + gamma_phi;
121  }
122 
123  return energy;
124  }
125 
126  bool _computeForceRaw(Real3D& force,
127  const Real3D& dist,
128  real distSqr) const {
129  real ffactor;
130  if(distSqr<=sqr_r_min){
131  real frac2 = auxCoef / distSqr;
132  real frac6 = frac2 * frac2 * frac2;
133  ffactor = frac6 * (ff1 * frac6 - ff2) * frac2;
134  }
135  else{
136  ffactor = alpha_phi * sin(alpha * distSqr + beta);
137  }
138  force = dist * ffactor;
139  return true;
140  }
