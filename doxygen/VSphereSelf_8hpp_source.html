119  real _computeEnergySqrRaw(real distSqr) const {
120  real sigma2 = distSqr;
121  real energy;
122  energy = e1*pow(1.0L*M_4PI3*sigma2, 1.0L*mth) \
123  + a1*NbNbNb/(sigma2*sigma2*sigma2) \
124  + a2dNb*sigma2;
125  return energy;
126  }

128  bool _computeForceRaw(Real3D& force,
129  const Real3D& dist,
130  real distSqr) const {
131 
132  real sigma = dist[0];
133  real sigma2 = distSqr;
134  force[1] = force[2] = 0.0;
135  force[0] = M_4PI*e1*pow(1.0L*M_4PI3*sigma2, 1.0L*mfh)*sigma \
136  + a16NbNbNb/(sigma2*sigma2*sigma2*sigma) \
137  - a22dNb*sigma;
138  return true;
