185  real _computeEnergy(const Real3D& r12, const Real3D& r32) const {
186  // 2 is central particle
187  real d12 = r12.abs();
188  real d32 = r32.abs();
189 
190  // probably it should be moved to AngularPotential.hpp
191  // should check does it good for cosine etc.
192  if ( d12 > R + D || d32 > R + D )
193  return 0.0;
194  else {
195  real fA = -B*exp(-lambda2*d12);
196  real fC_j = 0.0;
197  real fC_k = 0.0;
198  if (d12 < R - D)
199  fC_j = 1.0;
200  else {
201  real arg_sin_j = 0.5*Pi_2D*(d12-R);
202  fC_j = 0.5*(1.0 - sin(arg_sin_j));
203  }
204  if (d32 < R - D)
205  fC_k = 1.0;
206  else {
207  real arg_sin_k = 0.5*Pi_2D*(d32-R);
208  fC_k = 0.5*(1.0 - sin(arg_sin_k));
209  }
210  real cosTheta123 = (r12 * r32) / (d12 * d32);
211  real difCos = cosTheta123 - cosTheta0;
212  real difCos2 = difCos * difCos;
213  real g = gamma*(1.0 + c2/d2 - c2/(d2 + difCos2));
214  real zeta = fC_k*g*exp(pow(lambda3*(d12 - d32), m));
215  real b = pow(1.0 + pow(beta*zeta, n), -0.5/n);
216  real energy = fC_j*b*fA;
217 
218  return energy;
219  }

226  bool _computeForceRaw(Real3D& force12, Real3D& force32,
227  const Real3D& r12, const Real3D& r32) const{
228 
229  real d12 = r12.abs();
230  real d32 = r32.abs();
231 
232  if ( d12 > R + D || d32 > R + D ) {
233  force12 = 0.0;
234  force32 = 0.0;
235  return false;
236  }
237  else{
238  real inv_d12 = 1.0 / d12;
239  real inv_d32 = 1.0 / d32;
240 
241  Real3D e12 = r12 * inv_d12;
242  Real3D e32 = r32 * inv_d32;
243 
244  real fA = -B * exp(-lambda2*d12);
245  Real3D D12_fA = -lambda2 * fA * e12;
246  real fC_j = 0.0;
247  Real3D D12_fC_j = 0.0;
248  if (d12 < R - D) {
249  fC_j = 1.0;
250  D12_fC_j = 0.0;
251  }
252  else {
253  real arg_sin_j = 0.5*Pi_2D*(d12-R);
254  fC_j = 0.5*(1.0 - sin(arg_sin_j));
255  D12_fC_j = -0.5 *Pi_2D * cos(arg_sin_j) * e12;
256  }
257  real fC_k = 0.0;
258  Real3D D32_fC_k = 0.0;
259  if (d32 < R - D) {
260  fC_k = 1.0;
261  D32_fC_k = 0.0;
262  }
263  else {
264  real arg_sin_k = 0.5*Pi_2D*(d32-R);
265  fC_k = 0.5*(1.0 - sin(arg_sin_k));
266  D32_fC_k = -0.5 *Pi_2D * cos(arg_sin_k) * e32;
267  }
268 
269 
270  real cosTheta123 = (r12 * r32) * ( inv_d12 * inv_d32 );
271  real difCos = cosTheta123 - cosTheta0;
272  real difCos2 = difCos * difCos;
273 
274  real g = gamma*(1.0 + c2/d2 - c2/(d2 + difCos2));
275  real expTerm = exp(pow(lambda3*(d12 - d32), m));
276  real zeta = fC_k*g*expTerm;
277  real b_base = 1.0 + pow(beta*zeta, n);
278  real b = pow(b_base, -0.5/n);
279 
280  Real3D D12_cosTheta123 = (e32 - e12 * cosTheta123) * inv_d12;
281  Real3D D32_cosTheta123 = (e12 - e32 * cosTheta123) * inv_d32;
282  real factor_D_g = (2.0 * gamma * c2 * difCos)/pow(d2 + difCos2, 2);
283  Real3D D12_g = factor_D_g * D12_cosTheta123;
284  Real3D D32_g = factor_D_g * D32_cosTheta123;
285  real factor_D_expTerm = expTerm * pow(lambda3, m) * m * pow(d12 - d32, m-1);
286  Real3D D12_expTerm = factor_D_expTerm * e12;
287  Real3D D32_expTerm = -factor_D_expTerm * e32;
288 
289  Real3D D12_zeta = fC_k * (expTerm * D12_g + g * D12_expTerm);
290  Real3D D32_zeta = fC_k * (expTerm * D32_g + g * D32_expTerm)
291  + g * expTerm * D32_fC_k;
292  Real3D D12_b = -0.5 * pow(b_base, -0.5/n - 1) * beta * pow(beta*zeta, n-1) * D12_zeta;
293  Real3D D32_b = -0.5 * pow(b_base, -0.5/n - 1) * beta * pow(beta*zeta, n-1) * D32_zeta;
294 
295  force12 = D12_fC_j * b * fA + fC_j * D12_b * fA + fC_j * b * D12_fA;
296 
297  force32 = fC_j * fA * D32_b;
298 
299  return true;
300  }
