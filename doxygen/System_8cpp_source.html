95  void System::addInteraction(shared_ptr< interaction::Interaction > ia){
96  shortRangeInteractions.push_back(ia);
97 
98  // check if the cutoff of this interaction is bigger then maxCutoff
99  real cut = ia->getMaxCutoff();
100 
101  bool isCutoffInf = std::numeric_limits<real>::has_infinity &&
102  cut == std::numeric_limits<real>::infinity();
103 
104  if( !isCutoffInf && cut > maxCutoff) maxCutoff=cut;
105  }

107  void System::removeInteraction(int i){
108  esutil::Error err(comm);
109  if(i>=shortRangeInteractions.size()){
110  std::stringstream msg;
111  msg << "Probably you are trying to remove the interaction "<<i<<
112  " which does not exist. Check your script!";
113  err.setException( msg.str() );
114  }
115  err.checkException();
116 
117  size_t iIter = i;
118  real maxCutoffDelete = shortRangeInteractions[iIter]->getMaxCutoff();
119 
120  shortRangeInteractions.erase(shortRangeInteractions.begin()+i);
121 
122  // check if the maxCutoff changed or not
123  if(maxCutoffDelete>=maxCutoff){
124  maxCutoff=0.0;
125  for (size_t j = 0; j < shortRangeInteractions.size(); j++) {
126  real cut = shortRangeInteractions[j]->getMaxCutoff();
127  maxCutoff = std::max(maxCutoff, cut);
128  }
129  }
130  }
