00107   void System::removeInteraction(int i){
00108     esutil::Error err(comm);
00109     if(i>=shortRangeInteractions.size()){
00110       std::stringstream msg;
00111       msg << "Probably you are trying to remove the interaction "<<i<<
00112               " which does not exist. Check your script!";
00113       err.setException( msg.str() );
00114     }
00115     err.checkException();
00116     
00117     size_t iIter = i;
00118     real maxCutoffDelete = shortRangeInteractions[iIter]->getMaxCutoff();
00119     
00120     shortRangeInteractions.erase(shortRangeInteractions.begin()+i);
00121     
00122     // check if the maxCutoff changed or not
00123     if(maxCutoffDelete>=maxCutoff){
00124       maxCutoff=0.0;
00125       for (size_t j = 0; j < shortRangeInteractions.size(); j++) {
00126         real cut = shortRangeInteractions[j]->getMaxCutoff();
00127         maxCutoff = std::max(maxCutoff, cut);
00128       }
00129     }
00130   }
