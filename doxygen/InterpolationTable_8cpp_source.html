00176     void InterpolationTable::spline(const real *x, const real *y, int n,
00177                                     real yp1, real ypn, real *y2) {
00178       real *u = new real[n];
00179      
00180       if (yp1 > 0.99e30) {
00181         y2[0] = 0.0;
00182         u[0]  = 0.0;
00183       }
00184       else {
00185         y2[0] = -0.5;
00186         u[0]  = (3.0/(x[1]-x[0])) * ((y[1]-y[0]) / (x[1]-x[0]) - yp1);
00187       }
00188      
00189       for (int i = 1; i < n-1; i++) {
00190      
00191         real sig = (x[i] - x[i-1]) / (x[i+1]-x[i-1]);
00192         real p   = sig * y2[i-1] + 2.0;
00193      
00194         y2[i] = (sig-1.0) / p;
00195         u[i]  = (y[i+1] - y[i]) / (x[i+1] - x[i]) -
00196                 (y[i] - y[i-1]) / (x[i] - x[i-1]);
00197         u[i]  = (6.0*u[i] / (x[i+1]-x[i-1]) - sig*u[i-1]) / p;
00198       }
00199      
00200       real qn, un;
00201      
00202       if (ypn > 0.99e30) {
00203         qn = 0.0;
00204         un = 0.0;
00205       } else {
00206         qn = 0.5;
00207         un = (3.0 / (x[n-1]-x[n-2])) * (ypn - (y[n-1]-y[n-2]) / (x[n-1]-x[n-2]));
00208       }
00209      
00210       y2[n-1] = (un - qn * u[n-2]) / (qn * y2[n-2] + 1.0);
00211         
00212       for (int k = n-2; k >= 0; k--) {
00213         y2[k] = y2[k] * y2[k+1] + u[k];
00214       }
00215      
00216       delete [] u;
00217     }// spline
