63  void InterpolationTable::read(mpi::communicator comm, const char* file) {
64  int root = 0; // control processor
65 
66  if (comm.rank() == root) N = readFile(file, true); // dummy read
67 
68  mpi::broadcast(comm, N, root);
69 
70  if (N < 2) {
71  throw std::runtime_error("illegal file for tabulated potential");
72  }
73 
74  radius = new real[N];
75  energy = new real[N];
76  force = new real[N];
77 
78  if (comm.rank() == root) readFile(file, false);
79 
80  mpi::broadcast(comm, radius, N, root);
81  mpi::broadcast(comm, energy, N, root);
82  mpi::broadcast(comm, force, N, root);
83 
84  delta = radius[1] - radius[0];
85 
86  if (delta <= 0.0) {
87  LOG4ESPP_ERROR(theLogger, "illegal distance for entries")
88  }
89 
90  for (int i = 2; i < N; i++) {
91  real r = radius[i] - radius[i-1];
92  if (fabs(r - delta) > 0.0001) {
93  LOG4ESPP_ERROR(theLogger, "delta " << r << " not same as " << delta);
94  }
95  }
96 
97  int nbins = N - 1; // number of intervals is number of points - 1
98 
99  inner = radius[0];
100  real outer = radius[N-1];
101 
102  delta = (outer - inner) / nbins;
103 
104  LOG4ESPP_INFO(theLogger, "tab file has range " << inner << " - "
105  << outer << ", delta = " << delta);
106 
107  invdelta = 1.0 / delta;
108  deltasq6 = delta * delta / 6.0;
109 
110  energy2 = new real[N];
111  force2 = new real[N];
112 
113  // the derivatives of energy are known by the forces
114  real yp1 = -force[0];
115  real ypN = -force[N-1];
116 
117  // the derivatives of energies are approximated (if one uses tables not for energy/force tables)
118  //real yp1 = (energy[1] - energy[0]) / (radius[1] - radius[0]);
119  //real ypN = (energy[N-1] -energy [N-2]) / (radius[N-1] - radius[N-2]);
120 
121  spline(radius, energy, N, yp1, ypN, energy2);
122

176  void InterpolationTable::spline(const real *x, const real *y, int n,
177  real yp1, real ypn, real *y2) {
178  real *u = new real[n];
179 
180  if (yp1 > 0.99e30) {
181  y2[0] = 0.0;
182  u[0] = 0.0;
183  }
184  else {
185  y2[0] = -0.5;
186  u[0] = (3.0/(x[1]-x[0])) * ((y[1]-y[0]) / (x[1]-x[0]) - yp1);
187  }
188 
189  for (int i = 1; i < n-1; i++) {
190 
191  real sig = (x[i] - x[i-1]) / (x[i+1]-x[i-1]);
192  real p = sig * y2[i-1] + 2.0;
193 
194  y2[i] = (sig-1.0) / p;
195  u[i] = (y[i+1] - y[i]) / (x[i+1] - x[i]) -
196  (y[i] - y[i-1]) / (x[i] - x[i-1]);
197  u[i] = (6.0*u[i] / (x[i+1]-x[i-1]) - sig*u[i-1]) / p;
198  }
199 
200  real qn, un;
201 
202  if (ypn > 0.99e30) {
203  qn = 0.0;
204  un = 0.0;
205  } else {
206  qn = 0.5;
207  un = (3.0 / (x[n-1]-x[n-2])) * (ypn - (y[n-1]-y[n-2]) / (x[n-1]-x[n-2]));
208  }
209 
210  y2[n-1] = (un - qn * u[n-2]) / (qn * y2[n-2] + 1.0);
211 
212  for (int k = n-2; k >= 0; k--) {
213  y2[k] = y2[k] * y2[k+1] + u[k];
214  }
215 
216  delete [] u;
217  }// spline

226  real InterpolationTable::splineInterpolation(real r, const real* fn,
227  const real* fn2) const {
228 
229  int index = static_cast<int>((r - inner) * invdelta);
230 
231  if (index < 0) {
232  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
233  << inner << " - " << inner + (N - 1) * delta);
234  index = 0;
235  } else if (index >= N) {
236  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
237  << inner << " - " << inner + (N - 1) * delta);
238  index = N-1;
239  }
240 
241  real b = (r - radius[index]) * invdelta;
242  real a = 1.0 - b;
243  real f = a * fn[index] +
244  b * fn[index+1] +
245  ((a*a*a-a)*fn2[index] +
246  (b*b*b-b)*fn2[index+1]) *
247  deltasq6;
248 
249  // printf("interpoloate %f, a = %f, b = %f, fn = %f - %f, fn2 = %f - %f\n",
250  // r, a, b, fn[index], fn[index+1], fn2[index], fn2[index+1]);
251  return f;
252  }
