67  void Rattle::addBond(int pid1, int pid2, real constraintDist, real mass1, real mass2) {
68  if (mass2 > mass1) {
69  std::ostringstream msg;
70  msg << "In Rattle, the heavy atom should be listed before the hydrogen in each constrained bond" << std::endl;
71  throw std::runtime_error( msg.str() );
72  }
73  ConstrainedBond newbond;
74  newbond.pidHeavy = pid1;
75  newbond.pidHyd = pid2;
76  newbond.constraintDist2 = constraintDist*constraintDist;
77  newbond.invmassHeavy = 1.0/mass1;
78  newbond.invmassHyd = 1.0/mass2;
79  constrainedBonds.insert(std::make_pair(pid2,newbond));
80  constrainedBondsKeys.push_back(pid2);
81  }

83  void Rattle::saveOldPos() {
84  oldPos.clear();
85  System& system = getSystemRef();
86  boost::unordered_map<longint, ConstrainedBond>::iterator it;
87  //collect coordinates of bonds on this CPU
88  //loop over bonds
89  for (it=constrainedBonds.begin(); it != constrainedBonds.end(); it++ ) {
90  longint heavyID = (it->second).pidHeavy;
91  longint lightID = (it->second).pidHyd;
92  Particle* hp = system.storage->lookupAdrATParticle(heavyID);
93  if (hp) { //heavy atom is on this node
94  Particle* lp = system.storage->lookupAdrATParticle(lightID);
95  if (!lp) {
96  std::ostringstream msg;
97  msg << "In Rattle, cannot find light particle " << lightID << ", all light and heavy particles in a group of rigid bonds must be on the same node" << std::endl;
98  throw std::runtime_error( msg.str() );
99  }
100  oldPos.insert(std::make_pair(lightID,
101  std::pair<Real3D,Real3D>(hp->getPos(),lp->getPos())));
102  }
103  }

106  void Rattle::applyPositionConstraints() {
107 
108  int iteration = 0;
109  bool done = false;
110  real dt = integrator->getTimeStep();
111  System& system = getSystemRef();
112  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
113 
114  boost::unordered_map<longint, Real3D> currPosition;
115  boost::unordered_map<longint, Real3D> currVelocity;
116  boost::unordered_map<longint, bool> movedLastTime; //was this particle moved last time?
117  boost::unordered_map<longint, bool> movingThisTime; //is the particle being moved this time?
118 
119  if (oldPos.size() == 0) {return;} //no rigid bonds on this node
120 
121  //loop over bonds on this node
122  OldPos::iterator it;
123  for (it=oldPos.begin(); it != oldPos.end(); it++ ) {
124  longint pidHyd = it->first;
125  longint pidHeavy = constrainedBonds[pidHyd].pidHeavy;
126  //In the maps below, the values for each key pidHeavy may be initialised more than, if pidHeavy is involved in more than one constrained bond. This is not a problem
127  movedLastTime[pidHyd] = true;
128  movedLastTime[pidHeavy] = true;
129  movingThisTime[pidHyd] = false;
130  movingThisTime[pidHeavy] = false;
131  currPosition[pidHyd] = system.storage->lookupAdrATParticle(pidHyd)->getPos();
132  currPosition[pidHeavy] = system.storage->lookupAdrATParticle(pidHeavy)->getPos();
133  currVelocity[pidHyd] = system.storage->lookupAdrATParticle(pidHyd)->getV();
134  currVelocity[pidHeavy] = system.storage->lookupAdrATParticle(pidHeavy)->getV();
135  }
136 
137  //constraint interations
138  while (!done && iteration < maxit) {
139  done = true;
140  //loop over constrained bonds on this cpu
141  for (it=oldPos.begin(); it != oldPos.end(); it++) {
142  //get pids of the two particles in this bond
143  longint a = it->first; //pidHyd
144  longint b = constrainedBonds[a].pidHeavy;
145  if (movedLastTime[a] || movedLastTime[b]) {
146  //compare current distance to desired constraint distance
147  Real3D pab;
148  bc.getMinimumImageVectorBox(pab,currPosition[a],currPosition[b]); //a-b, current positions which change during iterations
149  real pabsq = pab.sqr();
150  real constraint_absq = constrainedBonds[a].constraintDist2;
151  real diffsq = pabsq - constraint_absq;
152  if (fabs(diffsq) > (constraint_absq*tol) ) {
153  //get ab vector before unconstrained position update
154  Real3D rab;
155  bc.getMinimumImageVectorBox(rab,oldPos[a].second,oldPos[a].first); //pos at time t (end of last timestep), a-b;
156  real rab_dot_pab = rab * pab; //r_ab(t) * r_ab,curr(t+dt)
157  if (rab_dot_pab < (constraint_absq*rptol)) { //i.e. if angle is too large
158  std::ostringstream msg;
159  msg << "Constraint failure in RATTLE" << std::endl;
160  throw std::runtime_error( msg.str() );
161  }
162  real rma = constrainedBonds[a].invmassHyd;
163  real rmb = constrainedBonds[a].invmassHeavy;
164  real gab = diffsq / (2.0 * (rma + rmb) * rab_dot_pab);
165  //direct constraint along bond vector at end of previous timestep
166  Real3D displ = gab * rab;
167  currPosition[a] -= rma * displ;
168  currPosition[b] += rmb * displ;
169 
170  displ /= dt;
171  currVelocity[a] -= rma * displ;
172  currVelocity[b] += rmb * displ;
173 
174  movingThisTime[a] = true;
175  movingThisTime[b] = true;
176  done = false;
177  }
178  }
179  }
180  boost::unordered_map<longint, bool>::iterator it3;
181  for (it3 = movedLastTime.begin(); it3 != movedLastTime.end(); it3++) {
182  longint id = it3->first;
183  movedLastTime[id] = movingThisTime[id];
184  movingThisTime[id] = false;
185  }
186 
187  iteration += 1;
188  }
189 
190  if (!done) {
191  std::ostringstream msg;
192  msg << "Too many position constraint iterations in Rattle" << std::endl;
193  throw std::runtime_error( msg.str() );
194  }
195 
196  //store new values for positions
197  boost::unordered_map<longint, Real3D>::iterator it2;
198  for (it2 = currPosition.begin(); it2 != currPosition.end(); it2++) {
199  longint id = it2->first;
200  Particle* p = system.storage->lookupAdrATParticle(id);
201  p->position() = currPosition[id];
202  p->velocity() = currVelocity[id];
203  }
204  }

206  void Rattle::applyVelocityConstraints() {
207 
208  int iteration = 0;
209  bool done = false;
210  real dt = integrator->getTimeStep();
211  System& system = getSystemRef();
212  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
213 
214  boost::unordered_map<longint, Real3D> currPosition;
215  boost::unordered_map<longint, Real3D> currVelocity;
216  boost::unordered_map<longint, bool> changedLastTime; //was this particle velocity changed last time?
217  boost::unordered_map<longint, bool> changingThisTime; //is the particle velocity being changed this time?
218 
219  //get all constrained bonds (light particles) on this CPU
220  //can't use contents of oldPos to identify which constrained bonds are on this CPU because particles may have changed CPU since applyPositionConstraints()
221  lightPart.clear();
222  boost::unordered_map<longint, ConstrainedBond>::iterator it3;
223  for (it3=constrainedBonds.begin(); it3 != constrainedBonds.end(); it3++ ) {
224  longint lightID = (it3->second).pidHyd;
225  Particle* lp = system.storage->lookupAdrATParticle(lightID);
226  if (lp) { //light atom is on this node
227  lightPart.push_back(lightID);
228  }
229  }
230 
231  //loop over constrained bonds on this node
232  std::vector<int>::iterator it;
233  for (it=lightPart.begin(); it != lightPart.end(); it++ ) {
234  longint pidHyd = *it;
235  longint pidHeavy = constrainedBonds[pidHyd].pidHeavy;
236  //In the maps below, the values for each key pidHeavy may be initialised more than, if pidHeavy is involved in more than one constrained bond. This is not a problem
237  changedLastTime[pidHyd] = true;
238  changedLastTime[pidHeavy] = true;
239  changingThisTime[pidHyd] = false;
240  changingThisTime[pidHeavy] = false;
241  currPosition[pidHyd] = system.storage->lookupAdrATParticle(pidHyd)->getPos();
242  currPosition[pidHeavy] = system.storage->lookupAdrATParticle(pidHeavy)->getPos();
243  currVelocity[pidHyd] = system.storage->lookupAdrATParticle(pidHyd)->getV();
244  currVelocity[pidHeavy] = system.storage->lookupAdrATParticle(pidHeavy)->getV();
245  }
246 
247  //constraint interations
248  while (!done && iteration < maxit) {
249  done = true;
250  //loop over constrained bonds on this cpu
251  for (it=lightPart.begin(); it != lightPart.end(); it++) {
252  //get pids of the two particles in this bond
253  longint a = *it; //pidHyd
254  longint b = constrainedBonds[a].pidHeavy;
255  if (changedLastTime[a] || changedLastTime[b]) {
256  Real3D vab = currVelocity[a] - currVelocity[b];
257  Real3D rab;
258  bc.getMinimumImageVectorBox(rab,currPosition[a],currPosition[b]);
259  real rab_dot_vab = rab * vab;
260  real rma = constrainedBonds[a].invmassHyd;
261  real rmb = constrainedBonds[a].invmassHeavy;
262  real constraint_absq = constrainedBonds[a].constraintDist2;
263  real gab = -1.0 * rab_dot_vab / ( (rma + rmb) * constraint_absq);
264  if (fabs(gab) > tol) {
265  Real3D deltav = gab * rab;
266  currVelocity[a] += rma * deltav;
267  currVelocity[b] -= rmb * deltav;
268 
269  changingThisTime[a] = true;
270  changingThisTime[b] = true;
271  done = false;
272  }
273  }
274  }
275 
276  boost::unordered_map<longint, bool>::iterator it3;
277  for (it3 = changedLastTime.begin(); it3 != changedLastTime.end(); it3++) {
278  longint id = it3->first;
279  changedLastTime[id] = changingThisTime[id];
280  changingThisTime[id] = false;
281  }
282 
283  iteration += 1;
284  }
285 
286  if (!done) {
287  std::ostringstream msg;
288  msg << "Too many velocity constraint iterations in Rattle" << std::endl;
289  throw std::runtime_error( msg.str() );
290  }
291 
292  //store new values for velocities
293  boost::unordered_map<longint, Real3D>::iterator it2;
294  for (it2 = currVelocity.begin(); it2 != currVelocity.end(); it2++) {
295  longint id = it2->first;
296  Particle* p = system.storage->lookupAdrATParticle(id);
297  p->velocity() = currVelocity[id];
298  }
299  }
