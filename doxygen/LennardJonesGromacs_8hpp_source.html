90  void preset1() {
91  real sig2 = sigma * sigma;
92  real sig6 = sig2 * sig2 * sig2;
93  ff1 = 48.0 * epsilon * sig6 * sig6;
94  ff2 = 24.0 * epsilon * sig6;
95  ef1 = 4.0 * epsilon * sig6 * sig6;
96  ef2 = 4.0 * epsilon * sig6;
97  r1sq = r1*r1;
98  }

100  void preset2() {
101  //A12 = -((4+12)*cutoff-r1*(1+12))/(pow(cutoff,2+12)*pow(cutoff-r1,2));
102  //B12 = ((3+12)*cutoff-r1*(1+12))/(pow(cutoff,2+12)*pow(cutoff-r1,3));
103  //C12 = 1.0/(12*pow(cutoff,12))-(A12/3)*pow(cutoff-r1,3)-(B12/4)*pow(cutoff-r1,4);
104  //A6 = -((4+6)*cutoff-r1*(1+6))/(pow(cutoff,2+6)*pow(cutoff-r1,2));
105  //B6 = ((3+6)*cutoff-r1*(1+6))/(pow(cutoff,2+6)*pow(cutoff-r1,3));
106  //C6 = 1.0/(6*pow(cutoff,6))-(A6/3)*pow(cutoff-r1,3)-(B6/4)*pow(cutoff-r1,4);
107  real t = cutoff - r1;
108  real r6inv = 1.0/pow(cutoff,6.0);
109  real r8inv = 1.0/pow(cutoff,8.0);
110  real t2inv = 1.0/(t*t);
111  real t3inv = t2inv/t;
112  real t3 = 1.0/t3inv;
113  real a6 = (7.0*r1 - 10.0*cutoff)*r8inv*t2inv;
114  real b6 = (9.0*cutoff - 7.0*r1)*r8inv*t3inv;
115  real a12 = (13.0*r1 - 16.0*cutoff)*r6inv*r8inv*t2inv;
116  real b12 = (15.0*cutoff - 13.0*r1)*r6inv*r8inv*t3inv;
117  real c6 = r6inv - t3*(6.0*a6/3.0 + 6.0*b6*t/4.0);
118  real c12 = r6inv*r6inv - t3*(12.0*a12/3.0 + 12.0*b12*t/4.0);
119  ljsw1 = ff1*a12 - ff2*a6;
120  ljsw2 = ff1*b12 - ff2*b6;
121  ljsw3 = -ef1*12.0*a12/3.0 + ef2*6.0*a6/3.0;
122  ljsw4 = -ef1*12.0*b12/4.0 + ef2*6.0*b6/4.0;
123  ljsw5 = -ef1*c12 + ef2*c6;
124  }

151  real _computeEnergySqrRaw(real distSqr) const {
152  real frac2 = sigma*sigma / distSqr;
153  real frac6 = frac2 * frac2 * frac2;
154  real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
155  if (distSqr > r1sq) {
156  real dr = sqrt(distSqr) - r1;
157  energy += dr*dr*dr*(ljsw3 + ljsw4*dr) + ljsw5;
158  }
159  return energy;
160  }

162  bool _computeForceRaw(Real3D& force,
163  const Real3D& dist,
164  real distSqr) const {
165  real frac2 = 1.0 / distSqr;
166  real frac6 = frac2 * frac2 * frac2;
167  //real ffactor = frac6 * (ff1 * frac6 - ff2) * frac2;
168  real ffactor = frac6 * (ff1 * frac6 - ff2);
169  if (distSqr > r1sq) {
170  real r = sqrt(distSqr);
171  real dr = r - r1;
172  ffactor += r*dr*dr*(ljsw1 + ljsw2*dr);
173  //ffactor += ff1*(A12*pow(dr,2)+B12*pow(dr,3))/r-ff2*(A6*pow(dr,2)+B6*pow(dr,3))/r;
174  }
175  force = dist * ffactor * frac2;
176  return true;
177  }
