00100       void preset2() {
00101         //A12 = -((4+12)*cutoff-r1*(1+12))/(pow(cutoff,2+12)*pow(cutoff-r1,2));
00102         //B12 =  ((3+12)*cutoff-r1*(1+12))/(pow(cutoff,2+12)*pow(cutoff-r1,3));
00103         //C12 = 1.0/(12*pow(cutoff,12))-(A12/3)*pow(cutoff-r1,3)-(B12/4)*pow(cutoff-r1,4);

00107         real t = cutoff - r1;
00108         real r6inv = 1.0/pow(cutoff,6.0);
00109         real r8inv = 1.0/pow(cutoff,8.0);
00110         real t2inv = 1.0/(t*t);
00111         real t3inv = t2inv/t;
00112         real t3 = 1.0/t3inv;
00113         real a6 = (7.0*r1 - 10.0*cutoff)*r8inv*t2inv;
00114         real b6 = (9.0*cutoff - 7.0*r1)*r8inv*t3inv;
00115         real a12 = (13.0*r1 - 16.0*cutoff)*r6inv*r8inv*t2inv;
00116         real b12 = (15.0*cutoff - 13.0*r1)*r6inv*r8inv*t3inv;
00117         real c6 = r6inv - t3*(6.0*a6/3.0 + 6.0*b6*t/4.0);
00118         real c12 = r6inv*r6inv - t3*(12.0*a12/3.0 + 12.0*b12*t/4.0);
00119         ljsw1 = ff1*a12 - ff2*a6;

00122         ljsw4 = -ef1*12.0*b12/4.0 + ef2*6.0*b6/4.0;

00151       real _computeEnergySqrRaw(real distSqr) const {
00152         real frac2 = sigma*sigma / distSqr;
00153         real frac6 = frac2 * frac2 * frac2;
00154         real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
00155         if (distSqr > r1sq) {
00156           real dr = sqrt(distSqr) - r1;
00157           energy += dr*dr*dr*(ljsw3 + ljsw4*dr) + ljsw5;
00158         }
00159         return energy;
