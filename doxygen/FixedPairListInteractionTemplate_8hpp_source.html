109  template < typename _Potential > inline void
110  FixedPairListInteractionTemplate < _Potential >::addForces() {
111  LOG4ESPP_INFO(_Potential::theLogger, "adding forces of FixedPairList");
112  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
113  real ltMaxBondSqr = fixedpairList->getLongtimeMaxBondSqr();
114  for (FixedPairList::PairList::Iterator it(*fixedpairList); it.isValid(); ++it) {
115  Particle &p1 = *it->first;
116  Particle &p2 = *it->second;
117  Real3D dist;
118  bc.getMinimumImageVectorBox(dist, p1.position(), p2.position());
119  Real3D force;
120  real d = dist.sqr();
121  if (d > ltMaxBondSqr) {
122  fixedpairList->setLongtimeMaxBondSqr(d);
123  ltMaxBondSqr = d;
124  }
125  if(potential->_computeForce(force, dist)) {
126  p1.force() += force;
127  p2.force() -= force;
128  LOG4ESPP_DEBUG(_Potential::theLogger, "p" << p1.id() << "(" << p1.position()[0] << "," << p1.position()[1] << "," << p1.position()[2] << ") "
129  << "p" << p2.id() << "(" << p2.position()[0] << "," << p2.position()[1] << "," << p2.position()[2] << ") "
130  << "dist=" << sqrt(dist*dist) << " "
131  << "force=(" << force[0] << "," << force[1] << "," << force[2] << ")" );
132  }
133  }

362  template < typename _Potential > inline void
363  FixedPairListInteractionTemplate < _Potential >::
364  computeVirialTensor(Tensor *w, int n){
365  LOG4ESPP_INFO(theLogger, "compute the virial tensor for the FixedPair List");
366 
367  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
368  Real3D Li = bc.getBoxL();
369  Tensor *wlocal = new Tensor[n];
370  for(int i=0; i<n; i++) wlocal[i] = Tensor(0.0);
371  for (FixedPairList::PairList::Iterator it(*fixedpairList);
372  it.isValid(); ++it) {
373  const Particle &p1 = *it->first;
374  const Particle &p2 = *it->second;
375  Real3D p1pos = p1.position();
376  Real3D p2pos = p2.position();
377 
378  int position1 = (int)( n * p1pos[2]/Li[2]);
379  int position2 = (int)( n * p1pos[2]/Li[2]);
380 
381  int maxpos = std::max(position1, position2);
382  int minpos = std::min(position1, position2);
383 
384  Real3D r21;
385  bc.getMinimumImageVectorBox(r21, p1pos, p2pos);
386  Real3D force;
387  Tensor ww;
388  if(potential->_computeForce(force, r21)) {
389  ww = Tensor(r21, force);
390  }
391 
392  int i = minpos + 1;
393  while(i<=maxpos){
394  wlocal[i] += ww;
395  i++;
396  }
397  }
398 
399  Tensor *wsum = new Tensor[n];
400  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, n, (double*)&wsum, std::plus<double>());
401 
402  for(int j=0; j<n; j++){
403  w[j] += wsum[j];
404  }
405 
406  delete [] wsum;
407  delete [] wlocal;
408  }
