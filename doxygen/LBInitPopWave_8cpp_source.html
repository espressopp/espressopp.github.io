33  void LBInitPopWave::createDenVel (real _rho0, Real3D _u0) {
34  if (mpiWorld->rank()==0) {
35  printf("Creating an initial configuration with uniform density %f and \nharmonic velocity wave "
36  "v_x = %f, v_y = %f and v_z(x) = %f * sin(2 \\pi *i/Nx)\n",
37  _rho0, _u0.getItem(0), _u0.getItem(1), _u0.getItem(2));
38  printf("-------------------------------------\n");
39  } else {
40  // do nothing
41  }
42 
43  real invCs2 = 1. / latticeboltzmann->getCs2();
44 
45  real invCs4 = invCs2*invCs2;
46  real scalp, value;
47  int _offset = latticeboltzmann->getHaloSkin();
48  Int3D _myPos = latticeboltzmann->getMyPos();
49  Int3D _nodeGrid = latticeboltzmann->getNodeGrid();
50  Int3D _Ni = latticeboltzmann->getNi(); // system size in lattice node
51  Int3D _myNi = latticeboltzmann->getMyNi(); // my local nodes
52  Int3D _globalNi = Int3D(0,0,0); // index of the first real node in cpu in the global lattice
53  int _numVels = latticeboltzmann->getNumVels();// number of velocities in the model
54 
55  for (int _dim = 0; _dim < 3; _dim++) {
56  _globalNi[_dim] = floor(_myPos[_dim] * _Ni[_dim] / _nodeGrid[_dim]);
57  }
58 
59  Real3D vel = _u0;
60 
61  // set initial velocity of the populations from Maxwell's distribution
62  for (int i = _offset; i < _myNi.getItem(0) - _offset; i++) {
63  // test the damping of a sin-like initial velocities:
64  _u0 = Real3D(vel.getItem(0),vel.getItem(1),vel.getItem(2) * sin (2. * M_PI * (_globalNi[0] + i - _offset) / _Ni[0]));
65  real trace = _u0*_u0*invCs2;
66  for (int j = _offset; j < _myNi.getItem(1) - _offset; j++) {
67  for (int k = _offset; k < _myNi.getItem(2) - _offset; k++) {
68  for (int l = 0; l < _numVels; l++) {
69  scalp = _u0 * latticeboltzmann->getCi(l);
70  value = 0.5 * latticeboltzmann->getEqWeight(l) * _rho0 * (2. + 2. * scalp * invCs2 + scalp * scalp * invCs4 - trace);
71  latticeboltzmann->setPops( Int3D(i,j,k), l, value );
72  latticeboltzmann->setGhostFluid( Int3D(i,j,k), l, 0.0);
73  }
74 
75  /* fill in den and j values for real and halo regions */
76  latticeboltzmann->setLBMom(Int3D(i,j,k),0,_rho0);
77  latticeboltzmann->setLBMom(Int3D(i,j,k),1,_u0[0]*_rho0);
78  latticeboltzmann->setLBMom(Int3D(i,j,k),2,_u0[1]*_rho0);
79  latticeboltzmann->setLBMom(Int3D(i,j,k),3,_u0[2]*_rho0);
80  }
81  }
82  }
83  latticeboltzmann->copyDenMomToHalo();
84  }
