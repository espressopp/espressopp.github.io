34  void LBInitPopWave::createDenVel (real _rho0, Real3D _u0) {
35  if (mpiWorld->rank()==0) {
36  printf("Creating an initial configuration with uniform density %f and \nharmonic velocity wave "
37  "v_x = %f, v_y = %f and v_z(x) = %f * sin(2 \\pi *i/Nx)\n",
38  _rho0, _u0.getItem(0), _u0.getItem(1), _u0.getItem(2));
39  printf("-------------------------------------\n");
40  } else {
41  // do nothing
42  }
43 
44  real invCs2 = 1. / latticeboltzmann->getCs2();
45 
46  real invCs4 = invCs2*invCs2;
47  real scalp, value;
48  int _offset = latticeboltzmann->getHaloSkin();
49  Int3D _myPosition = latticeboltzmann->getMyPosition();
50  Int3D _nodeGrid = latticeboltzmann->getNodeGrid();
51  Int3D _Ni = latticeboltzmann->getNi(); // system size in lattice node
52  Int3D _myNi = latticeboltzmann->getMyNi(); // my local nodes
53  Int3D _globalNi = Int3D(0,0,0); // index of the first real node in cpu in the global lattice
54  int _numVels = latticeboltzmann->getNumVels();// number of velocities in the model
55 
56  for (int _dim = 0; _dim < 3; _dim++) {
57  _globalNi[_dim] = floor(_myPosition[_dim] * _Ni[_dim] / _nodeGrid[_dim]);
58  }
59 
60  Real3D vel = _u0;
61 
62  // set initial velocity of the populations from Maxwell's distribution
63  for (int i = _offset; i < _myNi.getItem(0) - _offset; i++) {
64  // test the damping of a sin-like initial velocities:
65  _u0 = Real3D(vel.getItem(0),vel.getItem(1),vel.getItem(2) * sin (2. * M_PI * (_globalNi[0] + i - _offset) / _Ni[0]));
66  real trace = _u0*_u0*invCs2;
67  for (int j = _offset; j < _myNi.getItem(1) - _offset; j++) {
68  for (int k = _offset; k < _myNi.getItem(2) - _offset; k++) {
69  for (int l = 0; l < _numVels; l++) {
70  scalp = _u0 * latticeboltzmann->getCi(l);
71  value = 0.5 * latticeboltzmann->getEqWeight(l) * _rho0 * (2. + 2. * scalp * invCs2 + scalp * scalp * invCs4 - trace);
72  latticeboltzmann->setLBFluid(Int3D(i,j,k),l,value);
73  latticeboltzmann->setGhostFluid(Int3D(i,j,k),l,0.0);
74  }
75 
76  /* fill in den and j values for real and halo regions */
77  latticeboltzmann->setLBMom(Int3D(i,j,k),0,_rho0);
78  latticeboltzmann->setLBMom(Int3D(i,j,k),1,_u0[0]*_rho0);
79  latticeboltzmann->setLBMom(Int3D(i,j,k),2,_u0[1]*_rho0);
80  latticeboltzmann->setLBMom(Int3D(i,j,k),3,_u0[2]*_rho0);
81  }
82  }
83  }
84  latticeboltzmann->copyDenMomToHalo();
85  }
