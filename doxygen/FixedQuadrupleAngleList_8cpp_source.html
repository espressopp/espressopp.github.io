00034   /* We assume that the particle with pid2 is a real one, others can be ghosts.
00035    * It is in order to minimize the cize of the cell, because otherwise user should 
00036    * be sure that
00037    * (r4-r1) < cell size
00038    * distance between first and the last particle should be smaller then the cell size.
00039    * Now it is (r3-r1)
00040    * 
00041    * That means that the structure of our quadruple in our local list will be the next
00042    * multimap< pid2, pair<Triple < pid1, pid3, pid4 >, angle> >

00089         err.setException( msg.str() );

00092         std::stringstream msg;

00094         err.setException( msg.str() );
00095       }
00096       if (!p4) {
00097         std::stringstream msg;
00098         msg << "adding error: quadruple particle p4 " << pid4 << " does not exists here and cannot be added";
00099         err.setException( msg.str() );
00100       }
00101     }
00102     
00103     err.checkException();
00104     
00105     if(returnVal){
00106       // add the quadruple locally
00107       this->add(p1, p2, p3, p4);
00108 
00109       Real3D pos1 = p1->position();
00110       Real3D pos2 = p2->position();
00111       Real3D pos3 = p3->position();
00112       Real3D pos4 = p4->position();
00113 
00114       Real3D r21(0.0,0.0,0.0);
00115       system.bc->getMinimumImageVectorBox(r21, pos2, pos1);
00116       Real3D r32(0.0,0.0,0.0);
00117       system.bc->getMinimumImageVectorBox(r32, pos3, pos2);
00118       Real3D r43(0.0,0.0,0.0);
00119       system.bc->getMinimumImageVectorBox(r43, pos4, pos3);
00120 
00121       Real3D rijjk = r21.cross(r32); // [r21 x r32]
00122       Real3D rjkkn = r32.cross(r43); // [r32 x r43]
00123 
00124       real rijjk_sqr = rijjk.sqr();
00125       real rjkkn_sqr = rjkkn.sqr();
00126 
00127       real rijjk_abs = sqrt(rijjk_sqr);
00128       real rjkkn_abs = sqrt(rjkkn_sqr);
00129 
00130       real inv_rijjk = 1.0 / rijjk_abs;
00131       real inv_rjkkn = 1.0 / rjkkn_abs;
00132 
00133       // cosine between planes
00134       real cos_phi = (rijjk * rjkkn) * (inv_rijjk * inv_rjkkn);
00135       if (cos_phi > 1.0) cos_phi = 1.0;
00136       else if (cos_phi < -1.0) cos_phi = -1.0;
00137 
00138       quadruplesAngles.insert(std::make_pair(pid2, 
00139               std::make_pair(Triple<longint, longint, longint>(pid1, pid3, pid4),
00140                             acos(cos_phi))));
00141     }
00142     LOG4ESPP_INFO(theLogger, "added fixed quadruple to global quadruple list");
00143     
00144     return returnVal;
00145   }

00204     
00205     std::vector< longint > toSendInt;
00206     std::vector< real > toSendReal;
00207     // loop over the particle list
00208     for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
00209       longint pid = pit->id();
00210       
00211       // find all quadruples that involve this particle
00212       int n = quadruplesAngles.count(pid);
00213 
00214       if (n > 0) {
00215         std::pair<QuadruplesAngles::const_iterator, 
00216                 QuadruplesAngles::const_iterator> equalRange 
00217                 = quadruplesAngles.equal_range(pid);
00218 
00219         // first write the pid of this particle
00220         // then the number of partners (n)
00221         // and then the pids of the partners
00222         toSendInt.reserve(toSendInt.size()+3*n+1);
00223         toSendReal.reserve(toSendReal.size()+n);
00224         toSendInt.push_back(pid);
00225         toSendInt.push_back(n);
00226         for (QuadruplesAngles::const_iterator it = equalRange.first;
00227             it != equalRange.second; ++it) {
00228           toSendInt.push_back(it->second.first.first);
00229           toSendInt.push_back(it->second.first.second);
00230           toSendInt.push_back(it->second.first.third);
00231           toSendReal.push_back(it->second.second);
00232         }
00233 
00234         // delete all of these quadruples from the global list
00235         quadruplesAngles.erase(pid);
00236       }
00237     }
00238     // send the list
00239     buf.write(toSendInt);
00240     buf.write(toSendReal);
