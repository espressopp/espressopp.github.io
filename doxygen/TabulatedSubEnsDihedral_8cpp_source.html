95  void TabulatedSubEnsDihedral::computeColVarWeights(
96  const Real3D& dist21, const Real3D& dist32,
97  const Real3D& dist43, const bc::BC& bc){
98  // Sanity Check
99  if (weights.getDimension() == 0)
100  throw std::runtime_error("TabulatedSubEnsDihedral requires at least one interaction.");
101  // Compute the weights for each force field
102  // given the reference and instantaneous values of ColVars
103  setColVar(dist21, dist32, dist43, bc);
104  // Compute weights up to next to last FF
105  real maxWeight = 0.;
106  int maxWeightI = 0;
107  // Check first whether we're stuck in a surface
108  bool stuck = false;
109  for (int i=0; i<numInteractions; ++i) {
110  if (weights[i] > maxWeight) {
111  maxWeight = weights[i];
112  maxWeightI = i;
113  }
114  }
115  // Lock the surface until it's sampled >= 98% of the target prob.
116  if (weightCounts > 0 &&
117  maxWeightI < numInteractions-1 &&
118  maxWeight > 0. &&
119  weightSum[maxWeightI]/weightCounts < 0.98*targetProb[maxWeightI])
120  stuck = true;
121  if (!stuck) {
122  maxWeight = 0.;
123  maxWeightI = numInteractions-1;
124  for (int i=0; i<numInteractions-1; ++i) {
125  weights[i] = 1.;
126  real norm_d_i = 0.;
127  real norm_l_i = 0.;
128  for (int j=0; j<colVar.getDimension(); ++j) {
129  int k = 0;
130  // Choose between dihed, bonds, and angles
131  if (j <= 0+colVarDihedListSize) k = 0;
132  else if (j<1+colVarDihedListSize+colVarBondListSize) k = 1;
133  else k = 2;
134  real diff = colVar[j] - colVarRef[i][k];
135  if (k == 0) {
136  if (diff>M_PI) diff -= 2.0*M_PI;
137  if (diff<(-1.0*M_PI)) diff += 2.0*M_PI;
138  }
139  norm_d_i += pow(diff / colVarSd[k], 2);
140  norm_l_i += pow(colVarRef[i][3+k], 2);
141  }
142  if (norm_d_i > norm_l_i)
143  weights[i] = exp(- (sqrt(norm_d_i) - sqrt(norm_l_i)) / alpha);
144  if (weights[i] > maxWeight) {
145  maxWeight = weights[i];
146  maxWeightI = i;
147  }
148  }
149  for (int i=0; i<numInteractions-1; ++i) {
150  if (i != maxWeightI)
151  weights[i] = 0.;
152  else {
153  if (weightCounts > 0 &&
154  weights[i] > 0.01 &&
155  weightSum[i]/weightCounts < 0.98*targetProb[i]) {
156  weights[i] = 1.;
157  maxWeight = 1.;
158  }
159  }
160  }
161  if (maxWeightI == numInteractions-1)
162  maxWeight = 1.;
163  weights[numInteractions-1] = 1. - maxWeight;
164  }
165 
166  // Update weightSum
167  for (int i=0; i<numInteractions; ++i)
168  weightSum[i] += weights[i];
169  weightCounts += 1;
170  }

172  // Collective variables
173  void TabulatedSubEnsDihedral::setColVar(
174  const Real3D& r21, const Real3D& r32,
175  const Real3D& r43, const bc::BC& bc) {
176  colVar.setDimension(1);
177  colVar[0] = computePhi(r21, r32, r43);
178  int i=1;
179  if (colVarDihedList != nullptr) {
180  colVarDihedListSize = colVarDihedList->size();
181  // Now all dihedrals in colVarDihedList
182  for (FixedQuadrupleList::QuadrupleList::Iterator it(*colVarDihedList); it.isValid(); ++it) {
183  colVar.setDimension(i+1);
184  Particle &p1 = *it->first;
185  Particle &p2 = *it->second;
186  Particle &p3 = *it->third;
187  Particle &p4 = *it->fourth;
188  Real3D r21, r32, r43;
189  bc.getMinimumImageVectorBox(r21, p2.position(), p1.position());
190  bc.getMinimumImageVectorBox(r32, p3.position(), p2.position());
191  bc.getMinimumImageVectorBox(r43, p4.position(), p3.position());
192  colVar[i] = DihedralPotential::computePhi(r21, r32, r43);
193  i+=1;
194  }
195  }
196  if (colVarBondList != nullptr) {
197  // Now all bonds in colVarBondList
198  colVarBondListSize = colVarBondList->size();
199  for (FixedPairList::PairList::Iterator it(*colVarBondList); it.isValid(); ++it) {
200  colVar.setDimension(i+1);
201  Particle &p1 = *it->first;
202  Particle &p2 = *it->second;
203  Real3D dist12;
204  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
205  colVar[i] = sqrt(dist12 * dist12);
206  i+=1;
207  }
208  }
209  if (colVarAngleList != nullptr) {
210  // Now all angles in colVarAngleList
211  colVarAngleListSize = colVarAngleList->size();
212  for (FixedTripleList::TripleList::Iterator it(*colVarAngleList); it.isValid(); ++it) {
213  colVar.setDimension(i+1);
214  Particle &p1 = *it->first;
215  Particle &p2 = *it->second;
216  Particle &p3 = *it->third;
217  Real3D dist12, dist32;
218  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
219  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
220  real dist12_sqr = dist12 * dist12;
221  real dist32_sqr = dist32 * dist32;
222  real dist1232 = sqrt(dist12_sqr) * sqrt(dist32_sqr);
223  real cos_theta = dist12 * dist32 / dist1232;
224  colVar[i] = acos(cos_theta);
225  i+=1;
226  }
227  }
228  }
