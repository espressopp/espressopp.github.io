39  real Pressure::compute() const {
40 
41  System& system = getSystemRef();
42  mpi::communicator communic = *system.comm;
43 
44  // determine volume of the box
45  Real3D Li = system.bc->getBoxL();
46  real tripleV = 3.0 * Li[0] * Li[1] * Li[2];
47 
48  // compute the kinetic contriubtion (2/3 \sum 1/2mv^2)
49  real p_kinetic;
50  real v2sum = 0.0;
51  real v2 = 0.0;
52 
53  CellList realCells = system.storage->getRealCells();
54 
55  if (system.storage->getFixedTuples()){ // AdResS - hence, need to distinguish between CG and AT particles.
56 
57  shared_ptr<FixedTupleListAdress> fixedtupleList=system.storage->getFixedTuples();
58  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
59  Particle &vp = *cit;
60  FixedTupleListAdress::iterator it2;
61  it2 = fixedtupleList->find(&vp);
62 
63  if (it2 != fixedtupleList->end()) { // Are there atomistic particles for given CG particle? If yes, use those for calculation.
64  std::vector<Particle*> atList;
65  atList = it2->second;
66  for (std::vector<Particle*>::iterator it3 = atList.begin();
67  it3 != atList.end(); ++it3) {
68  Particle &at = **it3;
69  v2 += at.mass() * (at.velocity() * at.velocity());
70  }
71  }
72 
73  else{ // If not, use CG particle itself for calculation.
74  v2 += vp.mass() * (vp.velocity() * vp.velocity());
75  }
76  }
77 
78  }
79  else{
80 
81  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
82  const Particle& p = *cit;
83  v2 += p.mass() * (p.velocity() * p.velocity());
84  }
85 
86  }
87  boost::mpi::all_reduce(*getSystem()->comm, v2, v2sum, std::plus<real>());
88  //boost::mpi::all_reduce( communic, v2, v2sum, std::plus<real>());
89  p_kinetic = v2sum;
90 
91  // compute the short-range nonbonded contribution
92  real rij_dot_Fij = 0.0;
93  const InteractionList& srIL = system.shortRangeInteractions;
94  for (size_t j = 0; j < srIL.size(); j++) {
95  rij_dot_Fij += srIL[j]->computeVirial();
96  //std::cout << "srIL[" << j << "]: " << srIL[j]->computeVirial() << "\n";
97  }
98 
99  real p_nonbonded = rij_dot_Fij;
100 
101  //DEBUG (This pressure calculation seems incorrect at least for liquid water.
102  // Be careful when doing system with electrostatic interactions)
103  //return (p_kinetic + p_nonbonded) / tripleV; TEST FOR CONSTRAINTS
104  /*std::cout << "tripleV: " << tripleV << "\n";
105  std::cout << "p_kinetic: " << p_kinetic << "\n";
106  std::cout << "p_nonbonded: " << p_nonbonded << "\n";
107  std::cout << "p_kinetic/tripleV: " << p_kinetic/tripleV << "\n";
108  std::cout << "p_nonbonded/tripleV: " << p_nonbonded/tripleV << "\n"; */
109  //DEBUG
110 
111  return (p_kinetic + p_nonbonded) / tripleV;
112  //return ((p_kinetic/tripleV)*(3.0/2.0) + p_nonbonded/tripleV); //SETTLE CONSTAINTS
113  }
