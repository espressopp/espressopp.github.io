00097                 /* CALCULATION OF THE LOCAL MOMENTS */
00098                 void LBSite::calcLocalMoments (real *m) {
00099                         real f0,
00100                         f1p2, f1m2, f3p4, f3m4, f5p6, f5m6, f7p8, f7m8, f9p10, f9m10,
00101                         f11p12, f11m12, f13p14, f13m14, f15p16, f15m16, f17p18, f17m18;
00102                         
00103                         /* shorthand functions for "simplified" notation */
00104                         f0     =  getF_i(0);
00105                         f1p2   =  getF_i(1) +  getF_i(2);    f1m2 =  getF_i(1) -  getF_i(2);
00106                         f3p4   =  getF_i(3) +  getF_i(4);    f3m4 =  getF_i(3) -  getF_i(4);
00107                         f5p6   =  getF_i(5) +  getF_i(6);    f5m6 =  getF_i(5) -  getF_i(6);
00108                         f7p8   =  getF_i(7) +  getF_i(8);    f7m8 =  getF_i(7) -  getF_i(8);
00109                         f9p10  =  getF_i(9) + getF_i(10);   f9m10 =  getF_i(9) - getF_i(10);
00110                         f11p12 = getF_i(11) + getF_i(12);  f11m12 = getF_i(11) - getF_i(12);
00111                         f13p14 = getF_i(13) + getF_i(14);  f13m14 = getF_i(13) - getF_i(14);
00112                         f15p16 = getF_i(15) + getF_i(16);  f15m16 = getF_i(15) - getF_i(16);
00113                         f17p18 = getF_i(17) + getF_i(18);  f17m18 = getF_i(17) - getF_i(18);
00114                         
00115                         /* mass mode */
00116                         m[0] = f0 + f1p2 + f3p4 + f5p6 + f7p8 + f9p10 + f11p12 + f13p14 + f15p16 + f17p18;
00117                         
00118                         /* momentum modes */
00119                         m[1] = f1m2 +   f7m8 +  f9m10 + f11m12 + f13m14;
00120                         m[2] = f3m4 +   f7m8 -  f9m10 + f15m16 + f17m18;
00121                         m[3] = f5m6 + f11m12 - f13m14 + f15m16 - f17m18;
00122                         
00123                         /* stress modes */
00124                         m[4] = -f0 +   f7p8 + f9p10 + f11p12 + f13p14 + f15p16 + f17p18;
00125                         m[5] = 2.*f1p2 -   f3p4 -  f5p6 +   f7p8 +  f9p10 + f11p12 + f13p14 - 2.* (f15p16 + f17p18);
00126                         m[6] = f3p4 -   f5p6 +  f7p8 +  f9p10 - f11p12 - f13p14;
00127                         m[7] = f7p8 -  f9p10;
00128                         m[8] = f11p12 - f13p14;
00129                         m[9] = f15p16 - f17p18;
00130                         
00131                         /* kinetic (ghost) modes */
00132                         m[10] = -2.* f1m2 +   f7m8 +  f9m10 + f11m12 + f13m14;
00133                         m[11] = -2.* f3m4 +   f7m8 -  f9m10 + f15m16 + f17m18;
00134                         m[12] = -2.* f5m6 + f11m12 - f13m14 + f15m16 - f17m18;
00135                         m[13] = f7m8 +  f9m10 - f11m12 - f13m14;
00136                         m[14] = -f7m8 +  f9m10 + f15m16 + f17m18;
00137                         m[15] = f11m12 - f13m14 - f15m16 + f17m18;
00138                         m[16] = f0 - 2.* (f1p2 + f3p4 + f5p6) +   f7p8 +  f9p10
00139                         + f11p12 + f13p14 + f15p16 + f17p18;
00140                         m[17] = -2.* f1p2 +   f3p4 +   f5p6 +   f7p8 +  f9p10 + f11p12
00141                         + f13p14 -    2.* (f15p16 + f17p18);
00142                         m[18] = -f3p4 +   f5p6 +   f7p8 +  f9p10 - f11p12 - f13p14;
00143                 }

00147                 /* RELAXATION OF THE MOMENTS TO THEIR EQUILIBRIUM VALUES */
00148                 void LBSite::relaxMoments (real *m, int _extForceFlag, Real3D _f) {
00149                         // moments on the site //
00150                         real _invTauLoc = 1. / LatticePar::getTauLoc();
00151                         Real3D jLoc(m[1], m[2], m[3]);
00152                         jLoc *= LatticePar::getALoc();
00153                         jLoc *= _invTauLoc;
00154                         
00155                         // if we have external forces then modify the eq.fluxes //
00156                         if (_extForceFlag == 1) jLoc += 0.5 * _f; // when doing coupling, the flag is set to 1!
00157                         
00158                         real _invRhoLoc = 1. / m[0];
00159                         real pi_eq[6];
00160                         
00161                         pi_eq[0] =  jLoc.sqr()*_invRhoLoc;
00162                         pi_eq[1] =  (jLoc[0]*jLoc[0] - jLoc[1]*jLoc[1])*_invRhoLoc;
00163                         pi_eq[2] =  (3.*jLoc[0]*jLoc[0] - jLoc.sqr())*_invRhoLoc;
00164                         pi_eq[3] =  jLoc[0]*jLoc[1]*_invRhoLoc;
00165                         pi_eq[4] =  jLoc[0]*jLoc[2]*_invRhoLoc;
00166                         pi_eq[5] =  jLoc[1]*jLoc[2]*_invRhoLoc;
00167                         
00168                         real _gamma_b = getGammaBLoc();
00169                         real _gamma_s = getGammaSLoc();
00170                         real _gamma_odd = getGammaOddLoc();
00171                         real _gamma_even = getGammaEvenLoc();
00172                         
00173                         /* relax bulk mode */
00174                         m[4] = pi_eq[0] + _gamma_b * (m[4] - pi_eq[0]);
00175                         
00176                         /* relax shear modes */
00177                         m[5] = pi_eq[1] + _gamma_s * (m[5] - pi_eq[1]);
00178                         m[6] = pi_eq[2] + _gamma_s * (m[6] - pi_eq[2]);
00179                         m[7] = pi_eq[3] + _gamma_s * (m[7] - pi_eq[3]);
00180                         m[8] = pi_eq[4] + _gamma_s * (m[8] - pi_eq[4]);
00181                         m[9] = pi_eq[5] + _gamma_s * (m[9] - pi_eq[5]);
00182                         
00183                         /* relax odd modes */
00184                         m[10] *= _gamma_odd; m[11] *= _gamma_odd;       m[12] *= _gamma_odd;
00185                         m[13] *= _gamma_odd; m[14] *= _gamma_odd;       m[15] *= _gamma_odd;
00186                         
00187                         /* relax even modes */
00188                         m[16] *= _gamma_even; m[17] *= _gamma_even; m[18] *= _gamma_even;
00189                 }

00194                 void LBSite::thermalFluct (real *m) {
00195                         /* values of PhiLoc were already set in LatticeBoltzmann.cpp */
00196                         int _numVelsLoc = LatticePar::getNumVelsLoc();
00197                         real rootRhoLoc = sqrt(12.*m[0]); // factor 12. comes from usage of
00198                         // not gaussian but uniformly distributed random numbers
00199                         
00200                         for (int l = 4; l < _numVelsLoc; l++) {
00201                                 m[l] += rootRhoLoc*getPhiLoc(l)*((*LatticePar::rng)() - 0.5);
00202                         }
00203                         /*      for normally distributed random numbers change
00204                          in the previous lines sqrt(12.*getM_i(0)) -> sqrt(getM_i(0)) and
00205                          rootRhoLoc*getPhiLoc(l)*((*LatticePar::rng)() - 0.5)) -> 
00206                          rootRhoLoc*getPhiLoc(l)*(*LatticePar::normal())
00207                          */

00212                 void LBSite::applyForces (real *m, Real3D _f) {
00213 //                      Real3D _f = LBForce().getExtForceLoc() + LBForce().getCouplForceLoc();
00214                         
00215                         // set velocity _u
00216                         Real3D _u = _f;
00217                         _u *= 0.5;
00218                         _u[0] += m[1];  _u[1] += m[2]; _u[2] += m[3];
00219                         _u /= m[0];
00220                         
00221                         /* update momentum modes */
00222                         m[1] += _f[0];
00223                         m[2] += _f[1];
00224                         m[3] += _f[2];
00225                         
00226                         /* update stress modes */
00227                         // See def. of _sigma (Eq.198) in B.Dünweg & A.J.C.Ladd in Adv.Poly.Sci. 221, 89-166 (2009)
00228                         real _sigma[6];
00229                         real _gamma_b = getGammaBLoc();
00230                         real _gamma_s = getGammaSLoc();
00231                         real _gamma_sp = _gamma_s + 1.;
00232                         real _gamma_sph = 0.5 * _gamma_sp;
00233                         
00234                         real _scalp = _u*_f;
00235                         real _secTerm = (1./3.)*(_gamma_b - _gamma_s)*_scalp;
00236                         
00237                         _sigma[0] = _gamma_sp*_u[0]*_f[0] + _secTerm;
00238                         _sigma[1] = _gamma_sp*_u[1]*_f[1] + _secTerm;
00239                         _sigma[2] = _gamma_sp*_u[2]*_f[2] + _secTerm;
00240                         _sigma[3] = _gamma_sph*(_u[0]*_f[1]+_u[1]*_f[0]);
00241                         _sigma[4] = _gamma_sph*(_u[0]*_f[2]+_u[2]*_f[0]);
00242                         _sigma[5] = _gamma_sph*(_u[1]*_f[2]+_u[2]*_f[1]);
00243                         
00244                         m[4] += _sigma[0]+_sigma[1]+_sigma[2];
00245                         m[5] += 2.*_sigma[0]-_sigma[1]-_sigma[2];
00246                         m[6] += _sigma[1]-_sigma[2];
00247                         m[7] += _sigma[3];
00248                         m[8] += _sigma[4];
00249                         m[9] += _sigma[5];
00250                 }
