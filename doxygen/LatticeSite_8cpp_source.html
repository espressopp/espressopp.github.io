97  /* CALCULATION OF THE LOCAL MOMENTS */
98  void LBSite::calcLocalMoments (real *m) {
99  real f0,
100  f1p2, f1m2, f3p4, f3m4, f5p6, f5m6, f7p8, f7m8, f9p10, f9m10,
101  f11p12, f11m12, f13p14, f13m14, f15p16, f15m16, f17p18, f17m18;
102 
103  /* shorthand functions for "simplified" notation */
104  f0 = getF_i(0);
105  f1p2 = getF_i(1) + getF_i(2); f1m2 = getF_i(1) - getF_i(2);
106  f3p4 = getF_i(3) + getF_i(4); f3m4 = getF_i(3) - getF_i(4);
107  f5p6 = getF_i(5) + getF_i(6); f5m6 = getF_i(5) - getF_i(6);
108  f7p8 = getF_i(7) + getF_i(8); f7m8 = getF_i(7) - getF_i(8);
109  f9p10 = getF_i(9) + getF_i(10); f9m10 = getF_i(9) - getF_i(10);
110  f11p12 = getF_i(11) + getF_i(12); f11m12 = getF_i(11) - getF_i(12);
111  f13p14 = getF_i(13) + getF_i(14); f13m14 = getF_i(13) - getF_i(14);
112  f15p16 = getF_i(15) + getF_i(16); f15m16 = getF_i(15) - getF_i(16);
113  f17p18 = getF_i(17) + getF_i(18); f17m18 = getF_i(17) - getF_i(18);
114 
115  /* mass mode */
116  m[0] = f0 + f1p2 + f3p4 + f5p6 + f7p8 + f9p10 + f11p12 + f13p14 + f15p16 + f17p18;
117 
118  /* momentum modes */
119  m[1] = f1m2 + f7m8 + f9m10 + f11m12 + f13m14;
120  m[2] = f3m4 + f7m8 - f9m10 + f15m16 + f17m18;
121  m[3] = f5m6 + f11m12 - f13m14 + f15m16 - f17m18;
122 
123  /* stress modes */
124  m[4] = -f0 + f7p8 + f9p10 + f11p12 + f13p14 + f15p16 + f17p18;
125  m[5] = 2.*f1p2 - f3p4 - f5p6 + f7p8 + f9p10 + f11p12 + f13p14 - 2.* (f15p16 + f17p18);
126  m[6] = f3p4 - f5p6 + f7p8 + f9p10 - f11p12 - f13p14;
127  m[7] = f7p8 - f9p10;
128  m[8] = f11p12 - f13p14;
129  m[9] = f15p16 - f17p18;
130 
131  /* kinetic (ghost) modes */
132  m[10] = -2.* f1m2 + f7m8 + f9m10 + f11m12 + f13m14;
133  m[11] = -2.* f3m4 + f7m8 - f9m10 + f15m16 + f17m18;
134  m[12] = -2.* f5m6 + f11m12 - f13m14 + f15m16 - f17m18;
135  m[13] = f7m8 + f9m10 - f11m12 - f13m14;
136  m[14] = -f7m8 + f9m10 + f15m16 + f17m18;
137  m[15] = f11m12 - f13m14 - f15m16 + f17m18;
138  m[16] = f0 - 2.* (f1p2 + f3p4 + f5p6) + f7p8 + f9p10
139  + f11p12 + f13p14 + f15p16 + f17p18;
140  m[17] = -2.* f1p2 + f3p4 + f5p6 + f7p8 + f9p10 + f11p12
141  + f13p14 - 2.* (f15p16 + f17p18);
142  m[18] = -f3p4 + f5p6 + f7p8 + f9p10 - f11p12 - f13p14;
143  }

147  /* RELAXATION OF THE MOMENTS TO THEIR EQUILIBRIUM VALUES */
148  void LBSite::relaxMoments (real *m, int _extForceFlag, Real3D _f) {
149  // moments on the site //
150  real _invTauLoc = 1. / LatticePar::getTauLoc();
151  Real3D jLoc(m[1], m[2], m[3]);
152  jLoc *= LatticePar::getALoc();
153  jLoc *= _invTauLoc;
154 
155  // if we have external forces then modify the eq.fluxes //
156  if (_extForceFlag == 1) jLoc += 0.5 * _f; // when doing coupling, the flag is set to 1!
157 
158  real _invRhoLoc = 1. / m[0];
159  real pi_eq[6];
160 
161  pi_eq[0] = jLoc.sqr()*_invRhoLoc;
162  pi_eq[1] = (jLoc[0]*jLoc[0] - jLoc[1]*jLoc[1])*_invRhoLoc;
163  pi_eq[2] = (3.*jLoc[0]*jLoc[0] - jLoc.sqr())*_invRhoLoc;
164  pi_eq[3] = jLoc[0]*jLoc[1]*_invRhoLoc;
165  pi_eq[4] = jLoc[0]*jLoc[2]*_invRhoLoc;
166  pi_eq[5] = jLoc[1]*jLoc[2]*_invRhoLoc;
167 
168  real _gamma_b = getGammaBLoc();
169  real _gamma_s = getGammaSLoc();
170  real _gamma_odd = getGammaOddLoc();
171  real _gamma_even = getGammaEvenLoc();
172 
173  /* relax bulk mode */
174  m[4] = pi_eq[0] + _gamma_b * (m[4] - pi_eq[0]);
175 
176  /* relax shear modes */
177  m[5] = pi_eq[1] + _gamma_s * (m[5] - pi_eq[1]);
178  m[6] = pi_eq[2] + _gamma_s * (m[6] - pi_eq[2]);
179  m[7] = pi_eq[3] + _gamma_s * (m[7] - pi_eq[3]);
180  m[8] = pi_eq[4] + _gamma_s * (m[8] - pi_eq[4]);
181  m[9] = pi_eq[5] + _gamma_s * (m[9] - pi_eq[5]);
182 
183  /* relax odd modes */
184  m[10] *= _gamma_odd; m[11] *= _gamma_odd; m[12] *= _gamma_odd;
185  m[13] *= _gamma_odd; m[14] *= _gamma_odd; m[15] *= _gamma_odd;
186 
187  /* relax even modes */
188  m[16] *= _gamma_even; m[17] *= _gamma_even; m[18] *= _gamma_even;
189  }

193  /* ADDING THERMAL FLUCTUATIONS */
194  void LBSite::thermalFluct (real *m) {
195  /* values of PhiLoc were already set in LatticeBoltzmann.cpp */
196  int _numVelsLoc = LatticePar::getNumVelsLoc();
197  real rootRhoLoc = sqrt(12.*m[0]); // factor 12. comes from usage of
198  // not gaussian but uniformly distributed random numbers
199 
200  for (int l = 4; l < _numVelsLoc; l++) {
201  m[l] += rootRhoLoc*getPhiLoc(l)*((*LatticePar::rng)() - 0.5);
202  }
203  /* for normally distributed random numbers change
204  in the previous lines sqrt(12.*getM_i(0)) -> sqrt(getM_i(0)) and
205  rootRhoLoc*getPhiLoc(l)*((*LatticePar::rng)() - 0.5)) ->
206  rootRhoLoc*getPhiLoc(l)*(*LatticePar::normal())
207  */
208  }

212  void LBSite::applyForces (real *m, Real3D _f) {
213 // Real3D _f = LBForce().getExtForceLoc() + LBForce().getCouplForceLoc();
214 
215  // set velocity _u
216  Real3D _u = _f;
217  _u *= 0.5;
218  _u[0] += m[1]; _u[1] += m[2]; _u[2] += m[3];
219  _u /= m[0];
220 
221  /* update momentum modes */
222  m[1] += _f[0];
223  m[2] += _f[1];
224  m[3] += _f[2];
225 
226  /* update stress modes */
227  // See def. of _sigma (Eq.198) in B.Dünweg & A.J.C.Ladd in Adv.Poly.Sci. 221, 89-166 (2009)
228  real _sigma[6];
229  real _gamma_b = getGammaBLoc();
230  real _gamma_s = getGammaSLoc();
231  real _gamma_sp = _gamma_s + 1.;
232  real _gamma_sph = 0.5 * _gamma_sp;
233 
234  real _scalp = _u*_f;
235  real _secTerm = (1./3.)*(_gamma_b - _gamma_s)*_scalp;
236 
237  _sigma[0] = _gamma_sp*_u[0]*_f[0] + _secTerm;
238  _sigma[1] = _gamma_sp*_u[1]*_f[1] + _secTerm;
239  _sigma[2] = _gamma_sp*_u[2]*_f[2] + _secTerm;
240  _sigma[3] = _gamma_sph*(_u[0]*_f[1]+_u[1]*_f[0]);
241  _sigma[4] = _gamma_sph*(_u[0]*_f[2]+_u[2]*_f[0]);
242  _sigma[5] = _gamma_sph*(_u[1]*_f[2]+_u[2]*_f[1]);
243 
244  m[4] += _sigma[0]+_sigma[1]+_sigma[2];
245  m[5] += 2.*_sigma[0]-_sigma[1]-_sigma[2];
246  m[6] += _sigma[1]-_sigma[2];
247  m[7] += _sigma[3];
248  m[8] += _sigma[4];
249  m[9] += _sigma[5];
250  }

264

266  setF_i(1,m[0] +m[1] + 2.* (m[5] -m[10] -m[16] -m[17]));
267  setF_i(2,m[0] -m[1] + 2.* (m[5] +m[10] -m[16] -m[17]));
