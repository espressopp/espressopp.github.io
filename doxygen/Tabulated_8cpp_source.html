49  void Tabulated::setFilename(int itype, const char* _filename) {
50  boost::mpi::communicator world;
51  filename = _filename;
52 
53  if (itype == 1) { // create a new InterpolationLinear
54  table = make_shared <InterpolationLinear> ();
55  table->read(world, _filename);
56  }
57 
58  else if (itype == 2) { // create a new InterpolationAkima
59  table = make_shared <InterpolationAkima> ();

61  }
62 
63  else if (itype == 3) { // create a new InterpolationCubic
