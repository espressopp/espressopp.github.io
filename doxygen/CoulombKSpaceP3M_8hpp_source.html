319  void aliasing_sum(Int3D i, Real3D *nominator, real *denominator){
320  /*
321  * taken from Deserno's code
322  * calculates the aliasing sums in the nominator and denominator of the
323  expression for the optimal influence function (see Hockney/Eastwood:
324  8-22, p. 275).
325  N: is the n-vector for which the aliasing sum is to be performed.
326  *nominatorX,*nominatorY,*nominatorZ : x-, y-, and z-component of the
327  aliasing sum.
328  *denominator : aliasing sum in the denominator. */
329 
330  Real3D ms(mesh_shift[0][i[0]], mesh_shift[1][i[1]], mesh_shift[2][i[2]]);
331 
332  real sc;
333  Real3D nml(0.0);
334 
335  Real3D Minv;
336  for(int i=0;i<3;i++) Minv[i] = 1.0/(real)M[i];
337 
338  real ftr = pow( M_PIl / (alpha*sysL[0]), 2);
339 
340  *denominator = 0.0;
341  *nominator = Real3D(0.0);
342 
343  Int3D ijk(0);
344  Real3D x(0.0);
345  for ( ijk[0] = -brillouin; ijk[0] <= brillouin; ijk[0]++) {
346  for ( ijk[1] = -brillouin; ijk[1] <= brillouin; ijk[1]++) {
347  for ( ijk[2] = -brillouin; ijk[2] <= brillouin; ijk[2]++) {
348 
349  for(int i=0; i<3; i++){
350  nml[i] = ms[i] + M[i] * ijk[i];
351  x[i] = Minv[i] * nml[i];
352  }
353 
354  sc = pow( sinc(x), 2.0*P);
355 
356  *denominator += sc;
357 
358  real nml2 = nml.sqr();
359  real e = ftr * nml2;
360 
361  *nominator += ( ( e<30 ) ? sc*exp(-e)/nml2 : 0.0 ) * nml;
362  }
363  }
364  }
365  }

382  real sinc(real x){
383  real epsi = 0.1;
384 
385  real c2 = -0.1666666666667e-0;
386  real c4 = 0.8333333333333e-2;
387  real c6 = -0.1984126984127e-3;
388  real c8 = 0.2755731922399e-5;
389 
390  real PIx = M_PIl*x, PIx2;
391 
392  if (fabs(x)>epsi){
393  return sin(PIx)/PIx;
394  }
395  else {
396  PIx2 = PIx * PIx;
397  return 1.0 + PIx2*(c2+PIx2*(c4+PIx2*(c6+PIx2*c8)));
398  }

632  real asignment_f2(int i, real x, int P) {
633  mpi::communicator communic = *system->comm;
634  int this_node = communic.rank();
635  switch (P) {
636  case 1 : return 1.0;
637  case 2 : {
638  switch (i) {
639  case 0: return 0.5-x;
640  case 1: return 0.5+x;
641  default:
642  fprintf(stderr,"%d: Tried to access charge assignment "
643  "function of degree %d in scheme of order %d.\n",this_node,i,P);
644  return 0.0;
645  }
646  }
647  case 3 : {
648  switch (i) {
649  case 0: return 0.5 * (0.5 - x);
650  case 1: return 0.75 - pow(x, 2);
651  case 2: return 0.5 * pow(0.5 + x, 2);
652  default:
653  fprintf(stderr,"%d: Tried to access charge assignment function"
654  " of degree %d in scheme of order %d.\n",this_node,i,P);
655  return 0.0;
656  }
657  case 4 : {
