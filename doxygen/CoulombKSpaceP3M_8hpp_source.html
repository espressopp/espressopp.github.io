00319       void aliasing_sum(Int3D i, Real3D *nominator, real *denominator){
00320         /*
00321          * taken from Deserno's code
00322          *  calculates the aliasing sums in the nominator and denominator of the 
00323            expression for the optimal influence function (see  Hockney/Eastwood: 
00324            8-22, p. 275).
00325            N: is the n-vector for which the aliasing sum is to be performed.
00326            *nominatorX,*nominatorY,*nominatorZ : x-, y-, and z-component of the 
00327                                                  aliasing sum.
00328            *denominator : aliasing sum in the denominator. */
00329 
00330         Real3D ms(mesh_shift[0][i[0]], mesh_shift[1][i[1]], mesh_shift[2][i[2]]);
00331         
00332         real sc;
00333         Real3D nml(0.0);
00334         
00335         Real3D Minv;
00336         for(int i=0;i<3;i++) Minv[i] = 1.0/(real)M[i];
00337         
00338         real ftr = pow( M_PIl / (alpha*sysL[0]), 2);
00339 
00340         *denominator = 0.0;
00341         *nominator = Real3D(0.0);
00342         
00343         Int3D ijk(0);
00344         Real3D x(0.0);
00345         for ( ijk[0] = -brillouin; ijk[0] <= brillouin; ijk[0]++) {
00346           for ( ijk[1] = -brillouin; ijk[1] <= brillouin; ijk[1]++) {
00347             for ( ijk[2] = -brillouin; ijk[2] <= brillouin; ijk[2]++) {
00348               
00349               for(int i=0; i<3; i++){
00350                 nml[i] = ms[i] + M[i] * ijk[i];
00351                 x[i]   = Minv[i] * nml[i];                
00352               }
00353 
00354               sc  = pow( sinc(x), 2.0*P);
00355 
00356               *denominator += sc;
00357 
00358               real nml2 = nml.sqr();
00359               real e = ftr * nml2;
00360               
00361               *nominator += ( ( e<30 ) ? sc*exp(-e)/nml2 : 0.0 ) * nml;
00362             }
00363           }
00364         }
00365       }
