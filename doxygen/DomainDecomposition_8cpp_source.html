00593   void DomainDecomposition::prepareGhostCommunication() {
00594     // direction loop: x, y, z
00595     for (int coord = 0; coord < 3; ++coord) {
00596         // boundaries of area to send
00597         int leftBoundary[3], rightBoundary[3];
00598         /* boundaries perpendicular directions are the same for left/right send.
00599         We also send the ghost frame that we have already, so the data amount
00600         increase with each cycle.
00601 
00602         For a direction that was done already, i.e. is smaller than dir,
00603         we take the full ghost frame, otherwise only the inner frame.  */
00604         for (int offset = 1; offset <= 2; ++offset) {
00605             int otherCoord = (coord + offset) % 3;
00606             if (otherCoord < coord) {
00607                 leftBoundary[otherCoord] = 0;
00608                 rightBoundary[otherCoord] = cellGrid.getFrameGridSize(otherCoord);
00609             } else {
00610                 leftBoundary[otherCoord] = cellGrid.getInnerCellsBegin(otherCoord);
00611                 rightBoundary[otherCoord] = cellGrid.getInnerCellsEnd(otherCoord);
00612             }
00613         }
00614 
00615         //  lr loop: left right - loop
00616         for (int lr = 0; lr < 2; ++lr) {
00617             int dir = 2 * coord + lr;
00618 
00619             /* participating real particles from this node */
00620             LOG4ESPP_DEBUG(logger, "direction " << dir << " reals");
00621 
00622             if (lr == 0) {
00623                 leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
00624                 rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord) + cellGrid.getFrameWidth();
00625             } else {
00626                 leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord) - cellGrid.getFrameWidth();
00627                 rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
00628             }
00629             fillCells(commCells[dir].reals, leftBoundary, rightBoundary);
00630 
00631             /* participating ghosts from this node */
00632             LOG4ESPP_DEBUG(logger, "direction " << dir << " ghosts");
00633 
00634             if (lr == 0) {
00635                 leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
00636                 rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord) + cellGrid.getFrameWidth();
00637             } else {
00638                 leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord) - cellGrid.getFrameWidth();
00639                 rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
00640             }
00641             fillCells(commCells[dir].ghosts, leftBoundary, rightBoundary);

00649            << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);

00651     /* direction loop: x, y, z.
00652    Here we could in principle build in a one sided ghost
00653    communication, simply by taking the lr loop only over one
00654    value. */
00655     for (int _coord = 0; _coord < 3; ++_coord) {
00656       /* inverted processing order for ghost force communication,
00657         since the corner ghosts have to be collected via several
00658         nodes. We now add back the corner ghost forces first again
00659         to ghost forces, which only eventually go back to the real
00660         particle.

00668         int oppositeDir = 2 * coord + (1 - lr);
00669 
00670         Real3D shift(0, 0, 0);
00671 
00672         shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
00673 
00674         LOG4ESPP_DEBUG(logger, "direction " << dir);
00675 
00676         if (nodeGrid.getGridSize(coord) == 1) {
00677           LOG4ESPP_DEBUG(logger, "local communication");
00678 
00679           // copy operation, we have to receive as many cells as we send
00680           if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
00681             throw std::runtime_error("DomainDecomposition::doGhostCommunication: send/recv cell structure mismatch during local copy");
00682           }
00683 
00684           for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
00685             if (realToGhosts) {
00686               copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
00687             } else {
00688               addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
00689             }
00690           }
00691         }
00692         else {
00693           // exchange size information, if necessary
00694           if (sizesFirst) {
00695             LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");

00698             std::vector<longint> sendSizes, recvSizes;
00699             sendSizes.reserve(commCells[dir].reals.size());

00701               sendSizes.push_back(commCells[dir].reals[i]->particles.size());
00702             }

00705             // exchange sizes, odd-even rule

00711             }

00714                         << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
00715               getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
00716               getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
00717             }
00718 
00719             // resize according to received information
00720             for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
00721               commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
00722             }
00723             LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
00724           }
00725 
00726           // prepare send and receive buffers
00727           longint receiver, sender;
00728           outBuffer.reset();

00730             receiver = nodeGrid.getNodeNeighborIndex(dir);
00731             sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
00732             for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
00733               packPositionsEtc(outBuffer, *commCells[dir].reals[i], extradata, shift);
00734             }
00735           }
00736           else {
00737             receiver = nodeGrid.getNodeNeighborIndex(oppositeDir);
00738             sender = nodeGrid.getNodeNeighborIndex(dir);
00739             for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
00740               packForces(outBuffer, *commCells[dir].ghosts[i]);
00741             }
00742           }
00743 
00744           // exchange particles, odd-even rule
00745           if (nodeGrid.getNodePosition(coord) % 2 == 0) {
00746             outBuffer.send(receiver, DD_COMM_TAG);
00747             inBuffer.recv(sender, DD_COMM_TAG);
00748           } else {
00749             inBuffer.recv(sender, DD_COMM_TAG);
00750             outBuffer.send(receiver, DD_COMM_TAG);
00751           }
00752 
00753           // unpack received data
00754           if (realToGhosts) {
00755             for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
00756               unpackPositionsEtc(*commCells[dir].ghosts[i], inBuffer, extradata);
00757             }
00758           }
00759           else {

00763           }
