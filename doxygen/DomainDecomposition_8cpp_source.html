234  Real3D box_sizeL = getSystem() -> bc -> getBoxL();
235  real skinL = getSystem() -> getSkin();

237

241  real rc_skin = maxCutoffL + skinL;
242  int ix = (int)(box_sizeL[0] / (rc_skin * _nodeGrid[0]));
243  int iy = (int)(box_sizeL[1] / (rc_skin * _nodeGrid[1]));
244  int iz = (int)(box_sizeL[2] / (rc_skin * _nodeGrid[2]));
245  Int3D _newCellGrid(ix, iy, iz);
246 
247  // save all particles to temporary vector
248  std::vector<ParticleList> tmp_pl;
249  size_t _N = realCells.size();
250  tmp_pl.reserve( _N );
251  for(CellList::Iterator it(realCells); it.isValid(); ++it) {
252  tmp_pl.push_back((*it)->particles);
253  }
254 
255  // reset all cells info
256  invalidateGhosts();
257  cells.clear();
258  localCells.clear();
259  realCells.clear();
260  ghostCells.clear();
261  for(int i=0; i<6; i++){
262  commCells[i].reals.clear();
263  commCells[i].ghosts.clear();
264  }
265 
266  // creating new grids
267  createCellGrid(_nodeGrid, _newCellGrid);
268  initCellInteractions();
269  prepareGhostCommunication();
270 
271  // pushing the particles back to the empty cell ("do we have to check particles?")
272  for(int i=0; i<tmp_pl.size(); i++){
273  for (size_t p = 0; p < tmp_pl[i].size(); ++p) {
274  Particle& part = tmp_pl[i][p];
275  const Real3D& pos = part.position();

277  appendUnindexedParticle(sortCell->particles, part);
278  }
279  }

282  updateLocalParticles((*it)->particles);

302  cell->neighborCells.reserve(26);
303 
304  // loop all neighbor cells

381  void DomainDecomposition::decomposeRealParticles() {
382 
383  //std::cout << getSystem()->comm->rank() << ": " << " decomposeRealParticles\n";
384 
385  LOG4ESPP_DEBUG(logger, "starting, expected comm buffer size " << exchangeBufferSize);
386 
387  // allocate send/recv buffers. We use the size as we need maximally so far, to avoid reallocation
388  // TODO: This might be a problem when all particles are created on a single node initially!
389  ParticleList sendBufL;
390  sendBufL.reserve(exchangeBufferSize);
391  ParticleList sendBufR;
392  sendBufR.reserve(exchangeBufferSize);
393  ParticleList recvBufL;
394  recvBufL.reserve(exchangeBufferSize);
395  ParticleList recvBufR;
396  recvBufR.reserve(exchangeBufferSize);
397 
398  bool allFinished;
399  do {
400  bool finished = true;
401 
402  for (int coord = 0; coord < 3; ++coord) {
403  LOG4ESPP_DEBUG(logger, "starting with direction " << coord);
404 
405  if (nodeGrid.getGridSize(coord) > 1) {
406  for (std::vector<Cell*>::iterator it = realCells.begin(),
407  end = realCells.end(); it != end; ++it) {
408 
409  Cell &cell = **it;
410 
411  // do not use an iterator here, since we need to take out particles during the loop
412  for (size_t p = 0; p < cell.particles.size(); ++p) {
413  Particle &part = cell.particles[p];
414  const Real3D& pos = part.position();
415 
416  // check whether the particle is now "left" of the local domain
417  if (pos[coord] - cellGrid.getMyLeft(coord) < -ROUND_ERROR_PREC) {
418  LOG4ESPP_TRACE(logger, "send particle left " << part.id());
419  moveIndexedParticle(sendBufL, cell.particles, p);
420  // redo same particle since we took one out here, so it's a new one
421  --p;
422  }
423  // check whether the particle is now "right" of the local domain
424  else if (pos[coord] - cellGrid.getMyRight(coord) >= ROUND_ERROR_PREC) {
425  LOG4ESPP_TRACE(logger, "send particle right " << part.id());
426  moveIndexedParticle(sendBufR, cell.particles, p);
427  --p;
428  }
429  // Sort particles in cells of this node during last direction
430  else if (coord == 2) {
431  const Real3D& pos = part.position();
432  Cell *sortCell = mapPositionToCellChecked(pos);
433  if (sortCell != &cell) {
434  if (sortCell == 0) {
435  // particle is not in the local domain
436  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
437  << " @ " << pos <<
438  " is not inside node domain after neighbor exchange");
439  // isnan function is C99 only, x != x is only true if x == nan
440  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
441  // TODO: error handling
442  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
443  " has moved to outer space (one or more coordinates are nan)");
444  } else {
445  // particle stays where it is, and will be sorted in the next round
446  finished = false;
447  }
448  } else {
449  // particle is in the local domain
450  moveIndexedParticle(sortCell->particles, cell.particles, p);
451  --p;
452  }
453  }
454  }
455 
456  }
457  }
458 
459  // Exchange particles, odd-even rule
460  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
461  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
462  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
463  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
464  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
465  } else {
466  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
467  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
468  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
469  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
470  }
471 
472  // sort received particles to cells
473  if (appendParticles(recvBufL, 2 * coord) && coord == 2) finished = false;
474  if (appendParticles(recvBufR, 2 * coord + 1) && coord == 2) finished = false;
475 
476  // reset send/recv buffers
477  sendBufL.resize(0);
478  sendBufR.resize(0);
479  recvBufL.resize(0);
480  recvBufR.resize(0);
481 
482 
483  } else {
484  /* Single node direction case (no communication)
485  Fold particles that have left the box */
486  for (std::vector< Cell* >::iterator it = realCells.begin(),
487  end = realCells.end(); it != end; ++it) {
488  Cell &cell = **it;
489  // do not use an iterator here, since we have need to take out particles during the loop
490  for (size_t p = 0; p < cell.particles.size(); ++p) {
491  Particle &part = cell.particles[p];
492  getSystem()->bc->foldCoordinate(part.position(), part.image(), coord);
493  LOG4ESPP_TRACE(logger, "folded coordinate " << coord << " of particle " << part.id());
494 
495  if (coord == 2) {
496  Cell *sortCell = mapPositionToCellChecked(part.position());
497 
498  if (sortCell != &cell) {
499  if (sortCell == 0) {
500  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
501  << " @ " << part.position()
502  << " is not inside node domain after neighbor exchange");
503  const Real3D& pos = part.position();
504  // isnan function is C99 only, x != x is only true if x == nan
505  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
506  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
507  " has moved to outer space (one or more coordinates are nan)");
508  } else {
509  // particle stays where it is, and will be sorted in the next round
510  finished = false;
511  }
512  } else {
513  moveIndexedParticle(sortCell->particles, cell.particles, p);
514  --p;
515  }
516  }
517  }
518  }
519  }
520  }
521 
522  LOG4ESPP_DEBUG(logger, "done with direction " << coord);
523  }
524 
525  // Communicate wether particle exchange is finished
526  mpi::all_reduce(*getSystem()->comm, finished, allFinished, std::logical_and<bool>());
527  } while (!allFinished);
528 
529  exchangeBufferSize = std::max(exchangeBufferSize,
530  std::max(sendBufL.capacity(),
531  std::max(sendBufR.capacity(),
532  std::max(recvBufL.capacity(),
533  recvBufR.capacity()))));
534 
535  LOG4ESPP_DEBUG(logger, "finished exchanging particles, new send/recv buffer size " << exchangeBufferSize);
536 
537  LOG4ESPP_DEBUG(logger, "done");
538  }

593  void DomainDecomposition::prepareGhostCommunication() {
594  // direction loop: x, y, z
595  for (int coord = 0; coord < 3; ++coord) {
596  // boundaries of area to send
597  int leftBoundary[3], rightBoundary[3];
598  /* boundaries perpendicular directions are the same for left/right send.
599  We also send the ghost frame that we have already, so the data amount
600  increase with each cycle.
601 
602  For a direction that was done already, i.e. is smaller than dir,
603  we take the full ghost frame, otherwise only the inner frame. */
604  for (int offset = 1; offset <= 2; ++offset) {
605  int otherCoord = (coord + offset) % 3;
606  if (otherCoord < coord) {
607  leftBoundary[otherCoord] = 0;
608  rightBoundary[otherCoord] = cellGrid.getFrameGridSize(otherCoord);
609  } else {
610  leftBoundary[otherCoord] = cellGrid.getInnerCellsBegin(otherCoord);
611  rightBoundary[otherCoord] = cellGrid.getInnerCellsEnd(otherCoord);
612  }
613  }
614 
615  // lr loop: left right - loop
616  for (int lr = 0; lr < 2; ++lr) {
617  int dir = 2 * coord + lr;
618 
619  /* participating real particles from this node */
620  LOG4ESPP_DEBUG(logger, "direction " << dir << " reals");
621 
622  if (lr == 0) {
623  leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
624  rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord) + cellGrid.getFrameWidth();
625  } else {
626  leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord) - cellGrid.getFrameWidth();
627  rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
628  }
629  fillCells(commCells[dir].reals, leftBoundary, rightBoundary);
630 
631  /* participating ghosts from this node */
632  LOG4ESPP_DEBUG(logger, "direction " << dir << " ghosts");
633 
634  if (lr == 0) {
635  leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
636  rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord) + cellGrid.getFrameWidth();
637  } else {
638  leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord) - cellGrid.getFrameWidth();
639  rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
640  }
641  fillCells(commCells[dir].ghosts, leftBoundary, rightBoundary);
642  }
643  }
644  }

646  void DomainDecomposition::
647  doGhostCommunication(bool sizesFirst, bool realToGhosts, int extradata) {
648  LOG4ESPP_DEBUG(logger, "do ghost communication " << (sizesFirst ? "with sizes " : "")
649  << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);
650 
651  /* direction loop: x, y, z.
652  Here we could in principle build in a one sided ghost
653  communication, simply by taking the lr loop only over one
654  value. */
655  for (int _coord = 0; _coord < 3; ++_coord) {
656  /* inverted processing order for ghost force communication,
657  since the corner ghosts have to be collected via several
658  nodes. We now add back the corner ghost forces first again
659  to ghost forces, which only eventually go back to the real
660  particle.
661  */
662  int coord = realToGhosts ? _coord : (2 - _coord);
663  real curCoordBoxL = getSystem()->bc->getBoxL()[coord];
664 
665  // lr loop: left right
666  for (int lr = 0; lr < 2; ++lr) {
667  int dir = 2 * coord + lr;
668  int oppositeDir = 2 * coord + (1 - lr);
669 
670  Real3D shift(0, 0, 0);
671 
672  shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
673 
674  LOG4ESPP_DEBUG(logger, "direction " << dir);
675 
676  if (nodeGrid.getGridSize(coord) == 1) {
677  LOG4ESPP_DEBUG(logger, "local communication");
678 
679  // copy operation, we have to receive as many cells as we send
680  if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
681  throw std::runtime_error("DomainDecomposition::doGhostCommunication: send/recv cell structure mismatch during local copy");
682  }
683 
684  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
685  if (realToGhosts) {
686  copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
687  } else {
688  addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
689  }
690  }
691  }
692  else {
693  // exchange size information, if necessary
694  if (sizesFirst) {
695  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");
696 
697  // prepare buffers
698  std::vector<longint> sendSizes, recvSizes;
699  sendSizes.reserve(commCells[dir].reals.size());
700  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
701  sendSizes.push_back(commCells[dir].reals[i]->particles.size());
702  }
703  recvSizes.resize(commCells[dir].ghosts.size());
704 
705  // exchange sizes, odd-even rule
706  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
707  LOG4ESPP_DEBUG(logger, "sending to node " << nodeGrid.getNodeNeighborIndex(dir)
708  << ", then receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir));
709  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
710  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
711  }
712  else {
713  LOG4ESPP_DEBUG(logger, "receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir)
714  << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
715  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
716  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
717  }
718 
719  // resize according to received information
720  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
721  commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
722  }
723  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
724  }
725 
726  // prepare send and receive buffers
727  longint receiver, sender;
728  outBuffer.reset();
729  if (realToGhosts) {
730  receiver = nodeGrid.getNodeNeighborIndex(dir);
731  sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
732  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
733  packPositionsEtc(outBuffer, *commCells[dir].reals[i], extradata, shift);
734  }
735  }
736  else {
737  receiver = nodeGrid.getNodeNeighborIndex(oppositeDir);
738  sender = nodeGrid.getNodeNeighborIndex(dir);
739  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
740  packForces(outBuffer, *commCells[dir].ghosts[i]);
741  }
742  }
743 
744  // exchange particles, odd-even rule
745  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
746  outBuffer.send(receiver, DD_COMM_TAG);
747  inBuffer.recv(sender, DD_COMM_TAG);
748  } else {
749  inBuffer.recv(sender, DD_COMM_TAG);
750  outBuffer.send(receiver, DD_COMM_TAG);
751  }
752 
753  // unpack received data
754  if (realToGhosts) {
755  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
756  unpackPositionsEtc(*commCells[dir].ghosts[i], inBuffer, extradata);
757  }
758  }
759  else {
760  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
761  unpackAndAddForces(*commCells[dir].reals[i], inBuffer);
762  }
763  }
764  }
765  }
766  }
767  LOG4ESPP_DEBUG(logger, "ghost communication finished");
768  }
