87  void DomainDecomposition:: createCellGrid(const Int3D& _nodeGrid, const Int3D& _cellGrid) {
88  real myLeft[3];
89  real myRight[3];
90 
91  nodeGrid = NodeGrid(_nodeGrid, getSystem()->comm->rank(), getSystem()->bc->getBoxL());
92 
93  if (nodeGrid.getNumberOfCells() != getSystem()->comm->size()) {
94  throw NodeGridMismatch(_nodeGrid, getSystem()->comm->size());
95  }
96 
97  LOG4ESPP_INFO(logger, "my node grid position: "
98  << nodeGrid.getNodePosition(0) << " "
99  << nodeGrid.getNodePosition(1) << " "
100  << nodeGrid.getNodePosition(2) << " -> "
101  << getSystem()->comm->rank());
102 
103  LOG4ESPP_DEBUG(logger, "my neighbors: "
104  << nodeGrid.getNodeNeighborIndex(0) << "<->"
105  << nodeGrid.getNodeNeighborIndex(1) << ", "
106  << nodeGrid.getNodeNeighborIndex(2) << "<->"
107  << nodeGrid.getNodeNeighborIndex(3) << ", "
108  << nodeGrid.getNodeNeighborIndex(4) << "<->"
109  << nodeGrid.getNodeNeighborIndex(5));
110 
111  for (int i = 0; i < 3; ++i) {
112  myLeft[i] = nodeGrid.getMyLeft(i);
113  myRight[i] = nodeGrid.getMyRight(i);
114  }
115 
116  cellGrid = CellGrid(_cellGrid, myLeft, myRight, halfCellInt);
117 
118  LOG4ESPP_INFO(logger, "local box "
119  << myLeft[0] << "-" << myRight[0] << ", "
120  << myLeft[1] << "-" << myRight[1] << ", "
121  << myLeft[2] << "-" << myRight[2]);
122 
123  longint nLocalCells = 1;
124  longint nRealCells = 1;
125  for (int i = 0; i < 3; ++i) {
126  nRealCells *= cellGrid.getGridSize(i);
127  nLocalCells *= cellGrid.getFrameGridSize(i);

129

131 
132  realCells.reserve(nRealCells);
133  ghostCells.reserve(nLocalCells - nRealCells);
134 
135  markCells();
136 
137  LOG4ESPP_DEBUG(logger, "total # cells=" << nLocalCells
138  << ", # real cells=" << nRealCells
139  << ", frame cell grid = (" << cellGrid.getFrameGridSize(0)
140  << ", " << cellGrid.getFrameGridSize(1)

148 
149  for (int o = 0; o < cellGrid.getFrameGridSize(2); ++o) {
150  for (int n = 0; n < cellGrid.getFrameGridSize(1); ++n) {

235  void DomainDecomposition::cellAdjust(){
236  // create an appropriate cell grid
237  Real3D box_sizeL = getSystem() -> bc -> getBoxL();
238  real skinL = getSystem() -> getSkin();
239  real maxCutoffL = getSystem() -> maxCutoff;
240 
241  // nodeGrid is already defined
242  Int3D _nodeGrid(nodeGrid.getGridSize());
243  // new cellGrid
244  real rc_skin = maxCutoffL + skinL;
245  int ix = (int)(box_sizeL[0] / (rc_skin * _nodeGrid[0]));
246  int iy = (int)(box_sizeL[1] / (rc_skin * _nodeGrid[1]));
247  int iz = (int)(box_sizeL[2] / (rc_skin * _nodeGrid[2]));
248  Int3D _newCellGrid(ix, iy, iz);
249 
250  // save all particles to temporary vector
251  std::vector<ParticleList> tmp_pl;
252  size_t _N = realCells.size();
253  tmp_pl.reserve( _N );
254  for(CellList::Iterator it(realCells); it.isValid(); ++it) {
255  tmp_pl.push_back((*it)->particles);
256  }
257 
258  // reset all cells info
259  invalidateGhosts();
260  cells.clear();
261  localCells.clear();
262  realCells.clear();
263  ghostCells.clear();
264  for(int i=0; i<6; i++){
265  commCells[i].reals.clear();
266  commCells[i].ghosts.clear();
267  }
268 
269  // creating new grids
270  createCellGrid(_nodeGrid, _newCellGrid);
271  initCellInteractions();
272  prepareGhostCommunication();
273 
274  // pushing the particles back to the empty cell ("do we have to check particles?")
275  for(int i=0; i<tmp_pl.size(); i++){
276  for (size_t p = 0; p < tmp_pl[i].size(); ++p) {
277  Particle& part = tmp_pl[i][p];
278  const Real3D& pos = part.position();
279  Cell *sortCell = mapPositionToCellClipped(pos);
280  appendUnindexedParticle(sortCell->particles, part);
281  }
282  }
283 
284  for(CellList::Iterator it(realCells); it.isValid(); ++it) {
285  updateLocalParticles((*it)->particles);
286  }
287 
288  exchangeGhosts();
289 
292  onCellAdjust();
293 
294  onParticlesChanged();
295  }

297  void DomainDecomposition::initCellInteractions() {
298  LOG4ESPP_DEBUG(logger, "setting up neighbors for " << cells.size() << " cells");
299 
300  for (int o = cellGrid.getInnerCellsBegin(2); o < cellGrid.getInnerCellsEnd(2); ++o) {
301  for (int n = cellGrid.getInnerCellsBegin(1); n < cellGrid.getInnerCellsEnd(1); ++n) {
302  for (int m = cellGrid.getInnerCellsBegin(0); m < cellGrid.getInnerCellsEnd(0); ++m) {
303  longint cellIdx = cellGrid.mapPositionToIndex(m, n, o);
304  Cell *cell = &cells[cellIdx];
305 
306  LOG4ESPP_TRACE(logger, "setting up neighbors for cell " << cell - getFirstCell()
307  << " @ " << m << " " << n << " " << o);
308 
309  cell->neighborCells.reserve((2*halfCellInt + 1) * (2*halfCellInt + 1) * (2*halfCellInt + 1) - 1);
310 
311  // loop all neighbor cells
312  for (int p = o - halfCellInt; p <= o + halfCellInt; ++p) {
313  for (int q = n - halfCellInt; q <= n + halfCellInt; ++q) {
314  for (int r = m - halfCellInt; r <= m + halfCellInt; ++r) {
315  if (p != o || q != n || r != m) {
316  longint cell2Idx = cellGrid.mapPositionToIndex(r, q, p);
317  Cell *cell2 = &cells[cell2Idx];
318  cell->neighborCells.push_back(NeighborCellInfo(cell2, (cell2Idx<cellIdx)));
319 
320  LOG4ESPP_TRACE(logger, "neighbor cell " << cell2 - getFirstCell()
321  << " @ " << r << " " << q << " " << p << ((cell2Idx<cellIdx) ? " is" : " is not") << " taken" );
322  }
323  }
324  }
325  }
326  }
327  }
328  }
329 
330  LOG4ESPP_DEBUG(logger, "done");
331  }

359  bool DomainDecomposition::appendParticles(ParticleList &l, int dir) {
360  bool outlier = false;
361 
362  LOG4ESPP_DEBUG(logger, "got " << l.size() << " particles");
363 
364  for (ParticleList::iterator it = l.begin(), end = l.end(); it != end; ++it) {
365  Real3D& pos = it->position();
366 
367  if (nodeGrid.getBoundary(dir) != 0) {
368  getSystem()->bc->foldCoordinate(pos, it->image(), nodeGrid.convertDirToCoord(dir));
369  LOG4ESPP_TRACE(logger, "folded coordinate " << nodeGrid.convertDirToCoord(dir)
370  << " of particle " << it->id());
371  }
372 
373  longint cell;
374  if (cellGrid.mapPositionToCellCheckedAndClipped(cell, pos)) {
375  LOG4ESPP_TRACE(logger, "particle " << it->id()
376  << " @ " << pos << " is not inside node domain");
377  outlier = true;
378  }
379 
380  LOG4ESPP_TRACE(logger, "append part " << it->id() << " to cell "
381  << cell);
382 
383  appendIndexedParticle(cells[cell].particles, *it);
384  }
385  return outlier;

388  void DomainDecomposition::decomposeRealParticles() {
389 
390  //std::cout << getSystem()->comm->rank() << ": " << " decomposeRealParticles\n";
391 
392  LOG4ESPP_DEBUG(logger, "starting, expected comm buffer size " << exchangeBufferSize);
393 
394  // allocate send/recv buffers. We use the size as we need maximally so far, to avoid reallocation
395  // TODO: This might be a problem when all particles are created on a single node initially!
396  ParticleList sendBufL;
397  sendBufL.reserve(exchangeBufferSize);
398  ParticleList sendBufR;
399  sendBufR.reserve(exchangeBufferSize);
400  ParticleList recvBufL;
401  recvBufL.reserve(exchangeBufferSize);
402  ParticleList recvBufR;
403  recvBufR.reserve(exchangeBufferSize);
404 
405  bool allFinished;
406  do {
407  bool finished = true;
408 
409  for (int coord = 0; coord < 3; ++coord) {
410  LOG4ESPP_DEBUG(logger, "starting with direction " << coord);
411 
412  if (nodeGrid.getGridSize(coord) > 1) {
413  for (std::vector<Cell*>::iterator it = realCells.begin(),
414  end = realCells.end(); it != end; ++it) {
415 
416  Cell &cell = **it;
417 
418  // do not use an iterator here, since we need to take out particles during the loop
419  for (size_t p = 0; p < cell.particles.size(); ++p) {
420  Particle &part = cell.particles[p];
421  const Real3D& pos = part.position();
422 
423  // check whether the particle is now "left" of the local domain
424  if (pos[coord] - cellGrid.getMyLeft(coord) < -ROUND_ERROR_PREC) {
425  LOG4ESPP_TRACE(logger, "send particle left " << part.id());
426  moveIndexedParticle(sendBufL, cell.particles, p);
427  // redo same particle since we took one out here, so it's a new one
428  --p;
429  }
430  // check whether the particle is now "right" of the local domain
431  else if (pos[coord] - cellGrid.getMyRight(coord) >= ROUND_ERROR_PREC) {
432  LOG4ESPP_TRACE(logger, "send particle right " << part.id());
433  moveIndexedParticle(sendBufR, cell.particles, p);
434  --p;
435  }
436  // Sort particles in cells of this node during last direction
437  else if (coord == 2) {
438  const Real3D& pos = part.position();
439  Cell *sortCell = mapPositionToCellChecked(pos);
440  if (sortCell != &cell) {
441  if (sortCell == 0) {
442  // particle is not in the local domain
443  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
444  << " @ " << pos <<
445  " is not inside node domain after neighbor exchange");
446  // isnan function is C99 only, x != x is only true if x == nan
447  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
448  // TODO: error handling
449  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
450  " has moved to outer space (one or more coordinates are nan)");
451  } else {
452  // particle stays where it is, and will be sorted in the next round
453  finished = false;
454  }
455  } else {
456  // particle is in the local domain
457  moveIndexedParticle(sortCell->particles, cell.particles, p);
458  --p;
459  }
460  }
461  }
462 
463  }
464  }
465 
466  // Exchange particles, odd-even rule
467  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
468  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
469  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
470  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
471  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
472  } else {
473  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
474  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
475  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
476  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
477  }
478 
479  // sort received particles to cells
480  if (appendParticles(recvBufL, 2 * coord) && coord == 2) finished = false;
481  if (appendParticles(recvBufR, 2 * coord + 1) && coord == 2) finished = false;
482 
483  // reset send/recv buffers
484  sendBufL.resize(0);
485  sendBufR.resize(0);
486  recvBufL.resize(0);
487  recvBufR.resize(0);
488 
489 
490  } else {
491  /* Single node direction case (no communication)
492  Fold particles that have left the box */
493  for (std::vector< Cell* >::iterator it = realCells.begin(),
494  end = realCells.end(); it != end; ++it) {
495  Cell &cell = **it;
496  // do not use an iterator here, since we have need to take out particles during the loop
497  for (size_t p = 0; p < cell.particles.size(); ++p) {
498  Particle &part = cell.particles[p];
499  getSystem()->bc->foldCoordinate(part.position(), part.image(), coord);
500  LOG4ESPP_TRACE(logger, "folded coordinate " << coord << " of particle " << part.id());
501 
502  if (coord == 2) {
503  Cell *sortCell = mapPositionToCellChecked(part.position());
504 
505  if (sortCell != &cell) {
506  if (sortCell == 0) {
507  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
508  << " @ " << part.position()
509  << " is not inside node domain after neighbor exchange");
510  const Real3D& pos = part.position();
511  // isnan function is C99 only, x != x is only true if x == nan
512  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
513  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
514  " has moved to outer space (one or more coordinates are nan)");
515  } else {
516  // particle stays where it is, and will be sorted in the next round
517  finished = false;
518  }
519  } else {
520  moveIndexedParticle(sortCell->particles, cell.particles, p);
521  --p;
522  }
523  }
524  }
525  }
526  }
527  }
528 
529  LOG4ESPP_DEBUG(logger, "done with direction " << coord);
530  }
531 
532  // Communicate wether particle exchange is finished
533  mpi::all_reduce(*getSystem()->comm, finished, allFinished, std::logical_and<bool>());
534  } while (!allFinished);
535 
536  exchangeBufferSize = std::max(exchangeBufferSize,
537  std::max(sendBufL.capacity(),
538  std::max(sendBufR.capacity(),
539  std::max(recvBufL.capacity(),
540  recvBufR.capacity()))));
541 
542  LOG4ESPP_DEBUG(logger, "finished exchanging particles, new send/recv buffer size " << exchangeBufferSize);
543 
544  LOG4ESPP_DEBUG(logger, "done");
545  }

600  void DomainDecomposition::prepareGhostCommunication() {
601  // direction loop: x, y, z
602  for (int coord = 0; coord < 3; ++coord) {
603  // boundaries of area to send
604  int leftBoundary[3], rightBoundary[3];
605  /* boundaries perpendicular directions are the same for left/right send.
606  We also send the ghost frame that we have already, so the data amount
607  increase with each cycle.
608 
609  For a direction that was done already, i.e. is smaller than dir,
610  we take the full ghost frame, otherwise only the inner frame. */
611  for (int offset = 1; offset <= 2; ++offset) {
612  int otherCoord = (coord + offset) % 3;
613  if (otherCoord < coord) {
614  leftBoundary[otherCoord] = 0;
615  rightBoundary[otherCoord] = cellGrid.getFrameGridSize(otherCoord);
616  } else {
617  leftBoundary[otherCoord] = cellGrid.getInnerCellsBegin(otherCoord);
618  rightBoundary[otherCoord] = cellGrid.getInnerCellsEnd(otherCoord);
619  }
620  }
621 
622  // lr loop: left right - loop
623  for (int lr = 0; lr < 2; ++lr) {
624  int dir = 2 * coord + lr;
625 
626  /* participating real particles from this node */
627  LOG4ESPP_DEBUG(logger, "direction " << dir << " reals");
628 
629  if (lr == 0) {
630  leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
631  rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord) + cellGrid.getFrameWidth();
632  } else {
633  leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord) - cellGrid.getFrameWidth();
634  rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
635  }
636  fillCells(commCells[dir].reals, leftBoundary, rightBoundary);
637 
638  /* participating ghosts from this node */
639  LOG4ESPP_DEBUG(logger, "direction " << dir << " ghosts");
640 
641  if (lr == 0) {
642  leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
643  rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord) + cellGrid.getFrameWidth();
644  } else {
645  leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord) - cellGrid.getFrameWidth();
646  rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
647  }
648  fillCells(commCells[dir].ghosts, leftBoundary, rightBoundary);
649  }
650  }
651  }

653  void DomainDecomposition::
654  doGhostCommunication(bool sizesFirst, bool realToGhosts, int extradata) {
655  LOG4ESPP_DEBUG(logger, "do ghost communication " << (sizesFirst ? "with sizes " : "")
656  << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);
657 
658  /* direction loop: x, y, z.
659  Here we could in principle build in a one sided ghost
660  communication, simply by taking the lr loop only over one
661  value. */
662  for (int _coord = 0; _coord < 3; ++_coord) {
663  /* inverted processing order for ghost force communication,
664  since the corner ghosts have to be collected via several
665  nodes. We now add back the corner ghost forces first again
666  to ghost forces, which only eventually go back to the real
667  particle.
668  */
669  int coord = realToGhosts ? _coord : (2 - _coord);
670  real curCoordBoxL = getSystem()->bc->getBoxL()[coord];
671 
672  // lr loop: left right
673  for (int lr = 0; lr < 2; ++lr) {
674  int dir = 2 * coord + lr;
675  int oppositeDir = 2 * coord + (1 - lr);
676 
677  Real3D shift(0, 0, 0);
678 
679  shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
680 
681  LOG4ESPP_DEBUG(logger, "direction " << dir);
682 
683  if (nodeGrid.getGridSize(coord) == 1) {
684  LOG4ESPP_DEBUG(logger, "local communication");
685 
686  // copy operation, we have to receive as many cells as we send
687  if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
688  throw std::runtime_error("DomainDecomposition::doGhostCommunication: send/recv cell structure mismatch during local copy");
689  }
690 
691  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
692  if (realToGhosts) {
693  copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
694  } else {
695  addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
696  }
697  }
698  }
699  else {
700  // exchange size information, if necessary
701  if (sizesFirst) {
702  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");
703 
704  // prepare buffers
705  std::vector<longint> sendSizes, recvSizes;
706  sendSizes.reserve(commCells[dir].reals.size());
707  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
708  sendSizes.push_back(commCells[dir].reals[i]->particles.size());
709  }
710  recvSizes.resize(commCells[dir].ghosts.size());
711 
712  // exchange sizes, odd-even rule
713  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
714  LOG4ESPP_DEBUG(logger, "sending to node " << nodeGrid.getNodeNeighborIndex(dir)
715  << ", then receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir));
716  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
717  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
718  }
719  else {
720  LOG4ESPP_DEBUG(logger, "receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir)
721  << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
722  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
723  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
724  }
725 
726  // resize according to received information
727  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
728  commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
729  }
730  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
731  }
732 
733  // prepare send and receive buffers
734  longint receiver, sender;
735  outBuffer.reset();
736  if (realToGhosts) {
737  receiver = nodeGrid.getNodeNeighborIndex(dir);
738  sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
739  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
740  packPositionsEtc(outBuffer, *commCells[dir].reals[i], extradata, shift);
741  }
742  }
743  else {
744  receiver = nodeGrid.getNodeNeighborIndex(oppositeDir);
745  sender = nodeGrid.getNodeNeighborIndex(dir);
746  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
747  packForces(outBuffer, *commCells[dir].ghosts[i]);
748  }
749  }
750 
751  // exchange particles, odd-even rule
752  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
753  outBuffer.send(receiver, DD_COMM_TAG);
754  inBuffer.recv(sender, DD_COMM_TAG);
755  } else {
756  inBuffer.recv(sender, DD_COMM_TAG);
757  outBuffer.send(receiver, DD_COMM_TAG);
758  }
759 
760  // unpack received data
761  if (realToGhosts) {
762  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
763  unpackPositionsEtc(*commCells[dir].ghosts[i], inBuffer, extradata);
764  }
765  }
766  else {
767  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
768  unpackAndAddForces(*commCells[dir].reals[i], inBuffer);
769  }
770  }
771  }
772  }
773  }
774  LOG4ESPP_DEBUG(logger, "ghost communication finished");
775  }
