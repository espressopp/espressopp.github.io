155  template < class Derived >
156  inline real
157  AngularPotentialTemplate< Derived >::
158  computeEnergy(const Real3D& dist12, const Real3D& dist32) const {
159  real dist12Sqr = dist12 * dist12;
160  real dist32Sqr = dist32 * dist32;
161  real cos_theta = dist12 * dist32 / (sqrt(dist12Sqr) * sqrt(dist32Sqr));
162  return computeEnergy(acos(cos_theta));
163  }

181  template < class Derived >
182  inline real
183  AngularPotentialTemplate< Derived >::
184  _computeEnergy(const Real3D& dist12, const Real3D& dist32) const {
185 
186  real dist12_sqr = dist12 * dist12;
187  real dist32_sqr = dist32 * dist32;
188  if (dist12_sqr >= cutoffSqr || dist32_sqr >= cutoffSqr )
189  return 0.0;
190  else{
191  real cos_theta = dist12 * dist32 / (sqrt(dist12_sqr) * sqrt(dist32_sqr));
192  return _computeEnergy(acos(cos_theta));
193  }
