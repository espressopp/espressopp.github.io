270  template < class Derived >
271  inline real
272  AngularPotentialTemplate< Derived >::
273  computeEnergy(const Real3D& dist12, const Real3D& dist32) const {
274  real dist12Sqr = dist12 * dist12;
275  real dist32Sqr = dist32 * dist32;
276  real cos_theta = dist12 * dist32 / (sqrt(dist12Sqr) * sqrt(dist32Sqr));
277  return computeEnergy(acos(cos_theta));
278  }

296  template < class Derived >
297  inline real
298  AngularPotentialTemplate< Derived >::
299  _computeEnergy(const Real3D& dist12, const Real3D& dist32) const {
300 
301  real dist12_sqr = dist12 * dist12;
302  real dist32_sqr = dist32 * dist32;
303  if (dist12_sqr >= cutoffSqr || dist32_sqr >= cutoffSqr )
304  return 0.0;
305  else{
306  real cos_theta = dist12 * dist32 / (sqrt(dist12_sqr) * sqrt(dist32_sqr));
307  return _computeEnergy(acos(cos_theta));
308  }
309  }
