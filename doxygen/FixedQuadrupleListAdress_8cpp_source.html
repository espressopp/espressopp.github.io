00061 bool FixedQuadrupleListAdress::add(longint pid1, longint pid2, longint pid3, longint pid4) {
00062   // here we assume pid1 < pid2 < pid3 < pid4
00063   bool returnVal = true;
00064   System& system = storage->getSystemRef();
00065   esutil::Error err(system.comm);
00066 
00067   // ADD THE LOCAL QUADRUPLET
00068   Particle *p1 = storage->lookupAdrATParticle(pid1);
00069   Particle *p2 = storage->lookupAdrATParticle(pid2);
00070   Particle *p3 = storage->lookupAdrATParticle(pid3);
00071   Particle *p4 = storage->lookupAdrATParticle(pid4);
00072   if (!p1){
00073     // Particle does not exist here, return false
00074     returnVal = false;
00075   } else {
00076     if (!p2) {
00077       std::stringstream msg;
00078       msg << "Quadruple particle p2 " << pid2 << " does not exists here and cannot be added.";
00079       err.setException( msg.str() );
00080     }
00081     if (!p3) {
00082       std::stringstream msg;
00083       msg << "Quadruple particle p3 " << pid3 << " does not exists here and cannot be added.";
00084       err.setException( msg.str() );
00085     }
00086     if (!p4) {
00087       std::stringstream msg;
00088       msg << "Quadruple particle p4 " << pid4 << " does not exists here and cannot be added.";
00089       err.setException( msg.str() );
00090     }
00091   }
00092   err.checkException();
00093 
00094   if(returnVal){
00095     // add the quadruple locally
00096     this->add(p1, p2, p3, p4);
00097 
00098     // ADD THE GLOBAL QUADRUPLET
00099     // see whether the particle already has quadruples
00100     std::pair<GlobalQuadruples::const_iterator,
00101               GlobalQuadruples::const_iterator> equalRange
00102       = globalQuadruples.equal_range(pid1);
00103     if (equalRange.first == globalQuadruples.end()) {
00104       // if it hasn't, insert the new quadruple
00105       globalQuadruples.insert(std::make_pair(pid1,
00106         Triple<longint, longint, longint>(pid2, pid3, pid4)));
00107     } else {
00108       // otherwise test whether the quadruple already exists
00109       for (GlobalQuadruples::const_iterator it = equalRange.first; it != equalRange.second; ++it)
00110         if (it->second == Triple<longint, longint, longint>(pid2, pid3, pid4))
00111           // TODO: Quadruple already exists, generate error!
00112         ;
00113       // if not, insert the new quadruple
00114       globalQuadruples.insert(equalRange.first,
00115         std::make_pair(pid1, Triple<longint, longint, longint>(pid2, pid3, pid4)));
00116     }
00117   }
00118   LOG4ESPP_INFO(theLogger, "Added fixed quadruple to local quadruple list.");

00122 void FixedQuadrupleListAdress::beforeSendATParticles(std::vector<longint>& atpl, OutBuffer& buf) {
00123   std::vector< longint > toSend;
00124 
00125   // loop over the VP particle list
00126   for (std::vector<longint>::iterator it = atpl.begin(); it != atpl.end(); ++it) {
00127     longint pid = *it;
00128 
00129     LOG4ESPP_DEBUG(theLogger, "Send particle with pid " << pid << ", find pairs.");
00130 
00131     // find all triples that involve this particle
00132     int n = globalQuadruples.count(pid);
00133 
00134     if (n > 0) {
00135       std::pair<GlobalQuadruples::const_iterator,
00136         GlobalQuadruples::const_iterator> equalRange = globalQuadruples.equal_range(pid);
00137 
00138       // Data structure:
00139       // pid1 - N - n0 - n1 - ... - nN - pid2 - N ....
00140       toSend.reserve(toSend.size()+3*n+1);
00141       toSend.push_back(pid);
00142       toSend.push_back(n);
00143       for (GlobalQuadruples::const_iterator it = equalRange.first;
00144            it != equalRange.second; ++it) {
00145         toSend.push_back(it->second.first);
00146         toSend.push_back(it->second.second);
00147         toSend.push_back(it->second.third);
00148       }
00149 
00150       // Delete all of these triples from the global list.
00151       globalQuadruples.erase(equalRange.first, equalRange.second);
00152     }
00153   }
00154 
00155   // send the list
00156   buf.write(toSend);
00157   LOG4ESPP_INFO(theLogger, "Prepared fixed triple list before send particles.");
00158 }
