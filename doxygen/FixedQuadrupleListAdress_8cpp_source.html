61 bool FixedQuadrupleListAdress::add(longint pid1, longint pid2, longint pid3, longint pid4) {
62  // here we assume pid1 < pid2 < pid3 < pid4
63  bool returnVal = true;
64  System& system = storage->getSystemRef();
65  esutil::Error err(system.comm);
66 
67  // ADD THE LOCAL QUADRUPLET
68  Particle *p1 = storage->lookupAdrATParticle(pid1);
69  Particle *p2 = storage->lookupAdrATParticle(pid2);
70  Particle *p3 = storage->lookupAdrATParticle(pid3);
71  Particle *p4 = storage->lookupAdrATParticle(pid4);
72  if (!p1){
73  // Particle does not exist here, return false
74  returnVal = false;
75  } else {
76  if (!p2) {
77  std::stringstream msg;
78  msg << "Quadruple particle p2 " << pid2 << " does not exists here and cannot be added.";
79  err.setException( msg.str() );
80  }
81  if (!p3) {
82  std::stringstream msg;
83  msg << "Quadruple particle p3 " << pid3 << " does not exists here and cannot be added.";
84  err.setException( msg.str() );
85  }
86  if (!p4) {
87  std::stringstream msg;
88  msg << "Quadruple particle p4 " << pid4 << " does not exists here and cannot be added.";
89  err.setException( msg.str() );
90  }
91  }
92  err.checkException();
93 
94  if(returnVal){
95  // add the quadruple locally
96  this->add(p1, p2, p3, p4);
97 
98  // ADD THE GLOBAL QUADRUPLET
99  // see whether the particle already has quadruples
100  std::pair<GlobalQuadruples::const_iterator,
101  GlobalQuadruples::const_iterator> equalRange
102  = globalQuadruples.equal_range(pid1);
103  if (equalRange.first == globalQuadruples.end()) {
104  // if it hasn't, insert the new quadruple
105  globalQuadruples.insert(std::make_pair(pid1,
106  Triple<longint, longint, longint>(pid2, pid3, pid4)));
107  } else {
108  // otherwise test whether the quadruple already exists
109  for (GlobalQuadruples::const_iterator it = equalRange.first; it != equalRange.second; ++it)
110  if (it->second == Triple<longint, longint, longint>(pid2, pid3, pid4))
111  // TODO: Quadruple already exists, generate error!
112  ;
113  // if not, insert the new quadruple
114  globalQuadruples.insert(equalRange.first,
115  std::make_pair(pid1, Triple<longint, longint, longint>(pid2, pid3, pid4)));
116  }
117  }
118  LOG4ESPP_INFO(theLogger, "Added fixed quadruple to local quadruple list.");
119  return returnVal;
120 }

122 void FixedQuadrupleListAdress::beforeSendATParticles(std::vector<longint>& atpl, OutBuffer& buf) {
123  std::vector< longint > toSend;
124 
125  // loop over the VP particle list
126  for (std::vector<longint>::iterator it = atpl.begin(); it != atpl.end(); ++it) {
127  longint pid = *it;
128 
129  LOG4ESPP_DEBUG(theLogger, "Send particle with pid " << pid << ", find pairs.");
130 
131  // find all triples that involve this particle
132  int n = globalQuadruples.count(pid);
133 
134  if (n > 0) {
135  std::pair<GlobalQuadruples::const_iterator,
136  GlobalQuadruples::const_iterator> equalRange = globalQuadruples.equal_range(pid);
137 
138  // Data structure:
139  // pid1 - N - n0 - n1 - ... - nN - pid2 - N ....
140  toSend.reserve(toSend.size()+3*n+1);
141  toSend.push_back(pid);
142  toSend.push_back(n);
143  for (GlobalQuadruples::const_iterator it = equalRange.first;
144  it != equalRange.second; ++it) {
145  toSend.push_back(it->second.first);
146  toSend.push_back(it->second.second);
147  toSend.push_back(it->second.third);
148  }
149 
150  // Delete all of these triples from the global list.
151  globalQuadruples.erase(equalRange.first, equalRange.second);
152  }
153  }
154 
155  // send the list
156  buf.write(toSend);
157  LOG4ESPP_INFO(theLogger, "Prepared fixed triple list before send particles.");
158 }

165 void FixedQuadrupleListAdress::onParticlesChanged() {
166  // (re-)generate the local quadruple list from the global list
167  LOG4ESPP_INFO(theLogger, "Rebuild local bond list from global\n");
168 
169  System& system = storage->getSystemRef();
170  esutil::Error err(system.comm);
171 
172  this->clear();
173  longint lastpid1 = -1;
174  Particle *p1;
175  Particle *p2;
176  Particle *p3;
177  Particle *p4;
178  for (GlobalQuadruples::const_iterator it = globalQuadruples.begin();
179  it != globalQuadruples.end(); ++it) {
180  if (it->first != lastpid1) {
181  p1 = storage->lookupAdrATParticle(it->first);
182  if (p1 == NULL) {
183  std::stringstream msg;
184  msg << "Quadruple particle p1 " << it->first << " does not exists here.";
185  err.setException( msg.str() );
186  }
187  lastpid1 = it->first;
188  }
189  p2 = storage->lookupAdrATParticle(it->second.first);
190  if (p2 == NULL) {
191  std::stringstream msg;
192  msg << "Quadruple particle p2 " << it->second.first << " does not exists here.";
193  err.setException( msg.str() );
194  }
195  p3 = storage->lookupAdrATParticle(it->second.second);
196  if (p3 == NULL) {
197  std::stringstream msg;
198  msg << "Quadruple particle p3 " << it->second.second << " does not exists here.";
199  err.setException( msg.str() );
200  }
201  p4 = storage->lookupAdrATParticle(it->second.third);
202  if (p4 == NULL) {
203  std::stringstream msg;
204  msg << "Quadruple particle p4 " << it->second.third << " does not exists here.";
205  err.setException( msg.str() );
206  }
207  this->add(p1, p2, p3, p4);
208  }
209  err.checkException();
210  LOG4ESPP_INFO(theLogger, "Regenerated local fixed quadruple list from global list.");
211 }
