110  void FixedListComm::beforeSendParticles
111  (ParticleList& pl, OutBuffer& buf) {
112 
113  std::vector<longint> toSend;
114 
115  // loop over the particle list
116  for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
117  longint pid = pit->id();
118  LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
119 
120  // find all particles that involve this particle id
121  int n = globalLists.count(pid);
122  if (n > 0) {
123  std::pair<GlobalList::const_iterator,
124  GlobalList::const_iterator>
125  equalRange = globalLists.equal_range(pid);
126 
127  // get the length of the vector in the map
128  int l = equalRange.first->second.size();
129  toSend.reserve(toSend.size()+3+n*l);
130  // first write the pid of the first particle
131  toSend.push_back(pid);
132  // then the number of partners
133  toSend.push_back(n);
134  // then the size of the vector
135  toSend.push_back(l);
136  // and then the pids of the partners
137  for (GlobalList::const_iterator it = equalRange.first;
138  it != equalRange.second; ++it) {
139  // iterate through vector to add pids
140  for (pvec::const_iterator it2=it->second.begin();
141  it2!=it->second.end(); ++it2) {
142  toSend.push_back(*it2);
143  LOG4ESPP_DEBUG(theLogger, "send global bond: pid "
144  << pid << " and its vector");
145  }
146  }
147 
148  // delete all of these pairs from the global list
149  globalLists.erase(equalRange.first, equalRange.second);
150  }
151  }
152  // send the list
153  buf.write(toSend);
154  LOG4ESPP_INFO(theLogger, "prepared fixed pair list before send particles");
155  }

179  pids.push_back(received[i++]);
180  }
181  // add the bond to the global list

183  it = globalLists.insert(it, std::make_pair(pid1, pids));

185  }

216  }

218  }
219  // iterate through vector in map
220  for (pvec::const_iterator it2 = it->second.begin(); it2!=it->second.end(); ++it2) {
221  p = storage->lookupLocalParticle(*it2);
222  if (p == NULL) {
223  printf("SERIOUS ERROR: particle %d not available\n", *it2);
224  }
225  tmp.push_back(p);
226  }
227  // add the particles

231  tmp.clear();
232  }
