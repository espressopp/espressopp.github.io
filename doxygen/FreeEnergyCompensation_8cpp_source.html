00169           
00170           real CompEnergy = 0.0;
00171           real CompEnergySum = 0.0;
00172           System& system = getSystemRef();
00173 
00174           // iterate over CG particles
00175           CellList cells = system.storage->getRealCells();
00176           for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
00177 
00178               Table table = forces.find(cit->getType())->second;                                       
00179               if (table) {                 
00180                   Particle &vp = *cit;
00181                   real weight = vp.lambda();

00184               else{
00185                   std::cout << "ERROR: Using FEC Extension without providing table." << std::endl;
