95  void FreeEnergyCompensation::applyForce() {
96  LOG4ESPP_DEBUG(theLogger, "apply Free Energy Compensation force");
97 
98  System& system = getSystemRef();
99 
100  std::unordered_map<int, Table>::iterator tableIt;
101  Table table;
102  // iterate over CG particles
103  CellList cells = system.storage->getRealCells();
104  shared_ptr<FixedTupleListAdress> fixedtupleList = system.storage->getFixedTuples();
105  FixedTupleListAdress::iterator it2;
106  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
107  tableIt = forces.find(cit->getType());
108  if (tableIt != forces.end()) { //because there may be CG particles to which Free Energy Compensation is not applied
109  table = tableIt->second;
110  }
111 
112  if (table) {
113 
114  Particle &vp = *cit;
115  real weight = vp.lambda();
116 
117  if (weight != 1.0 && weight != 0.0){
118  real fforce = table->getForce(weight);
119 
120  Real3D dist3D(0.0, 0.0, 0.0);
121  if(sphereAdr){
122  dist3D = cit->getPos() - center;
123  real absdist3D = sqrt(dist3D.sqr());
124  dist3D *= fforce;
125  dist3D /= absdist3D;
126  }
127  else{
128  real dist = vp.position()[0]-center[0];
129  if ( dist >= 0.0 ) {dist = 1.0;}
130  else {dist = -1.0;}
131  fforce *=dist;
132  }
133 
134  it2 = fixedtupleList->find(&vp);
135  if (it2 != fixedtupleList->end()) { // Are there atomistic particles for given CG particle? If yes, use those for calculation.
136  std::vector<Particle*> atList;
137  atList = it2->second;
138  for (std::vector<Particle*>::iterator it3 = atList.begin();
139  it3 != atList.end(); ++it3) {
140  Particle &at = **it3;
141 
142  if(sphereAdr){
143  at.force() += vp.lambdaDeriv() * at.mass() * dist3D / (vp.mass() * ntrotter);
144  }
145  else{
146  at.force()[0] += vp.lambdaDeriv() * at.mass() * fforce / (vp.mass() * ntrotter);
147  }
148 
149  }
150  }
151  else{ // If not, use CG particle itself for calculation.
152  std::cout << "Particle " << vp.id() << " not found in tuples!" << std::endl << "It's unclear how FEC work when combining particles, which do change resolution with particles that don't." << std::endl;
153  exit(1);
154  return;
155  }
156  }
157  }
158  }

161  real FreeEnergyCompensation::computeCompEnergy() {
162  LOG4ESPP_DEBUG(theLogger, "compute Free Energy Compensation Energies");
163 
164  real CompEnergy = 0.0;
165  real CompEnergySum = 0.0;
166  System& system = getSystemRef();
167 
168  // iterate over CG particles
169  CellList cells = system.storage->getRealCells();
170  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
171 
172  Table table = forces.find(cit->getType())->second;
173  if (table) {
174  Particle &vp = *cit;
175  real weight = vp.lambda();
176  CompEnergy += table->getEnergy(weight);
177  }
178  else{
179  std::cout << "ERROR: Using FEC Extension without providing table." << std::endl;
180  exit(1);
181  return 0.0;
182  }
183 
184  }
185  mpi::all_reduce(*getSystem()->comm, CompEnergy, CompEnergySum, std::plus<real>());
186  return CompEnergySum;
187  }
