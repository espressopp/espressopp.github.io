89  void FreeEnergyCompensation::applyForce() {
90  LOG4ESPP_DEBUG(theLogger, "apply Free Energy Compensation force");
91 
92  System& system = getSystemRef();
93 
94  // iterate over CG particles
95  CellList cells = system.storage->getRealCells();
96  shared_ptr<FixedTupleListAdress> fixedtupleList = system.storage->getFixedTuples();
97  FixedTupleListAdress::iterator it2;
98  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
99 
100  Table table = forces.find(cit->getType())->second;
101  if (table) {
102 
103  Particle &vp = *cit;
104  real weight = vp.lambda();
105 
106  if (weight != 1.0 && weight != 0.0){
107  real fforce = table->getForce(weight);
108 
109  Real3D dist3D(0.0, 0.0, 0.0);
110  if(sphereAdr){
111  dist3D = cit->getPos() - center;
112  real absdist3D = sqrt(dist3D.sqr());
113  dist3D *= fforce;
114  dist3D /= absdist3D;
115  }
116  else{
117  real dist = vp.position()[0]-center[0];
118  if ( dist >= 0.0 ) {dist = 1.0;}
119  else {dist = -1.0;}
120  fforce *=dist;
121  }
122 
123  it2 = fixedtupleList->find(&vp);
124  if (it2 != fixedtupleList->end()) { // Are there atomistic particles for given CG particle? If yes, use those for calculation.
125  std::vector<Particle*> atList;
126  atList = it2->second;
127  for (std::vector<Particle*>::iterator it3 = atList.begin();
128  it3 != atList.end(); ++it3) {
129  Particle &at = **it3;
130 
131  if(sphereAdr){
132  at.force() += vp.lambdaDeriv() * at.mass() * dist3D / vp.mass();
133  }
134  else{
135  at.force()[0] += vp.lambdaDeriv() * at.mass() * fforce / vp.mass();
136  }
137 
138  }
139  }
140  else{ // If not, use CG particle itself for calculation.
141  std::cout << "Particle " << vp.id() << " not found in tuples!" << std::endl << "It's unclear how FEC work when combining particles, which do change resolution with particles that don't." << std::endl;
142  exit(1);
143  return;
144  }
145  }
146  }
147  else{
148  std::cout << "ERROR: Using FEC Extension without providing table." << std::endl;
149  exit(1);
150  return;
151  }
152 
153  }
154  }

156  real FreeEnergyCompensation::computeCompEnergy() {
157  LOG4ESPP_DEBUG(theLogger, "compute Free Energy Compensation Energies");
158 
159  real CompEnergy = 0.0;
160  real CompEnergySum = 0.0;
161  System& system = getSystemRef();
162 
163  // iterate over CG particles
164  CellList cells = system.storage->getRealCells();
165  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
166 
167  Table table = forces.find(cit->getType())->second;
168  if (table) {
169  Particle &vp = *cit;
170  real weight = vp.lambda();
171  CompEnergy += table->getEnergy(weight);
172  }
173  else{
174  std::cout << "ERROR: Using FEC Extension without providing table." << std::endl;
175  exit(1);
176  return 0.0;
177  }
178 
179  }
180  mpi::all_reduce(*getSystem()->comm, CompEnergy, CompEnergySum, std::plus<real>());
181  return CompEnergySum;
182  }
