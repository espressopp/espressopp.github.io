42  void DumpXYZ::dump(){
43 
44  shared_ptr<System> system = getSystem();
45  ConfigurationsExt conf( system );
46  conf.setUnfolded(unfolded);
47  conf.gather();
48 
49  if( system->comm->rank()==0 ){
50  ConfigurationExtPtr conf_real = conf.back();
51 
52  int num_of_particles = conf_real->getSize();
53 
54  char *ch_f_name = new char[file_name.length() + 1];
55  strcpy(ch_f_name, file_name.c_str());
56  ofstream myfile (ch_f_name, ios::out | ios::app);
57  if (myfile.is_open()){
58  myfile << num_of_particles << endl;
59 
60  Real3D Li = system->bc->getBoxL();
61 
62  // for noncubic simulation boxes
63  myfile << Li[0] * length_factor << " 0.0 0.0 0.0 "<<
64  Li[1] * length_factor << " 0.0 0.0 0.0 "<< Li[2] * length_factor;
65  // additional info to comment line
66  myfile << " currentStep " << integrator->getStep() << " lengthUnit "<< length_unit << endl;
67 
68  ConfigurationExtIterator cei = conf_real-> getIterator();
69  std::streamsize p = myfile.precision();
70  for (size_t i=0; i<num_of_particles; i++) {
71 
72  if (store_pids) {
73  myfile << cei.currentId() << " ";
74  }
75 
76  myfile << particleIDToType.find(cei.currentId())->second <<
77  " " << fixed << setprecision(10) << length_factor * cei.currentProperties()[0] << " "
78  << length_factor * cei.currentProperties()[1] <<
79  " " << length_factor * cei.currentProperties()[2];
80 
81  if (store_velocities) {
82  myfile << " " << length_factor * cei.currentProperties()[3] << " " << length_factor * cei.currentProperties()[4] <<
83  " " << length_factor * cei.currentProperties()[5];
84  }
85  myfile << endl;
86  myfile.unsetf(ios_base::fixed);
87  myfile << setprecision(p);
88 
89  cei.incrementIterator();
90  }
91 
92  myfile.close();
93  }
94  else cout << "Unable to open file: "<< file_name <<endl;
95 
96  delete [] ch_f_name;
97  }
98  }
