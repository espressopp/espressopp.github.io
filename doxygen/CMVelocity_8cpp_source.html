39  Real3D CMVelocity::computeRaw() {
40 
41  System& system = getSystemRef();
42 
43  real myMass;
44  Real3D myV;
45 
46  CellList realCells = system.storage->getRealCells();
47 
48  // Compute the local velocity
49  myMass=0.;
50  myV=0.;
51 
52  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
53  Real3D& velocity = cit->velocity();
54  real mass = cit->getMass();
55  myMass += mass;
56  myV += mass*velocity;
57  }
58 
59  real v_data[4], total_v[4];
60  v_data[0] = myMass;
61  v_data[1] = myV[0];
62  v_data[2] = myV[1];
63  v_data[3] = myV[2];
64 
65  // Compute the total velocity of the system
66  boost::mpi::all_reduce(*(system.comm), v_data, 4, total_v, std::plus<real>());
67 
68  v[0] = total_v[1]/total_v[0];
69  v[1] = total_v[2]/total_v[0];
70  v[2] = total_v[3]/total_v[0];
71 
72  return v;
73  }

75  void CMVelocity::reset() {
