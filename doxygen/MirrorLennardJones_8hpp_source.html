96  real _computeEnergySqrRaw(real distSqr) const {
97  if (distSqr<cutoffSqr/4.0) return shift;
98  real dist = sqrt(distSqr);
99  real rSqr = pow(cutoff-dist, 2);
100  real frac2 = sigma*sigma / rSqr;
