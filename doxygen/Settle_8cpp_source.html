00158     void Settle::settlep(longint molID){
00159 
00160         System& system = getSystemRef();
00161         const bc::BC& bc = *getSystemRef().bc;  // boundary conditions
00162         real dt = integrator->getTimeStep();
00163         real invdt = 1.0/dt;
00164 
00165         //std::cout << "\nsettling " << molID << "\n";
00166 
00167         // positions in previous time step
00168         OldPos::const_iterator pos = oldPos.find(molID);
00169         if (pos == oldPos.end()) {
00170                 std::cout << "WARNING: oldPos not found! Skipping settlep().\n";
00171                 return;
00172         }
00173 
00174         // retrieve particles from tuples
00175         Particle* vp = system.storage->lookupRealParticle(molID);
00176         FixedTupleListAdress::iterator it;
00177         it = fixedTupleList->find(&(*vp));
00178 
00179         Particle* O  = it->second.at(0);
00180         Particle* H1 = it->second.at(1);
00181         Particle* H2 = it->second.at(2);
00182 
00183         // --- Step1 A1' ---
00184         // vectors in the plane of the original positions
00185         // previous positions OHH
00186         Real3D b0 = pos->second.second - pos->second.first; // H1.pos - O.pos
00187         Real3D c0 = pos->second.third - pos->second.first; // H2.pos - O.pos
00188 
00189         // new center of mass
00190         // present positions OHH
00191         Real3D d0 = O->position() * mOrmT + ((H1->position() + H2->position())*mHrmT);
00192 
00193         Real3D a1 = O->position() - d0;
00194         Real3D b1 = H1->position() - d0;
00195         Real3D c1 = H2->position() - d0;
00196 
00197         // Vectors describing transformation from original coordinate system to
00198         // the 'primed' coordinate system
00199         Real3D n0 = b0.cross(c0);
00200         Real3D n1 = a1.cross(n0);
00201         Real3D n2 = n0.cross(n1);
00202 
00203         // unit vectors
00204         n0 = n0/n0.abs();
00205         n1 = n1/n1.abs();
00206         n2 = n2/n2.abs();
00207 
00208         b0 = Real3D(n1*b0, n2*b0, n0*b0); // note: b0.z is never referenced again
00209         c0 = Real3D(n1*c0, n2*c0, n0*c0); // note: c0.z is never referenced again
00210 
00211         // shuffle the order of operations around from the normal algorithm so
00212         // that we can double pump sqrt() with n2 and cosphi at the same time
00213         // these components are usually computed down in the canonical water block
00214         real A1Z = n0 * a1;
00215         b1 = Real3D(n1*b1, n2*b1, n0*b1);
00216         c1 = Real3D(n1*c1, n2*c1, n0*c1);
00217 
00218         // --- Step2 A2' ---
00219         // now we can compute positions of canonical water
00220         real sinphi = A1Z * rra;
00221         real tmp = 1.0 - sinphi*sinphi;
00222         real cosphi = sqrt(tmp);
00223         real sinpsi = (b1[2] - c1[2]) / (2.0 * rc * cosphi);
00224         tmp = 1.0 - sinpsi*sinpsi;
00225         real cospsi = sqrt(tmp);
00226 
00227         real rbphi = -rb * cosphi;
00228         real tmp1 = rc * sinpsi*sinphi;
00229         real tmp2 = rc * sinpsi*cosphi;
00230 
00231         Real3D a2(0, ra * cosphi, ra * sinphi);
00232         Real3D b2(-rc * cospsi, rbphi - tmp1, -rb * sinphi + tmp2);
00233         Real3D c2( rc * cosphi, rbphi + tmp1, -rb * sinphi - tmp2);
00234 
00235         // --- Step3 al, be, ga ---
00236         // there are no a0 terms because we've already subtracted the term off
00237         // when we first defined b0 and c0.
00238         real alpha = b2[0] * (b0[0] - c0[0]) + b0[1] * b2[1] + c0[1] * c2[1];
00239         real beta  = b2[0] * (c0[1] - b0[1]) + b0[0] * b2[1] + c0[0] * c2[1];
00240         real gama  = b0[0] * b1[1] - b1[0] * b0[1] + c0[0] * c1[1] - c1[0] * c0[1];
00241 
00242         real a2b2 = alpha*alpha + beta*beta;
00243         real sintheta = (alpha*gama - beta*sqrt(a2b2 - gama*gama))/a2b2;
00244 
00245 
00246         // --- Step4 A3' ---
00247         real costheta = sqrt(1.0 - sintheta*sintheta);
00248 
00249         Real3D a3(-a2[1] * sintheta,
00250                         a2[1] * costheta,
00251                         A1Z);
00252 
00253         Real3D b3(b2[0] * costheta - b2[1] * sintheta,
00254                         b2[0] * sintheta + b2[1] * costheta,
00255                         b1[2]);
00256 
00257         Real3D c3(-b2[0] * costheta - c2[1] * sintheta,
00258                         -b2[0] * sintheta + c2[1] * costheta,
00259                         c1[2]);
00260 
00261 
00262         // --- Step5 A3 ---
00263         // undo the transformation; generate new normal vectors from the transpose.
00264         Real3D m1(n1[0], n2[0], n0[0]);
00265         Real3D m2(n1[1], n2[1], n0[1]);
00266         Real3D m0(n1[2], n2[2], n0[2]);
00267 
00268         // new positions
00269         O->position() = Real3D(a3*m1, a3*m2, a3*m0) + d0;
00270         H1->position() = Real3D(b3*m1, b3*m2, b3*m0) + d0;
00271         H2->position() = Real3D(c3*m1, c3*m2, c3*m0) + d0;
00272 
00273         //get unconstrained velocities at v(t+dt)
00274         Real3D displ1,displ2,displ3;
00275         bc.getMinimumImageVectorBox(displ1,O->getPos(),pos->second.first); // pos after settle - pos at prev timestep
00276         Real3D vO=displ1*invdt;
00277         bc.getMinimumImageVectorBox(displ2,H1->getPos(),pos->second.second);
00278         Real3D vH1=displ2*invdt;
00279         bc.getMinimumImageVectorBox(displ3,H2->getPos(),pos->second.third);
00280         Real3D vH2=displ3*invdt;
00281         O->setV(vO);
00282         H1->setV(vH1);
00283         H2->setV(vH2);        
00284 
00285     }
