41  real _mO, real _mH, real _distHH, real _distOH)

54  // initialize settlep variables
55  real rmT = 1.0 / (mO + mH + mH);
56  mOrmT = mO * rmT;

158  void Settle::settlep(longint molID){
159 
160  System& system = getSystemRef();
161  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
162  real dt = integrator->getTimeStep();
163  real invdt = 1.0/dt;
164 
165  //std::cout << "\nsettling " << molID << "\n";
166 
167  // positions in previous time step
168  OldPos::const_iterator pos = oldPos.find(molID);
169  if (pos == oldPos.end()) {
170  std::cout << "WARNING: oldPos not found! Skipping settlep().\n";
171  return;
172  }
173 
174  // retrieve particles from tuples
175  Particle* vp = system.storage->lookupRealParticle(molID);
176  FixedTupleListAdress::iterator it;
177  it = fixedTupleList->find(&(*vp));
178 
179  Particle* O = it->second.at(0);
180  Particle* H1 = it->second.at(1);
181  Particle* H2 = it->second.at(2);
182 
183  // --- Step1 A1' ---
184  // vectors in the plane of the original positions
185  // previous positions OHH
186  Real3D b0 = pos->second.second - pos->second.first; // H1.pos - O.pos
187  Real3D c0 = pos->second.third - pos->second.first; // H2.pos - O.pos
188 
189  // new center of mass
190  // present positions OHH
191  Real3D d0 = O->position() * mOrmT + ((H1->position() + H2->position())*mHrmT);
192 
193  Real3D a1 = O->position() - d0;
194  Real3D b1 = H1->position() - d0;
195  Real3D c1 = H2->position() - d0;
196 
197  // Vectors describing transformation from original coordinate system to
198  // the 'primed' coordinate system
199  Real3D n0 = b0.cross(c0);
200  Real3D n1 = a1.cross(n0);
201  Real3D n2 = n0.cross(n1);
202 
203  // unit vectors
204  n0 = n0/n0.abs();
205  n1 = n1/n1.abs();
206  n2 = n2/n2.abs();
207 
208  b0 = Real3D(n1*b0, n2*b0, n0*b0); // note: b0.z is never referenced again
209  c0 = Real3D(n1*c0, n2*c0, n0*c0); // note: c0.z is never referenced again
210 
211  // shuffle the order of operations around from the normal algorithm so
212  // that we can double pump sqrt() with n2 and cosphi at the same time
213  // these components are usually computed down in the canonical water block
214  real A1Z = n0 * a1;
215  b1 = Real3D(n1*b1, n2*b1, n0*b1);
216  c1 = Real3D(n1*c1, n2*c1, n0*c1);
217 
218  // --- Step2 A2' ---
219  // now we can compute positions of canonical water
220  real sinphi = A1Z * rra;
221  real tmp = 1.0 - sinphi*sinphi;
222  real cosphi = sqrt(tmp);
223  real sinpsi = (b1[2] - c1[2]) / (2.0 * rc * cosphi);
224  tmp = 1.0 - sinpsi*sinpsi;
225  real cospsi = sqrt(tmp);
226 
227  real rbphi = -rb * cosphi;
228  real tmp1 = rc * sinpsi*sinphi;
229  real tmp2 = rc * sinpsi*cosphi;
230 
231  Real3D a2(0, ra * cosphi, ra * sinphi);
232  Real3D b2(-rc * cospsi, rbphi - tmp1, -rb * sinphi + tmp2);
233  Real3D c2( rc * cosphi, rbphi + tmp1, -rb * sinphi - tmp2);
234 
235  // --- Step3 al, be, ga ---
236  // there are no a0 terms because we've already subtracted the term off
237  // when we first defined b0 and c0.
238  real alpha = b2[0] * (b0[0] - c0[0]) + b0[1] * b2[1] + c0[1] * c2[1];
239  real beta = b2[0] * (c0[1] - b0[1]) + b0[0] * b2[1] + c0[0] * c2[1];
240  real gama = b0[0] * b1[1] - b1[0] * b0[1] + c0[0] * c1[1] - c1[0] * c0[1];
241 
242  real a2b2 = alpha*alpha + beta*beta;
243  real sintheta = (alpha*gama - beta*sqrt(a2b2 - gama*gama))/a2b2;
244 
245 
246  // --- Step4 A3' ---
247  real costheta = sqrt(1.0 - sintheta*sintheta);
248 
249  Real3D a3(-a2[1] * sintheta,
250  a2[1] * costheta,
251  A1Z);
252 
253  Real3D b3(b2[0] * costheta - b2[1] * sintheta,
254  b2[0] * sintheta + b2[1] * costheta,
255  b1[2]);
256 
257  Real3D c3(-b2[0] * costheta - c2[1] * sintheta,
258  -b2[0] * sintheta + c2[1] * costheta,
259  c1[2]);
260 
261 
262  // --- Step5 A3 ---
263  // undo the transformation; generate new normal vectors from the transpose.
264  Real3D m1(n1[0], n2[0], n0[0]);
265  Real3D m2(n1[1], n2[1], n0[1]);
266  Real3D m0(n1[2], n2[2], n0[2]);
267 
268  // new positions
269  O->position() = Real3D(a3*m1, a3*m2, a3*m0) + d0;
270  H1->position() = Real3D(b3*m1, b3*m2, b3*m0) + d0;
271  H2->position() = Real3D(c3*m1, c3*m2, c3*m0) + d0;
272 
273  //get unconstrained velocities at v(t+dt)
274  Real3D displ1,displ2,displ3;
275  bc.getMinimumImageVectorBox(displ1,O->getPos(),pos->second.first); // pos after settle - pos at prev timestep
276  Real3D vO=displ1*invdt;
277  bc.getMinimumImageVectorBox(displ2,H1->getPos(),pos->second.second);
278  Real3D vH1=displ2*invdt;
279  bc.getMinimumImageVectorBox(displ3,H2->getPos(),pos->second.third);
280  Real3D vH2=displ3*invdt;
281  O->setV(vO);
282  H1->setV(vH1);
283  H2->setV(vH2);
284 
285  }

287  void Settle::settlev(longint molID){
288 
289  //settlev never called, not necessarily debugged
290 
291  real dt = integrator->getTimeStep();
292  real invdt = 1.0/dt;
293 
294  System& system = getSystemRef();
295  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
296 
297  // retrieve particles from tuples
298  Particle* vp = system.storage->lookupRealParticle(molID);
299  FixedTupleListAdress::iterator it;
300  it = fixedTupleList->find(&(*vp));
301 
302  Particle* O = it->second.at(0);
303  Particle* H1 = it->second.at(1);
304  Particle* H2 = it->second.at(2);
305 
306  Real3D vO = O->getV();
307  Real3D vH1 = H1->getV();
308  Real3D vH2 = H2->getV();
309 
310  //get unit vectors along bonds
311  Real3D rab;
312  Real3D rbc;
313  Real3D rca;
314 
315  bc.getMinimumImageVectorBox(rab,H1->getPos(),O->getPos());
316  bc.getMinimumImageVectorBox(rbc,H2->getPos(),H1->getPos());
317  bc.getMinimumImageVectorBox(rca,O->getPos(),H2->getPos());
318 
319  real rab2 = rab.sqr();
320  real rbc2 = rbc.sqr();
321  real rca2 = rca.sqr();
322 
323  real rab_abs = sqrt(rab2);
324  real rbc_abs = sqrt(rbc2);
325  real rca_abs = sqrt(rca2);
326 
327  Real3D eab = rab/rab_abs;
328  Real3D ebc = rbc/rbc_abs;
329  Real3D eca = rca/rca_abs;
330 
331  //get relative velocities
332  //Real3D vab0r = H1->getV() - O->getV();
333  //Real3D vbc0r = H2->getV() - H1->getV();
334  //Real3D vca0r = O->getV() - H2->getV();
335  Real3D vab0r = vH1 - vO;
336  Real3D vbc0r = vH2 - vH1;
337  Real3D vca0r = vO - vH2;
338 
339  //get components of relative velocities along bonds
340  real vab0 = eab * vab0r;
341  real vbc0 = ebc * vbc0r;
342  real vca0 = eca * vca0r;
343 
344  real cosA = (rca2 + rab2 - rbc2)/(2*rca_abs*rab_abs); //angles depend on constrained geom, no need to recalc, rewrite code to just do it once
345  real cosB = (rbc2 + rab2 - rca2)/(2*rbc_abs*rab_abs); //A=109.527, B=C=35.2819
346  real cosC = (rbc2 + rca2 - rab2)/(2*rbc_abs*rca_abs);
347  real interm1 = 2*mOmH2 + twicemO*mH*cosA*cosB*cosC - 2*mH2*cosA*cosA - mO*mOmH*(cosB*cosB+cosC*cosC);
348  real d = dt * interm1 / (twicemH);
349 
350  real interm2 = vab0 * (2*mOmH - mO*cosC*cosC) +
351  vbc0 * (mH*cosC*cosA - mOmH*cosB) +
352  vca0 * (mO*cosB*cosC - twicemH*cosA);
353  real tauab = mO*interm2/d;
354 
355  real interm3 = vbc0 * (mOmH2 - mH2*cosA*cosA) +
356  vca0 * mO * (mH*cosA*cosB - mOmH*cosC) +
357  vab0 * mO * (mH*cosC*cosA - mOmH*cosB);
358  real taubc = interm3/d;
359 
360  real interm4 = vca0 * (2*mOmH - mO*cosB*cosB) +
361  vab0 * (mO*cosB*cosC - twicemH*cosA) +
362  vbc0 * (mH*cosA*cosB - mOmH*cosC);
363  real tauca = mO*interm4/d;
364 
365  Real3D newVa = vO + dt/twicemO*(tauab*eab - tauca*eca);
366  Real3D newVb = vH1 + dt/twicemH*(taubc*ebc - tauab*eab);
367  Real3D newVc = vH2 + dt/twicemH*(tauca*eca - taubc*ebc);
368 
369  O->setV(newVa);
370  H1->setV(newVb);
371  H2->setV(newVc);
372 
373  }
