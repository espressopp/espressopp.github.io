59  void DomainDecompositionNonBlocking::decomposeRealParticles() {
60 
61  //std::cout << getSystem()->comm->rank() << ": " << " decomposeRealParticles\n";
62 
63  LOG4ESPP_DEBUG(logger, "starting, expected comm buffer size " << exchangeBufferSize);
64 
65  // allocate send/recv buffers. We use the size as we need maximally so far, to avoid reallocation
66  // TODO: This might be a problem when all particles are created on a single node initially!
67  ParticleList sendBufL;
68  sendBufL.reserve(exchangeBufferSize);
69  ParticleList sendBufR;
70  sendBufR.reserve(exchangeBufferSize);
71  ParticleList recvBufL;
72  recvBufL.reserve(exchangeBufferSize);
73  ParticleList recvBufR;
74  recvBufR.reserve(exchangeBufferSize);
75 
76  bool allFinished;
77  do {
78  bool finished = true;
79 
80  for (int coord = 0; coord < 3; ++coord) {
81  LOG4ESPP_DEBUG(logger, "starting with direction " << coord);
82 
83  if (nodeGrid.getGridSize(coord) > 1) {
84  for (std::vector<Cell*>::iterator it = realCells.begin(),
85  end = realCells.end(); it != end; ++it) {
86 
87  Cell &cell = **it;
88 
89  // do not use an iterator here, since we need to take out particles during the loop
90  for (size_t p = 0; p < cell.particles.size(); ++p) {
91  Particle &part = cell.particles[p];
92  const Real3D& pos = part.position();
93 
94  // check whether the particle is now "left" of the local domain
95  if (pos[coord] - cellGrid.getMyLeft(coord) < -ROUND_ERROR_PREC) {
96  LOG4ESPP_TRACE(logger, "send particle left " << part.id());
97  moveIndexedParticle(sendBufL, cell.particles, p);
98  // redo same particle since we took one out here, so it's a new one
99  --p;
100  }
101  // check whether the particle is now "right" of the local domain
102  else if (pos[coord] - cellGrid.getMyRight(coord) >= ROUND_ERROR_PREC) {
103  LOG4ESPP_TRACE(logger, "send particle right " << part.id());
104  moveIndexedParticle(sendBufR, cell.particles, p);
105  --p;
106  }
107  // Sort particles in cells of this node during last direction
108  else if (coord == 2) {
109  const Real3D& pos = part.position();
110  Cell *sortCell = mapPositionToCellChecked(pos);
111  if (sortCell != &cell) {
112  if (sortCell == 0) {
113  // particle is not in the local domain
114  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
115  << " @ " << pos <<
116  " is not inside node domain after neighbor exchange");
117  // isnan function is C99 only, x != x is only true if x == nan
118  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
119  // TODO: error handling
120  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
121  " has moved to outer space (one or more coordinates are nan)");
122  } else {
123  // particle stays where it is, and will be sorted in the next round
124  finished = false;
125  }
126  } else {
127  // particle is in the local domain
128  moveIndexedParticle(sortCell->particles, cell.particles, p);
129  --p;
130  }
131  }
132  }
133 
134  }
135  }
136 
137  mpi::request reqs[4];
138 
139  // Exchange particles, odd-even rule
140  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
141  reqs[0]=isendParticles(outBufferL, sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
142  reqs[1]=irecvParticles_initiate( inBufferR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
143  reqs[2]=isendParticles(outBufferR, sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
144  reqs[3]=irecvParticles_initiate( inBufferL, nodeGrid.getNodeNeighborIndex(2 * coord));
145  } else {
146  reqs[0]=irecvParticles_initiate( inBufferR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
147  reqs[1]=isendParticles(outBufferL, sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
148  reqs[2]=irecvParticles_initiate( inBufferL, nodeGrid.getNodeNeighborIndex(2 * coord));
149  reqs[3]=isendParticles(outBufferR, sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
150  }
151 
152  mpi::wait_all(reqs, reqs + 4);
153 
154  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
155  irecvParticles_finish( inBufferR, recvBufR);
156  irecvParticles_finish( inBufferL, recvBufL);
157  } else {
158  irecvParticles_finish( inBufferR, recvBufR);
159  irecvParticles_finish( inBufferL, recvBufL);
160  }
161 
162  // sort received particles to cells
163  if (appendParticles(recvBufL, 2 * coord) && coord == 2) finished = false;
164  if (appendParticles(recvBufR, 2 * coord + 1) && coord == 2) finished = false;
165 
166  // reset send/recv buffers
167  sendBufL.resize(0);
168  sendBufR.resize(0);
169  recvBufL.resize(0);
170  recvBufR.resize(0);
171 
172 
173  } else {
174  /* Single node direction case (no communication)
175  Fold particles that have left the box */
176  for (std::vector< Cell* >::iterator it = realCells.begin(),
177  end = realCells.end(); it != end; ++it) {
178  Cell &cell = **it;
179  // do not use an iterator here, since we have need to take out particles during the loop
180  for (size_t p = 0; p < cell.particles.size(); ++p) {
181  Particle &part = cell.particles[p];
182  getSystem()->bc->foldCoordinate(part.position(), part.image(), coord);
183  LOG4ESPP_TRACE(logger, "folded coordinate " << coord << " of particle " << part.id());
184 
185  if (coord == 2) {
186  Cell *sortCell = mapPositionToCellChecked(part.position());
187 
188  if (sortCell != &cell) {
189  if (sortCell == 0) {
190  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
191  << " @ " << part.position()
192  << " is not inside node domain after neighbor exchange");
193  const Real3D& pos = part.position();
194  // isnan function is C99 only, x != x is only true if x == nan
195  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
196  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
197  " has moved to outer space (one or more coordinates are nan)");
198  } else {
199  // particle stays where it is, and will be sorted in the next round
200  finished = false;
201  }
202  } else {
203  moveIndexedParticle(sortCell->particles, cell.particles, p);
204  --p;
205  }
206  }
207  }
208  }
209  }
210  }
211 
212  LOG4ESPP_DEBUG(logger, "done with direction " << coord);
213  }
214 
215  // Communicate wether particle exchange is finished
216  mpi::all_reduce(*getSystem()->comm, finished, allFinished, std::logical_and<bool>());
217  } while (!allFinished);
218 
219  exchangeBufferSize = std::max(exchangeBufferSize,
220  std::max(sendBufL.capacity(),
221  std::max(sendBufR.capacity(),
222  std::max(recvBufL.capacity(),
223  recvBufR.capacity()))));
224 
225  LOG4ESPP_DEBUG(logger, "finished exchanging particles, new send/recv buffer size " << exchangeBufferSize);
226 
227  LOG4ESPP_DEBUG(logger, "done");
228  }

230  void DomainDecompositionNonBlocking::
231  doGhostCommunication(bool sizesFirst, bool realToGhosts, int extradata) {
232  LOG4ESPP_DEBUG(logger, "do ghost communication " << (sizesFirst ? "with sizes " : "")
233  << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);
234 
235  /* direction loop: x, y, z.
236  Here we could in principle build in a one sided ghost
237  communication, simply by taking the lr loop only over one
238  value. */
239  for (int _coord = 0; _coord < 3; ++_coord) {
240  /* inverted processing order for ghost force communication,
241  since the corner ghosts have to be collected via several
242  nodes. We now add back the corner ghost forces first again
243  to ghost forces, which only eventually go back to the real
244  particle.
245  */
246  int coord = realToGhosts ? _coord : (2 - _coord);
247  real curCoordBoxL = getSystem()->bc->getBoxL()[coord];
248 
249  // lr loop: left right
250  for (int lr = 0; lr < 2; ++lr) {
251  int dir = 2 * coord + lr;
252  int oppositeDir = 2 * coord + (1 - lr);
253 
254  Real3D shift(0, 0, 0);
255 
256  shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
257 
258  LOG4ESPP_DEBUG(logger, "direction " << dir);
259 
260  if (nodeGrid.getGridSize(coord) == 1) {
261  LOG4ESPP_DEBUG(logger, "local communication");
262 
263  // copy operation, we have to receive as many cells as we send
264  if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
265  throw std::runtime_error("DomainDecomposition::doGhostCommunication: send/recv cell structure mismatch during local copy");
266  }
267 
268  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
269  if (realToGhosts) {
270  copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
271  } else {
272  addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
273  }
274  }
275  }
276  else {
277  // exchange size information, if necessary
278  if (sizesFirst) {
279  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");
280 
281  // prepare buffers
282  std::vector<longint> sendSizes, recvSizes;
283  sendSizes.reserve(commCells[dir].reals.size());
284  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
285  sendSizes.push_back(commCells[dir].reals[i]->particles.size());
286  }
287  recvSizes.resize(commCells[dir].ghosts.size());
288 
289  mpi::request reqs[2];
290 
291  // exchange sizes, odd-even rule
292  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
293  LOG4ESPP_DEBUG(logger, "sending to node " << nodeGrid.getNodeNeighborIndex(dir)
294  << ", then receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir));
295  reqs[0]=getSystem()->comm->isend(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
296  reqs[1]=getSystem()->comm->irecv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
297  }
298  else {
299  LOG4ESPP_DEBUG(logger, "receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir)
300  << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
301  reqs[0]=getSystem()->comm->irecv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
302  reqs[1]=getSystem()->comm->isend(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
303  }
304 
305  mpi::wait_all(reqs, reqs + 2);
306 
307  // resize according to received information
308  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
309  commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
310  }
311  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
312  }
313 
314  // prepare send and receive buffers
315  longint receiver, sender;
316  outBufferG.reset();
317  if (realToGhosts) {
318  receiver = nodeGrid.getNodeNeighborIndex(dir);
319  sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
320  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
321  packPositionsEtc(outBufferG, *commCells[dir].reals[i], extradata, shift);
322  }
323  }
324  else {
325  receiver = nodeGrid.getNodeNeighborIndex(oppositeDir);
326  sender = nodeGrid.getNodeNeighborIndex(dir);
327  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
328  packForces(outBufferG, *commCells[dir].ghosts[i]);
329  }
330  }
331 
332  mpi::request reqs[2];
333 
334  // exchange particles, odd-even rule
335  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
336  reqs[0]=outBufferG.isend(receiver, DD_COMM_TAG);
337  reqs[1]=inBufferG.irecv(sender, DD_COMM_TAG);
338  } else {
339  reqs[0]=inBufferG.irecv(sender, DD_COMM_TAG);
340  reqs[1]=outBufferG.isend(receiver, DD_COMM_TAG);
341  }
342 
343  mpi::wait_all(reqs, reqs + 2);
344 
345  // unpack received data
346  if (realToGhosts) {
347  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
348  unpackPositionsEtc(*commCells[dir].ghosts[i], inBufferG, extradata);
349  }
350  }
351  else {
352  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
353  unpackAndAddForces(*commCells[dir].reals[i], inBufferG);
354  }
355  }
356  }
357  }
358  }
359  LOG4ESPP_DEBUG(logger, "ghost communication finished");
360  }

362  mpi::request DomainDecompositionNonBlocking::isendParticles(OutBuffer &data, ParticleList &list, longint node)
363  {
364  LOG4ESPP_DEBUG(logger, "initiate non blocking isend " << list.size() << " particles to " << node);
365  data.reset();
366  int size = list.size();
367  data.write(size);

385  void DomainDecompositionNonBlocking::irecvParticles_finish(InBuffer &data, ParticleList &list)
386  {
387  LOG4ESPP_DEBUG(logger, "finish non blocking irecv");
388  // ... and unpack
389  int size;
390  data.read(size);
391  int curSize = list.size();
392  LOG4ESPP_DEBUG(logger, "got " << size << " particles, have " << curSize);
393 
394  if (size > 0) {
395  list.resize(curSize + size);
396 
397  for (int i = 0; i < size; ++i) {
398  Particle *p = &list[curSize + i];
399  data.read(*p);
400  updateInLocalParticles(p);
401  }
402 
403  afterRecvParticles(list, data); // this also takes care of AdResS AT Particles
404  }
405  LOG4ESPP_DEBUG(logger, "done");
