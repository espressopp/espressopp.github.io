113  template < typename _AngularPotential > inline void
114  FixedTripleAngleListInteractionTemplate <_AngularPotential>::
115  addForces() {
116  LOG4ESPP_INFO(theLogger, "add forces computed by FixedTripleAngleList");
117  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
118  for (FixedTripleAngleList::TripleList::Iterator it(*fixedtripleList); it.isValid(); ++it) {
119  Particle &p1 = *it->first;
120  Particle &p2 = *it->second;
121  Particle &p3 = *it->third;
122  //const Potential &potential = getPotential(p1.type(), p2.type());
123  Real3D dist12, dist32;
124  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
125  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
126  Real3D force12, force32;
127 
128  real currentAngle = fixedtripleList->getAngle(p1.getId(), p2.getId(), p3.getId());
129 
130  potential->_computeForce(force12, force32, dist12, dist32, currentAngle);
131  p1.force() += force12;
132  p2.force() -= force12 + force32;
133  p3.force() += force32;
134  }
135  }

141 
142  const bc::BC& bc = *getSystemRef().bc;

148  //const Potential &potential = getPotential(p1.type(), p2.type());
149  Real3D dist12 = bc.getMinimumImageVector(p1.position(), p2.position());
150  Real3D dist32 = bc.getMinimumImageVector(p3.position(), p2.position());
151 
152  real currentAngle = fixedtripleList->getAngle(p1.getId(), p2.getId(), p3.getId());

154  e += potential->_computeEnergy(dist12, dist32, currentAngle);
155  }
156  real esum;
157  boost::mpi::all_reduce(*mpiWorld, e, esum, std::plus<real>());
158  return esum;
