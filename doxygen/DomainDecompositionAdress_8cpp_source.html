782 
783  // allocate send/recv buffers. We use the size as we need maximally so far, to avoid reallocation
784  // TODO: This might be a problem when all particles are created on a single node initially!
785  ParticleList sendBufL;
786  sendBufL.reserve(exchangeBufferSize);
787  ParticleList sendBufR;
788  sendBufR.reserve(exchangeBufferSize);
789  ParticleList recvBufL;
790  recvBufL.reserve(exchangeBufferSize);
791  ParticleList recvBufR;
792  recvBufR.reserve(exchangeBufferSize);
793 
794  bool allFinished;
795  do {
796  bool finished = true;
797 
798  for (int coord = 0; coord < 3; ++coord) {
799  LOG4ESPP_DEBUG(logger, "starting with direction " << coord);
800 
801  if (nodeGrid.getGridSize(coord) > 1) {
802  for (std::vector<Cell*>::iterator it = realCells.begin(),
803  end = realCells.end(); it != end; ++it) {
804 
805  Cell &cell = **it;
806 
807  // do not use an iterator here, since we need to take out particles during the loop
808  for (size_t p = 0; p < cell.particles.size(); ++p) {
809  Particle &part = cell.particles[p];
810  const Real3D& pos = part.position();
811 
812  // check whether the particle is now "left" of the local domain
813  if (pos[coord] - cellGrid.getMyLeft(coord) < -ROUND_ERROR_PREC) {
814  LOG4ESPP_TRACE(logger, "send particle left " << part.id());
815  moveIndexedParticle(sendBufL, cell.particles, p);
816  // redo same particle since we took one out here, so it's a new one
817  --p;
818  }
819  // check whether the particle is now "right" of the local domain
820  else if (pos[coord] - cellGrid.getMyRight(coord) >= ROUND_ERROR_PREC) {
821  LOG4ESPP_TRACE(logger, "send particle right " << part.id());
822  moveIndexedParticle(sendBufR, cell.particles, p);
823  --p;
824  }
825  // Sort particles in cells of this node during last direction
826  else if (coord == 2) {
827  const Real3D& pos = part.position();
828  Cell *sortCell = mapPositionToCellChecked(pos);
829  if (sortCell != &cell) {
830  if (sortCell == 0) {
831  // particle is not in the local domain
832  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
833  << " @ " << pos <<
834  " is not inside node domain after neighbor exchange");
835  // isnan function is C99 only, x != x is only true if x == nan
836  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
837  // TODO: error handling
838  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
839  " has moved to outer space (one or more coordinates are nan)");
840  } else {
841  // particle stays where it is, and will be sorted in the next round
842  finished = false;
843  }
844  } else {
845  // particle is in the local domain
846  moveIndexedParticle(sortCell->particles, cell.particles, p);
847  --p;
848  }
849  }
850  }
851 
852  }

854 
855  // Exchange particles, odd-even rule
856  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
857  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
858  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
859  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
860  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
861  } else {
862  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
863  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
864  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
865  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
866  }

869  if (appendParticles(recvBufL, 2 * coord ) && coord == 2) finished = false;
870  if (appendParticles(recvBufR, 2 * coord + 1) && coord == 2) finished = false;
871 
872  // reset send/recv buffers
873  sendBufL.resize(0);
874  sendBufR.resize(0);
875  recvBufL.resize(0);
876  recvBufR.resize(0);
877 
878 
879  } else {
880  /* Single node direction case (no communication)
881  Fold particles that have left the box */
882  for (std::vector< Cell* >::iterator it = realCells.begin(),
883  end = realCells.end(); it != end; ++it) {
884  Cell &cell = **it;
885  // do not use an iterator here, since we have need to take out particles during the loop
886  for (size_t p = 0; p < cell.particles.size(); ++p) {
887  Particle &part = cell.particles[p];
888  getSystem()->bc->foldCoordinate(part.position(), part.image(), coord);
889  LOG4ESPP_TRACE(logger, "folded coordinate " << coord << " of particle " << part.id());
890 
891  if (coord == 2) {
892  Cell *sortCell = mapPositionToCellChecked(part.position());
893 
894  if (sortCell != &cell) {
895  if (sortCell == 0) {
896  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
897  << " @ " << part.position()
898  << " is not inside node domain after neighbor exchange");
899  const Real3D& pos = part.position();
900  // isnan function is C99 only, x != x is only true if x == nan
901  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
902  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
903  " has moved to outer space (one or more coordinates are nan)");
904  } else {
905  // particle stays where it is, and will be sorted in the next round
906  finished = false;
907  }
908  } else {
909  moveIndexedParticle(sortCell->particles, cell.particles, p);
910  --p;
911  }
912  }

914  }

927  std::max(sendBufR.capacity(),
928  std::max(recvBufL.capacity(),
929  recvBufR.capacity()))));
930 
931  LOG4ESPP_DEBUG(logger, "finished exchanging particles, new send/recv buffer size " << exchangeBufferSize);
932 
933  LOG4ESPP_DEBUG(logger, "done");

1002  if (otherCoord < coord) {
1003  leftBoundary[otherCoord] = 0;
1004  rightBoundary[otherCoord] = cellGrid.getFrameGridSize(otherCoord);
1005  } else {

1007  rightBoundary[otherCoord] = cellGrid.getInnerCellsEnd(otherCoord);

1042  void DomainDecompositionAdress::
1043  doGhostCommunication(bool sizesFirst, bool realToGhosts, int extradata) {
1044  LOG4ESPP_DEBUG(logger, "do ghost communication " << (sizesFirst ? "with sizes " : "")
1045  << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);
1046 
1047  /* direction loop: x, y, z.
1048  Here we could in principle build in a one sided ghost
1049  communication, simply by taking the lr loop only over one
1050  value. */
1051  for (int _coord = 0; _coord < 3; ++_coord) {
1052  /* inverted processing order for ghost force communication,
1053  since the corner ghosts have to be collected via several
1054  nodes. We now add back the corner ghost forces first again
1055  to ghost forces, which only eventually go back to the real
1056  particle.
1057  */
1058  int coord = realToGhosts ? _coord : (2 - _coord);
1059  real curCoordBoxL = getSystem()->bc->getBoxL()[coord];
1060 
1061  // lr loop: left right
1062  for (int lr = 0; lr < 2; ++lr) {
1063  int dir = 2 * coord + lr;
1064  int oppositeDir = 2 * coord + (1 - lr);
1065 
1066  Real3D shift(0, 0, 0);
1067 
1068  shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
1069 
1070  LOG4ESPP_DEBUG(logger, "direction " << dir);
1071 
1072  if (nodeGrid.getGridSize(coord) == 1) {
1073  LOG4ESPP_DEBUG(logger, "local communication");
1074 
1075  // copy operation, we have to receive as many cells as we send
1076  if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
1077  throw std::runtime_error("DomainDecompositionAdress::doGhostCommunication: send/recv cell structure mismatch during local copy");
1078  }
1079 
1080  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
1081  if (realToGhosts) {
1082  copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
1083  } else {
1084  addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
1085  }
1086  }
1087  }
1088  else {
1089  // exchange size information, if necessary
1090  if (sizesFirst) {
1091  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");
1092 
1093  // prepare buffers
1094  std::vector<longint> sendSizes, recvSizes;
1095  sendSizes.reserve(commCells[dir].reals.size());
1096  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1097  sendSizes.push_back(commCells[dir].reals[i]->particles.size());
1098  }
1099  recvSizes.resize(commCells[dir].ghosts.size());
1100 
1101  // exchange sizes, odd-even rule
1102  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
1103  LOG4ESPP_DEBUG(logger, "sending to node " << nodeGrid.getNodeNeighborIndex(dir)
1104  << ", then receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir));
1105  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
1106  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
1107  }
1108  else {
1109  LOG4ESPP_DEBUG(logger, "receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir)
1110  << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
1111  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
1112  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
1113  }
1114 
1115  // resize according to received information
1116  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
1117  commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
1118  }
1119  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
1120  }
1121 
1122  // prepare send and receive buffers
1123  longint receiver, sender;
1124  outBuffer.reset();
1125  if (realToGhosts) {
1126  receiver = nodeGrid.getNodeNeighborIndex(dir);
1127  sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
1128  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1129  packPositionsEtc(outBuffer, *commCells[dir].reals[i], extradata, shift);
1130  }
1131  }
1132  else {
1133  receiver = nodeGrid.getNodeNeighborIndex(oppositeDir);
1134  sender = nodeGrid.getNodeNeighborIndex(dir);
1135  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
1136  packForces(outBuffer, *commCells[dir].ghosts[i]);
1137  }
1138  }
1139 
1140  // exchange particles, odd-even rule
1141  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
1142  outBuffer.send(receiver, DD_COMM_TAG);
1143  inBuffer.recv(sender, DD_COMM_TAG);
1144  } else {
1145  inBuffer.recv(sender, DD_COMM_TAG);
1146  outBuffer.send(receiver, DD_COMM_TAG);
1147  }
1148 
1149  // unpack received data
1150  if (realToGhosts) {
1151  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1152  unpackPositionsEtc(*commCells[dir].ghosts[i], inBuffer, extradata);
1153  }
1154  }
1155  else {
1156  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1157  unpackAndAddForces(*commCells[dir].reals[i], inBuffer);
1158  }
1159  }
1160  }
1161  }
1162  }
