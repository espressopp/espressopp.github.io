178  Real3D box_sizeL = getSystem() -> bc -> getBoxL();

191

195  tmp_pl.reserve( _N );

201  invalidateGhosts();
202  cells.clear();
203  localCells.clear();
204  realCells.clear();
205  ghostCells.clear();

207  commCells[i].reals.clear();

209  }
210 
211  // creating new grids
212  createCellGrid(_nodeGrid, _newCellGrid);
213  initCellInteractions();
214  prepareGhostCommunication();
215 
216  // pushing the particles back to the empty cell ("do we have to check particles?")

243  void DomainDecompositionAdress::initCellInteractions() {
244  LOG4ESPP_DEBUG(logger, "setting up neighbors for " << cells.size() << " cells");
245 
246  for (int o = cellGrid.getInnerCellsBegin(2); o < cellGrid.getInnerCellsEnd(2); ++o) {
247  for (int n = cellGrid.getInnerCellsBegin(1); n < cellGrid.getInnerCellsEnd(1); ++n) {
248  for (int m = cellGrid.getInnerCellsBegin(0); m < cellGrid.getInnerCellsEnd(0); ++m) {
249  longint cellIdx = cellGrid.mapPositionToIndex(m, n, o);
250  Cell *cell = &cells[cellIdx];
251 
252  LOG4ESPP_TRACE(logger, "setting up neighbors for cell " << cell - getFirstCell()
253  << " @ " << m << " " << n << " " << o);
254 
255  // there should be always 26 neighbors
256  cell->neighborCells.reserve(26);
257 
258  // loop all neighbor cells
259  for (int p = o - 1; p <= o + 1; ++p) {
260  for (int q = n - 1; q <= n + 1; ++q) {
261  for (int r = m - 1; r <= m + 1; ++r) {
262  if (p != o || q != n || r != m) {
263  longint cell2Idx = cellGrid.mapPositionToIndex(r, q, p);
264  Cell *cell2 = &cells[cell2Idx];
265  cell->neighborCells.push_back(NeighborCellInfo(cell2, (cell2Idx<cellIdx)));
266 
267  LOG4ESPP_TRACE(logger, "neighbor cell " << cell2 - getFirstCell()
268  << " @ " << r << " " << q << " " << p << ((cell2Idx<cellIdx) ? " is" : " is not") << " taken" );
269  }
270  }
271  }
272  }
273  }
274  }
275  }

277  LOG4ESPP_DEBUG(logger, "done");

775  void DomainDecompositionAdress::decomposeRealParticles() {
776 
777  //std::cout << getSystem()->comm->rank() << ": " << " decomposeRealParticles\n";
778 
779  LOG4ESPP_DEBUG(logger, "starting, expected comm buffer size " << exchangeBufferSize);
780 
781  // allocate send/recv buffers. We use the size as we need maximally so far, to avoid reallocation
782  // TODO: This might be a problem when all particles are created on a single node initially!
783  ParticleList sendBufL;
784  sendBufL.reserve(exchangeBufferSize);
785  ParticleList sendBufR;
786  sendBufR.reserve(exchangeBufferSize);
787  ParticleList recvBufL;
788  recvBufL.reserve(exchangeBufferSize);
789  ParticleList recvBufR;
790  recvBufR.reserve(exchangeBufferSize);
791 
792  bool allFinished;
793  do {
794  bool finished = true;
795 
796  for (int coord = 0; coord < 3; ++coord) {
797  LOG4ESPP_DEBUG(logger, "starting with direction " << coord);
798 
799  if (nodeGrid.getGridSize(coord) > 1) {
800  for (std::vector<Cell*>::iterator it = realCells.begin(),
801  end = realCells.end(); it != end; ++it) {
802 
803  Cell &cell = **it;
804 
805  // do not use an iterator here, since we need to take out particles during the loop
806  for (size_t p = 0; p < cell.particles.size(); ++p) {
807  Particle &part = cell.particles[p];
808  const Real3D& pos = part.position();
809 
810  // check whether the particle is now "left" of the local domain
811  if (pos[coord] - cellGrid.getMyLeft(coord) < -ROUND_ERROR_PREC) {
812  LOG4ESPP_TRACE(logger, "send particle left " << part.id());
813  moveIndexedParticle(sendBufL, cell.particles, p);
814  // redo same particle since we took one out here, so it's a new one
815  --p;
816  }
817  // check whether the particle is now "right" of the local domain
818  else if (pos[coord] - cellGrid.getMyRight(coord) >= ROUND_ERROR_PREC) {
819  LOG4ESPP_TRACE(logger, "send particle right " << part.id());
820  moveIndexedParticle(sendBufR, cell.particles, p);
821  --p;
822  }
823  // Sort particles in cells of this node during last direction
824  else if (coord == 2) {
825  const Real3D& pos = part.position();
826  Cell *sortCell = mapPositionToCellChecked(pos);
827  if (sortCell != &cell) {
828  if (sortCell == 0) {
829  // particle is not in the local domain
830  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
831  << " @ " << pos <<
832  " is not inside node domain after neighbor exchange");
833  // isnan function is C99 only, x != x is only true if x == nan
834  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
835  // TODO: error handling
836  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
837  " has moved to outer space (one or more coordinates are nan)");
838  } else {
839  // particle stays where it is, and will be sorted in the next round
840  finished = false;
841  }
842  } else {
843  // particle is in the local domain
844  moveIndexedParticle(sortCell->particles, cell.particles, p);
845  --p;
846  }
847  }
848  }
849 
850  }
851  }
852 
853  // Exchange particles, odd-even rule
854  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
855  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
856  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
857  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
858  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
859  } else {
860  recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
861  sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
862  recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
863  sendParticles(sendBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
864  }
865 
866  // sort received particles to cells
867  if (appendParticles(recvBufL, 2 * coord ) && coord == 2) finished = false;
868  if (appendParticles(recvBufR, 2 * coord + 1) && coord == 2) finished = false;
869 
870  // reset send/recv buffers
871  sendBufL.resize(0);
872  sendBufR.resize(0);
873  recvBufL.resize(0);
874  recvBufR.resize(0);
875 
876 
877  } else {
878  /* Single node direction case (no communication)
879  Fold particles that have left the box */
880  for (std::vector< Cell* >::iterator it = realCells.begin(),
881  end = realCells.end(); it != end; ++it) {
882  Cell &cell = **it;
883  // do not use an iterator here, since we have need to take out particles during the loop
884  for (size_t p = 0; p < cell.particles.size(); ++p) {
885  Particle &part = cell.particles[p];
886  getSystem()->bc->foldCoordinate(part.position(), part.image(), coord);
887  LOG4ESPP_TRACE(logger, "folded coordinate " << coord << " of particle " << part.id());
888 
889  if (coord == 2) {
890  Cell *sortCell = mapPositionToCellChecked(part.position());
891 
892  if (sortCell != &cell) {
893  if (sortCell == 0) {
894  LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
895  << " @ " << part.position()
896  << " is not inside node domain after neighbor exchange");
897  const Real3D& pos = part.position();
898  // isnan function is C99 only, x != x is only true if x == nan
899  if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
900  LOG4ESPP_ERROR(logger, "particle " << part.id() <<
901  " has moved to outer space (one or more coordinates are nan)");
902  } else {
903  // particle stays where it is, and will be sorted in the next round
904  finished = false;
905  }
906  } else {
907  moveIndexedParticle(sortCell->particles, cell.particles, p);
908  --p;
909  }
910  }
911  }
912  }
913  }
914  }
915 
916  LOG4ESPP_DEBUG(logger, "done with direction " << coord);
917  }
918 
919  // Communicate wether particle exchange is finished
920  mpi::all_reduce(*getSystem()->comm, finished, allFinished, std::logical_and<bool>());
921  } while (!allFinished);
922 
923  exchangeBufferSize = std::max(exchangeBufferSize,
924  std::max(sendBufL.capacity(),
925  std::max(sendBufR.capacity(),
926  std::max(recvBufL.capacity(),
927  recvBufR.capacity()))));
928 
929  LOG4ESPP_DEBUG(logger, "finished exchanging particles, new send/recv buffer size " << exchangeBufferSize);
930 
931  LOG4ESPP_DEBUG(logger, "done");
932  }

987  void DomainDecompositionAdress::prepareGhostCommunication() {
988  // direction loop: x, y, z
989  for (int coord = 0; coord < 3; ++coord) {
990  // boundaries of area to send
991  int leftBoundary[3], rightBoundary[3];
992  /* boundaries perpendicular directions are the same for left/right send.
993  We also send the ghost frame that we have already, so the data amount
994  increase with each cycle.
995 
996  For a direction that was done already, i.e. is smaller than dir,
997  we take the full ghost frame, otherwise only the inner frame. */
998  for (int offset = 1; offset <= 2; ++offset) {
999  int otherCoord = (coord + offset) % 3;
1000  if (otherCoord < coord) {
1001  leftBoundary[otherCoord] = 0;
1002  rightBoundary[otherCoord] = cellGrid.getFrameGridSize(otherCoord);
1003  } else {
1004  leftBoundary[otherCoord] = cellGrid.getInnerCellsBegin(otherCoord);
1005  rightBoundary[otherCoord] = cellGrid.getInnerCellsEnd(otherCoord);
1006  }
1007  }
1008 
1009  // lr loop: left right - loop
1010  for (int lr = 0; lr < 2; ++lr) {
1011  int dir = 2 * coord + lr;
1012 
1013  /* participating real particles from this node */
1014  LOG4ESPP_DEBUG(logger, "direction " << dir << " reals");
1015 
1016  if (lr == 0) {
1017  leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
1018  rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord) + cellGrid.getFrameWidth();
1019  } else {
1020  leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord) - cellGrid.getFrameWidth();
1021  rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
1022  }
1023  fillCells(commCells[dir].reals, leftBoundary, rightBoundary);
1024 
1025  /* participating ghosts from this node */
1026  LOG4ESPP_DEBUG(logger, "direction " << dir << " ghosts");
1027 
1028  if (lr == 0) {
1029  leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
1030  rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord) + cellGrid.getFrameWidth();
1031  } else {
1032  leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord) - cellGrid.getFrameWidth();
1033  rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
1034  }
1035  fillCells(commCells[dir].ghosts, leftBoundary, rightBoundary);
1036  }
1037  }

1040  void DomainDecompositionAdress::
1041  doGhostCommunication(bool sizesFirst, bool realToGhosts, int extradata) {
1042  LOG4ESPP_DEBUG(logger, "do ghost communication " << (sizesFirst ? "with sizes " : "")
1043  << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);
1044 
1045  /* direction loop: x, y, z.
1046  Here we could in principle build in a one sided ghost
1047  communication, simply by taking the lr loop only over one
1048  value. */
1049  for (int _coord = 0; _coord < 3; ++_coord) {
1050  /* inverted processing order for ghost force communication,
1051  since the corner ghosts have to be collected via several
1052  nodes. We now add back the corner ghost forces first again
1053  to ghost forces, which only eventually go back to the real
1054  particle.
1055  */
1056  int coord = realToGhosts ? _coord : (2 - _coord);
1057  real curCoordBoxL = getSystem()->bc->getBoxL()[coord];
1058 
1059  // lr loop: left right
1060  for (int lr = 0; lr < 2; ++lr) {
1061  int dir = 2 * coord + lr;
1062  int oppositeDir = 2 * coord + (1 - lr);
1063 
1064  Real3D shift(0, 0, 0);
1065 
1066  shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
1067 
1068  LOG4ESPP_DEBUG(logger, "direction " << dir);
1069 
1070  if (nodeGrid.getGridSize(coord) == 1) {
1071  LOG4ESPP_DEBUG(logger, "local communication");
1072 
1073  // copy operation, we have to receive as many cells as we send
1074  if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
1075  throw std::runtime_error("DomainDecompositionAdress::doGhostCommunication: send/recv cell structure mismatch during local copy");
1076  }
1077 
1078  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
1079  if (realToGhosts) {
1080  copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
1081  } else {
1082  addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
1083  }
1084  }
1085  }
1086  else {
1087  // exchange size information, if necessary
1088  if (sizesFirst) {
1089  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");
1090 
1091  // prepare buffers
1092  std::vector<longint> sendSizes, recvSizes;
1093  sendSizes.reserve(commCells[dir].reals.size());
1094  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1095  sendSizes.push_back(commCells[dir].reals[i]->particles.size());
1096  }
1097  recvSizes.resize(commCells[dir].ghosts.size());
1098 
1099  // exchange sizes, odd-even rule
1100  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
1101  LOG4ESPP_DEBUG(logger, "sending to node " << nodeGrid.getNodeNeighborIndex(dir)
1102  << ", then receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir));
1103  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
1104  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
1105  }
1106  else {
1107  LOG4ESPP_DEBUG(logger, "receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir)
1108  << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
1109  getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
1110  getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
1111  }
1112 
1113  // resize according to received information
1114  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
1115  commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
1116  }
1117  LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
1118  }
1119 
1120  // prepare send and receive buffers
1121  longint receiver, sender;
1122  outBuffer.reset();
1123  if (realToGhosts) {
1124  receiver = nodeGrid.getNodeNeighborIndex(dir);
1125  sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
1126  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1127  packPositionsEtc(outBuffer, *commCells[dir].reals[i], extradata, shift);
1128  }
1129  }
1130  else {
1131  receiver = nodeGrid.getNodeNeighborIndex(oppositeDir);
1132  sender = nodeGrid.getNodeNeighborIndex(dir);
1133  for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
1134  packForces(outBuffer, *commCells[dir].ghosts[i]);
1135  }
1136  }
1137 
1138  // exchange particles, odd-even rule
1139  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
1140  outBuffer.send(receiver, DD_COMM_TAG);
1141  inBuffer.recv(sender, DD_COMM_TAG);
1142  } else {
1143  inBuffer.recv(sender, DD_COMM_TAG);
1144  outBuffer.send(receiver, DD_COMM_TAG);
1145  }
1146 
1147  // unpack received data
1148  if (realToGhosts) {
1149  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1150  unpackPositionsEtc(*commCells[dir].ghosts[i], inBuffer, extradata);
1151  }
1152  }
1153  else {
1154  for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
1155  unpackAndAddForces(*commCells[dir].reals[i], inBuffer);
1156  }
1157  }
1158  }
1159  }
1160  }
1161  LOG4ESPP_DEBUG(logger, "ghost communication finished");
1162  }
