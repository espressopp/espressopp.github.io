00778 
00779     LOG4ESPP_DEBUG(logger, "starting, expected comm buffer size " << exchangeBufferSize);
00780 
00781     // allocate send/recv buffers. We use the size as we need maximally so far, to avoid reallocation
00782     // TODO: This might be a problem when all particles are created on a single node initially!
00783     ParticleList sendBufL;
00784     sendBufL.reserve(exchangeBufferSize);
00785     ParticleList sendBufR;
00786     sendBufR.reserve(exchangeBufferSize);
00787     ParticleList recvBufL;
00788     recvBufL.reserve(exchangeBufferSize);
00789     ParticleList recvBufR;
00790     recvBufR.reserve(exchangeBufferSize);
00791 
00792     bool allFinished;
00793     do {
00794   bool finished = true;
00795 
00796   for (int coord = 0; coord < 3; ++coord) {
00797     LOG4ESPP_DEBUG(logger, "starting with direction " << coord);
00798 
00799     if (nodeGrid.getGridSize(coord) > 1) {
00800       for (std::vector<Cell*>::iterator it = realCells.begin(),
00801          end = realCells.end(); it != end; ++it) {
00802 
00803         Cell &cell = **it;
00804 
00805         // do not use an iterator here, since we need to take out particles during the loop
00806         for (size_t p = 0; p < cell.particles.size(); ++p) {
00807             Particle &part = cell.particles[p];
00808             const Real3D& pos = part.position();
00809 
00810             // check whether the particle is now "left" of the local domain
00811             if (pos[coord] - cellGrid.getMyLeft(coord) < -ROUND_ERROR_PREC) {
00812                 LOG4ESPP_TRACE(logger, "send particle left " << part.id());
00813                 moveIndexedParticle(sendBufL, cell.particles, p);
00814                 // redo same particle since we took one out here, so it's a new one
00815                 --p;
00816             }
00817             // check whether the particle is now "right" of the local domain
00818             else if (pos[coord] - cellGrid.getMyRight(coord) >= ROUND_ERROR_PREC) {
00819                 LOG4ESPP_TRACE(logger, "send particle right " << part.id());
00820                 moveIndexedParticle(sendBufR, cell.particles, p);
00821                 --p;
00822             }
00823             // Sort particles in cells of this node during last direction
00824             else if (coord == 2) {
00825                 const Real3D& pos = part.position();
00826                 Cell *sortCell = mapPositionToCellChecked(pos);
00827                 if (sortCell != &cell) {
00828                     if (sortCell == 0) {
00829                         // particle is not in the local domain
00830                         LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
00831                                 << " @ " << pos <<
00832                                 " is not inside node domain after neighbor exchange");
00833                         // isnan function is C99 only, x != x is only true if x == nan
00834                         if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
00835                             // TODO: error handling
00836                             LOG4ESPP_ERROR(logger, "particle " << part.id() <<
00837                                     " has moved to outer space (one or more coordinates are nan)");
00838                         } else {
00839                             // particle stays where it is, and will be sorted in the next round
00840                             finished = false;
00841                         }
00842                     } else {
00843                         // particle is in the local domain
00844                         moveIndexedParticle(sortCell->particles, cell.particles, p);
00845                         --p;
00846                     }
00847                 }
00848             }
00849 
00850         }
00851       }
00852 
00853       // Exchange particles, odd-even rule
00854       if (nodeGrid.getNodePosition(coord) % 2 == 0) {

00856         recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));

00858         recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
00859       } else {
00860         recvParticles(recvBufR, nodeGrid.getNodeNeighborIndex(2 * coord + 1));
00861         sendParticles(sendBufL, nodeGrid.getNodeNeighborIndex(2 * coord));
00862         recvParticles(recvBufL, nodeGrid.getNodeNeighborIndex(2 * coord));

00865

00870       // reset send/recv buffers
00871       sendBufL.resize(0);
00872       sendBufR.resize(0);
00873       recvBufL.resize(0);
00874       recvBufR.resize(0);
00875 
00876 
00877     } else {

00882         Cell &cell = **it;
00883         // do not use an iterator here, since we have need to take out particles during the loop
00884         for (size_t p = 0; p < cell.particles.size(); ++p) {
00885             Particle &part = cell.particles[p];
00886             getSystem()->bc->foldCoordinate(part.position(), part.image(), coord);
00887             LOG4ESPP_TRACE(logger, "folded coordinate " << coord << " of particle " << part.id());
00888 
00889             if (coord == 2) {
00890                 Cell *sortCell = mapPositionToCellChecked(part.position());
00891 
00892                 if (sortCell != &cell) {
00893                     if (sortCell == 0) {
00894                         LOG4ESPP_DEBUG(logger, "take another loop: particle " << part.id()
00895                                 << " @ " << part.position()
00896                                 << " is not inside node domain after neighbor exchange");
00897                         const Real3D& pos = part.position();
00898                         // isnan function is C99 only, x != x is only true if x == nan
00899                         if (pos[0] != pos[0] || pos[1] != pos[1] || pos[2] != pos[2]) {
00900                             LOG4ESPP_ERROR(logger, "particle " << part.id() <<
00901                                     " has moved to outer space (one or more coordinates are nan)");
00902                         } else {
00903                             // particle stays where it is, and will be sorted in the next round
00904                             finished = false;
00905                         }
00906                     } else {
00907                         moveIndexedParticle(sortCell->particles, cell.particles, p);

00987   void DomainDecompositionAdress::prepareGhostCommunication() {
00988     // direction loop: x, y, z
00989     for (int coord = 0; coord < 3; ++coord) {
00990         // boundaries of area to send
00991         int leftBoundary[3], rightBoundary[3];
00992         /* boundaries perpendicular directions are the same for left/right send.
00993         We also send the ghost frame that we have already, so the data amount
00994         increase with each cycle.
00995 
00996         For a direction that was done already, i.e. is smaller than dir,
00997         we take the full ghost frame, otherwise only the inner frame.  */
00998         for (int offset = 1; offset <= 2; ++offset) {
00999             int otherCoord = (coord + offset) % 3;
01000             if (otherCoord < coord) {
01001                 leftBoundary[otherCoord] = 0;
01002                 rightBoundary[otherCoord] = cellGrid.getFrameGridSize(otherCoord);
01003             } else {
01004                 leftBoundary[otherCoord] = cellGrid.getInnerCellsBegin(otherCoord);
01005                 rightBoundary[otherCoord] = cellGrid.getInnerCellsEnd(otherCoord);
01006             }
01007         }
01008 
01009         //  lr loop: left right - loop
01010         for (int lr = 0; lr < 2; ++lr) {
01011             int dir = 2 * coord + lr;
01012 
01013             /* participating real particles from this node */
01014             LOG4ESPP_DEBUG(logger, "direction " << dir << " reals");
01015 
01016             if (lr == 0) {
01017                 leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
01018                 rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord) + cellGrid.getFrameWidth();
01019             } else {
01020                 leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord) - cellGrid.getFrameWidth();
01021                 rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
01022             }
01023             fillCells(commCells[dir].reals, leftBoundary, rightBoundary);
01024 
01025             /* participating ghosts from this node */
01026             LOG4ESPP_DEBUG(logger, "direction " << dir << " ghosts");
01027 
01028             if (lr == 0) {
01029                 leftBoundary[coord] = cellGrid.getInnerCellsEnd(coord);
01030                 rightBoundary[coord] = cellGrid.getInnerCellsEnd(coord) + cellGrid.getFrameWidth();
01031             } else {
01032                 leftBoundary[coord] = cellGrid.getInnerCellsBegin(coord) - cellGrid.getFrameWidth();
01033                 rightBoundary[coord] = cellGrid.getInnerCellsBegin(coord);
01034             }
01035             fillCells(commCells[dir].ghosts, leftBoundary, rightBoundary);
01036         }
01037     }
01038   }

01040   void DomainDecompositionAdress::
01041   doGhostCommunication(bool sizesFirst, bool realToGhosts, int extradata) {
01042     LOG4ESPP_DEBUG(logger, "do ghost communication " << (sizesFirst ? "with sizes " : "")
01043            << (realToGhosts ? "reals to ghosts " : "ghosts to reals ") << extradata);
01044 
01045     /* direction loop: x, y, z.
01046    Here we could in principle build in a one sided ghost
01047    communication, simply by taking the lr loop only over one
01048    value. */
01049     for (int _coord = 0; _coord < 3; ++_coord) {
01050   /* inverted processing order for ghost force communication,
01051      since the corner ghosts have to be collected via several
01052      nodes. We now add back the corner ghost forces first again
01053      to ghost forces, which only eventually go back to the real
01054      particle.
01055   */
01056   int coord = realToGhosts ? _coord : (2 - _coord);
01057   real curCoordBoxL = getSystem()->bc->getBoxL()[coord];
01058 
01059   // lr loop: left right
01060   for (int lr = 0; lr < 2; ++lr) {
01061     int dir         = 2 * coord + lr;
01062     int oppositeDir = 2 * coord + (1 - lr);
01063 
01064     Real3D shift(0, 0, 0);
01065 
01066     shift[coord] = nodeGrid.getBoundary(dir) * curCoordBoxL;
01067 
01068     LOG4ESPP_DEBUG(logger, "direction " << dir);
01069 
01070     if (nodeGrid.getGridSize(coord) == 1) {
01071       LOG4ESPP_DEBUG(logger, "local communication");
01072 
01073       // copy operation, we have to receive as many cells as we send
01074       if (commCells[dir].ghosts.size() != commCells[dir].reals.size()) {
01075         throw std::runtime_error("DomainDecompositionAdress::doGhostCommunication: send/recv cell structure mismatch during local copy");
01076       }
01077 
01078       for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
01079         if (realToGhosts) {
01080           copyRealsToGhosts(*commCells[dir].reals[i], *commCells[dir].ghosts[i], extradata, shift);
01081         } else {
01082           addGhostForcesToReals(*commCells[dir].ghosts[i], *commCells[dir].reals[i]);
01083         }
01084       }
01085     }
01086     else {
01087       // exchange size information, if necessary
01088       if (sizesFirst) {
01089         LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes");
01090 
01091         // prepare buffers
01092         std::vector<longint> sendSizes, recvSizes;
01093         sendSizes.reserve(commCells[dir].reals.size());
01094         for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
01095           sendSizes.push_back(commCells[dir].reals[i]->particles.size());
01096         }
01097         recvSizes.resize(commCells[dir].ghosts.size());
01098 
01099         // exchange sizes, odd-even rule
01100         if (nodeGrid.getNodePosition(coord) % 2 == 0) {
01101           LOG4ESPP_DEBUG(logger, "sending to node " << nodeGrid.getNodeNeighborIndex(dir)
01102                      << ", then receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir));
01103           getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
01104           getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
01105         }
01106         else {
01107           LOG4ESPP_DEBUG(logger, "receiving from node " << nodeGrid.getNodeNeighborIndex(oppositeDir)
01108                      << ", then sending to node " << nodeGrid.getNodeNeighborIndex(dir));
01109           getSystem()->comm->recv(nodeGrid.getNodeNeighborIndex(oppositeDir), DD_COMM_TAG, &(recvSizes[0]), recvSizes.size());
01110           getSystem()->comm->send(nodeGrid.getNodeNeighborIndex(dir), DD_COMM_TAG, &(sendSizes[0]), sendSizes.size());
01111         }
01112 
01113         // resize according to received information
01114         for (int i = 0, end = commCells[dir].ghosts.size(); i < end; ++i) {
01115           commCells[dir].ghosts[i]->particles.resize(recvSizes[i]);
01116         }
01117         LOG4ESPP_DEBUG(logger, "exchanging ghost cell sizes done");
01118       }
01119 
01120       // prepare send and receive buffers
01121       longint receiver, sender;
01122       outBuffer.reset();
01123       if (realToGhosts) {
01124         receiver = nodeGrid.getNodeNeighborIndex(dir);
01125         sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
01126         for (int i = 0, end = commCells[dir].reals.size(); i < end; ++i) {
01127           packPositionsEtc(outBuffer, *commCells[dir].reals[i], extradata, shift);
01128         }
01129       }
01130       else {
