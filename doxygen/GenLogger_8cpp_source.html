98 static void setLoggerLevel (string& name, string& value)
99 
100 {
101  Logger::Level level = Logger::WARN;
102 
103  if (value == "TRACE") {
104  level = Logger::TRACE;
105  } else if (value == "DEBUG") {
106  level = Logger::DEBUG;
107  } else if (value == "INFO") {
108  level = Logger::INFO;
109  } else if (value == "WARN"){
110  level = Logger::WARN;
111  } else if (value == "ERROR"){
112  level = Logger::ERROR;
113  } else if (value == "FATAL") {
114  level = Logger::FATAL;
115  } else {
116  LOG4ESPP_ERROR(myLogger, "logger " << name << " = " << value << ", unknown level");
117  return;
118  }
119 
120  if (name == "root") {
121  Logger::getRoot().setLevel(level, true);
122  } else {
123  Logger::getInstance(name).setLevel(level, true);
124  }
125 
126  LOG4ESPP_INFO(myLogger, "logger " << name << " = " << value);
127 }

146 /********************************************************************
147 * help routine: eval_entry *
148 ********************************************************************/
149 
150 static int eval_entry (char* line, int length, char *filename)
151 
152 {
153  line[length] = '\0';
154 
155  string myLine = line;
156 
157  // check for an empty line
158 
159  string::size_type firstPos = myLine.find_first_not_of(" ", 0);
160 
161 #ifdef DEBUGGING
162  printf("pos of first relevant char = %d\n", firstPos);
163 #endif
164 
165  if (string::npos == firstPos) return 0;
166 
167  // check for a comment line
168 
169 #ifdef DEBUGGING
170  printf("first relevant char = %c\n", myLine[firstPos]);
171 #endif
172 
173  if (myLine[firstPos] == '#') return 0;
174 
175  // check for an equal sign in the line
176 
177  string::size_type equalPos = myLine.find_first_of("=", 0);
178 
179  if (string::npos == equalPos) return 0;
180 
181  // now find the name without blanks, e.g. "atom vec = " is only atom
182 
183  string::size_type lastPos = myLine.find_first_of(" =", firstPos);
184  string name = myLine.substr(firstPos, lastPos - firstPos);
185 
186  firstPos = myLine.find_first_not_of(" ", equalPos+1);
187  lastPos = myLine.find_first_of(" ", firstPos);
188 
189 #ifdef DEBUGGING
190  printf("value at %d - %d\n", firstPos, lastPos);
191 #endif
192 
193  if (string::npos == lastPos) lastPos = myLine.length();
194 
195  for (string::size_type i = firstPos; i <= lastPos; i++) myLine[i] = toupper(myLine[i]);
196 
197  string value = myLine.substr(firstPos, lastPos - firstPos);
198 
199  setLoggerLevel (name, value);
200 
201  return 1;
202 }

204 /********************************************************************
205 * help routine: read_config for reading a logger configuration *
206 ********************************************************************/

208 static int read_config (char *fname) {
209  FILE *config_file;
210 
211  char buffer [180];
212  int buf_len;
213  char c;
214  char eof;
215  int no_entries; /* number of relevant entries */
216 
217  config_file = fopen (fname, "r");
218 
219  if (config_file == NULL) {
220  LOG4ESPP_ERROR(myLogger, "config: could not open config file " << fname);
221  return 0;
222  }
223 
224  buf_len = 0;
225  eof = EOF;
226  no_entries = 0;
227 
228  while ((c = fgetc (config_file)) != eof)
229 
230  { if (c == '\n')
231 
232  { /* new line, evaluate current line */
233 
234  no_entries += eval_entry (buffer, buf_len, fname);
235  buf_len = 0;
236  }
237 
238  else buffer [buf_len++] = c;
239  }
240 
241  if (buf_len > 0)
242 
243  no_entries += eval_entry (buffer, buf_len, fname);
244 
245  fclose (config_file);
246 
247  return no_entries;
248 }
