00094     void BerendsenBarostat::barostat(){
00095       LOG4ESPP_DEBUG(theLogger, "equilibrating the pressure");
00096 
00097       System& system = getSystemRef();
00098       
00099       Pressure Pcurrent(getSystem());
00100       
00101       real P = Pcurrent.compute();  // calculating the current pressure in system
00102       
00103       real mu3 = 1 + pref * (P - P0);
00104       
00105       Error err(system.comm);
00106       if(mu3<0.0){
00107         stringstream msg;
00108         msg << "Scaling coefficient is <0 (Berendsen barostat). mu^3="<<mu3;
00109         err.setException( msg.str() );
00110         err.checkException();
00111       }
00112       
00113       real mu = pow(  mu3, exponent );  // calculating the current scaling parameter
00114       
00115       Real3D mu3D( 1.0 );
00116       for(int i=0;i<3;i++)
00117         if( fixed[i]>0 )
00118           mu3D[i]=mu;
00119 
00120       system.scaleVolume( mu3D, true );
00121     }
