69  real _computeEnergy(const Particle& p1, const Particle& p2) const {
70  Real3D dist = p1.position() - p2.position();

75  bool _computeForce(Real3D& force, const Particle &p1, const Particle &p2) const {
76  Real3D dist = p1.position() - p2.position();
77  real abs_dist = dist.abs();
78  real sqr_dist = dist.sqr();
79 
80  real forceFactor = prefactor * p1.mass() * p2.mass() / (abs_dist * sqr_dist);
81  force = dist * forceFactor;
82  return true;
