66  python::list Autocorrelation::compute() {
67  size_t M = getListSize();
68 
69  System& system = getSystemRef();
70 
71  int n_nodes = system.comm -> size();
72  int this_node = system.comm -> rank();
73 
74  // TODO it could be a problem if n_nodes > total_num !!!
75  unsigned int num_m[n_nodes];
76  unsigned int num_mH[n_nodes];
77 
78  if(this_node == 0){
79 
80  // it is 1+2+3+...+M
81  double local_num = ( (double)M * (double)(M + 1) / 2.0) / (double)n_nodes + 1.0;
82 
83  for(int i=0; i<n_nodes; i++){
84  double max_num = (i+1) * local_num;
85 
86  unsigned int m_max = (unsigned int) ( ( sqrt(1.0+8.0*max_num) - 1.0 )/2. );
87 
88  unsigned int lastNum = (i==0) ? 0 : num_mH[i-1];
89 
90  if(m_max-lastNum==0)
91  num_mH[i] = num_mH[i-1] + 1;
92  else
93  num_mH[i] = m_max;
94 
95  if(num_mH[i] > M) num_mH[i] = M;
96  }
97 
98  for(int i=0; i<n_nodes-1; i++)
99  num_m[i] = M - num_mH[n_nodes-2-i];
100  num_m[n_nodes-1] = M;
101  }
102 
103  boost::mpi::broadcast(*system.comm, num_m, n_nodes, 0);
104 
105  // now num_m[i], i - cpu number, is a number of series in "for" statement for each cpu
106 
107  unsigned int min_m = (this_node==0) ? 0 : num_m[this_node-1];
108  unsigned int max_m = num_m[this_node];
109 
110  real* Z;
111  Z = new real[M];
112 
113  cout<< "calculating autocorrelation.." << endl;
114  int perc=0;
115  real denom = 100.0 / (real)(max_m-min_m);
116  for(unsigned int m=min_m; m<max_m; m++){
117  Z[m] = 0.0;
118  for(unsigned int n=0; n<M-m; n++){
119  Z[m] += getValue(n + m) * getValue(n);
120  }
121 
122  /*
123  * additional calculations slow down routine but from the other hand
124  * it helps to monitor progress
125  */
126  if(system.comm->rank()==0){
127  perc = (int)(m*denom);
128  if(perc%5==0){
129  cout<<"calculation progress (autocorrelation): "<< perc << " %\r" <<flush;
130  }
131  }
132  }
133 
134  if(system.comm->rank()==0)
135  cout<<"calculation progress (autocorrelation): 100 %" <<endl;
136 
137  real coef = 3.0; // only if value is Real3D
138 
139  for(unsigned int m=min_m; m<max_m; m++){
140  Z[m] /= ( (real)(M-m)*coef );
141  }
142 
143  // TODO probably could be done nicer. gather doesn't work with different length of array
144  unsigned long int MM = M * n_nodes;
145  real* totZ = new real[MM];
146  boost::mpi::gather(*system.comm, Z, M, totZ, 0);
147 
148  python::list pyli;
149 
150  if(this_node == 0){
151  int count = 0;
152  for(int m=0; m<M; m++){
153  if( m >= num_m[count] ) count ++;
154  pyli.append( totZ[M*count + m] );
155  }
156  }
157 
158  delete [] Z;
159  Z = NULL;
160  delete [] totZ;
161  totZ = NULL;
162 
163  return pyli;
164  }
