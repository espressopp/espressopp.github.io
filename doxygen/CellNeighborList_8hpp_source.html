48  CellNeighborList(shared_ptr<storage::Storage>& storage)
49  {
50  CellList const& localCells = storage->getLocalCells();
51  CellList const& realCells = storage->getRealCells();
52  size_t numCells = localCells.size();
53  const Cell* cell0 = localCells[0];
54 
55  // Approximate the number of neighbors using the first real cell
56  size_t nnbrs0 = 0;
57  for(NeighborCellInfo& nc: realCells[0]->neighborCells)
58  if(!nc.useForAllPairs) nnbrs0++;
59 
60  // Reserve the maximum number of cells
61  Super::resize(nnbrs0+2, numCells);
62 
63  // copy neighbor information, skip cells with no neighbors
64  size_t irow = 0;
65  for(size_t icell=0; icell<numCells; icell++) {
66  size_t jnbr=0;
67  for(NeighborCellInfo& nc: localCells[icell]->neighborCells)
68  if(!nc.useForAllPairs) this->at(irow,jnbr++) = nc.cell - cell0;
69  if(jnbr>0){
70  this->cellId(irow) = icell;
71  this->numNeighbors(irow) = jnbr;
72  ++irow;
73  }
74  }
75 
76  // truncate empty rows
77  if(irow<numCells) Super::resize(Super::size_n(), irow);
78  }
