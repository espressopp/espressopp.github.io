58  void VelocityVerletOnRadius::integrate1() {
59  System& system = getSystemRef();
60  CellList realCells = system.storage->getRealCells();
61  // loop over all particles of the local cells
62  real dt = integrator->getTimeStep();
63  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
64  LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() << ", forceRadius = " << cit->fradius());
65  }
66  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
67  //LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() << ", radius = " << cit->radius());
68  real dtfm = 0.5 * dt / radialDampingMass;
69  // Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * f(t)
70  cit->vradius() += dtfm * cit->fradius();
71  // Propagate radius (only NVT): p(t + dt) = p(t) + dt * v(t+0.5*dt)
72  real deltaP = cit->vradius();
73  deltaP *= dt;
74  cit->radius() += deltaP;
75  }
76  }
