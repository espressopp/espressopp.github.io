00058     void VelocityVerletOnRadius::integrate1() {
00059       System& system = getSystemRef();
00060       CellList realCells = system.storage->getRealCells();
00061       // loop over all particles of the local cells
00062       real dt = integrator->getTimeStep();
00063       for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00064             LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() << ", forceRadius = " << cit->fradius());
00065       }
00066      for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00067         //LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() <<  ", radius = " << cit->radius());
00068         real dtfm = 0.5 * dt / radialDampingMass;
00069         // Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * f(t)
00070         cit->vradius() += dtfm * cit->fradius();
00071         // Propagate radius (only NVT): p(t + dt) = p(t) + dt * v(t+0.5*dt)
00072         real deltaP = cit->vradius();
00073         deltaP *= dt;
00074         cit->radius() += deltaP;
00075       }
00076     }
