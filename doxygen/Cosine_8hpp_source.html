81  real a11, a12, a22;
82 
83  // theta is the angle between r_{12} and r_{32}
84 
85  real cos_theta = dist12 * dist32 / (dist12_magn * dist32_magn);
86  if(cos_theta < -1.0) cos_theta = -1.0;
87  if(cos_theta > 1.0) cos_theta = 1.0;
88 
89  real sin_theta = sqrt(1.0 - cos_theta*cos_theta);
90 
91  a11 = cos_theta / dist12_sqr;
92  a12 = -1.0 / (dist12_magn * dist32_magn);

94 
95  if (sin_theta > SMALL_EPSILON) { //sin_theta is always positive

98  else {
99  force12 = Real3D(0.0,0.0,0.0);
100  force32 = Real3D(0.0,0.0,0.0);
101  }
