38  /* SET BODY FORCE */
39  void LBInitConstForce::setForce(Real3D _force)
40  {
41  int _id = 0; // _id shows whether force had been cancelled, set or added
42  int _offset = latticeboltzmann->getHaloSkin();
43  Int3D _Ni = latticeboltzmann->getMyNi();
44 
45  /* external forces loop */
46  for (int i = _offset; i < _Ni.getItem(0)-_offset; i++) {
47  for (int j = _offset; j < _Ni.getItem(1)-_offset; j++) {
48  for (int k = _offset; k < _Ni.getItem(2)-_offset; k++) {
49  // set local forces and general flag
50  if (_force != Real3D(0.,0.,0.)) {
51  latticeboltzmann->setDoExtForce(true);
52  latticeboltzmann->setExtForceLoc(Int3D(i,j,k),_force);
53  _id = 1;
54  } else {
55  latticeboltzmann->setDoExtForce(false);
56  latticeboltzmann->setExtForceLoc(Int3D(i,j,k),Real3D(0.,0.,0.));
57  }
58  }
59  }
60  }
61  printForce(_force, _id);
62  }

64  /* ADD BODY FORCE TO EXISTING FORCES (IF ANY) */
65  void LBInitConstForce::addForce(Real3D _force)
66  {
67  int _id = 0;
68  int _offset = latticeboltzmann->getHaloSkin();
69  Int3D _Ni = latticeboltzmann->getMyNi();
70 
71  Real3D existingforce;
72 
73  /* external forces loop */
74  for (int i = _offset; i < _Ni.getItem(0)-_offset; i++) {
75  for (int j = _offset; j < _Ni.getItem(1)-_offset; j++) {
76  for (int k = _offset; k < _Ni.getItem(2)-_offset; k++) {
77  existingforce = latticeboltzmann->getExtForceLoc(Int3D(i,j,k));
