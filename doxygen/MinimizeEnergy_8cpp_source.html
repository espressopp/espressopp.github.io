53  bool MinimizeEnergy::run(int max_steps, bool verbose) {
54  bool retval = false;
55  System &system = getSystemRef();
56  storage::Storage &storage = *system.storage;
57  real skin_half = 0.5 * system.getSkin();
58  dp_sqr_max_ = 0.0;
59  f_max_sqr_ = std::numeric_limits<real>::max();
60 
61  // Before start make sure that particles are on the right processor
62  if (resort_flag_) {
63  LOG4ESPP_DEBUG(theLogger, "storage.decompose")
64  storage.decompose();
65  resort_flag_ = false;
66  }
67 
68  updateForces();
69 
70  LOG4ESPP_INFO(theLogger,
71  "starting energy minimalization loop (iters=" << max_steps << ")");
72 
73  if (verbose) {
74  std::cout << "Minimize energy" << std::endl;
75  std::cout << " current force_max = " << sqrt(f_max_sqr_) << std::endl;
76  std::cout << " f_tol = " << sqrt(ftol_sqr_) << std::endl;
77  std::cout << " max_steps = " << max_steps << std::endl;
78  std::cout << " max displacement = " << max_displacement_ << std::endl;
79  }
80  int iters = 0;
81  for (; iters < max_steps && f_max_sqr_ > ftol_sqr_; iters++) {
82  steepestDescentStep();
83 
84  dp_MAX += sqrt(dp_sqr_max_);
85 
86  resort_flag_ = dp_MAX > skin_half;
87  LOG4ESPP_INFO(theLogger, "maxDist = " << dp_MAX << ", skin/2 = " << skin_half);
88 
89  if (resort_flag_) {
90  LOG4ESPP_INFO(theLogger, "Particles will be decomposed.");
91  dp_MAX = 0.;
92  storage.decompose();
93  LOG4ESPP_INFO(theLogger, "Particles have been decomposed.");
94  resort_flag_ = false;
95  }
96 
97  updateForces();
98 
99  if (verbose)
100  std::cout << nstep_ << ": f_max^2=" << f_max_sqr_ << " max_dp^2=" << dp_sqr_max_ << std::endl;
101 
102  nstep_++;
103  }
104 
105  if (verbose) {
106  std::cout << "Minimize energy finished" << std::endl;
107  std::cout << " current force_max = " << sqrt(f_max_sqr_) << std::endl;
108  std::cout << " run for steps = " << iters << std::endl;
109  std::cout << " max displacement^2 = " << dp_sqr_max_ << std::endl;
110  if (f_max_sqr_ > ftol_sqr_) {
111  std::cout << "WARNING: the current max force is greater than the ftol=" << sqrt(ftol_sqr_);
112  std::cout << " The system might required additional run of energy minimization." << std::endl;
113  }
114  }
115  retval = (f_max_sqr_ < ftol_sqr_);
116 
117  LOG4ESPP_INFO(theLogger,
118  "finished run, f_max_sqr_^2=" << f_max_sqr_ << " max_displ^2=" << dp_sqr_max_);
119  return retval;
120  }

123  void MinimizeEnergy::updateForces() {
124  LOG4ESPP_INFO(theLogger,
125  "update ghosts, calculate forces and collect ghost forces");
126  System& system = getSystemRef();
127 
128  system.storage->updateGhosts();
129 
130  LOG4ESPP_INFO(theLogger, "calculate forces");
131  // First initialize
132  CellList localCells = system.storage->getLocalCells();
133  for(CellListIterator cit(localCells); !cit.isDone(); ++cit) {
134  cit->force() = 0.0;
135  }
136 
137  // Calculate force component from all interactions.
138  const InteractionList& srIL = system.shortRangeInteractions;
139 
140  for (size_t i = 0; i < srIL.size(); i++) {
141  LOG4ESPP_INFO(theLogger, "compute forces for srIL " << i << " of " << srIL.size());
142  srIL[i]->addForces();
143  }
144  // Collect forces from ghost particles.
145  system.storage->collectGhostForces();
146 
147  // Get max force in the system.
148  LOG4ESPP_DEBUG(theLogger, "get max force in the system");
149  real f_max = -std::numeric_limits<real>::max();
150  CellList realCells = system.storage->getRealCells();
151  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
152  f_max = std::max(f_max, cit->force().sqr());
153  }
154  mpi::all_reduce(*system.comm, f_max, f_max_sqr_, boost::mpi::maximum<real>());
155  }

161  void MinimizeEnergy::steepestDescentStep() {
162  LOG4ESPP_INFO(theLogger, "steepestDescent single step");
163  System& system = getSystemRef();
164 
165  real f_sqr, dp, dp_sqr;
166  real f_max = sqrt(f_max_sqr_);
167  real dp_sqr_max = std::numeric_limits<real>::min();
168 
169  // Iterate over only real particles.
170  CellList realCells = system.storage->getRealCells();
171  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
172 
173  if (variable_step_flag_) {
174  dp_sqr = 0.;
175  for (int i = 0; i < 3; i++) { // Perhaps it can be done better.
176  dp = max_displacement_ * cit->force()[i]/f_max;
177  dp_sqr += dp*dp;
178 
179  // Update position component by dp.
180  cit->position()[i] += dp;
181  }
182  dp_sqr_max = std::max(dp_sqr_max, dp_sqr);
183 
184  } else {
185  dp_sqr = 0.;
186  for (int i = 0; i < 3; i++) { // Perhaps it can be done better.
187  dp = gamma_ * cit->force()[i];
188  if (fabs(dp) > max_displacement_)
189  dp = sgn<real>(dp)*max_displacement_;
190  dp_sqr += dp*dp;
191 
192  // Update position component by dp.
193  cit->position()[i] += dp;
194  }
195  dp_sqr_max = std::max(dp_sqr_max, dp_sqr);
196  }
197  }
198 
199  LOG4ESPP_INFO(theLogger, "steepestDescentStep calculating dp_sqr_max");
200  mpi::all_reduce(*system.comm, dp_sqr_max, dp_sqr_max_, boost::mpi::maximum<real>());
201  }
