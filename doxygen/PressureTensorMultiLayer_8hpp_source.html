00081           real zminBC = pos[2] - dz;
00082           real zmaxBC = pos[2] + dz;
00083 
00084           // boundary condition for orthorhombic box
00085           bool boundary = false;
00086           if(zminBC<0.0){
00087             zminBC += Li[2];
00088             boundary = true;

00090           else if(zmaxBC>=Li[2]){
00091             zmaxBC -= Li[2];
00092             boundary = true;
00093           }
00094

00096           int maxpos = (int)( zmaxBC/lZ );
00097 
00098           if(boundary){

00101               vvlocal[i] += vvt;
00102             }
00103             for(int i = minpos+1; i<n; i++){
00104               vvlocal[i] += vvt;
00105             }
00106           }
00107           else{
00108             for(int i = minpos+1; i<=maxpos; i++){
