151  template < typename _Potential1, typename _Potential2 > inline void
152  VerletListHadressATATInteractionTemplate < _Potential1, _Potential2 >::
153  addForces() {
154  LOG4ESPP_INFO(theLogger, "add forces computed by the Verlet List");
155 
156  std::set<Particle*> cgZone = verletList->getCGZone();
157  std::set<Particle*> adrZone = verletList->getAdrZone();
158 
159  for (std::set<Particle*>::iterator it=adrZone.begin();
160  it != adrZone.end(); ++it) {
161  Particle &p = **it;
162  // intitialize energy diff AA-CG
163  if (p.lambda()<0.9999999 && p.lambda()>0.0000001) {
164  energydiff[&p]=0.0;
165  }
166  // energydiff[&p]=0.0;
167  }
168 
169  // We only compute the atomistic contribution of all forces in this template. Hence, we do not loop over the pairs in the CG region.
170 
171  // Compute forces (AT and VP) of Pairs inside AdResS zone
172  for (PairList::Iterator it(verletList->getAdrPairs()); it.isValid(); ++it) {
173  real w1, w2;
174  // these are the two VP interacting
175  Particle &p1 = *it->first;
176  Particle &p2 = *it->second;
177 
178  w1 = p1.lambda();
179  w2 = p2.lambda();
180  real w12 = (w1 + w2)/2.0; // H-AdResS
181 
182  // force between AT particles
183  if (w12 != 0.0) { // calculate AT force if both VP are outside CG region (HY-HY, HY-AT, AT-AT)
184  FixedTupleListAdress::iterator it3;
185  FixedTupleListAdress::iterator it4;
186  it3 = fixedtupleList->find(&p1);
187  it4 = fixedtupleList->find(&p2);
188 
189  if (it3 != fixedtupleList->end() && it4 != fixedtupleList->end()) {
190 
191  std::vector<Particle*> atList1;
192  std::vector<Particle*> atList2;
193  atList1 = it3->second;
194  atList2 = it4->second;
195 
196  for (std::vector<Particle*>::iterator itv = atList1.begin();
197  itv != atList1.end(); ++itv) {
198 
199  Particle &p3 = **itv;
200 
201  for (std::vector<Particle*>::iterator itv2 = atList2.begin();
202  itv2 != atList2.end(); ++itv2) {
203 
204  Particle &p4 = **itv2;
205 
206  // AT forces
207  Real3D force(0.0, 0.0, 0.0);
208  const Potential1 &potential1 = getPotential1(p3.type(), p4.type());
209  if(potential1._computeForce(force, p3, p4)) {
210  force *= w12;
211  p3.force() += force;
212  p4.force() -= force;
213  }
214  // H-AdResS - Drift Term part 2
215  // Compute AT energies of particles in the hybrid and store and subtract in map energydiff
216  if(w12!=1.0) { //at least one particle in hybrid region => need to do the energy calculation
217  const real energyat = potential1._computeEnergy(p3, p4);
218  if(w1!=1.0) { // if particle one is in hybrid region
219  energydiff[&p1] -= energyat; // subtract AT energy for virtual particle 1
220  }
221  if(w2!=1.0) { // if particle two is in hybrid region
222  energydiff[&p2] -= energyat; // subtract AT energy for virtual particle 2
223  }
224  }
225  const Potential2 &potential2 = getPotential2(p3.type(), p4.type());
226  if(potential2._computeForce(force, p3, p4)) {
227  force *= w12;
228  p3.force() += force;
229  p4.force() -= force;
230  }
231  // H-AdResS - Drift Term part 2
232  // Compute AT energies of particles in the hybrid and store and subtract in map energydiff
233  if(w12!=1.0) { //at least one particle in hybrid region => need to do the energy calculation
234  const real energyat = potential2._computeEnergy(p3, p4);
235  if(w1!=1.0) { // if particle one is in hybrid region
236  energydiff[&p1] -= energyat; // subtract AT energy for virtual particle 1
237  }
238  if(w2!=1.0) { // if particle two is in hybrid region
239  energydiff[&p2] -= energyat; // subtract AT energy for virtual particle 2
240  }
241  }
242 
243  }
244 
245  }
246  }
247  else { // this should not happen
248  std::stringstream ss;
249  ss << "One of the VP particles not found in tuples: " << p1.id() << "-" << p1.ghost() << ", " << p2.id() << "-" << p2.ghost() << " (" << p1.position() << ") (" << p2.position();
250  throw std::runtime_error(ss.str());
251  }
252  }
253  }
254 
255  // H-AdResS - Drift Term part 3
256  // Iterate over all particles in the hybrid region and calculate drift force
257  for (std::set<Particle*>::iterator it=adrZone.begin();
258  it != adrZone.end(); ++it) { // Iterate over all particles
259  Particle &vp = **it;
260  real w = vp.lambda();
261 
262  if(w<0.9999999 && w>0.0000001) { // only chose those in the hybrid region
263  // calculate distance to nearest adress particle or center
264  std::vector<Real3D*>::iterator it2 = verletList->getAdrPositions().begin();
265  Real3D pa = **it2; // position of adress particle
266  Real3D mindriftforce(0.0, 0.0, 0.0);
267  verletList->getSystem()->bc->getMinimumImageVector(mindriftforce, vp.position(), pa);
268  real min1sq = 0.0;
269  if(verletList->getAdrRegionType()) {
270  min1sq = mindriftforce.sqr();
271  }
272  else {
273  min1sq = mindriftforce[0]*mindriftforce[0];
274  }
275  ++it2;
276  for (; it2 != verletList->getAdrPositions().end(); ++it2) {
277  pa = **it2;
278  Real3D driftforce(0.0, 0.0, 0.0);
279  verletList->getSystem()->bc->getMinimumImageVector(driftforce, vp.position(), pa);
280  real distsq1 = 0.0;
281  if(verletList->getAdrRegionType()) {
282  distsq1 = driftforce.sqr();
283  }
284  else {
285  distsq1 = driftforce[0]*driftforce[0];
286  }
287  if (distsq1 < min1sq) {
288  min1sq = distsq1;
289  mindriftforce = driftforce;
290  }
291  }
292  min1sq = sqrt(min1sq); // distance to nearest adress particle or center
293 
294  if(verletList->getAdrRegionType()) {
295  mindriftforce = (1.0/min1sq)*mindriftforce; // normalized driftforce vector
296  // mindriftforce *= (0.5 * energydiff.find(&vp)->second); // get the energy differences which were calculated previously and put in drift force
297  mindriftforce *= (0.5 * energydiff[&vp]);
298  mindriftforce *= vp.lambdaDeriv();
299  vp.force() += mindriftforce;
300  }
301  else {
302  real mindriftforceX = (1.0/min1sq)*mindriftforce[0]; // normalized driftforce vector
303  // mindriftforceX *= (0.5 * energydiff.find(&vp)->second); // get the energy differences which were calculated previously and put in drift force
304  mindriftforceX *= (0.5 * energydiff[&vp]);
305  mindriftforceX *= vp.lambdaDeriv();
306  Real3D driftforceadd(mindriftforceX,0.0,0.0);
307  vp.force() += driftforceadd;
308  }
309  vp.drift() += 0.5 * energydiff.find(&vp)->second;
310  }
311 
312  }
313 
314  // energydiff.clear(); // clear the energy difference map
315  }

318  template < typename _Potential1, typename _Potential2 >
319  inline real
320  VerletListHadressATATInteractionTemplate < _Potential1, _Potential2 >::
321  computeEnergy() {
322  LOG4ESPP_INFO(theLogger, "compute energy of the Verlet list pairs");
323 
324  real e = 0.0;
325  for (PairList::Iterator it(verletList->getAdrPairs());
326  it.isValid(); ++it) {
327  Particle &p1 = *it->first;
328  Particle &p2 = *it->second;
329  real w1 = p1.lambda();
330  real w2 = p2.lambda();
331  real w12 = (w1 + w2)/2.0;
332 
333  FixedTupleListAdress::iterator it3;
334  FixedTupleListAdress::iterator it4;
335  it3 = fixedtupleList->find(&p1);
336  it4 = fixedtupleList->find(&p2);
337 
338  if (it3 != fixedtupleList->end() && it4 != fixedtupleList->end()) {
339  std::vector<Particle*> atList1;
340  std::vector<Particle*> atList2;
341  atList1 = it3->second;
342  atList2 = it4->second;
343 
344  for (std::vector<Particle*>::iterator itv = atList1.begin();
345  itv != atList1.end(); ++itv) {
346 
347  Particle &p3 = **itv;
348  for (std::vector<Particle*>::iterator itv2 = atList2.begin();
349  itv2 != atList2.end(); ++itv2) {
350  Particle &p4 = **itv2;
351 
352  // AT energies
353  const Potential1 &potential1 = getPotential1(p3.type(), p4.type());
354  const Potential2 &potential2 = getPotential2(p3.type(), p4.type());
355  e += w12*(potential1._computeEnergy(p3, p4) + potential2._computeEnergy(p3, p4));
356  }
357  }
358  }
359  }
360 
361  real esum;
362  boost::mpi::all_reduce(*getVerletList()->getSystem()->comm, e, esum, std::plus<real>());
363  return esum;
364  }
