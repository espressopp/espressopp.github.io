86  void preset() {
87  real sig2 = sigma * sigma;
88  real sig6 = sig2 * sig2 * sig2;
89  ff1 = 48.0 * epsilon * sig6 * sig6;
90  ff2 = 24.0 * epsilon * sig6;
91  ef1 = 4.0 * epsilon * sig6 * sig6;
92  ef2 = 4.0 * epsilon * sig6;
93  }

113  real _computeEnergySqrRaw(real distSqr) const {
114  real frac2 = sigma*sigma / distSqr;
115  real frac6 = frac2 * frac2 * frac2;
116  real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
117  return energy;
118  }
119 
120  bool _computeForceRaw(Real3D& force, const Real3D& dist, real distSqr) const
121  {
122  real frac2 = 1.0 / distSqr;
123  real frac6 = frac2 * frac2 * frac2;
124  real ffactor = frac6 * (ff1 * frac6 - ff2) * frac2;
125  force = dist * ffactor;
126  return true;
127  }
