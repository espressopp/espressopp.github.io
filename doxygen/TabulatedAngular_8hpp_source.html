82  bool _computeForceRaw(Real3D& force12, Real3D& force32,
83  const Real3D& dist12, const Real3D& dist32) const {
84  if (table) {
85  real dist12_sqr = dist12 * dist12;
86  real dist32_sqr = dist32 * dist32;
87  real dist1232 = sqrt(dist12_sqr) * sqrt(dist32_sqr);
88  real cos_theta = dist12 * dist32 / dist1232;
89 
90  real a = table->getForce(acos(cos_theta));
91 
92  a*=1.0/(sqrt(1.0-cos_theta*cos_theta));
93 
94  real a11 = a * cos_theta / dist12_sqr;
95  real a12 = -a / dist1232;
96  real a22 = a * cos_theta / dist32_sqr;
97 
98  force12 = a11 * dist12 + a12 * dist32;
99  force32 = a22 * dist32 + a12 * dist12;
100  }
101  else {
102  LOG4ESPP_DEBUG(theLogger, "Tabulate angular potential table not available.");
103  return false;
104  }
105  return true;
106  }
