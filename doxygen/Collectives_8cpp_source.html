41 struct UniqueReduce: public std::binary_function< int, int, int > {
42  static const int NotHere = -1;

58 int espressopp::esutil::Collectives::locateItem(bool here, int controller, communicator world) {
59  int node = here ? world.rank() : UniqueReduce::NotHere;
60 
61  if (world.rank() != controller) {
