00107     void TDforce::applyForce() {
00108           LOG4ESPP_DEBUG(theLogger, "apply TD force");
00109 
00110           System& system = getSystemRef();
00111           const bc::BC& bc = *getSystemRef().bc;  // boundary conditions
00112 
00113           // iterate over CG particles
00114           CellList cells = system.storage->getRealCells();
00115           for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
00116 
00117               Table table;
00118               if (forces.find(cit->getType())!=forces.end()) { //because there may be CG particles to which TD force is not applied
00119                 table = forces.find(cit->getType())->second; //TODO shouldn't do find twice
00120               }
00121           
00122               if (table) {
00123 
00124 
00125                   if (!(verletList->getAdrCenterSet())) { //moving adress region
00126 //TODO if length(adrPositions) > 1 print warning (for the moment only works when there's only one particle in adrPositions)
00127                     center=**(verletList->adrPositions.begin());
00128                   }
00129 
00130                   if (sphereAdr){ // spherical adress region
00131                      // calculate distance from reference point
00132                      Real3D dist3D;
00133                      bc.getMinimumImageVectorBox(dist3D,cit->getPos(),center); //pos - center
00134                      real dist = sqrt(dist3D.sqr());
00135 
00136                      if (dist>0.0) {
00137                        // read fforce from table
00138                        real fforce = table->getForce(dist);
00139                        fforce /= dist;
00140 
00141                        // substract td force
00142                        cit->force() += (dist3D * fforce);
00143                      }
00144                   } else {
00145                      // use this if you need 1-dir force only!
00146                      real d1 = cit->getPos()[0] - center[0];
00147                      real d1abs = fabs(d1);
00148                      real force = table->getForce(d1abs);
00149 //std::cout<<cit->id()<<" "<<cit->force()<<" "<<cit->getPos()[0]<<" "<<center[0]<<" "<<d1<<" "<<d1abs<<" "<<force<<std::endl;
00150                      if (d1>0.0) {
00151                        cit->force()[0] += force;
00152                      } else {
00153                        cit->force()[0] -= force;
00154                      }
00155 //std::cout<<cit->id()<<" "<<cit->force()<<std::endl;
00156                   }
00157               }
00158           }
00159     }
