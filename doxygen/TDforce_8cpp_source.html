78  void TDforce::addForce(int itype, const char* _filename, int type) {
79  boost::mpi::communicator world;
80  filename = _filename;
81  Table table;
82 
83  if (itype == 1) { // create a new InterpolationLinear
84  table = make_shared <interaction::InterpolationLinear> ();
85  table->read(world, _filename);
86  }
87 
88  else if (itype == 2) { // create a new InterpolationAkima
89  table = make_shared <interaction::InterpolationAkima> ();
90  table->read(world, _filename);
91  }
92 
93  else if (itype == 3) { // create a new InterpolationCubic
94  table = make_shared <interaction::InterpolationCubic> ();
95  table->read(world, _filename);
96  }
97 
98  if (forces.count(type) == 0) {
99  forces.insert(std::make_pair(type,table));
100  } else {
101  forces[type].reset(); // decrease use_counter by one
102  forces[type] = table;
103  }
104  }

120  void TDforce::applyForce() {
121  LOG4ESPP_DEBUG(theLogger, "apply TD force");
122 
123  System& system = getSystemRef();
124  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
125 
126  std::unordered_map<int, Table>::iterator tableIt;
127  Table table;
128  // iterate over CG particles
129  CellList cells = system.storage->getRealCells();
130  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
131  tableIt = forces.find(cit->getType());
132  if (tableIt != forces.end()) { //because there may be CG particles to which TD force is not applied
133  table = tableIt->second;
134  }
135 
136  if (table) {
137 
138  if (!(verletList->getAdrCenterSet())) { // adress regions defined based on particles
139 
140  if (sphereAdr){
141 
142  if(verletList->getAdrList().size() > 1){
143 
144  if ((startdist == 0.0) && (enddist == 0.0)){
145  std::cout << "In TDforce: Trying to apply TD force with moving, particle-based spherical AdResS region. However, both startdist and enddist set to 0. This does not make sense.\n";
146  exit(1);
147  return;
148  }
149 
150  real buffer = 0.000001;
151  real width = fabs(enddist - startdist + 2.0*buffer); // width of region where TD force acts
152  std::vector<Real3D*>::iterator it2 = verletList->getAdrPositions().begin(); // get positions
153  Real3D pa = **it2;
154  Real3D dist3D;
155  bc.getMinimumImageVectorBox(dist3D,cit->getPos(), pa); // calculate vector between particle and first center
156 
157  Real3D mindist3D = dist3D; // set mindist3D before loop
158 
159  real dist3Dabs = sqrt(dist3D.sqr()); // calculate absolute distance
160 
161  real weight = 0.0; // initialize weight
162 
163  // weighting scheme: only particles that are in a hybrid region contribute
164  if ((dist3Dabs < enddist + buffer) && (dist3Dabs > startdist - buffer)){
165  weight = 1.0 - (dist3Dabs - startdist - buffer) / width; // 0 at edge to CG region, 1 at edge to atomistic region
166 
167  // Direction modification at the edges
168  weight = pow(weight, edgeweightmultiplier);
169  }
170  else{
171  weight = 0.0; // particle outside of hybrid region
172  }
173 
174  Real3D direction = dist3D*weight/dist3Dabs; // normalized but weighted direction vector
175 
176  // Loop over all other centers
177  ++it2;
178  for (; it2 != verletList->getAdrPositions().end(); ++it2) {
179  pa = **it2;
180  verletList->getSystem()->bc->getMinimumImageVector(dist3D, cit->getPos(), pa); // calculate vector between particle and other centers
181  if (dist3D.sqr() < mindist3D.sqr()) mindist3D = dist3D; // make it the minimum if shortest length so far
182  dist3Dabs = sqrt(dist3D.sqr()); // calculate absolute distance
183 
184  // calculate weights again, as above
185  if ((dist3Dabs < enddist + buffer) && (dist3Dabs > startdist - buffer)){
186  weight = 1.0 - (dist3Dabs - startdist - buffer) / width;
187 
188  // Direction modification at the edges
189  weight = pow(weight, edgeweightmultiplier);
190  }
191  else{
192  weight = 0.0;
193  }
194 
195  //norm += weight; // add to normalization constant
196  direction += dist3D*weight/dist3Dabs; // add to direction vector
197  }
198 
199  real mindist3Dabs = sqrt(mindist3D.sqr()); // calculate overall smallest absolute distance
200 
201  // apply td force
202  if ( (mindist3Dabs > startdist) && (mindist3Dabs < enddist) && (direction.sqr() > 0.0) ) {
203  real fforce = table->getForce(mindist3Dabs);
204  fforce /= sqrt(direction.sqr());
205  cit->force() += (direction * fforce);
206  }
207 
208  }
209  else{
210  // calculate distance from reference particle
211  center=**(verletList->adrPositions.begin());
212  Real3D dist3D;
213  bc.getMinimumImageVectorBox(dist3D,cit->getPos(),center); //pos - center
214  real dist = sqrt(dist3D.sqr());
215 
216  if (dist>0.0) {
217  // read fforce from table
218  real fforce = table->getForce(dist);
219  fforce /= dist;
220 
221  // substract td force
222  cit->force() += (dist3D * fforce);
223  }
224  }
225 
226  }
227  else{
228  std::cout << "In TDforce: Trying to use moving AdResS region with adaptive region slab geometry. This is not implemented and doesn't make much sense anyhow.\n";
229  exit(1);
230  return;
231  }
232 
233  }
234  else{
235 
236  if (sphereAdr){ // spherical adress region
237  // calculate distance from reference point
238  Real3D dist3D;
239  bc.getMinimumImageVectorBox(dist3D,cit->getPos(),center); //pos - center
240  real dist = sqrt(dist3D.sqr());
241 
242  if (dist>0.0) {
243  // read fforce from table
244  real fforce = table->getForce(dist);
245  fforce /= dist;
246 
247  // substract td force
248  cit->force() += (dist3D * fforce);
249  }
250  } else {
251  // use this if you need 1-dir force only!
252  real d1 = cit->getPos()[0] - center[0];
253  real d1abs = fabs(d1);
254  real force = table->getForce(d1abs);
255  if (d1>0.0) {
256  cit->force()[0] += force;
257  } else {
258  cit->force()[0] -= force;
259  }
260  }
261 
262  }
263  }
264  }

324  }
