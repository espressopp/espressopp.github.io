107  void TDforce::applyForce() {
108  LOG4ESPP_DEBUG(theLogger, "apply TD force");
109 
110  System& system = getSystemRef();
111  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
112 
113  // iterate over CG particles
114  CellList cells = system.storage->getRealCells();
115  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
116 
117  Table table;
118  if (forces.find(cit->getType())!=forces.end()) { //because there may be CG particles to which TD force is not applied
119  table = forces.find(cit->getType())->second; //TODO shouldn't do find twice
120  }
121 
122  if (table) {
123 
124 
125  if (!(verletList->getAdrCenterSet())) { //moving adress region
126 //TODO if length(adrPositions) > 1 print warning (for the moment only works when there's only one particle in adrPositions)
127  center=**(verletList->adrPositions.begin());
128  }
129 
130  if (sphereAdr){ // spherical adress region
131  // calculate distance from reference point
132  Real3D dist3D;
133  bc.getMinimumImageVectorBox(dist3D,cit->getPos(),center); //pos - center
134  real dist = sqrt(dist3D.sqr());
135 
136  if (dist>0.0) {
137  // read fforce from table
138  real fforce = table->getForce(dist);
139  fforce /= dist;
140 
141  // substract td force
142  cit->force() += (dist3D * fforce);
143  }
144  } else {
145  // use this if you need 1-dir force only!
146  real d1 = cit->getPos()[0] - center[0];
147  real d1abs = fabs(d1);
148  real force = table->getForce(d1abs);
149 //std::cout<<cit->id()<<" "<<cit->force()<<" "<<cit->getPos()[0]<<" "<<center[0]<<" "<<d1<<" "<<d1abs<<" "<<force<<std::endl;
150  if (d1>0.0) {
151  cit->force()[0] += force;
152  } else {
153  cit->force()[0] -= force;
154  }
155 //std::cout<<cit->id()<<" "<<cit->force()<<std::endl;
156  }
157  }
158  }
159  }
