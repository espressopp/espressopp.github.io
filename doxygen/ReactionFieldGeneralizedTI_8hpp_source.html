56  void initialize() {
57  real krc = kappa*rc;
58  real tmp1 = (epsilon1 - 4.0*epsilon2) *
59  (1.0 + krc) - 2.0*epsilon2 * krc*krc;
60  real tmp2 = (epsilon1 + 2.0*epsilon2) *
61  (1.0 + krc) + epsilon2 * krc*krc;
62  rc3 = pow(rc,3);
63  rc2 = pow(rc,2);
64  B1 = tmp1/tmp2;
65  B1 = (1.0+B1) / rc3;
66  B1_half = B1/2.0;
67  crf = 3*epsilon2/(rc*(2*epsilon2+1.0));
68  prefactor/=epsilon1;
69  complLambdaTI = 1.0 - lambdaTI;
70  }

126  bool checkTIpair(size_t pid1, size_t pid2) const {
127  if (annihilate) {
128  //if at least one of pid1 and pid2 in pidsTI (OR)
129  if ((pidsTI.find(pid1) != pidsTI.end()) or (pidsTI.find(pid2) != pidsTI.end())) {
130  return true;
131  } else {
132  return false;
133  }
134  } else { //decouple
135  //if pid1 or pid2 in pidsTI, but not both (XOR)
136  if ((pidsTI.find(pid1) != pidsTI.end()) != (pidsTI.find(pid2) != pidsTI.end())) {
137  return true;
138  } else {
139  return false;
140  }
141  }

144  real _computeEnergy(const Particle& p1, const Particle& p2) const {
145  Real3D dist = p1.position() - p2.position();
146  real distSqr = dist.sqr();
147  if (distSqr>rc2) return 0.0;
148  real qq = p1.q()*p2.q();
149  if (checkTIpair(p1.id(),p2.id())) {
150  qq *= complLambdaTI; //(1-lambda)*qAi*qAj
151  }
152  /* Note: this implementation counts minus integral of the force as energy
153  which is not the same as the full electrostatic energy
154  See the original paper Tironi et al, J. Chem. Phys. 102 , 5451 (1995)
155  for details*/
156 
157  real energy = prefactor*qq * (1.0 / sqrt(distSqr) - B1_half*distSqr -crf);
158  return energy;
159  }

161  bool _computeForce(Real3D& force, const Particle &p1,
162  const Particle &p2) const {
163  Real3D dist = p1.position() - p2.position();
164  real r2 = dist.sqr();
165  if (r2>rc2) return true;
166  real r = sqrt(r2);
167  real qq = p1.q()*p2.q();
168  if (checkTIpair(p1.id(),p2.id())) {
169  qq *= complLambdaTI; //(1-lambda)*qAi*qAj
170  }
171  real ffactor = prefactor*qq* (1.0/(r*r2) + B1);
172  force = dist * ffactor;
173  return true;
174 
175  }

177  real _computeEnergyDeriv(const Particle& p1, const Particle& p2) const {
178  Real3D dist = p1.position() - p2.position();
179  real distSqr = dist.sqr();
180  if (distSqr>rc2) return 0.0;
181  if (checkTIpair(p1.id(),p2.id())) {
182  real qq = p1.q()*p2.q();
183  real dhdl = -1.0 * prefactor * qq * (1.0 / sqrt(distSqr) - B1_half*distSqr -crf);
184  return dhdl;
185  } else {
186  return 0.0;
187  }
188  }
