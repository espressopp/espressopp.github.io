67  bool _computeForceRaw(Real3D& force12,
68  Real3D& force32,
69  const Real3D& r12,
70  const Real3D& r32, real _theta0) const {
71 
72  real dist12_sqr = r12.sqr();
73  real dist32_sqr = r32.sqr();
74  real dist12_magn = sqrt(dist12_sqr);
75  real dist32_magn = sqrt(dist32_sqr);
76  real dU_dtheta, a11, a12, a22;
77  real inv_dist1232 = 1.0/(dist12_magn * dist32_magn);
78 
79  real cos_theta = r12 * r32 * inv_dist1232;
80  if(cos_theta < -1.0) cos_theta = -1.0;
81  else if(cos_theta > 1.0) cos_theta = 1.0;
82  real sin_theta = sqrt(1.0 - cos_theta * cos_theta);
83 
84  dU_dtheta = - KpK * (acos(cos_theta) - _theta0) / sin_theta;
85 
86  a11 = dU_dtheta * cos_theta / dist12_sqr;
87  a12 = -dU_dtheta * inv_dist1232;
88  a22 = dU_dtheta * cos_theta / dist32_sqr;
89 
90  force12 = a11 * r12 + a12 * r32;
91  force32 = a22 * r32 + a12 * r12;
92  return true;
93  }
