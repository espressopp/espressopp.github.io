00092       couplecount++;
00093       if (couplecount < coupling) {

00110 
00111       for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00112         Real3D vel = cit->velocity();
00113         EKin_local += cit->mass() * (vel * vel); //0.5 *

00115       EKin_local *= 0.5;
00116 
00117       boost::mpi::all_reduce(*getSystem()->comm, EKin_local, EKin, std::plus<real>());

00119 
00120       DegreesOfFreedom   = 1.5 * NPart; //3.0/2.0

00122       SkalingFaktor      = sqrt(temperature / currentTemperature);

00124       for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00125         cit->velocity() *= SkalingFaktor;
