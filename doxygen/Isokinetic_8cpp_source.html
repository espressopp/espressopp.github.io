89  void Isokinetic::rescaleVelocities() {
90  LOG4ESPP_DEBUG(theLogger, "rescaleVelocities");
91 
92  couplecount++;
93  if (couplecount < coupling) {
94  return;
95  } else {
96  couplecount = 0;
97  }

99  int NPart_local, NPart;
100  real EKin = 0.0;
101  real EKin_local = 0.0;
102  real DegreesOfFreedom;
103  real currentTemperature;
104  real SkalingFaktor;
105 
106  System& system = getSystemRef();
107  CellList realCells = system.storage->getRealCells();
108 
109  NPart_local = system.storage->getNRealParticles();
110 
111  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
112  Real3D vel = cit->velocity();
113  EKin_local += cit->mass() * (vel * vel); //0.5 *
114  }
115  EKin_local *= 0.5;
116 
117  boost::mpi::all_reduce(*getSystem()->comm, EKin_local, EKin, std::plus<real>());
118  boost::mpi::all_reduce(*getSystem()->comm, NPart_local, NPart, std::plus<int>());
119 
120  DegreesOfFreedom = 1.5 * NPart; //3.0/2.0
121  currentTemperature = EKin / DegreesOfFreedom;
122  SkalingFaktor = sqrt(temperature / currentTemperature);
123 
124  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
125  cit->velocity() *= SkalingFaktor;
126  }
127 
128  }
