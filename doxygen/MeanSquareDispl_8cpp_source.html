00053       //creating vector which stores particleIDs for each CPU
00054       vector<longint> localIDs; 
00055       for (map<size_t,int>::const_iterator itr=idToCpu.begin(); itr!=idToCpu.end(); ++itr) {
00056         size_t i = itr->first;

00066         Real3D posCOM = Real3D(0.0,0.0,0.0);

00082       }
00083       
00084       // MSD calculation
00085       int perc=0, perc1=0;
00086       real denom = 100.0 / (real)M;

00092           for (vector<longint>::iterator itr=localIDs.begin(); itr!=localIDs.end(); ++itr) {
00093             size_t i = *itr;
00094             
00095             Real3D pos1 = getConf(n + m)->getCoordinates(i); // - centerOfMassList[n+m];
00096             Real3D pos2 = getConf(n)->getCoordinates(i); //     - centerOfMassList[n];
00097             Real3D delta = pos2 - pos1;
00098             Z[m] += delta.sqr();
00099           }
00100         }
00101         if(print_progress && system.comm->rank()==0){
00102           perc = (int)(m*denom);
00103           if(perc%5==0){
00104             cout<<"calculation progress (mean square displacement): "<< perc << " %\r"<<flush;
00105           }
00106         }
00107       }
00108       if(system.comm->rank()==0)
00109         cout<<"calculation progress (mean square displacement): 100%"<<endl;
00110       //summation of results from different CPUs
00111       boost::mpi::all_reduce( *system.comm, Z, M, totZ, plus<real>() );
00112       
00113       for(int m=0; m<M; m++){
00114         totZ[m] /= (real)(M - m);
00115       }
00116       
00117       real inv_coef = 1.0 / (6.0 * num_of_part);
00118       
00119       for(int m=0; m<M; m++){
00120         totZ[m] *= inv_coef;
00121         pyli.append( totZ[m] );

00123

00125       Z = NULL;
00126       delete [] totZ;
00127       totZ = NULL;
00128
