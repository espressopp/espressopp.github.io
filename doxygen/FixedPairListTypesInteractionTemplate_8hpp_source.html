44 //In FixedPairListInteractionTemplate.hpp, one potential is used for the entire FixedPairList.
45 //Here, the potential used for each FixedPair comes from an array of potentials, as a function of the types of the two particles in the FixedPair (i.e. same as how it works in VerletListInteractionTemplate.hpp)
46 //This is needed e.g. for non-bonded fixed-pair interactions, such as 1-4 LJ and Coulomb interactions in protein forcefields.

48 //Unlike in FixedPairListInteractionTemplate.hpp, no ltMaxBondSqr variable is calculated here. (The FixedPairListInteractionTemplate.hpp variable seems to be unused.)

79  void
80  setPotential(int type1, int type2, const Potential &potential) {
81  // typeX+1 because i<ntypes
82  ntypes = std::max(ntypes, std::max(type1+1, type2+1));
83  potentialArray.at(type1, type2) = potential;
84  if (type1 != type2) { // add potential in the other direction
85  potentialArray.at(type2, type1) = potential;
86  }
87  }
