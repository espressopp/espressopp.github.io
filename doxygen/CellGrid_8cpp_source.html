39 CellGrid::CellGrid(const Int3D& _size, const real _myLeft[3], const real _myRight[3], int _frame):
40  Grid(_size[0]+2*_frame, _size[1]+2*_frame, _size[2]+2*_frame),
41  frame(_frame), extraSize(2*frame)
42 {
43  if (_size[0] <= 0 || _size[1] <= 0 || _size[2] <= 0) {
44  throw CellGridIllegal();
45  }
46 
47  for(int i = 0; i < 3; ++i) {
48  myLeft[i] = _myLeft[i];
49  myRight[i] = _myRight[i];
50 
51  cellSize[i] = (myRight[i] - myLeft[i])/real(_size[i]);
52  invCellSize[i] = 1.0 / cellSize[i];
53  }
54

99 longint CellGrid::mapPositionToCellChecked(const Real3D& pos) const
100 {
101  int cpos[3];
102 
103  for(int i = 0; i < 3; ++i) {
104  real lpos = pos[i] - myLeft[i];
105 
106  cpos[i] = static_cast<int>(lpos*invCellSize[i]) + frame;
107 
108  /* particles outside our box. Still take them if
109  VERY close or nonperiodic boundary */
110  if (cpos[i] < frame) {
111  if (lpos >= -ROUND_ERROR_PREC) {
112  cpos[i] = frame;

115  return noCell;
