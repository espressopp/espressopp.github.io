00143     void LangevinBarostat::updVolume(){
00144       real dt = integrator->getTimeStep();
00145       
00146       System& system = getSystemRef();
00147       esutil::Error err(system.comm);
00148       
00149       // The volume is scaled according to the equations V(t+1/2*dt) = V(t) + 1/2*dt*V';
00150       // V' = d*V*pe/W
00151       // in order to accelerate  1.5 * dt := 0.5dt * 3
00152       real volScale = 1 + dt * 1.5 * momentum_mass;
00153       real scale_factor = 0;
00154       if( volScale < 0.0){
00155         stringstream msg;
00156         msg << "Scaling coefficient is <0 (Langevin barostat). coef="<<volScale;
00157         err.setException( msg.str() );
00158         err.checkException();
00159       }
00160 
00161       scale_factor = pow( volScale, 1./3.);  // calculating the current scaling parameter
00162       
00163       system.scaleVolume( scale_factor, false);
00164     }

00168      *  TODO now it is valid for nonbonded systems because the value of degreees of freedom

00172     void LangevinBarostat::updVolumeMomentum(){
00173       real dt_2 = 0.5 * integrator->getTimeStep();
00174 
00175       // dt_2 is timestep/2. 
00176       System& system = getSystemRef();
00177       Real3D Li = system.bc -> getBoxL(); // getting the system size
00178       real V = Li[0] * Li[1] * Li[2];     // system volume
00179       real m3V = 3 * V;
00180       
00181       // get a random value and distribute the same value over all of the CPUs
00182       mpi::communicator communic = *system.comm;
00183       real rannum; 
00184       if (communic.rank() == 0) rannum = (*rng)()-0.5; //rannum = rng->normal();
00185       mpi::broadcast(communic, rannum, 0);
00186       
00187       // compute the kinetic contribution (2/3 \sum 1/2mv^2)
00188       // it's not efficient to use the analysis.Pressure because of double calculation of m*v*v
00189       real v2sum;
00190       real v2 = 0.0;
00191       CellList realCells = system.storage->getRealCells();
00192       for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00193         const Particle& p = *cit;
00194         v2 += p.mass() * (p.velocity() * p.velocity());
00195       }
00196       mpi::all_reduce( communic, v2, v2sum, std::plus<real>());
00197       real p_kinetic = v2sum;
00198       
00199       // compute the short-range nonbonded contribution
00200       real rij_dot_Fij = 0.0;
00201       const InteractionList& srIL = system.shortRangeInteractions;
00202       for (size_t j = 0; j < srIL.size(); j++) {
00203         rij_dot_Fij += srIL[j]->computeVirial();
00204       }
00205       real p_nonbonded = rij_dot_Fij;
00206       // TODO optimization is needed, some terms are the same at the begin and at the end of integration
00207       real X = p_kinetic + p_nonbonded;
00208       /*
00209        * in order not to do double calculations X is not divided by 3V and term in the next line
00210        * 3V * (X - externalPressure) is replaced by (X - 3V * externalPressure)
00211        * local momentum derivative; in order to optimize term pref6 * v2sum could be coupled with X
00212        */
00213       real pe_deriv = (X - m3V * externalPressure) + pref6 * v2sum + pref4 * momentum + pref5 * rannum;
00214       
00215       //cout<< "externalPressure: " << externalPressure << endl;
00216       
00217       momentum += dt_2 * pe_deriv;
00218       momentum_mass = momentum / mass; // momentum is normalized by mass already
00219     }

00254     void LangevinBarostat::initialize(){
00255       // calculate the prefactors
00256       LOG4ESPP_INFO(theLogger, "init, gammaP = " << gammaP << 
00257                                ", external pressure = " << externalPressure <<
00258                                ", fictitious mass = " << mass );
00259 
00260       // TODO implement Nf for the system with constarins
00261       /* Nf - degrees of freedom. For N particles in d-dimentional system without constrains
00262        * Nf = d * N
00263        */
00264       // determine the number of local particles and total particles
00265       System& system = getSystemRef();
00266       int Nsum = 0;
00267       int N = system.storage->getNRealParticles();
00268       boost::mpi::all_reduce(*mpiWorld, N, Nsum, std::plus<int>());
00269       
00270       pref6 = 1./(double)Nsum;
00271       
00272       pref3 = - (1+pref6);
00273       
00274       // pressure friction prefactor
00275       pref4 = -gammaP;
00276       
00277       real dt = integrator->getTimeStep();
00278       
00279       // uniform distribution prefactor. (it can be used instead of normal distribution)
00280       pref5 = sqrt( 8.0 * desiredTemperature * gammaP * mass / dt );
00281       //pref5 = sqrt( 8.0 * gammaP * mass / dt );
00282     }
