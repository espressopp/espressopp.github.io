143  void LangevinBarostat::updVolume(){
144  real dt = integrator->getTimeStep();
145 
146  System& system = getSystemRef();
147  esutil::Error err(system.comm);
148 
149  // The volume is scaled according to the equations V(t+1/2*dt) = V(t) + 1/2*dt*V';
150  // V' = d*V*pe/W
151  // in order to accelerate 1.5 * dt := 0.5dt * 3
152  real volScale = 1 + dt * 1.5 * momentum_mass;
153  real scale_factor = 0;
154  if( volScale < 0.0){
155  stringstream msg;
156  msg << "Scaling coefficient is <0 (Langevin barostat). coef="<<volScale;
157  err.setException( msg.str() );
158  err.checkException();
159  }
160 
161  scale_factor = pow( volScale, 1./3.); // calculating the current scaling parameter
162 
163  system.scaleVolume( scale_factor, false);
164  }
165 
166  // TODO should be optimized!!!
167  /*
168  * TODO now it is valid for nonbonded systems because the value of degreees of freedom
169  * Nf = 3*N, N - number of particles, 3 - d-dimensional system (d=3). Thus d/Nf is
170  * replaced by 1/N.
171  */
172  void LangevinBarostat::updVolumeMomentum(){
173  real dt_2 = 0.5 * integrator->getTimeStep();
174 
175  // dt_2 is timestep/2.
176  System& system = getSystemRef();
177  Real3D Li = system.bc -> getBoxL(); // getting the system size
178  real V = Li[0] * Li[1] * Li[2]; // system volume
179  real m3V = 3 * V;
180 
181  // get a random value and distribute the same value over all of the CPUs
182  mpi::communicator communic = *system.comm;
183  real rannum;
184  if (communic.rank() == 0) rannum = (*rng)()-0.5; //rannum = rng->normal();
185  mpi::broadcast(communic, rannum, 0);
186 
187  // compute the kinetic contribution (2/3 \sum 1/2mv^2)
188  // it's not efficient to use the analysis.Pressure because of double calculation of m*v*v
189  real v2sum;
190  real v2 = 0.0;
191  CellList realCells = system.storage->getRealCells();
192  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
193  const Particle& p = *cit;
194  v2 += p.mass() * (p.velocity() * p.velocity());
195  }
196  mpi::all_reduce( communic, v2, v2sum, std::plus<real>());
197  real p_kinetic = v2sum;
198 
199  // compute the short-range nonbonded contribution
200  real rij_dot_Fij = 0.0;
201  const InteractionList& srIL = system.shortRangeInteractions;
202  for (size_t j = 0; j < srIL.size(); j++) {
203  rij_dot_Fij += srIL[j]->computeVirial();
204  }
205  real p_nonbonded = rij_dot_Fij;
206  // TODO optimization is needed, some terms are the same at the begin and at the end of integration
207  real X = p_kinetic + p_nonbonded;
208  /*
209  * in order not to do double calculations X is not divided by 3V and term in the next line
210  * 3V * (X - externalPressure) is replaced by (X - 3V * externalPressure)
211  * local momentum derivative; in order to optimize term pref6 * v2sum could be coupled with X
212  */
213  real pe_deriv = (X - m3V * externalPressure) + pref6 * v2sum + pref4 * momentum + pref5 * rannum;
214 
215  //cout<< "externalPressure: " << externalPressure << endl;
216 
217  momentum += dt_2 * pe_deriv;
218  momentum_mass = momentum / mass; // momentum is normalized by mass already
219  }

222  LOG4ESPP_DEBUG(theLogger, "barostating");

233  void LangevinBarostat::updDisplacement(real& maxSqDist){
234  System& system = getSystemRef();
235  CellList cells = system.storage->getRealCells();
236  real dt = integrator->getTimeStep();
237 
238  real coef = dt * momentum_mass;
239  for(CellListIterator cit(cells); !cit.isDone(); ++cit){
240  Particle& p = *cit;
241  Real3D delta = coef * p.position();
242  p.position() += delta;
243  real sqDist = delta * delta;
244  maxSqDist = std::max(maxSqDist, sqDist);
245  }
246  }

254  void LangevinBarostat::initialize(){
255  // calculate the prefactors
256  LOG4ESPP_INFO(theLogger, "init, gammaP = " << gammaP <<
257  ", external pressure = " << externalPressure <<
258  ", fictitious mass = " << mass );
259 
260  // TODO implement Nf for the system with constarins
261  /* Nf - degrees of freedom. For N particles in d-dimentional system without constrains
262  * Nf = d * N
263  */
264  // determine the number of local particles and total particles
265  System& system = getSystemRef();
266  int Nsum = 0;
267  int N = system.storage->getNRealParticles();
268  boost::mpi::all_reduce(*mpiWorld, N, Nsum, std::plus<int>());
269 
270  pref6 = 1./(double)Nsum;
271 
272  pref3 = - (1+pref6);
273 
274  // pressure friction prefactor
275  pref4 = -gammaP;
276 
277  real dt = integrator->getTimeStep();
278 
279  // uniform distribution prefactor. (it can be used instead of normal distribution)
280  pref5 = sqrt( 8.0 * desiredTemperature * gammaP * mass / dt );
281  //pref5 = sqrt( 8.0 * gammaP * mass / dt );
282  }
