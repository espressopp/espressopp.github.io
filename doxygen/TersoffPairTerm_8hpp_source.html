90

108  bool _computeForceRaw(Real3D& force, const Real3D& r12, real distSqr) const {
109 
110  real d12 = sqrt(distSqr);
111  real ffactor = 0.0;
112 
113  if (d12 > R + D) {
114 // force = r12 * ffactor;
115  force = 0.0;
116  }
117  else {
118  real inv_d12 = 1.0 / d12;
119  real fR = A * exp(-lambda1*d12);
120  if (d12 < R - D)
121  ffactor = lambda1*fR;
122  else {
123  real arg_sin = 0.5 * Pi_2D * (d12-R);
124  real fC = 0.5 * (1.0 - sin(arg_sin));
125  real fC_r = -0.5 * Pi_2D * cos(arg_sin);
126  ffactor = -fC_r * fR + lambda1 * fR * fC;
127  }
128  force = r12 * inv_d12 * ffactor;
129  }
130  return true;
131  }
