00054       myV=0.;
00055       for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00056 
00057         // replaced position with velocity
00058         Real3D& velocity = cit->velocity();

00065       real v_data[4], total_v[4];
00066       v_data[0] = myMass;
00067       v_data[1] = myV[0];

00069       v_data[3] = myV[2];
00070 
00071       // Compute the total velocity of the system
00072       boost::mpi::all_reduce(*(system.comm), v_data, 4, total_v, std::plus<real>());
00073 
00074       v[0] = total_v[1]/total_v[0];
00075       v[1] = total_v[2]/total_v[0];
