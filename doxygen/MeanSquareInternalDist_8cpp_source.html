00034     /* 
00035      * calculates the mean square internal distances
00036      * 
00037      * example of how to call it from python:
00038      * 
00039      * msid = espressopp.analysis.MeanSquareInternalDist(system,chainlength)
00040      * msid.gather()            # one can use gatherFromFile instead
00041      * result = msid.compute()  # computes the msid of the last snapshot added
00042      *
00043      * file = open("msid.dat","w")
00044      *   for i in range(chainlength-1):
00045      *   line = "%d %f %f\n" % (i+1,result[i],result[i]/(i+1)) # n, <r^2>, <r^2>/n
00046      *   file.write(line)
00047      * file.close()
00048      *

00051       python::list MeanSquareInternalDist::compute() const{
00052                          
00053       python::list pyli;
00054       
00055       System& system = getSystemRef();
00056             
00057       //creating vector which stores particleIDs for each CPU
00058       vector<longint> localIDs; // localIDs is a vector with PID of particles in each CPU and it contains whole chains
00059       for (map<size_t,int>::const_iterator itr=idToCpu.begin(); itr!=idToCpu.end(); ++itr) {
00060         size_t i = itr->first;
00061         int whichCPU = itr->second;
00062         if(system.comm->rank()==whichCPU){
00063           localIDs.push_back(i); 
00064         }
00065       }
00066       sort(localIDs.begin(), localIDs.end()); //sorts entries from low to high - should not be necessary as long as the above iterator iterates in ascending order according to the keys
00067                 
00068       // MSID calculation
00069       
00070       int M = getListSize(); //number of snapshots/configurations
00071                  
00072       real intdist_sum = 0;
00073                   
00074       int N_chains = localIDs.size() / chainlength; // N_chains is number of chains in this cpu and chainlength is a global variable from ConfigsParticleDecomp.hpp
00075 
00076       for (int n=1; n<chainlength;n++){ // loop over chainlength
00077 
00078           real intdist = 0.0;
00079       
00080           for (int j=0; j<N_chains;j++){
00081               
00082               real sumdistsq = 0.0 ;
00083                                     
00084               for (int i=j*chainlength; i<(j*chainlength)+chainlength-n;i++){
00085                   Real3D pos1 = getConf(M-1)->getCoordinates(localIDs[i]);   // compute the MSID of the last added snapshot
00086                   Real3D pos2 = getConf(M-1)->getCoordinates(localIDs[i+n]); // compute the MSID of the last added snapshot                                                     
00087                   Real3D delta = pos2 - pos1;   //vector with the distances from i to chainlength-n
00088                   sumdistsq += delta.sqr();// dx*dx+dy*dy+dz*dz
00089               }
00090               
00091               intdist += sumdistsq/real(chainlength-n);  //sum of the internal distances (chainlength-n) of the chains in this cpu
00092           }
00093                     
00094           boost::mpi::reduce(*system.comm, intdist, intdist_sum, std::plus<real>(), 0);
00095           
00096           if (system.comm->rank() == 0) { 
00097               int chains_total = num_of_part / chainlength;
00098               pyli.append( intdist_sum / real(chains_total)); 
00099           }          
00100       } 
00101       return pyli;
00102     }
