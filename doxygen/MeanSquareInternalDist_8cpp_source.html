34  /*
35  * calculates the mean square internal distances
36  *
37  * example of how to call it from python:
38  *
39  * msid = espressopp.analysis.MeanSquareInternalDist(system,chainlength)
40  * msid.gather() # one can use gatherFromFile instead
41  * result = msid.compute() # computes the msid of the last snapshot added

45  * line = "%d %f %f\n" % (i+1,result[i],result[i]/(i+1)) # n, <r^2>, <r^2>/n

51  python::list MeanSquareInternalDist::compute() const{
52 
53  python::list pyli;
54 
55  System& system = getSystemRef();
56 
57  //creating vector which stores particleIDs for each CPU
58  vector<longint> localIDs; // localIDs is a vector with PID of particles in each CPU and it contains whole chains
59  for (map<size_t,int>::const_iterator itr=idToCpu.begin(); itr!=idToCpu.end(); ++itr) {
60  size_t i = itr->first;
61  int whichCPU = itr->second;
62  if(system.comm->rank()==whichCPU){
63  localIDs.push_back(i);
64  }
65  }
66  sort(localIDs.begin(), localIDs.end()); //sorts entries from low to high - should not be necessary as long as the above iterator iterates in ascending order according to the keys
67 
68  // MSID calculation
69 
70  int M = getListSize(); //number of snapshots/configurations
71 
72  real intdist_sum = 0;
73 
74  int N_chains = localIDs.size() / chainlength; // N_chains is number of chains in this cpu and chainlength is a global variable from ConfigsParticleDecomp.hpp
75 
76  for (int n=1; n<chainlength;n++){ // loop over chainlength
77 
78  real intdist = 0.0;
79 
80  for (int j=0; j<N_chains;j++){
81 
82  real sumdistsq = 0.0 ;
83 
84  for (int i=j*chainlength; i<(j*chainlength)+chainlength-n;i++){
85  Real3D pos1 = getConf(M-1)->getCoordinates(localIDs[i]); // compute the MSID of the last added snapshot
86  Real3D pos2 = getConf(M-1)->getCoordinates(localIDs[i+n]); // compute the MSID of the last added snapshot
87  Real3D delta = pos2 - pos1; //vector with the distances from i to chainlength-n
88  sumdistsq += delta.sqr();// dx*dx+dy*dy+dz*dz
89  }
90 
91  intdist += sumdistsq/real(chainlength-n); //sum of the internal distances (chainlength-n) of the chains in this cpu
92  }
93 
94  boost::mpi::reduce(*system.comm, intdist, intdist_sum, std::plus<real>(), 0);
95 
96  if (system.comm->rank() == 0) {
97  int chains_total = num_of_part / chainlength;
98  pyli.append( intdist_sum / real(chains_total));
99  }
100  }
101  return pyli;
102  }
