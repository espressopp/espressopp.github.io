67  bool FixedTripleListAdress::add(longint pid1, longint pid2, longint pid3) {
68  bool returnVal = true;
69  System& system = storage->getSystemRef();
70  esutil::Error err(system.comm);
71 
72  //if (pid1 > pid2)
73  //std::swap(pid1, pid2);
74 
75  // ADD THE LOCAL TRIPLES
76  Particle *p1 = storage->lookupAdrATParticle(pid1);
77  Particle *p2 = storage->lookupAdrATParticle(pid2);
78  Particle *p3 = storage->lookupAdrATParticle(pid3);
79 
80  // middle particle is the reference particle and must exist here
81  if (!p2){
82  // particle does not exists here (some other CPU must have it)
83  returnVal = false;
84  }
85  else{
86  if (!p1) {
87  std::stringstream msg;
88  msg << "atomistic triple particle p1 " << pid1 << " does not exists here and cannot be added";
89  msg << " triplet: " << pid1 << "-" << pid2 << "-" << pid3;
90  err.setException( msg.str() );
91  }
92 
93  if (!p3) {
94  std::stringstream msg;
95  msg << "atomistic triple particle p3 " << pid3 << " does not exists here and cannot be added";
96  msg << " triplet: " << pid1 << "-" << pid2 << "-" << pid3;
97  err.setException( msg.str() );
98  }
99  }
100  err.checkException();
101 
102  if(returnVal){
103  // add the triple locally
104  this->add(p1, p2, p3);
105 
106  // ADD THE GLOBAL TRIPLET
107  // see whether the particle already has triples
108  std::pair<GlobalTriples::const_iterator,
109  GlobalTriples::const_iterator> equalRange
110  = globalTriples.equal_range(pid2);
111  if (equalRange.first == globalTriples.end()) {
112  // if it hasn't, insert the new triple
113  globalTriples.insert(std::make_pair(pid2,
114  std::pair<longint, longint>(pid1, pid3)));
115  }
116  else {
117  // otherwise test whether the triple already exists
118  for (GlobalTriples::const_iterator it = equalRange.first;
119  it != equalRange.second; ++it)
120  if (it->second == std::pair<longint, longint>(pid1, pid3))
121  // TODO: Triple already exists, generate error!
122  ;
123  // if not, insert the new triple
124  globalTriples.insert(equalRange.first, std::make_pair(pid2, std::pair<longint, longint>(pid1, pid3)));
125  }
126  LOG4ESPP_INFO(theLogger, "added fixed pair to global pair list");
127  }
128  return returnVal;

132  void FixedTripleListAdress::beforeSendATParticles(std::vector<longint>& atpl,
133  OutBuffer& buf) {
134 
135  //std::cout << "beforeSendATParticles() fixed pl (size " << atpl.size() << ")\n";
136 
137  std::vector< longint > toSend;
138 
139  // loop over the VP particle list
140  for (std::vector<longint>::iterator it = atpl.begin();
141  it != atpl.end(); ++it) {
142  longint pid = *it;
143 
144  LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
145 
146  // find all triples that involve this particle
147  int n = globalTriples.count(pid);
148 
149  if (n > 0) {
150  std::pair<GlobalTriples::const_iterator,
151  GlobalTriples::const_iterator> equalRange
152  = globalTriples.equal_range(pid);
153 
154  // first write the pid of this particle
155  // then the number of partners (n)
156  // and then the pids of the partners
157  toSend.reserve(toSend.size()+2*n+1);
158  toSend.push_back(pid);
159  toSend.push_back(n);
160  for (GlobalTriples::const_iterator it = equalRange.first;
161  it != equalRange.second; ++it) {
162  toSend.push_back(it->second.first);
163  toSend.push_back(it->second.second);
164  }
165 
166  // delete all of these triples from the global list
167  globalTriples.erase(equalRange.first, equalRange.second);
168  }
169  }
170 
171  // send the list
172  buf.write(toSend);
173  LOG4ESPP_INFO(theLogger, "prepared fixed triple list before send particles");
174  }

191 
192  this->clear();

198  for (GlobalTriples::const_iterator it = globalTriples.begin();
199  it != globalTriples.end(); ++it) {
200 
201  //printf("lookup global triple %d %d %d\n", it->first, it->second.first, it->second.second);
202 
203  if (it->first != lastpid2) {
204  p2 = storage->lookupAdrATParticle(it->first);

206  std::stringstream msg;

208  err.setException( msg.str() );
209  }
210  lastpid2 = it->first;
211  }
212  p1 = storage->lookupAdrATParticle(it->second.first);
213  if (p1 == NULL) {
214  std::stringstream msg;
215  msg<< "atomistic triple particle p1 " << it->second.first << " does not exists here";
216  err.setException( msg.str() );

218  p3 = storage->lookupAdrATParticle(it->second.second);
219  if (p3 == NULL) {
220  std::stringstream msg;
