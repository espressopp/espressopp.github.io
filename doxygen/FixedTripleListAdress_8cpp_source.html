00132   void FixedTripleListAdress::beforeSendATParticles(std::vector<longint>& atpl,
00133           OutBuffer& buf) {
00134 
00135         //std::cout << "beforeSendATParticles() fixed pl (size " << atpl.size() << ")\n";
00136 
00137         std::vector< longint > toSend;
00138 
00139         // loop over the VP particle list
00140         for (std::vector<longint>::iterator it = atpl.begin();
00141                 it != atpl.end(); ++it) {
00142           longint pid = *it;
00143 
00144           LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
00145 
00146           // find all triples that involve this particle
00147           int n = globalTriples.count(pid);
00148 
00149           if (n > 0) {
00150             std::pair<GlobalTriples::const_iterator,
00151             GlobalTriples::const_iterator> equalRange
00152                 = globalTriples.equal_range(pid);
00153 
00154             // first write the pid of this particle
00155             // then the number of partners (n)
00156             // and then the pids of the partners
00157             toSend.reserve(toSend.size()+2*n+1);
00158             toSend.push_back(pid);
00159             toSend.push_back(n);
00160             for (GlobalTriples::const_iterator it = equalRange.first;
00161                     it != equalRange.second; ++it) {
00162                 toSend.push_back(it->second.first);
00163                 toSend.push_back(it->second.second);
00164             }
00165 
00166             // delete all of these triples from the global list
00167             globalTriples.erase(equalRange.first, equalRange.second);
00168           }
00169         }
00170 
00171         // send the list
00172         buf.write(toSend);
00173         LOG4ESPP_INFO(theLogger, "prepared fixed triple list before send particles");
00174   }
