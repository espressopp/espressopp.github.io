58  bool FixedLocalTupleList::
59  addTuple(boost::python::list& tuple) {
60  bool returnVal = true;
61  System& system = storage->getSystemRef();
62  esutil::Error err(system.comm);
63 
64  Particle* vp, *at;
65  longint pidK; // the pid used as key
66  std::vector<Particle*> tmp; // temporary vector
67  std::vector<longint> pids; //used to extract from tuple;
68  std::vector<longint> pidstmp; // temporary vector
69 
70  for (int i = 0; i < boost::python::len(tuple); ++i) {
71  pids.push_back(boost::python::extract<int>(tuple[i]));
72  }
73 
74  tuple::iterator it = pids.begin();
75  vp = storage->lookupRealParticle(*it);
76  if (!vp) {// Particle does not exist here, return false
77  returnVal = false;
78  } else {
79  pidK = *it; // first pid is key
80  for (++it; it != pids.end(); ++it) {
81  at = storage->lookupLocalParticle(*it);
82  if (!at) { // Particle does not exist here, return false
83  std::stringstream msg;
84  msg << "ERROR: particle " << *it << " not found \n";
85  err.setException(msg.str());
86  returnVal = false;
87  break;
88  }
89  tmp.push_back(at);
90  pidstmp.push_back(*it); // pidK is not in this vector
91  }
92  }
93  err.checkException();
94 
95  if (returnVal) {
96  this->add(vp, tmp); // add to TupleList
97  globalTuples.insert(make_pair(pidK, pidstmp));
98  LOG4ESPP_INFO(theLogger, "added fixed tuple to global tuples");
99  }
100 
101  tmp.clear();
102  pids.clear();
103  pidstmp.clear();
104 
105 
106  return returnVal;
107  }

125  void FixedLocalTupleList::
126  beforeSendParticles(ParticleList& pl, OutBuffer& buf) {
127  std::vector<longint> toSend;
128  // loop over the particle list
129  for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
130  longint pidK = pit->id();
131  LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pidK << ", find tuples");
132 
133  // find particle that involves this particle id
134  GlobalTuples::const_iterator it = globalTuples.find(pidK);
135  if (it != globalTuples.end()) {
136  int s = it->second.size();
137  toSend.reserve(toSend.size() + s + 2);
138  // first write the pid of the first particle
139  toSend.push_back(pidK);
140  // write the size of the vector
141  toSend.push_back(s);
142 
143  // iterate through vector and add pids
144  for (tuple::const_iterator it2 = it->second.begin();
145  it2 != it->second.end(); ++it2) {
146  toSend.push_back(*it2);
147  }
148  // delete this pid from the global list
149  globalTuples.erase(pidK);
150  LOG4ESPP_DEBUG(theLogger, "Erase pid " << pidK);
151  }
152  }
153  buf.write(toSend);
154  }

156  void FixedLocalTupleList::
157  afterRecvParticles(ParticleList &pl, InBuffer& buf) {

172  n = received[i++];

175  int pidV = received[i++];
176  pids.push_back(pidV);
177  }
178  it = globalTuples.insert(it, std::make_pair(pidK, pids));
179  pids.clear();
180  LOG4ESPP_DEBUG(theLogger, "Insert pid " << pidK);

186  void FixedLocalTupleList::onParticlesChanged() {
187  // TODO errors should be thrown in a nice way
188  LOG4ESPP_INFO(theLogger, "rebuild local particle list from global tuples\n");
189  System& system = storage->getSystemRef();
190  esutil::Error err(system.comm);
191 
192  this->clear();

194  std::vector<Particle*> tmp;
195 
196  GlobalTuples::const_iterator it = globalTuples.begin();
197 
198  for (;it != globalTuples.end(); ++it) {
199  if (globalTuples.count(it->first) > 1) {
200  std::stringstream msg;
201  msg << " particle in tuple " << it->first << " is duplicated";
202  err.setException( msg.str() );
203  }
204  vp = storage->lookupRealParticle(it->first);
205  if (vp == NULL) {
206  std::stringstream msg;
207  msg << " particle in tuple " << it->first << " does not exists here";
208  err.setException( msg.str() );
209  }
210 
211  // iterate through vector in map
212  for (tuple::const_iterator it2 = it->second.begin(); it2 != it->second.end(); ++it2) {
213  at = storage->lookupLocalParticle(*it2);
214  if (at == NULL) {
215  std::stringstream msg;

217  err.setException( msg.str() );
218  }
219  tmp.push_back(at);
220  }
221  this->add(vp, tmp);
222  tmp.clear();
223  }
