00084   void VerletList::rebuild()
00085   {
00086     //real cutVerlet = cut + getSystem() -> getSkin();
00087     cutVerlet = cut + getSystem() -> getSkin();
00088     cutsq = cutVerlet * cutVerlet;
00089     
00090     vlPairs.clear();
00091 
00092     // add particles to adress zone

00094     LOG4ESPP_DEBUG(theLogger, "local cell list size = " << cl.size());
00095     for (CellListAllPairsIterator it(cl); it.isValid(); ++it) {
00096       checkPair(*it->first, *it->second);
00097       LOG4ESPP_DEBUG(theLogger, "checking particles " << it->first->id() << " and " << it->second->id());
00098     }
00099     
00100     builds++;
00101     LOG4ESPP_DEBUG(theLogger, "rebuilt VerletList (count=" << builds << "), cutsq = " << cutsq

00108   void VerletList::checkPair(Particle& pt1, Particle& pt2)
00109   {
00110 
00111     Real3D d = pt1.position() - pt2.position();
00112     real distsq = d.sqr();
00113 
00114     LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
00115                    << " @ " << pt1.position() 
00116                    << " - p2: " << pt2.id() << " @ " << pt2.position()
00117                    << " -> distsq = " << distsq);
00118 
00119     if (distsq > cutsq) return;
00120 
00121     // see if it's in the exclusion list (both directions)
00122     if (exList.count(std::make_pair(pt1.id(), pt2.id())) == 1) return;
00123     if (exList.count(std::make_pair(pt2.id(), pt1.id())) == 1) return;
00124 
00125     vlPairs.add(pt1, pt2); // add pair to Verlet List
00126   }
