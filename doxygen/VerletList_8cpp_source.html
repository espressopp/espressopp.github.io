89  void VerletList::rebuild()
90  {
91  timer.reset();
92  real currTime = timer.getElapsedTime();
93 
94  //real cutVerlet = cut + getSystem() -> getSkin();
95  cutVerlet = cut + getSystem() -> getSkin();
96  cutsq = cutVerlet * cutVerlet;
97 
98  vlPairs.clear();
99 
100  if(useBuffers) {
101  rebuildUsingBuffers(exList.size(), useSOA);
102  } else {
103  // add particles to adress zone
104  CellList cl = getSystem()->storage->getRealCells();
105  LOG4ESPP_DEBUG(theLogger, "local cell list size = " << cl.size());
106  for (CellListAllPairsIterator it(cl); it.isValid(); ++it) {
107  checkPair(*it->first, *it->second);
108  LOG4ESPP_DEBUG(theLogger, "checking particles " << it->first->id() << " and " << it->second->id());
109  }
110  }
111 
112  builds++;
113  timeRebuild += timer.getElapsedTime() - currTime;
114  LOG4ESPP_DEBUG(theLogger, "rebuilt VerletList (count=" << builds << "), cutsq = " << cutsq
115  << " local size = " << vlPairs.size());
116  }

120  template< bool USE_EXCLUSION_LIST, bool USE_SOA >
121  void VerletList::_rebuildUsingBuffers()
122  {
123  const CellList& realCells = getSystem()->storage->getRealCells();
124  const size_t numRealCells = realCells.size();
125 
126  // stores the range of neighbor particles belonging to cell i: with end=c_range[i]
127  std::vector<int> c_range;
128  c_range.reserve(numRealCells);
129 
130  // get the number of particles in all neighbor cells
131  size_t c_reserve = 0;
132  for(size_t icell=0; icell<numRealCells; icell++) {
133  size_t row_reserve = 0;
134  for(NeighborCellInfo& nc: realCells[icell]->neighborCells) {
135  if(!nc.useForAllPairs) {
136  row_reserve += nc.cell->particles.size();
137  }
138  }
139  c_reserve += row_reserve;
140  c_range.push_back(c_reserve);
141  }
142 
143  // resize buffer
144  if(c_reserve > c_p.size()) {
145  size_t c_resize = 2*c_reserve;
146  c_p.resize(c_resize);
147  if(USE_SOA){
148  c_x.resize(c_resize);
149  c_y.resize(c_resize);
150  c_z.resize(c_resize);
151  } else {
152  c_pos.resize(c_resize);
153  }
154  c_type.resize(c_resize);
155  if(USE_EXCLUSION_LIST){
156  c_id.resize(c_resize);
157  }
158  }
159 
160  // fill buffer
161  size_t ip = 0;
162  for(size_t icell=0; icell<numRealCells; icell++) {
163  size_t end = c_range[icell];
164  for(NeighborCellInfo& nc: realCells[icell]->neighborCells) {
165  if(!nc.useForAllPairs) {
166  for(Particle& p: nc.cell->particles) {
167  c_p[ip] = &p;
168  if(USE_EXCLUSION_LIST)
169  c_id[ip] = p.id();
170  c_type[ip] = p.type();
171  if(USE_SOA){
172  const Real3D& pos = p.position();
173  c_x[ip] = pos[0];
174  c_y[ip] = pos[1];
175  c_z[ip] = pos[2];
176  } else {
177  c_pos[ip] = p.position();
178  }
179  ip++;
180  }
181  }
182  }
183  if(ip!=end) std::runtime_error("Range mismatch.");
184  }
185 
186  // rebuild neighbor list
187  size_t start=0;
188  for(size_t icell=0; icell<numRealCells; icell++) {
189  size_t end=c_range[icell];
190  ParticleList& particles = realCells[icell]->particles;
191  size_t numParticles = particles.size();
192  for(size_t p1=0; p1<numParticles; p1++) {
193  Particle& part1 = particles[p1];
194 
195  // self-loop
196  for(size_t p2=p1+1; p2<numParticles; p2++) {
197  Particle& part2 = particles[p2];
198  checkPair(part1, part2);
199  }
200 
201  Real3D p1_pos;
202  real x1, y1, z1;
203  if(USE_SOA) {
204  const Real3D& pos = part1.position();
205  x1 = pos[0];
206  y1 = pos[1];
207  z1 = pos[2];
208  } else {
209  p1_pos = part1.position();
210  }
211  size_t id1;
212  if(USE_EXCLUSION_LIST) {
213  id1 = part1.id();
214  }
215  const size_t type1 = part1.type();
216 
217  // neighbor-loop
218  for(size_t p2=start; p2<end; p2++) {
219  real distsq;
220  if(USE_SOA) {
221  real dist_x = x1 - c_x[p2];
222  real dist_y = y1 - c_y[p2];
223  real dist_z = z1 - c_z[p2];
224  distsq = dist_x*dist_x + dist_y*dist_y + dist_z*dist_z;
225  } else {
226  Real3D d = p1_pos - c_pos[p2];
227  distsq = d.sqr();
228  }
229 
230  bool addpair = true;
231  if (distsq > cutsq) addpair = false;
232 
233  if(USE_EXCLUSION_LIST) {
234  size_t const& id2 = c_id[p2];
235  if (exList.count(std::make_pair(id1, id2)) == 1) addpair = false;
236  if (exList.count(std::make_pair(id2, id1)) == 1) addpair = false;
237  }
238 
239  if(addpair) {
240  max_type = std::max(max_type, std::max(type1, c_type[p2]));
241  vlPairs.add(&part1,c_p[p2]);
242  }
243  }
244  }
245  start = end;
246  }
247  }

251  void VerletList::checkPair(Particle& pt1, Particle& pt2)
252  {
253 
254  Real3D d = pt1.position() - pt2.position();
255  real distsq = d.sqr();
256 
257  LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
258  << " @ " << pt1.position()
259  << " - p2: " << pt2.id() << " @ " << pt2.position()
260  << " -> distsq = " << distsq);
261 
262  if (distsq > cutsq) return;
263 
264  // see if it's in the exclusion list (both directions)
265  if (exList.count(std::make_pair(pt1.id(), pt2.id())) == 1) return;
266  if (exList.count(std::make_pair(pt2.id(), pt1.id())) == 1) return;
267 
268  max_type = std::max(max_type, std::max(pt1.type(), pt2.type()));
269  vlPairs.add(pt1, pt2); // add pair to Verlet List
270  }
