84  void VerletList::rebuild()
85  {
86  //real cutVerlet = cut + getSystem() -> getSkin();
87  cutVerlet = cut + getSystem() -> getSkin();
88  cutsq = cutVerlet * cutVerlet;
89 
90  vlPairs.clear();
91 
92  // add particles to adress zone
93  CellList cl = getSystem()->storage->getRealCells();
94  LOG4ESPP_DEBUG(theLogger, "local cell list size = " << cl.size());
95  for (CellListAllPairsIterator it(cl); it.isValid(); ++it) {
96  checkPair(*it->first, *it->second);
97  LOG4ESPP_DEBUG(theLogger, "checking particles " << it->first->id() << " and " << it->second->id());
98  }
99 
100  builds++;
101  LOG4ESPP_DEBUG(theLogger, "rebuilt VerletList (count=" << builds << "), cutsq = " << cutsq
102  << " local size = " << vlPairs.size());
103  }

108  void VerletList::checkPair(Particle& pt1, Particle& pt2)
109  {
110 
111  Real3D d = pt1.position() - pt2.position();
112  real distsq = d.sqr();
113 
114  LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
115  << " @ " << pt1.position()
116  << " - p2: " << pt2.id() << " @ " << pt2.position()
117  << " -> distsq = " << distsq);
118 
119  if (distsq > cutsq) return;
120 
121  // see if it's in the exclusion list (both directions)
122  if (exList.count(std::make_pair(pt1.id(), pt2.id())) == 1) return;
123  if (exList.count(std::make_pair(pt2.id(), pt1.id())) == 1) return;
124 
125  vlPairs.add(pt1, pt2); // add pair to Verlet List
126  }
