316  void copyAsGhost(const Particle& src, int extradata, const Real3D& shift) {
317 
318  src.r.copyShifted(r, shift);
319  if (extradata & DATA_PROPERTIES) {
320  p = src.p;
321  }
322  if (extradata & DATA_MOMENTUM) {
323  m = src.m;
324  }
325  if (extradata & DATA_LOCAL) {
326  l = src.l;
327  }
328  ghost() = 1;
329  }
