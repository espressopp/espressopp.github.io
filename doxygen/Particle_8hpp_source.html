318  void copyAsGhost(const Particle& src, int extradata, const Real3D& shift) {
319 
320  src.r.copyShifted(r, shift);
321  if (extradata & DATA_PROPERTIES) {
322  p = src.p;
323  }
324  if (extradata & DATA_MOMENTUM) {
325  m = src.m;
326  }
327  if (extradata & DATA_LOCAL) {
328  l = src.l;
329  }
330  ghost() = 1;
331  }
