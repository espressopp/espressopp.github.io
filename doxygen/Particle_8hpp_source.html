375  void copyAsGhost(const Particle& src, int extradata, const Real3D& shift) {
376 
377  src.r.copyShifted(r, shift);
378  if (extradata & DATA_PROPERTIES) {
379  p = src.p;
380  }
381  if (extradata & DATA_MOMENTUM) {
382  m = src.m;
383  }
384  if (extradata & DATA_LOCAL) {
385  l = src.l;
386  }
387  ghost() = 1;
388  }
