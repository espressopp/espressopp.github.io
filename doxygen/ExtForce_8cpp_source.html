89  void ExtForce::applyForceToGroup() {
90  LOG4ESPP_DEBUG(theLogger, "applying external force to particle group of size " << particleGroup->size());
91  for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
92  LOG4ESPP_DEBUG(theLogger, "applying external force to particle " << it->getId());
93  it->force() += extForce;
94  }
95  }

97  void ExtForce::applyForceToAll() {
98  System& system = getSystemRef();
99  CellList realCells = system.storage->getRealCells();
100  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
101  cit->force() += extForce;
