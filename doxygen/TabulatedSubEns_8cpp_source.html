99  void TabulatedSubEns::computeColVarWeights(const Real3D& dist,
100  const bc::BC& bc){
101  // Sanity Check
102  if (weights.getDimension() == 0)
103  throw std::runtime_error("TabulatedSubEns requires at least one interaction.");
104  // Compute the weights for each force field
105  // given the reference and instantaneous values of ColVars
106  setColVar(dist, bc);
107  // Compute weights up to next to last FF
108  real maxWeight = 0.;
109  int maxWeightI = 0;
110  // Check first whether we're stuck in a surface
111  bool stuck = false;
112  for (int i=0; i<numInteractions; ++i) {
113  if (weights[i] > maxWeight) {
114  maxWeight = weights[i];
115  maxWeightI = i;
116  }
117  }
118  // Lock the surface until it's sampled >= 98% of the target prob.
119  if (weightCounts > 0 &&
120  maxWeightI < numInteractions-1 &&
121  maxWeight > 0. &&
122  weightSum[maxWeightI]/weightCounts < 0.98*targetProb[maxWeightI])
123  stuck = true;
124  if (!stuck) {
125  maxWeight = 0.;
126  maxWeightI = numInteractions-1;
127  for (int i=0; i<numInteractions-1; ++i) {
128  weights[i] = 1.;
129  real norm_d_i = 0.;
130  real norm_l_i = 0.;
131  for (int j=0; j<colVar.getDimension(); ++j) {
132  int k = 0;
133  // Choose between bond, angle, and dihed
134  if (j <= 0+colVarBondListSize) k = 0;
135  else if (j<1+colVarBondListSize+colVarAngleListSize) k = 1;
136  else k = 2;
137  real diff = colVar[j] - colVarRef[i][k];
138  if (k == 2) {
139  if (diff>M_PI) diff -= 2.0*M_PI;
140  if (diff<(-1.0*M_PI)) diff += 2.0*M_PI;
141  }
142  norm_d_i += pow(diff / colVarSd[k], 2);
143  norm_l_i += pow(colVarRef[i][3+k], 2);
144  }
145  if (norm_d_i > norm_l_i)
146  weights[i] = exp(- (sqrt(norm_d_i) - sqrt(norm_l_i)) / alpha);
147  if (weights[i] > maxWeight) {
148  maxWeight = weights[i];
149  maxWeightI = i;
150  }
151  }
152  for (int i=0; i<numInteractions-1; ++i) {
153  if (i != maxWeightI)
154  weights[i] = 0.;
155  else {
156  if (weightCounts > 0 &&
157  weights[i] > 0.01 &&
158  weightSum[i]/weightCounts < 0.98*targetProb[i]) {
159  weights[i] = 1.0;
160  maxWeight = 1.;
161  }
162  }
163  }
164  if (maxWeightI == numInteractions-1)
165  maxWeight = 1.;
166  weights[numInteractions-1] = 1. - maxWeight;
167  }
168 
169  // Update weightSum
170  for (int i=0; i<numInteractions; ++i)
171  weightSum[i] += weights[i];
172  weightCounts += 1;
173  }

175  // Collective variables
176  void TabulatedSubEns::setColVar(const Real3D& dist, const bc::BC& bc) {
177  colVar.setDimension(1);
178  colVar[0] = sqrt(dist*dist);
179  int i=1;
180  if (colVarBondList != nullptr) {
181  // Now all bonds in colVarBondList
182  colVarBondListSize = colVarBondList->size();
183  for (FixedPairList::PairList::Iterator it(*colVarBondList); it.isValid(); ++it) {
184  colVar.setDimension(i+1);
185  Particle &p1 = *it->first;
186  Particle &p2 = *it->second;
187  Real3D dist12;
188  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
189  colVar[i] = sqrt(dist12 * dist12);
190  i+=1;
191  }
192  }
193  if (colVarAngleList != nullptr) {
194  // Now all angles in colVarAngleList
195  colVarAngleListSize = colVarAngleList->size();
196  for (FixedTripleList::TripleList::Iterator it(*colVarAngleList); it.isValid(); ++it) {
197  colVar.setDimension(i+1);
198  Particle &p1 = *it->first;
199  Particle &p2 = *it->second;
200  Particle &p3 = *it->third;
201  Real3D dist12, dist32;
202  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
203  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
204  real dist12_sqr = dist12 * dist12;
205  real dist32_sqr = dist32 * dist32;
206  real dist1232 = sqrt(dist12_sqr) * sqrt(dist32_sqr);
207  real cos_theta = dist12 * dist32 / dist1232;
208  colVar[i] = acos(cos_theta);
209  i+=1;
210  }
211  }
212  if (colVarDihedList != nullptr) {
213  colVarDihedListSize = colVarDihedList->size();
214  // Now all dihedrals in colVarDihedList
215  for (FixedQuadrupleList::QuadrupleList::Iterator it(*colVarDihedList); it.isValid(); ++it) {
216  colVar.setDimension(i+1);
217  Particle &p1 = *it->first;
218  Particle &p2 = *it->second;
219  Particle &p3 = *it->third;
220  Particle &p4 = *it->fourth;
221  Real3D r21, r32, r43;
222  bc.getMinimumImageVectorBox(r21, p2.position(), p1.position());
223  bc.getMinimumImageVectorBox(r32, p3.position(), p2.position());
224  bc.getMinimumImageVectorBox(r43, p4.position(), p3.position());
225  colVar[i] = DihedralPotential::computePhi(r21, r32, r43);
226  i+=1;
227  }
228  }
229  }
