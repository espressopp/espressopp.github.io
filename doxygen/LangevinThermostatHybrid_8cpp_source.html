131  {
132  LOG4ESPP_DEBUG(theLogger, "thermalizeAdr");

166  void LangevinThermostatHybrid::frictionThermo(Particle& p,real weight)
167  {
168  real massf = sqrt(p.mass());
169 
170  // get a random value for each vector component
171 
172  Real3D ranval((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
173 
174  if (weight<1.0 && weight>0.0) { //hybrid region
175  p.force() += pref1hy * p.velocity() * p.mass() +
176  pref2hy * ranval * massf;
177  } else if (weight==1.0) {
178  p.force() += pref1 * p.velocity() * p.mass() +
179  pref2 * ranval * massf;
180  } else {
181  p.force() += pref1cg * p.velocity() * p.mass() +
182  pref2cg * ranval * massf;
183  }
184 
185  LOG4ESPP_TRACE(theLogger, "new force of p = " << p.force());
186  }

188  void LangevinThermostatHybrid::initialize()
189  { // calculate the prefactors
190 
191  real timestep = integrator->getTimeStep();
192 
193  LOG4ESPP_INFO(theLogger, "init, timestep = " << timestep <<
194  ", gamma = " << gamma <<
195  ", gammahy = " << gammahy <<
196  ", gammacg = " << gammacg <<
197  ", temperature = " << temperature);
198 
199  pref1 = -gamma;
200  pref2 = sqrt(24.0 * temperature * gamma / timestep);
201 
202  pref1hy = -gammahy;
203  pref2hy = sqrt(24.0 * temperature * gammahy / timestep);
204 
205  pref1cg = -gammacg;
206  pref2cg = sqrt(24.0 * temperature * gammacg / timestep);
207 
208  }
