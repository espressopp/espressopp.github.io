00166     void LangevinThermostatHybrid::frictionThermo(Particle& p,real weight)
00167     {
00168       real massf = sqrt(p.mass());
00169 
00170       // get a random value for each vector component
00171 
00172       Real3D ranval((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
00173 
00174       if (weight<1.0 && weight>0.0) { //hybrid region
00175         p.force() += pref1hy * p.velocity() * p.mass() +
00176                    pref2hy * ranval * massf;
00177       } else if (weight==1.0) {
00178         p.force() += pref1 * p.velocity() * p.mass() +
00179                    pref2 * ranval * massf;
00180       } else {
00181         p.force() += pref1cg * p.velocity() * p.mass() +
00182                    pref2cg * ranval * massf;
00183       }
00184 
00185       LOG4ESPP_TRACE(theLogger, "new force of p = " << p.force());
00186     }

00190

00202       pref1hy = -gammahy;
