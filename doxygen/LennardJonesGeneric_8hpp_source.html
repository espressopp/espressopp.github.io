118  real _computeEnergySqrRaw(real distSqr) const {
119  real dist = sqrt(distSqr);
120  real sig_over_r = sigma / dist;
121  real ffA = pow(sig_over_r, a);
122  real ffB = pow(sig_over_r, b);
123  real energy = 4.0 * epsilon * (ffA - ffB);
124  return energy;
125  }
126 
127  bool _computeForceRaw(Real3D& force, const Real3D& dist, real distSqr) const {
128  real invdist = 1.0 / sqrt(distSqr);
129  real ffA = pow(invdist, a+2);
130  real ffB = pow(invdist, b+2);
131  real ffactor = 4.0 * epsilon * (ff1*ffA - ff2*ffB);
132  force = dist * ffactor;
133  return true;
134  }
