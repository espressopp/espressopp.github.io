112  template < typename _Potential > inline real
113  VSphereSelfInteractionTemplate < _Potential >::
114  computeEnergy() {
115  LOG4ESPP_INFO(theLogger, "compute energy of the VSphere Self potential");
116  System& system = getSystemRef();
117  CellList realCells = system.storage->getRealCells();
118  // loop over all particles of the real cells
119  real e = 0.0;
120  Real3D radius = 0.0;
121  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
122  radius[0] = cit->radius();
123  e += potential->_computeEnergy(radius);
124  }
125  real esum;
126  boost::mpi::all_reduce(*mpiWorld, e, esum, std::plus<real>());
127  return esum;
