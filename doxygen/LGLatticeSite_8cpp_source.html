00174       // IF ONE USES DEFAULT D3Q19 MODEL
00175 
00176       /* density on the site */
00177       real rhoLoc = getM_i(0);
00178 
00179       /* moments on the site */
00180       Real3D jLoc(getM_i(1), getM_i(2), getM_i(3));
00181       jLoc *= (aLocal / tauLocal);
00182 
00183       /* if we have external forces then modify the eq.fluxes */
00184       if (_extForceFlag == 1) {
00185         jLoc[0] += 0.5*getExtForceLoc().getItem(0);
00186         jLoc[1] += 0.5*getExtForceLoc().getItem(1);
00187         jLoc[2] += 0.5*getExtForceLoc().getItem(2);
00188       }
00189 
00190       /* eq. stress modes */
00191       setMeq_i (4, jLoc.sqr() / rhoLoc);
00192       setMeq_i (5, (jLoc[0]*jLoc[0] - jLoc[1]*jLoc[1]) / rhoLoc);
00193       setMeq_i (6, (3*jLoc[0]*jLoc[0] - jLoc.sqr()) / rhoLoc);
00194       setMeq_i (7, jLoc[0]*jLoc[1] / rhoLoc);
00195       setMeq_i (8, jLoc[0]*jLoc[2] / rhoLoc);
00196       setMeq_i (9, jLoc[1]*jLoc[2] / rhoLoc);
00197 
00198       /* kinetic (ghost) modes */
00199       setMeq_i (10, 0.); setMeq_i (11, 0.);
00200       setMeq_i (12, 0.); setMeq_i (13, 0.);
00201       setMeq_i (14, 0.); setMeq_i (15, 0.);
00202       setMeq_i (16, 0.); setMeq_i (17, 0.);

00252     void LGSite::applyForces(int _numVels) {
00253       Real3D _f;
00254       _f = getExtForceLoc();
00255 
00256       // set velocity _u
00257       Real3D _u (getM_i(1) + 0.5*_f[0], getM_i(2) + 0.5*_f[1], getM_i(3) + 0.5*_f[2]);
00258       _u /= getM_i(0);
00259 
00260       /* update momentum modes */
00261       addM_i(1, _f[0]);
00262       addM_i(2, _f[1]);
00263       addM_i(3, _f[2]);
00264 
00265       /* update stress modes */
00266       // See def. of _sigma (Eq.198) in B.DÃ¼nweg & A.J.C.Ladd in Adv.Poly.Sci. 221, 89-166 (2009)
00267       real _sigma[6];
00268       real _gamma_b, _gamma_s;
00269       real _scalp;
00270       _gamma_b = getGammaBLoc();
00271       _gamma_s = getGammaSLoc();
00272 
00273       _scalp = _u*_f;
00274       _sigma[0] = (1. + _gamma_s)*_u[0]*_f[0] + 1./3.*(_gamma_b - _gamma_s)*_scalp;
00275       _sigma[1] = (1. + _gamma_s)*_u[1]*_f[1] + 1./3.*(_gamma_b - _gamma_s)*_scalp;
00276       _sigma[2] = (1. + _gamma_s)*_u[2]*_f[2] + 1./3.*(_gamma_b - _gamma_s)*_scalp;
00277       _sigma[3] = 1./2.*(1. + _gamma_s)*(_u[0]*_f[1]+_u[1]*_f[0]);
00278       _sigma[4] = 1./2.*(1. + _gamma_s)*(_u[0]*_f[2]+_u[2]*_f[0]);
00279       _sigma[5] = 1./2.*(1. + _gamma_s)*(_u[1]*_f[2]+_u[2]*_f[1]);
00280 
00281       addM_i(4, _sigma[0]+_sigma[1]+_sigma[2]);
00282       addM_i(5, 2*_sigma[0]-_sigma[1]-_sigma[2]);
00283       addM_i(6, _sigma[1]-_sigma[2]);
00284       addM_i(7, _sigma[3]);
00285       addM_i(8, _sigma[4]);
00286       addM_i(9, _sigma[5]);
00287     }
