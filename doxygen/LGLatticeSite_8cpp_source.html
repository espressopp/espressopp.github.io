120  /* CALCULATION OF THE LOCAL MOMENTS */
121  void LGSite::calcLocalMoments() {
122  // IF ONE USES DEFAULT D3Q19 MODEL
123  real f0,
124  f1p2, f1m2, f3p4, f3m4, f5p6, f5m6, f7p8, f7m8, f9p10, f9m10,
125  f11p12, f11m12, f13p14, f13m14, f15p16, f15m16, f17p18, f17m18;
126  /* shorthand functions for "simplified" notation */
127  f0 = getF_i(0);
128  f1p2 = getF_i(1) + getF_i(2); f1m2 = getF_i(1) - getF_i(2);
129  f3p4 = getF_i(3) + getF_i(4); f3m4 = getF_i(3) - getF_i(4);
130  f5p6 = getF_i(5) + getF_i(6); f5m6 = getF_i(5) - getF_i(6);
131  f7p8 = getF_i(7) + getF_i(8); f7m8 = getF_i(7) - getF_i(8);
132  f9p10 = getF_i(9) + getF_i(10); f9m10 = getF_i(9) - getF_i(10);
133  f11p12 = getF_i(11) + getF_i(12); f11m12 = getF_i(11) - getF_i(12);
134  f13p14 = getF_i(13) + getF_i(14); f13m14 = getF_i(13) - getF_i(14);
135  f15p16 = getF_i(15) + getF_i(16); f15m16 = getF_i(15) - getF_i(16);
136  f17p18 = getF_i(17) + getF_i(18); f17m18 = getF_i(17) - getF_i(18);
137 
138  /* mass mode */
139  setM_i (0, f0 + f1p2 + f3p4 + f5p6 + f7p8 + f9p10 + f11p12 + f13p14 + f15p16 + f17p18);
140 
141  /* momentum modes */
142  setM_i (1, f1m2 + f7m8 + f9m10 + f11m12 + f13m14);
143  setM_i (2, f3m4 + f7m8 - f9m10 + f15m16 + f17m18);
144  setM_i (3, f5m6 + f11m12 - f13m14 + f15m16 - f17m18);
145 
146  /* stress modes */
147  setM_i (4, -f0 + f7p8 + f9p10 + f11p12 + f13p14 + f15p16 + f17p18);
148  setM_i (5, 2.*f1p2 - f3p4 - f5p6 + f7p8 + f9p10 + f11p12 + f13p14 - 2.* (f15p16 + f17p18));
149  setM_i (6, f3p4 - f5p6 + f7p8 + f9p10 - f11p12 - f13p14);
150  setM_i (7, f7p8 - f9p10);
151  setM_i (8, f11p12 - f13p14);
152  setM_i (9, f15p16 - f17p18);
153 
154  /* kinetic (ghost) modes */
155  setM_i (10, -2.* f1m2 + f7m8 + f9m10 + f11m12 + f13m14);
156  setM_i (11, -2.* f3m4 + f7m8 - f9m10 + f15m16 + f17m18);
157  setM_i (12, -2.* f5m6 + f11m12 - f13m14 + f15m16 - f17m18);
158  setM_i (13, f7m8 + f9m10 - f11m12 - f13m14);
159  setM_i (14, -f7m8 + f9m10 + f15m16 + f17m18);
160  setM_i (15, f11m12 - f13m14 - f15m16 + f17m18);
161  setM_i (16, f0 - 2.* (f1p2 + f3p4 + f5p6) + f7p8 + f9p10
162  + f11p12 + f13p14 + f15p16 + f17p18);
163  setM_i (17, -2.* f1p2 + f3p4 + f5p6 + f7p8 + f9p10 + f11p12
164  + f13p14 - 2.* (f15p16 + f17p18));
165  setM_i (18, -f3p4 + f5p6 + f7p8 + f9p10 - f11p12 - f13p14);

172  /* CALCULATION OF THE EQUILIBRIUM MOMENTS */
173  void LGSite::calcEqMoments(int _extForceFlag) {
174  // IF ONE USES DEFAULT D3Q19 MODEL
175 
176  /* density on the site */
177  real rhoLoc = getM_i(0);
178 
179  /* moments on the site */
180  Real3D jLoc(getM_i(1), getM_i(2), getM_i(3));
181  jLoc *= (aLocal / tauLocal);
182 
183  /* if we have external forces then modify the eq.fluxes */
184  if (_extForceFlag == 1) {
185  jLoc[0] += 0.5*getExtForceLoc().getItem(0);
186  jLoc[1] += 0.5*getExtForceLoc().getItem(1);
187  jLoc[2] += 0.5*getExtForceLoc().getItem(2);
188  }
189 
190  /* eq. stress modes */
191  setMeq_i (4, jLoc.sqr() / rhoLoc);
192  setMeq_i (5, (jLoc[0]*jLoc[0] - jLoc[1]*jLoc[1]) / rhoLoc);
193  setMeq_i (6, (3*jLoc[0]*jLoc[0] - jLoc.sqr()) / rhoLoc);
194  setMeq_i (7, jLoc[0]*jLoc[1] / rhoLoc);
195  setMeq_i (8, jLoc[0]*jLoc[2] / rhoLoc);
196  setMeq_i (9, jLoc[1]*jLoc[2] / rhoLoc);
197 
198  /* kinetic (ghost) modes */
199  setMeq_i (10, 0.); setMeq_i (11, 0.);
200  setMeq_i (12, 0.); setMeq_i (13, 0.);
201  setMeq_i (14, 0.); setMeq_i (15, 0.);
202  setMeq_i (16, 0.); setMeq_i (17, 0.);
203  setMeq_i (18, 0.);
204  }

206  /* RELAXATION OF THE MOMENTS TO THEIR EQUILIBRIUM VALUES */
207  void LGSite::relaxMoments (int _numVels) {
208  real pi_eq[6];
209 
210  pi_eq[0] = getMeq_i(4); pi_eq[1] = getMeq_i(5); pi_eq[2] = getMeq_i(6);
211  pi_eq[3] = getMeq_i(7); pi_eq[4] = getMeq_i(8); pi_eq[5] = getMeq_i(9);
212 
213  real _gamma_b, _gamma_s, _gamma_odd, _gamma_even;
214  _gamma_b = getGammaBLoc();
215  _gamma_s = getGammaSLoc();
216  _gamma_odd = getGammaOddLoc();
217  _gamma_even = getGammaEvenLoc();
218 
219  /* relax bulk mode */
220  setM_i (4, pi_eq[0] + _gamma_b * (getM_i(4) - pi_eq[0]));
221 
222  /* relax shear modes */
223  setM_i (5, pi_eq[1] + _gamma_s * (getM_i(5) - pi_eq[1]));
224  setM_i (6, pi_eq[2] + _gamma_s * (getM_i(6) - pi_eq[2]));
225  setM_i (7, pi_eq[3] + _gamma_s * (getM_i(7) - pi_eq[3]));
226  setM_i (8, pi_eq[4] + _gamma_s * (getM_i(8) - pi_eq[4]));
227  setM_i (9, pi_eq[5] + _gamma_s * (getM_i(9) - pi_eq[5]));
228 
229  /* relax odd modes */
230  scaleM_i (10, _gamma_odd); scaleM_i (11, _gamma_odd); scaleM_i (12, _gamma_odd);
231  scaleM_i (13, _gamma_odd); scaleM_i (14, _gamma_odd); scaleM_i (15, _gamma_odd);
232 
233  /* relax even modes */
234  scaleM_i (16, _gamma_even); scaleM_i (17, _gamma_even); scaleM_i (18, _gamma_even);
235  }

252  void LGSite::applyForces(int _numVels) {
253  Real3D _f;
254  _f = getExtForceLoc();
255 
256  // set velocity _u
257  Real3D _u (getM_i(1) + 0.5*_f[0], getM_i(2) + 0.5*_f[1], getM_i(3) + 0.5*_f[2]);
258  _u /= getM_i(0);
259 
260  /* update momentum modes */
261  addM_i(1, _f[0]);
262  addM_i(2, _f[1]);
263  addM_i(3, _f[2]);
264 
265  /* update stress modes */
266  // See def. of _sigma (Eq.198) in B.Dünweg & A.J.C.Ladd in Adv.Poly.Sci. 221, 89-166 (2009)
267  real _sigma[6];
268  real _gamma_b, _gamma_s;
269  real _scalp;
270  _gamma_b = getGammaBLoc();
271  _gamma_s = getGammaSLoc();
272 
273  _scalp = _u*_f;
274  _sigma[0] = (1. + _gamma_s)*_u[0]*_f[0] + 1./3.*(_gamma_b - _gamma_s)*_scalp;
275  _sigma[1] = (1. + _gamma_s)*_u[1]*_f[1] + 1./3.*(_gamma_b - _gamma_s)*_scalp;
276  _sigma[2] = (1. + _gamma_s)*_u[2]*_f[2] + 1./3.*(_gamma_b - _gamma_s)*_scalp;
277  _sigma[3] = 1./2.*(1. + _gamma_s)*(_u[0]*_f[1]+_u[1]*_f[0]);
278  _sigma[4] = 1./2.*(1. + _gamma_s)*(_u[0]*_f[2]+_u[2]*_f[0]);
279  _sigma[5] = 1./2.*(1. + _gamma_s)*(_u[1]*_f[2]+_u[2]*_f[1]);
280 
281  addM_i(4, _sigma[0]+_sigma[1]+_sigma[2]);
282  addM_i(5, 2*_sigma[0]-_sigma[1]-_sigma[2]);
283  addM_i(6, _sigma[1]-_sigma[2]);
284  addM_i(7, _sigma[3]);
285  addM_i(8, _sigma[4]);
286  addM_i(9, _sigma[5]);
287  }
