105 void LangevinThermostatOnGroup::frictionThermo(Particle &p) {
106  real massf = sqrt(p.mass());
107 
108  // get a random value for each vector component
109 
110  Real3D ranval((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
111 
112  p.force() += pref1 * p.velocity() * p.mass() +
113  pref2 * ranval * massf;
114 
115  LOG4ESPP_TRACE(theLogger, "new force of p = " << p.force());
116 }

118 void LangevinThermostatOnGroup::initialize() { // calculate the prefactors
119  real timestep = integrator->getTimeStep();
120 
121  pref1 = -gamma;
122  pref2 = sqrt(24.0 * temperature * gamma / timestep);
123 
124  LOG4ESPP_INFO(theLogger, "init, timestep = " << timestep <<
125  ", gamma = " << gamma <<
126  ", temperature = " << temperature << " pref2=" << pref2);
127 }
