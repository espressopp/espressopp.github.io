51  void initialize() {
52  real krc = kappa*rc;
53  real tmp1 = (epsilon1 - 4.0*epsilon2) *
54  (1.0 + krc) - 2.0*epsilon2 * krc*krc;
55  real tmp2 = (epsilon1 + 2.0*epsilon2) *
56  (1.0 + krc) + epsilon2 * krc*krc;
57  rc3 = pow(rc,3);
58  rc2 = pow(rc,2);
59  B1 = tmp1/tmp2;
60  B1 = (1.0+B1) / rc3;
61  B1_half = B1/2.0;
62  //B0 = (epsilon1 - 2.0*epsilon2 * (1.0+krc))/(epsilon2*(1.0+krc));
63  crf = 3*epsilon2/(rc*(2*epsilon2+1.0));
64  //B0 = (1.0+B0) / rc;
65  prefactor/=epsilon1;
66  }

132  real _computeEnergy(const Particle& p1, const Particle& p2) const {
133  Real3D dist = p1.position() - p2.position();
134  real distSqr = dist.sqr();
135  //std::cout << "qq" << distSqr << " " << p1.id() << " " << p2.id() << "\n";
136  if (distSqr>rc2) return 0.0;
137  real qq = p1.q()*p2.q();
138  /* Note: this implementation counts minus integral of the force as energy
139  which is not the same as the full electrostatic energy
140  See the original paper Tironi et al, J. Chem. Phys. 102 , 5451 (1995)
141  for details*/
142 
143 
144 
145  real energy = prefactor*qq * (1.0 / sqrt(distSqr) - B1_half*distSqr -crf);
146  return energy;
