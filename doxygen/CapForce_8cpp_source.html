127  void CapForce::applyForceCappingToGroup() {
128  LOG4ESPP_DEBUG(theLogger, "applying force capping to particle group of size " << particleGroup->size());
129 
130  if (absCapping) {
131  real capfsq = absCapForce * absCapForce;
132  for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
133  LOG4ESPP_DEBUG(theLogger, "applying scalar force capping to particle " << it->getId());
134  real fsq = it->force().sqr();
135  if (fsq > capfsq) {
136  real scaling = sqrt(capfsq / fsq);
137  Real3D& f=it->force();
138  for (int dir=0; dir<3; dir++) {
139  f[dir] *= scaling;
140  }
141  }
142  }
143  } else {
144  for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
145  LOG4ESPP_DEBUG(theLogger, "applying vector force capping to particle " << it->getId());
146  Real3D& f=it->force();
147  for (int dir=0; dir<3; dir++) {
148  if (f[dir]>0 && f[dir]>capForce[dir]) {
149  f[dir] = capForce[dir];
150  }
151  else if (f[dir]<0 && f[dir] < -capForce[dir]) {
152  f[dir] = -capForce[dir];
153  }
154  }
155  }
156  }
157  }

159  void CapForce::applyForceCappingToAll() {
160  LOG4ESPP_DEBUG(theLogger, "applying force capping to all particles");
161 
162  System& system = getSystemRef();
163  CellList realCells = system.storage->getRealCells();
164  if (absCapping) {
165  real capfsq = absCapForce * absCapForce;
166  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
167  real fsq = cit->force().sqr();
168  if (fsq > capfsq) {
169  real scaling = sqrt(capfsq / fsq);
170  Real3D& f=cit->force();
171  for (int dir=0; dir<3; dir++) f[dir] *= scaling;
172  //std::cout << "Force Capping applied on particle " << cit->getId() << "\n";
173  }
174  }
175  } else {
176  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
177  Real3D& f=cit->force();
178  for (int dir=0; dir<3; dir++) {
179  if (f[dir]>0 && f[dir]>capForce[dir]) {
180  f[dir] = capForce[dir];
181  }
182  else if (f[dir]<0 && f[dir] < -capForce[dir]) {
183  f[dir] = -capForce[dir];
184  }
185  }
186  }
187  }
188 
189  if (adress && absCapping){
190  real capfsq2 = absCapForce * absCapForce;
191  ParticleList& adrATparticles = system.storage->getAdrATParticles();
192  for (std::vector<Particle>::iterator it = adrATparticles.begin();
193  it != adrATparticles.end(); it++) {
194  real fsq2 = it->force().sqr();
