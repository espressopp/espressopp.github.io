00127      void CapForce::applyForceCappingToGroup() {
00128        LOG4ESPP_DEBUG(theLogger, "applying force capping to particle group of size " << particleGroup->size());
00129 
00130        if (absCapping) {
00131          real capfsq = absCapForce * absCapForce;
00132          for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
00133            LOG4ESPP_DEBUG(theLogger, "applying scalar force capping to particle " << it->getId());
00134            real fsq = it->force().sqr();
00135            if (fsq > capfsq) {
00136              real scaling = sqrt(capfsq / fsq);
00137              Real3D& f=it->force();
00138              for (int dir=0; dir<3; dir++) {
00139                 f[dir] *= scaling;
00140              }
00141            }
00142          }
00143        } else {
00144          for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
00145            LOG4ESPP_DEBUG(theLogger, "applying vector force capping to particle " << it->getId());
00146            Real3D& f=it->force();
00147            for (int dir=0; dir<3; dir++) {
00148              if (f[dir]>0 && f[dir]>capForce[dir]) {
00149                f[dir] = capForce[dir];
00150              }
00151              else if (f[dir]<0 && f[dir] < -capForce[dir]) {
00152                f[dir] = -capForce[dir];
00153              }
00154            }
00155          }
00156        }
