127  void CapForce::applyForceCappingToGroup() {
128  LOG4ESPP_DEBUG(theLogger, "applying force capping to particle group of size " << particleGroup->size());
129 
130  if (absCapping) {
131  real capfsq = absCapForce * absCapForce;
132  for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
133  LOG4ESPP_DEBUG(theLogger, "applying scalar force capping to particle " << it->getId());
134  real fsq = it->force().sqr();
135  if (fsq > capfsq) {
136  real scaling = sqrt(capfsq / fsq);
137  Real3D& f=it->force();
138  for (int dir=0; dir<3; dir++) {
139  f[dir] *= scaling;
140  }
141  }
142  }
143  } else {
144  for (ParticleGroup::iterator it=particleGroup->begin(); it != particleGroup->end(); it++ ) {
145  LOG4ESPP_DEBUG(theLogger, "applying vector force capping to particle " << it->getId());
146  Real3D& f=it->force();
147  for (int dir=0; dir<3; dir++) {
148  if (f[dir]>0 && f[dir]>capForce[dir]) {
149  f[dir] = capForce[dir];
150  }
151  else if (f[dir]<0 && f[dir] < -capForce[dir]) {
152  f[dir] = -capForce[dir];
153  }
154  }
155  }
156  }
157  }
