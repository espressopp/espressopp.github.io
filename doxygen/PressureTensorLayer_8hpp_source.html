53  /*
54  * calculate a pressure for z0 layer
55  * particles in [z-dz, z+dz] slab will be taken into account
56  */
57  Tensor computeRaw() {
58  System& system = getSystemRef();
59  Real3D Li = system.bc->getBoxL();
60 
61  // determine the local volume size for kinetic part
62  real A = Li[0] * Li[1];
63  real V = A * (2 * dz);
64 
65  // compute the kinetic contribution (2/3 \sum 1/2mv^2)
66  Tensor vvlocal(0.0);
67  CellList realCells = system.storage->getRealCells();
68 
69  real zmin = z0 - dz;
70  real zmax = z0 + dz;
71 
72  // boundary condition for orthorhombic box
73  real zminBC = zmin;
74  real zmaxBC = zmax;
75  bool condition2 = false;
76  if(zminBC<0.0){
77  zminBC += Li[2];
78  condition2 = true;
79  }
80  else if(zmaxBC>=Li[2]){
81  zmaxBC -= Li[2];
82  condition2 = true;
83  }
84 
85  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
86  Real3D pos = cit->position();
87 
88  bool condition;
89  if(condition2){
90  condition = (pos[2]>=zminBC || pos[2]<zmaxBC);
91  }
92  else{
93  condition = (pos[2]>=zminBC && pos[2]<zmaxBC);
94  }
95 
96  if( condition ){
97  real mass = cit->mass();
98  Real3D& vel = cit->velocity();
99  vvlocal += mass * Tensor(vel, vel);
100  }
101  }
102  Tensor vv(0.0);
103  boost::mpi::all_reduce(*mpiWorld, (double*)&vvlocal,6, (double*)&vv, std::plus<double>());
104 
105  vv = vv / V;
106 
107  // compute the short-range nonbonded contribution
108  Tensor w(0.0);
109  const InteractionList& srIL = system.shortRangeInteractions;
110  for (size_t j = 0; j < srIL.size(); j++) {
111  srIL[j]->computeVirialTensor(w, z0);
112  }
113 
114  w = w / A;
115 
116  return ( vv + w );
117  }

150  void updateAverage(Tensor res) {
151  // compare e. g. Knuth TAOCP vol 2, 3rd edition, page 232
152  if (nMeasurements > 0) {
153  if (nMeasurements == 1) {
154  newAverage = res;
155  lastAverage = newAverage;
156  } else {
157  newAverage = lastAverage + (res - lastAverage) / nMeasurements;
158  for (int i=0; i<6; i++) {
159  newVariance[i] = lastVariance[i] + (res[i] - lastAverage[i]) * (res[i] - newAverage[i]);
160  }
161  lastAverage = newAverage;
162  lastVariance = newVariance;
163  }
164  }
165  return;
166  }
