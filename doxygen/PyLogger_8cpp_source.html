210 void PyLogger::setPythonLevel(int pyLevel)
211 
212 {
213  if (pyLevel == pyNOTSET) {
214  setFlag = false;
215  } else if (pyLevel == pyTRACE) {
216  setLevel(TRACE, true);
217  } else if (pyLevel == pyDEBUG) {
218  setLevel(DEBUG, true);
219  } else if (pyLevel == pyINFO) {
220  setLevel(INFO, true);
221  } else if (pyLevel == pyWARN) {
222  setLevel(WARN, true);
223  } else if (pyLevel == pyERROR) {
224  setLevel(ERROR, true);
225  } else if (pyLevel == pyFATAL) {
226  setLevel(FATAL, true);
227  } else {
228  printf("ERROR: setPythonLevel for %s: %d is unknown log level of Python\n",
229  getFullName().c_str(), pyLevel);
