63  bool FixedTupleList::
64  addTuple(boost::python::list& tuple) {
65  bool returnVal = true;
66  System& system = storage->getSystemRef();
67  esutil::Error err(system.comm);
68 
69  Particle* vp, *at;
70  longint pidK; // the pid used as key
71  std::vector<Particle*> tmp; // temporary vector
72  std::vector<longint> pids; //used to extract from tuple;
73  std::vector<longint> pidstmp; // temporary vector
74 
75 
76  for (int i = 0; i < boost::python::len(tuple); ++i) {
77  pids.push_back(boost::python::extract<int>(tuple[i]));
78  }
79 
80  tuple::iterator it = pids.begin();
81  vp = storage->lookupRealParticle(*it);
82  if (!vp) {// Particle does not exist here, return false
83  returnVal = false;
84  } else {
85  pidK = *it; // first pid is key
86  for (++it; it != pids.end(); ++it) {
87  at = storage->lookupLocalParticle(*it);
88  if (!at) { // Particle does not exist here, return false
89  std::stringstream msg;
90  msg << "ERROR: particle " << *it << " not found \n";
91  err.setException(msg.str());
92  returnVal = false;
93  break;
94  }
95  tmp.push_back(at);
96  // std::cout << " add: " << *it << "\n";
97  pidstmp.push_back(*it); // pidK is not in this vector
98  }
99  }
100  err.checkException();
101 
102  if (returnVal) {
103  this->add(vp, tmp); // add to TupleList
104  globalTuples.insert(make_pair(pidK, pidstmp));
105  }
106  LOG4ESPP_INFO(theLogger, "added fixed tuple to global tuples");
107 
108  tmp.clear();
109  pids.clear();
110  pidstmp.clear();
111 
112 
113  return returnVal;
114  }

132  void FixedTupleList::
133  beforeSendParticles(ParticleList& pl, OutBuffer& buf) {
134  std::vector<longint> toSend;
135  // loop over the particle list
136  for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
137  longint pidK = pit->id();
138  LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pidK << ", find tuples");
139 
140  // find particle that involves this particle id
141  GlobalTuples::const_iterator it = globalTuples.find(pidK);
142  if (it != globalTuples.end()) {
143 
144  // first write the pid of the first particle
145  buf.write(pidK);
146 
147  // write the size of the vector
148  int s = it->second.size();
149  buf.write(s);
150  toSend.reserve(s);
151 
152  // iterate through vector and add pids
153  for (tuple::const_iterator it2 = it->second.begin();
154  it2 != it->second.end(); ++it2) {
155  Particle* tp = storage->lookupLocalParticle(*it2);
156  buf.write(*tp);
157  toSend.push_back(*it2);
158  }
159  // delete this pid from the global list
160  globalTuples.erase(pidK);
161 
162  }
163  }
164  buf.write(toSend);
165  }

167  void FixedTupleList::

183  pids.push_back(p.id());
184  }
185  it = globalTuples.insert(it, std::make_pair(pidK, pids));
186  pids.clear();

194  System& system = storage->getSystemRef();
