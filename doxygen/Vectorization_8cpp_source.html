98  void Vectorization::befCalcForces()
99  {
100  if(mode==ESPP_VEC_AOS)
101  {
102  auto& f = particleArray.force;
103  real* el = &(f[0].x);
104  size_t end = 4*f.size();
105 
106  #pragma vector always
107  #pragma vector aligned
108  #pragma ivdep
109  for(size_t i=0; i<end; i++) el[i] = 0.0;
110  }
111  else
112  {
113  auto& f_x = particleArray.f_x;
114  auto& f_y = particleArray.f_y;
115  auto& f_z = particleArray.f_z;
116  std::fill(f_x.begin(),f_x.end(),0.0);
117  std::fill(f_y.begin(),f_y.end(),0.0);
118  std::fill(f_z.begin(),f_z.end(),0.0);
119  }
120 
121  // overwrite particleArray positon data
122  particleArray.updateFromPositionOnly(getSystem()->storage->getLocalCells());
123  }
