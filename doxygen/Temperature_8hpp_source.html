53  real compute_real() const {
54  int myN, systemN;
55  real sumT = 0.0;
56  real v2sum = 0.0;
57  System& system = getSystemRef();
58  int count = 0;
59 
60  if (system.storage->getFixedTuples()){ // AdResS - hence, need to distinguish between CG and AT particles.
61  shared_ptr<FixedTupleListAdress> fixedtupleList=system.storage->getFixedTuples();
62  CellList realCells = system.storage->getRealCells();
63  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) { // Iterate over all (CG) particles.
64  Particle &vp = *cit;
65  FixedTupleListAdress::iterator it2;
66  it2 = fixedtupleList->find(&vp);
67 
68  if (it2 != fixedtupleList->end()) { // Are there atomistic particles for given CG particle? If yes, use those for calculation.
69  std::vector<Particle*> atList;
70  atList = it2->second;
71  for (std::vector<Particle*>::iterator it3 = atList.begin();
72  it3 != atList.end(); ++it3) {
73  Particle &at = **it3;
74  if (!has_types || valid_type_ids.count(at.type())) {
75  Real3D vel = at.velocity();
76  v2sum += at.mass() * (vel * vel);
77  count += 1;
78  }
79  }
80  } else { // If not, use CG particle itself for calculation.
81  if (!has_types || valid_type_ids.count(cit->type())) {
82  Real3D vel = cit->velocity();
83  v2sum += cit->mass() * (vel * vel);
84  count += 1;
85  }
86  }
87 
88  }
89 
90  myN = count;
91  } else { // No AdResS - just iterate over all particles
92  CellList realCells = system.storage->getRealCells();
93  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
94  Real3D vel = cit->velocity();
95  if (!has_types || valid_type_ids.count(cit->type())) {
96  v2sum += cit->mass() * (vel * vel);
97  count += 1;
98  }
99  }
100 
101  myN = count;
102  }
103 
104  mpi::all_reduce(*getSystem()->comm, v2sum, sumT, std::plus<real>());
105  mpi::all_reduce(*getSystem()->comm, myN, systemN, std::plus<int>());
106 
107  eKin_ = 0.5*sumT;
108  return sumT / (3.0 * systemN);
109  }
