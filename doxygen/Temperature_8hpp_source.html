00045       int myN, systemN;

00051           int count = 0;
00052           shared_ptr<FixedTupleListAdress> fixedtupleList=system.storage->getFixedTuples();
00053           CellList realCells = system.storage->getRealCells();
00054           for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {  // Iterate over all (CG) particles.              
00055             Particle &vp = *cit;
00056             FixedTupleListAdress::iterator it2;
00057             it2 = fixedtupleList->find(&vp);
00058             
00059             if (it2 != fixedtupleList->end()) {  // Are there atomistic particles for given CG particle? If yes, use those for calculation.
00060                   std::vector<Particle*> atList;
00061                   atList = it2->second;
00062                   for (std::vector<Particle*>::iterator it3 = atList.begin();
00063                                        it3 != atList.end(); ++it3) {
00064                       Particle &at = **it3;
00065                       Real3D vel = at.velocity();
00066                       v2sum += at.mass() * (vel * vel);
00067                       count += 1;
00068                   }  
00069             }
00070             
00071             else{   // If not, use CG particle itself for calculation.
00072                   Real3D vel = cit->velocity();
00073                   v2sum += cit->mass() * (vel * vel);
00074                   count += 1;
00075             }
00076             
00077           }
00078           
00079           myN = count;
00080           //std::cout << "COUNT " << count << std::endl;
00081       }
00082       
00083       else{  // No AdResS - just iterate over all particles          
00084           CellList realCells = system.storage->getRealCells();
00085           for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00086             Real3D vel = cit->velocity();

00091       }
00092
