79  void _computeForceRaw(Real3D& force1,
80  Real3D& force2,
81  Real3D& force3,
82  Real3D& force4,
83  const Real3D& dist21,
84  const Real3D& dist32,
85  const Real3D& dist43) const {
86  if (table) {
87  // compute phi
88  real dist21_sqr = dist21 * dist21;
89  real dist32_sqr = dist32 * dist32;
90  real dist43_sqr = dist43 * dist43;
91  real dist21_magn = sqrt(dist21_sqr);
92  real dist32_magn = sqrt(dist32_sqr);
93  real dist43_magn = sqrt(dist43_sqr);
94 
95  // cos0
96  real sb1 = 1.0 / dist21_sqr;
97  real sb2 = 1.0 / dist32_sqr;
98  real sb3 = 1.0 / dist43_sqr;
99  real rb1 = sqrt(sb1);
100  real rb3 = sqrt(sb3);
101  real c0 = dist21 * dist43 * rb1 * rb3;
102 
103 
104  // 1st and 2nd angle
105  real ctmp = dist21 * dist32;
106  real r12c1 = 1.0 / (dist21_magn * dist32_magn);
107  real c1mag = ctmp * r12c1;
108 
109  ctmp = (-1.0 * dist32) * dist43;
110  real r12c2 = 1.0 / (dist32_magn * dist43_magn);
111  real c2mag = ctmp * r12c2;
112 
113 
114  //cos and sin of 2 angles and final cos
115  real sin2 = 1.0 - c1mag * c1mag;
116  if (sin2 < 0) sin2 = 0.0;
117  real sc1 = sqrt(sin2);
118  sc1 = 1.0 / sc1;
119 
120  sin2 = 1.0 - c2mag * c2mag;
121  if (sin2 < 0) sin2 = 0.0;
122  real sc2 = sqrt(sin2);
123  sc2 = 1.0 / sc2;
124 
125  real s1 = sc1 * sc1;
126  real s2 = sc2 * sc2;
127  real s12 = sc1 * sc2;
128  real c = (c0 + c1mag * c2mag) * s12;
129 
130  Real3D cc = dist21.cross(dist32);
131  real cmag = sqrt(cc * cc);
132  real dx = cc * dist43 / cmag / dist43_magn;
133 
134  if (c > 1.0) c = 1.0;
135  else if (c < -1.0) c = -1.0;
136 
137  // phi
138  real phi = acos(c);
139  if (dx < 0.0) phi *= -1.0;
140 
141  //phi = 1.0; //testing
142 
143  // read table
144  real a = table->getForce(phi);
145 
146  c = c * a;
147  s12 = s12 * a;
148 
149  real a11 = c * sb1 * s1;
150  real a22 = -sb2 * (2.0 * c0 * s12 - c * (s1 + s2));
151  real a33 = c * sb3 * s2;
152  real a12 = -r12c1 * (c1mag * c * s1 + c2mag * s12);
153  real a13 = -rb1 * rb3 * s12;
154  real a23 = r12c2 * (c2mag * c * s2 + c1mag * s12);
155 
156  Real3D sf2 = a12 * dist21 + a22 * dist32 + a23 * dist43;
157 
158  force1 = a11 * dist21 + a12 * dist32 + a13 * dist43;
159  force2 = (-1.0 * sf2) - force1;
160  force4 = a13 * dist21 + a23 * dist32 + a33 * dist43;
161  force3 = sf2 - force4;
162  }
163  else {
164  throw std::runtime_error("Tabulated dihedral potential table not available.");
165  }
166 
167  }
