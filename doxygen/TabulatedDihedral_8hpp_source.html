90  Real3D prod(Real3D a, Real3D b) const {
91  Real3D res(0.0, 0.0, 0.0);
92  for(int i=0; i<3; i++){
93  for(int j=0; j<3; j++){
94  res[i]+=(1.0-d(i, j))*a[j]*b[j];

100  void _computeForceRaw(Real3D& force1,
101  Real3D& force2,
102  Real3D& force3,
103  Real3D& force4,
104  const Real3D& r21,
105  const Real3D& r32,
106  const Real3D& r43) const {
107  if (table) {
108  Real3D retF[4];
109 
110  Real3D rijjk = r21.cross(r32); // [r21 x r32]
111  Real3D rjkkn = r32.cross(r43); // [r32 x r43]
112 
113  real rijjk_sqr = rijjk.sqr();
114  real rjkkn_sqr = rjkkn.sqr();
115 
116  real rijjk_abs = sqrt(rijjk_sqr);
117  real rjkkn_abs = sqrt(rjkkn_sqr);
118 
119  real inv_rijjk = 1.0 / rijjk_abs;
120  real inv_rjkkn = 1.0 / rjkkn_abs;
121 
122  // cosine between planes
123  real cos_phi = (rijjk * rjkkn) * (inv_rijjk * inv_rjkkn);
124  real _phi = acos(cos_phi);
125  if (cos_phi > 1.0) {
126  cos_phi = 1.0;
127  _phi = 1e-10; //not 0.0, because 1.0/sin(_phi) would cause a singularity
128  } else if (cos_phi < -1.0) {
129  cos_phi = -1.0;
130  _phi = M_PI-1e-10;
131  }
132 
133  //get sign of phi
134  //positive if (rij x rjk) x (rjk x rkn) is in the same direction as rjk, negative otherwise (see DLPOLY manual)
135  Real3D rcross = rijjk.cross(rjkkn); //(rij x rjk) x (rjk x rkn)
136  real signcheck = rcross * r32;
137  if (signcheck < 0.0) _phi *= -1.0;
138 
139  // read table
140  real a = table->getForce(_phi);
141 
142  real coef1 = -(1.0/sin(_phi)) * a;
143 
144  real A1 = inv_rijjk * inv_rjkkn;
145  real A2 = inv_rijjk * inv_rijjk;
146  real A3 = inv_rjkkn * inv_rjkkn;
147 
148  Real3D p3232 ( prod(r32,r32) );
149  Real3D p3243 ( prod(r32,r43) );
150  Real3D p2132 ( prod(r21,r32) );
151  Real3D p2143 ( prod(r21,r43) );
152  Real3D p2121 ( prod(r21,r21) );
153  Real3D p4343 ( prod(r43,r43) );
154 
155  // we have 4 particles 1,2,3,4
156  for(int l=0; l<4; l++){
157  Real3D B1, B2, B3;
158 
159  for(int i=0; i<3; i++){
160  B1[i]= r21[i] * ( p3232[i] * (d(l,2)-d(l,3)) + p3243[i] * (d(l,2)-d(l,1)) ) +
161  r32[i] * ( p2132[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,0)) ) +
162  r43[i] * ( p2132[i] * (d(l,2)-d(l,1)) + p3232[i] * (d(l,0)-d(l,1)) ) +
163  2.0 * r32[i] * p2143[i] * (d(l,1)-d(l,2));
164 
165  B2[i]= 2.0 * r21[i] * ( p3232[i] * (d(l,1)-d(l,0)) + p2132[i] * (d(l,1)-d(l,2)) ) +
166  2.0 * r32[i] * ( p2121[i] * (d(l,2)-d(l,1)) + p2132[i] * (d(l,0)-d(l,1)));
167 
168  B3[i]= 2.0 * r43[i] * ( p3232[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,2)) ) +
169  2.0 * r32[i] * ( p4343[i] * (d(l,2)-d(l,1)) + p3243[i] * (d(l,2)-d(l,3)));
170  }
171 
172  retF[l] = coef1 * ( A1*B1 - 0.5*cos_phi*(A2*B2 + A3*B3) );
173  }
174 
175  force1 = retF[0];
176  force2 = retF[1];
177  force3 = retF[2];
178  force4 = retF[3];
179 
180  }
181  else {
182  throw std::runtime_error("Tabulated dihedral potential table not available.");
183  }
184 
185  }
