00134     void DPDThermostat::frictionThermoDPD(Particle& p1, Particle& p2) {
00135       Real3D r = p1.position() - p2.position();
00136       real dist2 = r.sqr();
00137       if(dist2 < current_cutoff_sqr && gamma > 0.0){
00138         real dist = sqrt(dist2);
00139       
00140         real omega = 1-dist/current_cutoff;
00141         real omega2 = omega*omega;
00142 
00143         // standard DPD part
00144         real veldiff = (p1.velocity() - p2.velocity()) * r;
00145         real friction = pref1 * omega2 * veldiff;
00146         real noise = pref2 * omega * ((*rng)() - 0.5);
00147 
00148         Real3D f = (noise - friction) * r;
00149         p1.force() += f;
00150         p2.force() -= f;
00151       }
00152     }

00154     void DPDThermostat::frictionThermoTDPD(Particle& p1, Particle& p2) {
00155 
00156                 Real3D dist3D = p1.position() - p2.position();
00157                 real distsq = dist3D.sqr();
00158                 real omega2 = 1.0 / distsq;
00159                 real omega = sqrt(omega2);
00160 
00161                 // transverse DPD part
00162                 if (tgamma > 0.0) {
00163                   real distinv = omega;
00164 
00165                   Real3D noisevec((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
00166 
00167                   // damping, random force
00168                   Real3D f_damp(0.0, 0.0, 0.0), f_rand(0.0, 0.0, 0.0);
00169 
00170                   // proj matrix
00171                   Real3D *matrix = new Real3D[3];
00172                   matrix[0] = Real3D(distsq, 0.0, 0.0);
00173                   matrix[1] = Real3D(0.0, distsq, 0.0);
00174                   matrix[2] = Real3D(0.0, 0.0, distsq);
00175 
00176                   for (int i = 0; i < 3; i++) {
00177                           for (int j = 0; j < 3; j++) {
00178                                   matrix[i][j] -= dist3D[i] * dist3D[j];
00179                                   f_damp[i] += matrix[i][j] * (p1.velocity()[j] - p2.velocity()[j]);
00180                                   f_rand[i] += matrix[i][j] * noisevec[j];
00181                           }
00182                           f_damp[i] *= pref3 * omega2;
00183                           f_rand[i] *= pref4 * omega * distinv;
00184                   }
00185 
00186                   delete [] matrix;
00187 
00188                   Real3D f = f_rand - f_damp;
00189 
00190                   p1.force() += f;
00191                   p2.force() -= f;
00192                 }
00193         }
