134  void DPDThermostat::frictionThermoDPD(Particle& p1, Particle& p2) {
135  Real3D r = p1.position() - p2.position();
136  real dist2 = r.sqr();
137  if(dist2 < current_cutoff_sqr && gamma > 0.0){
138  real dist = sqrt(dist2);
139 
140  real omega = 1-dist/current_cutoff;
141  real omega2 = omega*omega;
142 
143  // standard DPD part
144  real veldiff = (p1.velocity() - p2.velocity()) * r;
145  real friction = pref1 * omega2 * veldiff;
146  real noise = pref2 * omega * ((*rng)() - 0.5);
147 
148  Real3D f = (noise - friction) * r;
149  p1.force() += f;
150  p2.force() -= f;
151  }
152  }

154  void DPDThermostat::frictionThermoTDPD(Particle& p1, Particle& p2) {
155 
156  Real3D dist3D = p1.position() - p2.position();
157  real distsq = dist3D.sqr();
158  real omega2 = 1.0 / distsq;
159  real omega = sqrt(omega2);
160 
161  // transverse DPD part
162  if (tgamma > 0.0) {
163  real distinv = omega;
164 
165  Real3D noisevec((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
166 
167  // damping, random force
168  Real3D f_damp(0.0, 0.0, 0.0), f_rand(0.0, 0.0, 0.0);
169 
170  // proj matrix
171  Real3D *matrix = new Real3D[3];
172  matrix[0] = Real3D(distsq, 0.0, 0.0);
173  matrix[1] = Real3D(0.0, distsq, 0.0);
174  matrix[2] = Real3D(0.0, 0.0, distsq);
175 
176  for (int i = 0; i < 3; i++) {
177  for (int j = 0; j < 3; j++) {
178  matrix[i][j] -= dist3D[i] * dist3D[j];
179  f_damp[i] += matrix[i][j] * (p1.velocity()[j] - p2.velocity()[j]);
180  f_rand[i] += matrix[i][j] * noisevec[j];
181  }
182  f_damp[i] *= pref3 * omega2;
183  f_rand[i] *= pref4 * omega * distinv;
184  }
185 
186  delete [] matrix;
187 
188  Real3D f = f_rand - f_damp;
189 
190  p1.force() += f;
191  p2.force() -= f;
192  }
193  }

195  void DPDThermostat::initialize() {
196  // calculate the prefactors
197  System& system = getSystemRef();
198  current_cutoff = verletList->getVerletCutoff() - system.getSkin();
199  current_cutoff_sqr = current_cutoff*current_cutoff;
200 
201  real timestep = integrator->getTimeStep();
202 
203  LOG4ESPP_INFO(theLogger, "init, timestep = " << timestep <<
204  ", gamma = " << gamma <<
205  ", tgamma = " << tgamma <<
206  ", temperature = " << temperature);
207 
208  pref1 = gamma;
209  pref2 = sqrt(24.0 * temperature * gamma/timestep);
210  pref3 = tgamma;
211  pref4 = sqrt(24.0 * temperature * tgamma);
212  }
