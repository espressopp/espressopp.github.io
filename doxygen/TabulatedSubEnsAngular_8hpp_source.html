160  bool _computeForceRaw(Real3D& force12, Real3D& force32,
161  const Real3D& dist12, const Real3D& dist32) const {
162  real dist12_sqr = dist12 * dist12;
163  real dist32_sqr = dist32 * dist32;
164  real dist1232 = sqrt(dist12_sqr) * sqrt(dist32_sqr);
165  real cos_theta = dist12 * dist32 / dist1232;
166  real theta = acos(cos_theta);
167 
168  real a = 0.;
169  for (int i=0; i<numInteractions; ++i)
170  a += weights[i] * tables[i]->getForce(theta);
171 
172  a*=1.0/(sqrt(1.0-cos_theta*cos_theta));
173 
174  real a11 = a * cos_theta / dist12_sqr;
175  real a12 = -a / dist1232;
176  real a22 = a * cos_theta / dist32_sqr;
177 
178  force12 = a11 * dist12 + a12 * dist32;
179  force32 = a22 * dist32 + a12 * dist12;
180  return true;
181  }
