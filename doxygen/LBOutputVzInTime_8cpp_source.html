00043       // test output in a console

00046 
00047       real _denLoc = 0.;
00048       real _jzLoc = 0.;
00049 
00050       FILE * vzInTimeFile;
00051 
00052       /* define lattice site to output its density and velocity into vzInTime file.
00053        * here we use a site situated at a quater of the simulation box length Nx,
00054        * while y=0 and z=0 (i = 0.25*Nx and indexes _j and _k are set to zero).
00055        */
00056       int _i = (int) _Ni.getItem(0) * 0.25;
00057       int _j = 0;
00058       int _k = 0;
00059 
00060       // creating a profile based on the current populations
00061       for (int l = 0; l < _numVels; l++) {
00062         _denLoc += latticeboltzmann->getLBFluid(Int3D(_i,_j,_k),l);
00063         _jzLoc += latticeboltzmann->getLBFluid(Int3D(_i,_j,_k),l) * latticeboltzmann->getCi(l).getItem(2);
00064       }
00065 
00066       vzInTimeFile = fopen ("vz_in_time.dat","a");
00067       fprintf (vzInTimeFile, "%9d %9.6f %9.6f  \n", _step, _denLoc, _jzLoc/_denLoc);
00068       fclose (vzInTimeFile);
00069     }
