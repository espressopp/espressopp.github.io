85  inline real InterpolationAkima::splineInterpolation(real r,
86  const real* p0, const real* p1, const real* p2, const real* p3) const {
87  int index;
88  index = static_cast<int>((r - inner) * invdelta);
89  if (index < 0) {
90  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
91  << inner << " - " << inner + (N - 1) * delta);
92  index = 0;
93  }
94  if (index >= N) {
95  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
96  << inner << " - " << inner + (N - 1) * delta);
97  index = N-1;
98  }
99 
100  real z = r - radius[index];
101  int zz2 = z*z;
102  return p0[index] +
103  p1[index] * z +
104  p2[index] * zz2 +
105  p3[index] * zz2 * z;
106  }

108  inline real InterpolationAkima::getSlope(real m1, real m2, real m3, real m4) {
109  if ((m1 == m2) && (m3 == m4)) {
110  return (m2 + m3) / 2.0;
111  } else {
112  return (fabs(m4 - m3) * m2 + fabs(m2 - m1) * m3) /
113  (fabs(m4 - m3) + fabs(m2 - m1));
114  }
115  }
