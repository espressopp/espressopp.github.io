170  template < typename _Potential > inline real
171  CellListAllPairsInteractionTemplate < _Potential >::
172  computeVirial() {
173  LOG4ESPP_INFO(theLogger, "computed virial for all pairs in the cell lists");
174 
175  real w = 0.0;
176  for (iterator::CellListAllPairsIterator it(storage->getRealCells());
177  it.isValid(); ++it) {
178  Particle &p1 = *it->first;
179  Particle &p2 = *it->second;
180  int type1 = p1.type();
181  int type2 = p2.type();
182  const Potential &potential = getPotential(type1, type2);
183 
184  Real3D force(0.0, 0.0, 0.0);

186  Real3D dist = p1.position() - p2.position();
187  w = w + dist * force;
188  }
189  }

192  real wsum;
193  boost::mpi::all_reduce(*mpiWorld, w, wsum, std::plus<real>());

256  template < typename _Potential > inline void
257  CellListAllPairsInteractionTemplate < _Potential >::
258  computeVirialTensor(Tensor *wij, int n) {
259  LOG4ESPP_INFO(theLogger, "computed virial tensor for all pairs in the cell lists");
260 
261  const bc::BC& bc = *storage->getSystemRef().bc; // boundary conditions
262  Real3D Li = bc.getBoxL();
263  Tensor *wlocal = new Tensor[n];
264  for(int i=0; i<n; i++) wlocal[i] = Tensor(0.0);
265  for (iterator::CellListAllPairsIterator it(storage->getRealCells());
266  it.isValid(); ++it) {
267  const Particle &p1 = *it->first;
268  const Particle &p2 = *it->second;
269  Real3D p1pos = p1.position();
270  Real3D p2pos = p2.position();
271 
272  int position1 = (int)( n * p1pos[2]/Li[2]);
273  int position2 = (int)( n * p1pos[2]/Li[2]);
274 
275  int maxpos = std::max(position1, position2);
276  int minpos = std::min(position1, position2);
277 
278  const Potential &potential = getPotential(p1.type(), p2.type());
279 
280  Real3D force(0.0, 0.0, 0.0);
281  Tensor ww;
282  if(potential._computeForce(force, p1, p2)) {
283  Real3D r21;
284  bc.getMinimumImageVectorBox(r21, p1pos, p2pos);
285  ww = Tensor(r21, force);
286  }
287 
288  int i = minpos + 1;
289  while(i<=maxpos){
290  wlocal[i] += ww;
291  i++;
292  }
293  }
294 
295  // reduce over all CPUs
296  Tensor *wsum = new Tensor[n];
297  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, n, (double*)&wsum, std::plus<double>());
298 
299  for(int j=0; j<n; j++){
300  wij[j] += wsum[j];
301  }
302 
303  delete [] wsum;
304  delete [] wlocal;
305  }
