120  // at this point we are ready to prepare the kvector[], it can be done just once at the begin
121  void preset(){
122  // TODO it could be parallelized too
123  Real3D Li = system -> bc -> getBoxL(); // getting the system size
124  Lx = Li[0];
125  Ly = Li[1];
126  Lz = Li[2];
127 
128  real skmax = kmax / min(Lx, min(Ly,Lz));
129  real skmaxsq = skmax * skmax ; // we choose the biggest cutoff
130 
131  rclx = M_2PI / Lx;
132  rcly = M_2PI / Ly;
133  rclz = M_2PI / Lz;
134 
135  force_prefac[0] = prefactor * (-2.0) * rclx;
136  force_prefac[1] = prefactor * (-2.0) * rcly;
137  force_prefac[2] = prefactor * (-2.0) * rclz;
138 
139  // precalculate factors
140  real invAlpha2 = 1.0 / (alpha * alpha);
141  real B = M_PI2 * invAlpha2; // PI^2 / alpha^2
142  real inv2alpha2 = 0.5 * invAlpha2; // 1.0 / (2*alpha^2)
143  real V = M_2PI * Lx * Ly * Lz;
144 
145  /* calculate the k-vector array */
146  int ksq, kx2, ky2, kz2;
147  real rksq, rkx2, rky2, rkz2;
148  real rLx2 = 1. / ( Lx * Lx );
149  real rLy2 = 1. / ( Ly * Ly );
150  real rLz2 = 1. / ( Lz * Lz );
151  real rk2PIx, rk2PIy, rk2PIz;
152  kVectorLength = 0;
153  // clear all vectors
154  kvector.clear();
155  kxfield.clear();
156  kyfield.clear();
157  kzfield.clear();
158 
159  kx_ind.clear();
160  ky_ind.clear();
161  kz_ind.clear();
162  virialPref.clear();
163  virialTensorPref.clear();
164  for(int kx = 0; kx <= kmax; kx++){
165  kx2 = kx * kx;
166  rkx2 = kx2 * rLx2;
167  rk2PIx = kx * rclx;
168  for(int ky = -kmax; ky <= kmax; ky++){
169  ky2 = ky * ky;
170  rky2 = ky2 * rLy2;
171  rk2PIy = ky * rcly;
172  for(int kz = -kmax; kz <= kmax; kz++) {
173  kz2 = kz * kz;
174  rkz2 = kz2 * rLz2;
175  rk2PIz = kz * rclz;
176 
177  ksq = kx2+ky2+kz2;
178  rksq = rkx2+rky2+rkz2;
179 
180  if( (rksq < skmaxsq) && (ksq !=0) ){
181  kvector.push_back( exp( -rksq * B)/( rksq * V) );
182 
183  kxfield.push_back( kx );
184  kyfield.push_back( ky );
185  kzfield.push_back( kz );
186 
187  kx_ind.push_back( kx );
188  ky_ind.push_back( kmax + ky );
189  kz_ind.push_back( kmax + kz );
190 
191  // the tensor should be: deltaKronecker(i,j) - 2*hi*hj / h^2 - hi*hj / (2*alfa^2)
192  Real3D h(rk2PIx, rk2PIy, rk2PIz);
193  real h2 = h * h;
194  Tensor hh(h, h); // it is tensor: hi*hj
195 
196  virialPref.push_back( 1 - h2 * inv2alpha2 );
197  virialTensorPref.push_back( I - 2 * hh / h2 - hh * inv2alpha2 );
198 
199  kVectorLength++;
200  }
201  }
202  }
203  }
204 
205  //cout <<"node: "<< system->comm->rank() << " kVectorLength: "<< kVectorLength<< " kmax: "<< skmax <<endl;
206  if(sum != NULL){
207  delete [] sum;
208  sum = NULL;
209  }
210  if(totsum != NULL){
211  delete [] totsum;
212  totsum = NULL;
213  }
214  sum = new dcomplex[kVectorLength];
215  totsum = new dcomplex[kVectorLength];

261  // compute force and energy
262  void exponentPrecalculation(CellList realcells){
263  /* Calculation of k space sums */
264  // -1, 0, 1
265  int j=0; // auxiliary variable, particle counter
266  for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
267  Particle &p = *it;
268 
269  eikx[ 0][j] = dcomplex(1.0, 0.0);
270  eiky[kmax+0][j] = dcomplex(1.0, 0.0);
271  eikz[kmax+0][j] = dcomplex(1.0, 0.0);
272 
273  eikx[ 1][j] = dcomplex( cos( rclx * p.position()[0] ), sin( rclx * p.position()[0] ) );
274  eiky[kmax+1][j] = dcomplex( cos( rcly * p.position()[1] ), sin( rcly * p.position()[1] ) );
275  eikz[kmax+1][j] = dcomplex( cos( rclz * p.position()[2] ), sin( rclz * p.position()[2] ) );
276 
277  eiky[kmax-1][j] = conj( eiky[kmax+1][j] );
278  eikz[kmax-1][j] = conj( eikz[kmax+1][j] );
279 
280  j++;
281  }
282 
283  // calculation of the rest terms
284  for (int k=2; k<=kmax; k++) {
285  j=0;
286  for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
287  Particle &p = *it;
288 
289  eikx[ k][j] = eikx[ k-1][j] * eikx[ 1][j];
290 
291  eiky[kmax+k][j] = eiky[kmax+k-1][j] * eiky[kmax+1][j];
292  eiky[kmax-k][j] = conj( eiky[kmax+k][j] );
293 
294  eikz[kmax+k][j] = eikz[kmax+k-1][j] * eikz[kmax+1][j];
295  eikz[kmax-k][j] = conj( eikz[kmax+k][j] );
296 
297  j++;
298  }
299  }
300 
301  real q; // particle charge, auxiliary
302  for (int k=0; k<kVectorLength; k++) {
303  sum[k] = dcomplex(0.0, 0.0);
304  totsum[k] = dcomplex(0.0, 0.0);
305  j=0;
306  for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
307  Particle &p = *it;
308  if( (q = p.q())!=0 ){
309  eik[k][j] = q * ( eikx[ kx_ind[k] ][j] * eiky[ ky_ind[k] ][j] * eikz[ kz_ind[k] ][j] );
310  sum[k] += eik[k][j];
311  }
312  j++;
313  }
314  }
315 
316  mpi::all_reduce( *system -> comm, sum, kVectorLength, totsum, plus<dcomplex>() );
317  }

319  real _computeEnergy(CellList realcells){
320  // exponent array
321  exponentPrecalculation(realcells);
322 
323  mpi::communicator communic = *system->comm;
324 
325  int n_nodes = communic.size();
326  int this_node = communic.rank();
327 
328  // TODO it could be a problem if n_nodes > kVectorLength
329  int numk = kVectorLength / n_nodes + 1;
330  int mink = this_node * numk;
331  int maxk = mink + numk;
332  if(maxk>kVectorLength) maxk = kVectorLength;
333  real fact;
334  real node_energy = 0;
335  for(int k=mink; k<maxk; k++) {
336  if (kxfield[k]==0)
337  fact=1.0;
338  else
339  fact=2.0;
340  node_energy += fact * kvector[k] * norm( totsum[k] );
341  }
342  //cout <<"node: "<< this_node << " node energy: "<< node_energy << " fact: "<< fact<< " kmax: "<< kVectorLength <<endl;
343 //exit(0);
344  real energy = 0;
345  mpi::all_reduce( communic, node_energy, energy, plus<real>() );
346 
347  /* self energy correction */
348  energy -= sum_q2 * alpha * M_1_SQRTPI;
349 
350  energy *= prefactor;
351 
352  return energy;
353  }

356  bool _computeForce(CellList realcells){
357  // exponent array
358  exponentPrecalculation(realcells);
359 
360  real fact; // factor due to the symmetry
361  for (int k=0; k<kVectorLength; k++) {
362  if ( kxfield[k] == 0)
363  fact=1.0;
364  else
365  fact=2.0;
366 
367  dcomplex tff = fact * kvector[k] * totsum[k]; // auxiliary complex factor
368  int j=0;
369  for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
370  Particle& p = *it;
371 
372  real tf = 0.0;
373  if( p.q()!=0 ){
374  tf= imag( tff * conj( eik[k][j] ) );
375 
376  p.force()[0] += force_prefac[0] * tf * kxfield[k];
377  p.force()[1] += force_prefac[1] * tf * kyfield[k];
378  p.force()[2] += force_prefac[2] * tf * kzfield[k];
379  }
380  j++;
381  }
382 
383  }
384 
385  return true;
