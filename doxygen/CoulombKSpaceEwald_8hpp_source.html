00120       // at this point we are ready to prepare the kvector[], it can be done just once at the begin
00121       void preset(){
00122         // TODO it could be parallelized too
00123         Real3D Li = system -> bc -> getBoxL(); // getting the system size
00124         Lx = Li[0];
00125         Ly = Li[1];
00126         Lz = Li[2];
00127        
00128         real skmax = kmax / min(Lx, min(Ly,Lz)); 
00129         real skmaxsq = skmax * skmax ; // we choose the biggest cutoff 
00130         
00131         rclx = M_2PI / Lx;
00132         rcly = M_2PI / Ly;
00133         rclz = M_2PI / Lz;
00134         
00135         force_prefac[0] = prefactor * (-2.0) * rclx;
00136         force_prefac[1] = prefactor * (-2.0) * rcly;
00137         force_prefac[2] = prefactor * (-2.0) * rclz;
00138         
00139         // precalculate factors
00140         real invAlpha2 = 1.0 / (alpha * alpha);
00141         real B = M_PI2 * invAlpha2; // PI^2 / alpha^2
00142         real inv2alpha2 = 0.5 * invAlpha2; // 1.0 / (2*alpha^2)
00143         real V = M_2PI * Lx * Ly * Lz;
00144         
00145         /* calculate the k-vector array */
00146         int ksq, kx2, ky2, kz2;
00147         real rksq, rkx2, rky2, rkz2;
00148         real rLx2 = 1. / ( Lx * Lx );
00149         real rLy2 = 1. / ( Ly * Ly );
00150         real rLz2 = 1. / ( Lz * Lz );
00151         real rk2PIx, rk2PIy, rk2PIz;
00152         kVectorLength = 0;
00153         // clear all vectors
00154         kvector.clear();
00155         kxfield.clear();
00156         kyfield.clear();
00157         kzfield.clear();
00158 
00159         kx_ind.clear();
00160         ky_ind.clear();
00161         kz_ind.clear();
00162         virialPref.clear();
00163         virialTensorPref.clear();
00164         for(int kx = 0; kx <= kmax; kx++){
00165           kx2  = kx * kx;
00166           rkx2 = kx2 * rLx2;
00167           rk2PIx = kx * rclx;
00168           for(int ky = -kmax; ky <= kmax; ky++){
00169             ky2  = ky * ky;
00170             rky2 = ky2 * rLy2;
00171             rk2PIy = ky * rcly;
00172             for(int kz = -kmax; kz <= kmax; kz++) {
00173               kz2  = kz * kz;
00174               rkz2 = kz2 * rLz2;
00175               rk2PIz = kz * rclz;
00176               
00177               ksq  = kx2+ky2+kz2;
00178               rksq = rkx2+rky2+rkz2;
00179               
00180               if( (rksq < skmaxsq) && (ksq !=0) ){
00181                 kvector.push_back(  exp( -rksq * B)/( rksq * V)  );
00182                 
00183                 kxfield.push_back( kx );
00184                 kyfield.push_back( ky );
00185                 kzfield.push_back( kz );
00186                 
00187                 kx_ind.push_back(        kx );
00188                 ky_ind.push_back( kmax + ky );
00189                 kz_ind.push_back( kmax + kz );
00190                 
00191                 // the tensor should be: deltaKronecker(i,j) - 2*hi*hj / h^2 - hi*hj / (2*alfa^2)
00192                 Real3D h(rk2PIx, rk2PIy, rk2PIz);
00193                 real h2 = h * h;
00194                 Tensor hh(h, h); // it is tensor: hi*hj
00195                 
00196                 virialPref.push_back( 1 - h2 * inv2alpha2 );
00197                 virialTensorPref.push_back( I - 2 * hh / h2 - hh * inv2alpha2 );
00198                 
00199                 kVectorLength++;
00200               }
00201             }
00202           }
00203         }
00204         
00205  //cout <<"node:  "<< system->comm->rank() <<  " kVectorLength: "<< kVectorLength<< "   kmax: "<< skmax  <<endl;
00206         if(sum != NULL){
00207           delete [] sum;
00208           sum = NULL;
00209         }
00210         if(totsum != NULL){
00211           delete [] totsum;
00212           totsum = NULL;
00213         }
00214         sum = new dcomplex[kVectorLength]; 
00215         totsum = new dcomplex[kVectorLength];
00216         
00217         getParticleNumber();
00218         
00219       }

00262       void exponentPrecalculation(CellList realcells){
00263         /* Calculation of k space sums */
00264         // -1, 0, 1
00265         int j=0; // auxiliary variable, particle counter
00266         for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
00267           Particle &p = *it;
00268           
00269           eikx[     0][j] = dcomplex(1.0, 0.0);
00270           eiky[kmax+0][j] = dcomplex(1.0, 0.0);
00271           eikz[kmax+0][j] = dcomplex(1.0, 0.0);
00272 
00273           eikx[     1][j] = dcomplex( cos( rclx * p.position()[0] ), sin( rclx * p.position()[0] ) );
00274           eiky[kmax+1][j] = dcomplex( cos( rcly * p.position()[1] ), sin( rcly * p.position()[1] ) );
00275           eikz[kmax+1][j] = dcomplex( cos( rclz * p.position()[2] ), sin( rclz * p.position()[2] ) );
00276 
00277           eiky[kmax-1][j] = conj( eiky[kmax+1][j] );
00278           eikz[kmax-1][j] = conj( eikz[kmax+1][j] );
00279          
00280           j++;
00281         }
00282         
00283         // calculation of the rest terms
00284         for (int k=2; k<=kmax; k++) {
00285           j=0;
00286           for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
00287             Particle &p = *it;
00288 
00289             eikx[     k][j] = eikx[     k-1][j] * eikx[     1][j];
00290             
00291             eiky[kmax+k][j] = eiky[kmax+k-1][j] * eiky[kmax+1][j];
00292             eiky[kmax-k][j] = conj( eiky[kmax+k][j] );
00293 
00294             eikz[kmax+k][j] = eikz[kmax+k-1][j] * eikz[kmax+1][j];
00295             eikz[kmax-k][j] = conj( eikz[kmax+k][j] );
00296 
00297             j++;
00298           }
00299         }
00300         
00301         real q; // particle charge, auxiliary
00302         for (int k=0; k<kVectorLength; k++) {
00303           sum[k]    = dcomplex(0.0, 0.0);
00304           totsum[k] = dcomplex(0.0, 0.0);
00305           j=0;
00306           for (iterator::CellListIterator it(realcells); !it.isDone(); ++it) {
00307             Particle &p = *it;
00308             if( (q = p.q())!=0 ){
00309               eik[k][j] = q  *  ( eikx[ kx_ind[k] ][j] * eiky[ ky_ind[k] ][j] * eikz[ kz_ind[k] ][j] );
00310               sum[k] += eik[k][j];
00311             }
00312             j++;
00313           }
00314         }
00315         
00316         mpi::all_reduce( *system -> comm, sum, kVectorLength, totsum, plus<dcomplex>() );
00317       }
