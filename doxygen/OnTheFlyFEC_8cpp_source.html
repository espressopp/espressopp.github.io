122  // Gather Statistics for "steps" steps
123  void OnTheFlyFEC::gatherStats() {
124  LOG4ESPP_DEBUG(theLogger, "gather Stats for OnTheFlyFEC");
125 
126  System& system = getSystemRef();
127 
128  gapcounter += 1;
129 
130  // Only go in after gap steps
131  // Nikita: just changed the '=' to '==' as I think it was initially meant
132  if (gapcounter == gap){
133 
134  // Only go in if counter below steps
135  if (counter < (steps/gap)){
136 
137  // iterate over CG particles
138  CellList cells = system.storage->getRealCells();
139  shared_ptr<FixedTupleListAdress> fixedtupleList = system.storage->getFixedTuples();
140  FixedTupleListAdress::iterator it2;
141  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
142 
143  Particle &vp = *cit;
144  real weight = vp.lambda();
145 
146  if (weight < 0.9999999 && weight > 0.0000001){
147  //std::cout << "weight: " << weight << "\n";
148  //std::cout << "floor(weight*bins): " << floor(weight*bins) << "\n";
149 
150  real driftterm = vp.drift();
151 
152  real dist = vp.position()[0]-center[0];
153  if ( dist < 0.0 ) {driftterm = (-1.0)*driftterm ;}
154 
155  NumbersAtoms[int(floor(weight*bins))] += 1;
156  EnergyDiff[int(floor(weight*bins))] += driftterm;
157  }
158  }
159  counter += 1;
160 
161  }
162 
163  gapcounter = 0;
164 
165  }
166 
167  }
