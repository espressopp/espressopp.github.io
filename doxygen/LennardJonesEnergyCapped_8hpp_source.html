110  real _computeEnergySqrRaw(real distSqr) const {
111 
112  if (distSqr > capradSqr) {
113  real frac2 = sigma*sigma / distSqr;
114  real frac6 = frac2 * frac2 * frac2;
115  real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
116  return energy;
117  }
118  else {
119  real frac2 = sigma*sigma / capradSqr;
120  real frac6 = frac2 * frac2 * frac2;
121  real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
122  return energy;
123  }
124  }

126  bool _computeForceRaw(Real3D& force,
