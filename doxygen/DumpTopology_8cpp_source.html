41 void DumpTopology::saveDataToBuffer() {
42  // Format: <step1><pair_idx_0><size><pid1><pid2><pid3><pid4><pair_idx_1><size><pid...><step2>..
43  int idx = 0;
44  for (std::vector<shared_ptr<FixedPairList> >::iterator it = fpls_.begin(); it != fpls_.end(); ++it) {
45  std::vector<longint> pairs = (*it)->getPairList();
46  fpl_buffer_.push_front(integrator_->getStep());
47  fpl_buffer_.push_front(idx);
48  fpl_buffer_.push_front(2);
49  fpl_buffer_.push_front(pairs.size()/2);
50  for (std::vector<longint>::iterator itt = pairs.begin(); itt != pairs.end();) {
51  fpl_buffer_.push_front(*(itt++));
52  fpl_buffer_.push_front(*(itt++));
53  }
54  idx++;
55  }
56 
57  // Save data from triplet.
58  idx = 0;
59  for (std::vector<shared_ptr<FixedTripleList> >::iterator it = ftls_.begin(); it != ftls_.end(); ++it) {
60  std::vector<longint> triples = (*it)->getTripleList();
61  fpl_buffer_.push_front(integrator_->getStep());
62  fpl_buffer_.push_front(idx);
63  fpl_buffer_.push_front(3);
64  fpl_buffer_.push_front(triples.size()/3);
65  for (std::vector<longint>::iterator itt = triples.begin(); itt != triples.end();) {
66  fpl_buffer_.push_front(*(itt++));
67  fpl_buffer_.push_front(*(itt++));
68  fpl_buffer_.push_front(*(itt++));
69  }
70  idx++;
71  }
72 
73  // Save data from quadruplets
74  idx = 0;
75  for (std::vector<shared_ptr<FixedQuadrupleList> >::iterator it = fqls_.begin(); it != fqls_.end(); ++it) {
76  std::vector<longint> quadruples = (*it)->getQuadrupleList();
77  fpl_buffer_.push_front(integrator_->getStep());
78  fpl_buffer_.push_front(idx);
79  fpl_buffer_.push_front(4);
80  fpl_buffer_.push_front(quadruples.size()/4);
81  for (std::vector<longint>::iterator itt = quadruples.begin(); itt != quadruples.end();) {
82  fpl_buffer_.push_front(*(itt++));
83  fpl_buffer_.push_front(*(itt++));
84  fpl_buffer_.push_front(*(itt++));
85  fpl_buffer_.push_front(*(itt++));
86  }
87  idx++;
88  }
89 }
