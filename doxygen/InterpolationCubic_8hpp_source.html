88  inline real InterpolationCubic::splineInterpolation(real r, const real* fn,
89  const real* fn2) const {
90  int index = static_cast<int>((r - inner) * invdelta);
91 
92  if (index < 0) {
93  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
94  << inner << " - " << inner + (N - 1) * delta);
95  index = 0;
96  }
97  else if (index >= N) {
98  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
99  << inner << " - " << inner + (N - 1) * delta);
100  index = N-1;
101  }
102 
103  real b = (r - radius[index]) * invdelta;
104  real a = 1.0 - b;
105  real f = a * fn[index] +
106  b * fn[index+1] +
107  ((a*a*a-a)*fn2[index] +
108  (b*b*b-b)*fn2[index+1]) *
109  deltasq6;
110 
111  // printf("interpoloate %f, a = %f, b = %f, fn = %f - %f, fn2 = %f - %f\n",
112  // r, a, b, fn[index], fn[index+1], fn2[index], fn2[index+1]);
113  return f;
114  }
