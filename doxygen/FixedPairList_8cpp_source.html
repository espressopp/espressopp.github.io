00157   void FixedPairList::beforeSendParticles(ParticleList& pl, OutBuffer& buf) {
00158     std::vector< longint > toSend;
00159     // loop over the particle list
00160     for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
00161       longint pid = pit->id();
00162       
00163       LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
00164 
00165       // find all pairs that involve this particle
00166       
00167       int n = globalPairs.count(pid);
00168 
00169       if (n > 0) {
00170         std::pair<GlobalPairs::const_iterator,
00171           GlobalPairs::const_iterator> equalRange
00172           = globalPairs.equal_range(pid);
00173 
00174         // first write the pid of the first particle
00175         // then the number of partners
00176         // and then the pids of the partners
00177         toSend.reserve(toSend.size()+n+1);
00178         toSend.push_back(pid);
00179         toSend.push_back(n);
00180         for (GlobalPairs::const_iterator it = equalRange.first;
00181              it != equalRange.second; ++it) {
00182           toSend.push_back(it->second);
00183           LOG4ESPP_DEBUG(theLogger, "send global bond: pid "
00184                        << pid << " and partner " << it->second);
00185         }
00186 
00187         // delete all of these pairs from the global list
00188         globalPairs.erase(equalRange.first, equalRange.second);
00189         // std::cout << "erasing particle " << pid << " from here" << std::endl;
00190       }
00191     }
00192     // send the list
00193     buf.write(toSend);
00194     LOG4ESPP_INFO(theLogger, "prepared fixed pair list before send particles");
00195   }
