49  Tensor computeRaw() {
50  System& system = getSystemRef();
51 
52  // determine volume of the box
53  Real3D Li = system.bc->getBoxL();
54  real V = Li[0] * Li[1] * Li[2];
55 
56  // compute the kinetic contribution (2/3 \sum 1/2mv^2)
57  Tensor vvLocal(0.0);
58  Tensor vv;
59 
60  CellList realCells = system.storage->getRealCells();
61  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
62  real mass = cit->mass();
63  Real3D& vel = cit->velocity();
64  vvLocal += mass * Tensor(vel, vel);
65  }
66 
67  boost::mpi::all_reduce(*mpiWorld, (double*)&vvLocal,6, (double*)&vv, std::plus<double>());
68 
69  // compute the short-range nonbonded contribution
70  Tensor wij(0.0);
71  // virial is already reduced
72  const InteractionList& srIL = system.shortRangeInteractions;
73  for (size_t j = 0; j < srIL.size(); j++) {
74  srIL[j]->computeVirialTensor(wij);
75  }
76 
77  return (vv + wij) / V;
78  }

90  python::list getAverageValue() {

92  Tensor res;

111  void updateAverage(Tensor res) {
112  // compare e. g. Knuth TAOCP vol 2, 3rd edition, page 232
113  if (nMeasurements > 0) {
114  if (nMeasurements == 1) {
115  newAverage = res;
116  lastAverage = newAverage;
117  } else {
118  newAverage = lastAverage + (res - lastAverage) / nMeasurements;
119  for (int i=0; i<6; i++) {
120  newVariance[i] = lastVariance[i] + (res[i] - lastAverage[i]) * (res[i] - newAverage[i]);
121  }
122  lastAverage = newAverage;
123  lastVariance = newVariance;
124  }
125  }
126  return;
