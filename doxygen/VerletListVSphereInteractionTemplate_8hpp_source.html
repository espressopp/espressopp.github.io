255  VerletListVSphereInteractionTemplate < _Potential >::
256  computeVirialTensor(Tensor& w, real z) {
257  LOG4ESPP_INFO(theLogger, "compute the virial tensor for the Verlet List");
258 
259  System& system = verletList->getSystemRef();
260  Real3D Li = system.bc->getBoxL();
261 
262  real rc_cutoff = verletList->getVerletCutoff();
263 
264  // boundaries should be taken into account
265  bool ghost_layer = false;
266  real zghost = -100.0;
267  if(z<rc_cutoff){
268  zghost = z + Li[2];
269  ghost_layer = true;
270  }
271  else if(z>=Li[2]-rc_cutoff){
272  zghost = z - Li[2];
273  ghost_layer = true;
274  }
275 
276  Tensor wlocal(0.0);
277  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
278  Particle &p1 = *it->first;
279  Particle &p2 = *it->second;
280  Real3D p1pos = p1.position();
281  Real3D p2pos = p2.position();
282 
283 
284  if( (p1pos[2]>z && p2pos[2]<z) ||
285  (p1pos[2]<z && p2pos[2]>z) ||
286  (ghost_layer &&
287  ((p1pos[2]>zghost && p2pos[2]<zghost) ||
288  (p1pos[2]<zghost && p2pos[2]>zghost))
289  )
290  ){
291  int type1 = p1.type();
292  int type2 = p2.type();
293  const Potential &potential = getPotential(type1, type2);
294 
295  Real3D force(0.0, 0.0, 0.0);
296  real fsi, fsj;
297  if(potential._computeForce(force, fsi, fsj, p1, p2)) {
298  //TODO think of how to incorporate sigmaij-force into virial calculation
299  Real3D r21 = p1pos - p2pos;
300  wlocal += Tensor(r21, force) / fabs(r21[2]);
301  }
302  }
303  }
304

312  // the first layer has coordinate 0.0 the last - (Lz - Lz/n)
313  template < typename _Potential > inline void
314  VerletListVSphereInteractionTemplate < _Potential >::
315  computeVirialTensor(Tensor *w, int n) {
316  LOG4ESPP_INFO(theLogger, "compute the virial tensor for the Verlet List");
317 
318  System& system = verletList->getSystemRef();
319  Real3D Li = system.bc->getBoxL();
320 
321  real z_dist = Li[2] / float(n); // distance between two layers
322  Tensor *wlocal = new Tensor[n];
323  for(int i=0; i<n; i++) wlocal[i] = Tensor(0.0);
324  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
325  Particle &p1 = *it->first;
326  Particle &p2 = *it->second;
327  int type1 = p1.type();
328  int type2 = p2.type();
329  Real3D p1pos = p1.position();
330  Real3D p2pos = p2.position();
331 
332  const Potential &potential = getPotential(type1, type2);
333 
334  Real3D force(0.0, 0.0, 0.0);
335  Tensor ww;
336  real fsi, fsj;
337  if(potential._computeForce(force, fsi, fsj, p1, p2)) {
338  //TODO think of how to incorporate sigmaij-force into virial calculation
339  Real3D r21 = p1pos - p2pos;
340  ww = Tensor(r21, force) / fabs(r21[2]);
341 
342  int position1 = (int)( p1pos[2]/z_dist );
343  int position2 = (int)( p2pos[2]/z_dist );
344 
345  int maxpos = std::max(position1, position2);
346  int minpos = std::min(position1, position2);
347 
348  // boundaries should be taken into account
349  bool boundaries1 = false;
350  bool boundaries2 = false;
351  if(minpos < 0){
352  minpos += n;
353  boundaries1 =true;
354  }
355  if(maxpos >=n){
356  maxpos -= n;
357  boundaries2 =true;
358  }
359 
360  if(boundaries1 || boundaries2){
361  for(int i = 0; i<=maxpos; i++){
362  wlocal[i] += ww;
363  }
364  for(int i = minpos+1; i<n; i++){
365  wlocal[i] += ww;
366  }
367  }
368  else{
369  for(int i = minpos+1; i<=maxpos; i++){
370  wlocal[i] += ww;
371  }
372  }
373  }
374  }
375 
376  // reduce over all CPUs
377  Tensor *wsum = new Tensor[n];
378  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, n, (double*)&wsum, std::plus<double>());
379 
380  for(int j=0; j<n; j++){
381  w[j] += wsum[j];
382  }
383 
384  delete [] wsum;
385  delete [] wlocal;
386  }
