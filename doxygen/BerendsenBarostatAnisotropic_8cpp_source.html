00083     void BerendsenBarostatAnisotropic::barostat(){
00084       LOG4ESPP_DEBUG(theLogger, "equilibrating the pressure");
00085 
00086       System& system = getSystemRef();
00087       
00088       PressureTensor Pcurrent(getSystem());
00089       
00090       Tensor Ptensor = Pcurrent.computeRaw();  // calculating the current pressure in system
00091       //std::cout << Ptensor << std::endl;
00092       Real3D P = Real3D(Ptensor[0], Ptensor[1], Ptensor[2]); //take the diagonal elements, xx, yy, zz from the ordered pressure tensor, ignore xy etc since only orthorhombic cells are considered so far //FIXME
00093       
00094       Real3D mu3 = Real3D(1) + pref * (P - P0)/3.0;// calculating the current scaling parameter // this is the tensorial form in Berendsen's paper from 1984
00095       
00096       Error err(system.comm);
00097       if(mu3[0]<0.0 || mu3[1]<0.0 || mu3[2]<0.0){
00098         stringstream msg;
00099         msg << "Scaling coefficient is <0 (Berendsen barostat). mu^3="<<mu3;
00100         err.setException( msg.str() );
00101         err.checkException();
00102       }
00103       
00104       system.scaleVolume( mu3, true );
00105     }
