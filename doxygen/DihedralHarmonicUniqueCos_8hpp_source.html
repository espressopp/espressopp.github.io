78 
79  Real3D retF[4];
80 
81  Real3D rijjk = r21.cross(r32); // [r21 x r32]
82  Real3D rjkkn = r32.cross(r43); // [r32 x r43]
83 
84  real rijjk_sqr = rijjk.sqr();
85  real rjkkn_sqr = rjkkn.sqr();
86 
87  real rijjk_abs = sqrt(rijjk_sqr);
88  real rjkkn_abs = sqrt(rjkkn_sqr);
89 
90  real inv_rijjk = 1.0 / rijjk_abs;
91  real inv_rjkkn = 1.0 / rjkkn_abs;
92 
93  // cosine between planes
94  real cos_phi = (rijjk * rjkkn) * (inv_rijjk * inv_rjkkn);
95  if (cos_phi > 1.0) cos_phi = 1.0;
96  else if (cos_phi < -1.0) cos_phi = -1.0;
97 
98  real coef1 = (-2.0) * K * (cos_phi - cos(_phi0));
99 
100  real A1 = inv_rijjk * inv_rjkkn;
101  real A2 = inv_rijjk * inv_rijjk;
102  real A3 = inv_rjkkn * inv_rjkkn;
103 
104  Real3D p3232 ( prod(r32,r32) );
105  Real3D p3243 ( prod(r32,r43) );
106  Real3D p2132 ( prod(r21,r32) );
107  Real3D p2143 ( prod(r21,r43) );
108  Real3D p2121 ( prod(r21,r21) );
109  Real3D p4343 ( prod(r43,r43) );
110 
111  // we have 4 particles 1,2,3,4
112  for(int l=0; l<4; l++){
113  Real3D B1, B2, B3;
114 
115  for(int i=0; i<3; i++){
116  B1[i]= r21[i] * ( p3232[i] * (d(l,2)-d(l,3)) + p3243[i] * (d(l,2)-d(l,1)) ) +
117  r32[i] * ( p2132[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,0)) ) +
118  r43[i] * ( p2132[i] * (d(l,2)-d(l,1)) + p3232[i] * (d(l,0)-d(l,1)) ) +
119  2.0 * r32[i] * p2143[i] * (d(l,1)-d(l,2));
120 
121  B2[i]= 2.0 * r21[i] * ( p3232[i] * (d(l,1)-d(l,0)) + p2132[i] * (d(l,1)-d(l,2)) ) +
122  2.0 * r32[i] * ( p2121[i] * (d(l,2)-d(l,1)) + p2132[i] * (d(l,0)-d(l,1)));
123 
124  B3[i]= 2.0 * r43[i] * ( p3232[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,2)) ) +
125  2.0 * r32[i] * ( p4343[i] * (d(l,2)-d(l,1)) + p3243[i] * (d(l,2)-d(l,3)));
126  }
127 
128  retF[l] = coef1 * ( A1*B1 - 0.5*cos_phi*(A2*B2 + A3*B3) );
129  }
130 
131  force1 = retF[0];
132  force2 = retF[1];
133  force3 = retF[2];
134  force4 = retF[3];
135  }
