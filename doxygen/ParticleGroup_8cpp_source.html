78  void ParticleGroup::beforeSendParticles(ParticleList& pl, class OutBuffer& buf) {
79  // remove all particles that move to a different node
80  for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
81  longint pid = pit->id();
82

91  void ParticleGroup::afterRecvParticles(ParticleList& pl, class InBuffer& buf) {

106  void ParticleGroup::onParticlesChanged() {
107  std::map<longint, Particle*>::iterator p;
108  std::list<longint> remove;
109 
110  // update all pointers
111  for (p = active.begin(); p != active.end(); ++p) {
112  if(!(p->second = storage->lookupRealParticle(p->first))) {
113  fprintf(stderr, "ParticleGroup: non local particle\n");
114  remove.push_back(p->first); // \todo do we need to check for ghosts here
115  }
116  }
117  // remove ghosts, check how iterators behave on erase,
118  // eventually this can be done in the loop
119  while(!remove.empty()) {
120  active.erase(remove.front());
121  remove.pop_front();
122  }
