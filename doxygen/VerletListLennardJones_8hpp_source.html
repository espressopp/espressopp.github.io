179  template <bool ONETYPE, bool VEC_MODE_AOS>
180  inline void
181  VerletListLennardJones::
182  addForces_impl()
183  {
184  {
185  real ff1_, ff2_, cutoffSqr_;
186  if(ONETYPE){
187  ff1_ = ffs[0].ff1;
188  ff2_ = ffs[0].ff2;
189  cutoffSqr_ = cutoffSqr[0];
190  }
191 
192  auto& particleArray = verletList->getParticleArray();
193  auto& neighborList = verletList->getNeighborList();
194 
195  const Real3DInt *pa_pos = particleArray.position.data();
196  Real4D *pa_force = particleArray.force.data();
197 
198  const ulongint* __restrict pa_type = particleArray.type.data();
199  const real* __restrict pa_p_x = particleArray.p_x.data();
200  const real* __restrict pa_p_y = particleArray.p_y.data();
201  const real* __restrict pa_p_z = particleArray.p_z.data();
202  real* __restrict pa_f_x = particleArray.f_x.data();
203  real* __restrict pa_f_y = particleArray.f_y.data();
204  real* __restrict pa_f_z = particleArray.f_z.data();
205 
206  {
207  const auto* __restrict plist = neighborList.plist.data();
208  const auto* __restrict prange = neighborList.prange.data();
209  const auto* __restrict nplist = neighborList.nplist.data();
210  const int ip_max = neighborList.plist.size();
211 
212  int in_min=0;
213  for(int ip=0; ip<ip_max; ip++)
214  {
215  int p = plist[ip];
216  int p_lookup;
217  real p_x, p_y, p_z;
218 
219  if(VEC_MODE_AOS)
220  {
221  p_x = pa_pos[p].x;
222  p_y = pa_pos[p].y;
223  p_z = pa_pos[p].z;
224  if(!ONETYPE){
225  p_lookup = pa_pos[p].t*np_types;
226  }
227  }
228  else
229  {
230  if(!ONETYPE){
231  p_lookup = pa_type[p]*np_types;
232  }
233  p_x = pa_p_x[p];
234  p_y = pa_p_y[p];
235  p_z = pa_p_z[p];
236  }
237 
238  real f_x = 0.0;
239  real f_y = 0.0;
240  real f_z = 0.0;
241 
242  const int in_max=prange[ip];
243 
244  #pragma vector always
245  #pragma vector aligned
246  #pragma ivdep
247  for(int in=in_min; in<in_max; in++)
248  {
249  auto np_ii = nplist[in];
250  {
251  int np_lookup;
252  real dist_x, dist_y, dist_z;
253  if(VEC_MODE_AOS)
254  {
255  dist_x = p_x - pa_pos[np_ii].x;
256  dist_y = p_y - pa_pos[np_ii].y;
257  dist_z = p_z - pa_pos[np_ii].z;
258  if(!ONETYPE) np_lookup = pa_pos[np_ii].t+p_lookup;
259  }
260  else
261  {
262  dist_x = p_x - pa_p_x[np_ii];
263  dist_y = p_y - pa_p_y[np_ii];
264  dist_z = p_z - pa_p_z[np_ii];
265  if(!ONETYPE) np_lookup = pa_type[np_ii]+p_lookup;
266  }
267 
268  real distSqr = dist_x*dist_x + dist_y*dist_y + dist_z*dist_z;
269  if(!ONETYPE){
270  cutoffSqr_ = cutoffSqr[np_lookup];
271  }
272 
273  #if defined(ESPP_VECTOR_MASK)
274  if(distSqr <= cutoffSqr_)
275  #endif
276  {
277  real frac2 = 1.0 / distSqr;
278  real frac6 = frac2 * frac2 * frac2;
279  real ffactor;
280 
281  if(ONETYPE)
282  ffactor = ff1_ * frac6 - ff2_;
283  else
284  ffactor = ffs[np_lookup].ff1 * frac6 - ffs[np_lookup].ff2;
285 
286  #if !defined(ESPP_VECTOR_MASK)
287  if(distSqr > cutoffSqr_) ffactor = 0.0;
288  #endif
289 
290  ffactor = frac6 * ffactor * frac2;
291 
292  f_x += dist_x * ffactor;
293  f_y += dist_y * ffactor;
294  f_z += dist_z * ffactor;
295 
296  if(VEC_MODE_AOS)
297  {
298  auto& np_force = pa_force[np_ii];
299  np_force.x -= dist_x * ffactor;
300  np_force.y -= dist_y * ffactor;
301  np_force.z -= dist_z * ffactor;
302  }
303  else
304  {
305  pa_f_x[np_ii] -= dist_x * ffactor;
306  pa_f_y[np_ii] -= dist_y * ffactor;
307  pa_f_z[np_ii] -= dist_z * ffactor;
308  }
309 
310  }
311  }
312  }
313  if(VEC_MODE_AOS)
314  {
315  auto& p_force = pa_force[p];
316  p_force.x += f_x;
317  p_force.y += f_y;
318  p_force.z += f_z;
319  }
320  else
321  {
322  pa_f_x[p] += f_x;
323  pa_f_y[p] += f_y;
324  pa_f_z[p] += f_z;
325  }
326 
327  in_min = in_max;
328  }
329 
330  }
331  }

459  // local pressure tensor for layer, plane is defined by z coordinate
460  inline void
461  VerletListLennardJones::
462  computeVirialTensor(Tensor& w, real z) {
463  LOG4ESPP_DEBUG(_Potential::theLogger, "loop over verlet list pairs and sum up virial tensor over one z-layer");
464 
465  System& system = verletList->getSystemRef();
466  Real3D Li = system.bc->getBoxL();
467 
468  real rc_cutoff = verletList->getVerletCutoff();
469 
470  // boundaries should be taken into account
471  bool ghost_layer = false;
472  real zghost = -100.0;
473  if(z<rc_cutoff){
474  zghost = z + Li[2];
475  ghost_layer = true;
476  }
477  else if(z>=Li[2]-rc_cutoff){
478  zghost = z - Li[2];
479  ghost_layer = true;
480  }
481 
482  Tensor wlocal(0.0);
483  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
484  Particle &p1 = *it->first;
485  Particle &p2 = *it->second;
486  Real3D p1pos = p1.position();
487  Real3D p2pos = p2.position();
488 
489 
490  if( (p1pos[2]>z && p2pos[2]<z) ||
491  (p1pos[2]<z && p2pos[2]>z) ||
492  (ghost_layer &&
493  ((p1pos[2]>zghost && p2pos[2]<zghost) ||
494  (p1pos[2]<zghost && p2pos[2]>zghost))
495  )
496  ){
497  int type1 = p1.type();
498  int type2 = p2.type();
499  const Potential &potential = getPotential(type1, type2);
500 
501  Real3D force(0.0, 0.0, 0.0);
502  if(potential._computeForce(force, p1, p2)) {
503  Real3D r21 = p1pos - p2pos;
504  wlocal += Tensor(r21, force) / fabs(r21[2]);
505  }
506  }
507  }
508 
509  // reduce over all CPUs
510  Tensor wsum(0.0);
511  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, 6, (double*)&wsum, std::plus<double>());
512  w += wsum;
513  }

515  // it will calculate the pressure in 'n' layers along Z axis
516  // the first layer has coordinate 0.0 the last - (Lz - Lz/n)
517  inline void
518  VerletListLennardJones::
519  computeVirialTensor(Tensor *w, int n) {
520  LOG4ESPP_DEBUG(_Potential::theLogger, "loop over verlet list pairs and sum up virial tensor in bins along z-direction");
521 
522  System& system = verletList->getSystemRef();
523  Real3D Li = system.bc->getBoxL();
524 
525  real z_dist = Li[2] / float(n); // distance between two layers
526  Tensor *wlocal = new Tensor[n];
527  for(int i=0; i<n; i++) wlocal[i] = Tensor(0.0);
528  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
529  Particle &p1 = *it->first;
530  Particle &p2 = *it->second;
531  int type1 = p1.type();
532  int type2 = p2.type();
533  Real3D p1pos = p1.position();
534  Real3D p2pos = p2.position();
535 
536  const Potential &potential = getPotential(type1, type2);
537 
538  Real3D force(0.0, 0.0, 0.0);
539  Tensor ww;
540  if(potential._computeForce(force, p1, p2)) {
541  Real3D r21 = p1pos - p2pos;
542  ww = Tensor(r21, force) / fabs(r21[2]);
543 
544  int position1 = (int)( p1pos[2]/z_dist );
545  int position2 = (int)( p2pos[2]/z_dist );
546 
547  int maxpos = std::max(position1, position2);
548  int minpos = std::min(position1, position2);
549 
550  // boundaries should be taken into account
551  bool boundaries1 = false;
552  bool boundaries2 = false;
553  if(minpos < 0){
554  minpos += n;
555  boundaries1 =true;
556  }
557  if(maxpos >=n){
558  maxpos -= n;
559  boundaries2 =true;
560  }
561 
562  if(boundaries1 || boundaries2){
563  for(int i = 0; i<=maxpos; i++){
564  wlocal[i] += ww;
565  }
566  for(int i = minpos+1; i<n; i++){
567  wlocal[i] += ww;
568  }
569  }
570  else{
571  for(int i = minpos+1; i<=maxpos; i++){
572  wlocal[i] += ww;
573  }
574  }
575  }
576  }
577 
578  // reduce over all CPUs
579  Tensor *wsum = new Tensor[n];
580  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, n, (double*)&wsum, std::plus<double>());
581 
582  for(int j=0; j<n; j++){
583  w[j] += wsum[j];
584  }
585 
586  delete [] wsum;
587  delete [] wlocal;
588  }
