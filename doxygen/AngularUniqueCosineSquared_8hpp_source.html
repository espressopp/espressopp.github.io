58  real dist12_sqr = r12.sqr();
59  real dist32_sqr = r32.sqr();
60  real dist12_magn = sqrt(dist12_sqr);
61  real dist32_magn = sqrt(dist32_sqr);
62  real inv_dist1232 = 1.0 / (dist12_magn * dist32_magn);
63 
64  real cos_theta = r12 * r32 * inv_dist1232;
65  if(cos_theta < -1.0) cos_theta = -1.0;
66  else if(cos_theta > 1.0) cos_theta = 1.0;
67 
68  real dU_dtheta = 2.0 * K * (cos_theta - cos(_theta0));
69 
70  real a11 = dU_dtheta * cos_theta / dist12_sqr;
71  real a12 = -dU_dtheta * inv_dist1232;
72  real a22 = dU_dtheta * cos_theta / dist32_sqr;
73 
74  force12 = a11 * r12 + a12 * r32;
75  force32 = a22 * r32 + a12 * r12;
