38  void LBInitPeriodicForce::setForce( Real3D _force ) {
39  int _printId = 0;
40  int _offset = latticeboltzmann->getHaloSkin();
41  Int3D _Ni = latticeboltzmann->getNi(); // system size in lattice node
42  Int3D _myNi = latticeboltzmann->getMyNi(); // my local nodes
43  Int3D _globIdx = latticeboltzmann->findGlobIdx(); // first lb site global index
44 
45  real ampFz = _force[2]; // amplitude of sin force in z-direction
46  real perCompX; // periodicity function with x-dir dependence
47  Real3D perF; // gravity forces in x- and y-dir; in z: amplitude
48  // of force periodic with sin in x, i.e.
49  // perForce_z = ampFz * sin ( x );
50 
51  // loop over real lattice nodes //
52  real coef = 2. * M_PI / _Ni[0];
53  for (int i = _offset; i < _myNi[0] - _offset; i++) {
54  perCompX = sin( coef * (_globIdx[0] + i - _offset) );
55  perF = Real3D (_force[0], _force[1], ampFz * perCompX);
56 
57  for (int j = _offset; j < _myNi[1] - _offset; j++) {
58  for (int k = _offset; k < _myNi[2] - _offset; k++) {
59  latticeboltzmann->setExtForceLoc( Int3D(i,j,k), perF );
60  }
61  }
62  }
63 
64  // take care of the force flag for lb //
65  if (_force.sqr() > ROUND_ERROR_PREC) {
66  _printId = 1;
67  latticeboltzmann->setDoExtForce( true );
68  printForce(_force, _printId );
69  }
70  }

72  void LBInitPeriodicForce::addForce(Real3D _force) {
73  int _printId = 0;
74  int _offset = latticeboltzmann->getHaloSkin();
75  Int3D _Ni = latticeboltzmann->getNi(); // system size in lattice node
76  Int3D _myNi = latticeboltzmann->getMyNi(); // my local nodes
77  Int3D _globIdx = latticeboltzmann->findGlobIdx();// first lb site global index
78 
79  bool fOnSite = false, fTot = false;
80  real ampFz = _force[2]; // amplitude of sin force in z-direction
81  real perCompX; // periodicity function with x-dir dependence
82  Real3D newF, totF; // gravity forces in x- and y-dir; in z: amplitude
83  // of force periodic with sin in x, i.e.
84  // perForce_z = ampFz * sin ( x );
85 
86  // loop over real lattice nodes //
87  real coef = 2. * M_PI / _Ni[0];
88  for (int i = _offset; i < _myNi[0] - _offset; i++) {
89  perCompX = sin( coef * (_globIdx[0] + i - _offset) );
90  newF = Real3D (_force[0], _force[1], ampFz * perCompX);
91 
92  for (int j = _offset; j < _myNi[1] - _offset; j++) {
93  for (int k = _offset; k < _myNi[2] - _offset; k++) {
94  totF = latticeboltzmann->getExtForceLoc( Int3D(i,j,k) ) + newF;
95 
96  // check if total force is greater than zero
97  if (totF.sqr() > ROUND_ERROR_PREC) {
98  fOnSite = true;
99  _printId = 2;
100  latticeboltzmann->setExtForceLoc( Int3D(i,j,k), totF );
101  } else {
102  latticeboltzmann->setExtForceLoc( Int3D(i,j,k), Real3D(0.) );
103  }
104 
105  fTot = fTot | fOnSite; // if at least one site has force -> fTot true
106  }
107  }
108  }
109 
110  // take care of the force flag for lb //
111  latticeboltzmann->setDoExtForce( fTot );
112  printForce(_force, _printId );
113  }
