98 void StochasticVelocityRescaling::initialize() {
99  LOG4ESPP_INFO(theLogger, "init, coupling = " << coupling <<
100  ", external temperature = " << temperature);
101  real dt = integrator->getTimeStep();
102  pref = coupling / dt;
103 
104  System& system = getSystemRef();
105  NPart_local = system.storage->getNRealParticles();
106  boost::mpi::all_reduce(*getSystem()->comm, NPart_local, NPart,
107  std::plus<int>());
108  DegreesOfFreedom = 3.0 * NPart; //TODO this is _only_ true for simple system without any constraints
109  //calculate the reference kinetic energy based on reference temperature 'temperature'
110  EKin_ref = 0.5 * temperature * BOLTZMANN * DegreesOfFreedom;
111 }

113 void StochasticVelocityRescaling::rescaleVelocities() {
114  LOG4ESPP_DEBUG(theLogger, "rescaleVelocities");
115 
116  real EKin = 0.0;
117  real EKin_local = 0.0;
118  real EKin_new = 0.0;
119  real ScalingFactor;
120 
121  System& system = getSystemRef();
122  CellList realCells = system.storage->getRealCells();
123 
124  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
125  Real3D vel = cit->velocity();
126  EKin_local += 0.5 * cit->mass() * (vel * vel); //FIXME do not forget that his is only correct for velocity-verlet; leap-frog would require adjustments
127  }
128 
129  boost::mpi::all_reduce(*getSystem()->comm, EKin_local, EKin,
130  std::plus<real>());
131 
132 
133  if(getSystem()->comm->rank() == 0){
134  EKin_new = stochasticVR_pullEkin(EKin, EKin_ref, DegreesOfFreedom, pref,
135  rng);
136  // it should always be larger than 0
137  if (EKin_new <= 0)
138  throw std::runtime_error(
139  "EKin_new in StochasticVelocityRescaling::rescaleVelocities() is equal or smaller than 0");
140  ScalingFactor = sqrt(EKin_new / EKin);
141  }
142 
143  boost::mpi::broadcast(*getSystem()->comm, ScalingFactor, 0);
144 
145  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
146  cit->velocity() *= ScalingFactor;
147  }
148 
149 }

151 real StochasticVelocityRescaling::stochasticVR_sumGaussians(const int n) {
152 
154  /* real tmp, sum = 0.0;
155  for (int j = 0; j < n; j++) {
156  tmp = rng->normal();
157  sum += tmp * tmp;
158  }
159  return sum;*/

161  real rr;
162  if (n == 0)
163  return 0.0;
164  else if (n == 1) {
165  rr = rng->normal();
166  return rr * rr;
167  } else if (n % 2 == 0) {
168  return 2.0 * gammaDist->drawNumber(n / 2);

175 real StochasticVelocityRescaling::stochasticVR_pullEkin(real Ekin,
176  real Ekin_ref, int dof, real taut, shared_ptr<esutil::RNG> rng) {
177  real factor, rr;
178 
179  /*
180  the parameter taut is the coupling time divided by the timestep, i.e.
181  even a strong coupling (low coupling time) with a large timestep should be
182  in the order of 0.1 / 0.01 = 10
183  make sure it's reasonably large
184  */
185  if (taut < 0.1)
186  throw std::runtime_error("taut in stochasticVR_pullEkin is very low."
187  "If that is intended, change the code");
188  else {
189  factor = exp(-1.0 / taut);
190  }
191  rr = rng->normal();
192  return Ekin
193  + (1.0 - factor)
194  * (Ekin_ref * (stochasticVR_sumGaussians(dof - 1) + rr * rr)
195  / dof - Ekin)
196  + 2.0 * rr * sqrt(Ekin * Ekin_ref / dof * (1.0 - factor) * factor);
197 }

204 real GammaDistributionNR2nd::drawNumber(const unsigned int ia) {
205  real x, v1, v2, y, am, s, e;
206 
207  if (ia < 1)
208  throw std::runtime_error(
209  "Error in routine stochasticVR_gammaDeviate2nd"); //TODO: how do I get __FUNCTION__ in here?
210  if (ia < 6) {
211  x = 1.0;
212  for (int j=0; j <= ia; j++)
213  x = x * (*rng)();
214  x = -log(x);
215  } else {
216  do {
217  do {
218  do {
219  v1 = 2.0 * (*rng)() - 1.0;
220  v2 = 2.0 * (*rng)() - 1.0;
221  } while (v1 * v1 + v2 * v2 > 1.0);
222  y = v2 / v1;
223  am = ia - 1;
224  s = sqrt(2.0 * am + 1.0);
225  x = s * y + am;
226  } while (x <= 0.0);
227  e = (1.0 + y * y) * exp(am * log(x / am) - s * y);
228  } while ((*rng)() > e);
229  }
230  return x;
231 }

234 real GammaDistributionNR3rd::drawNumber(const unsigned int ia) {
235  real a1, a2, x, v, u;
236  if (ia <= 0)
237  throw std::runtime_error(
238  "Error in routine stochasticVR_gammaDeviate3rd");
239  a1 = ia - 1.0 / 3.0;
240  a2 = 1.0 / sqrt(9.0 * a1);
241  do {
242  do {
243  x = rng->normal();
244  v = 1.0 + a2 * x;
245  } while (v <= 0.0);
246  v = v * v * v;
247  u = (*rng)();
248  } while ((u > 1.0 - 0.331 * x * x * x * x)
249  && log(u) > 0.5 * x * x + a1 * (1.0 - v + log(v)));
250  return a1 * v;
251 }
