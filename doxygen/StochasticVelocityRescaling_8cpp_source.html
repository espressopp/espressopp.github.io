00175 real StochasticVelocityRescaling::stochasticVR_pullEkin(real Ekin,
00176                 real Ekin_ref, int dof, real taut, shared_ptr<esutil::RNG> rng) {
00177         real factor, rr;
00178 
00179         /*
00180          the parameter taut is the coupling time divided by the timestep, i.e.
00181          even a strong coupling (low coupling time) with a large timestep should be
00182          in the order of 0.1 / 0.01 = 10
00183          make sure it's reasonably large
00184          */
00185         if (taut < 0.1)
00186                 throw std::runtime_error("taut in stochasticVR_pullEkin is very low."
00187                                 "If that is intended, change the code");
00188         else {
00189                 factor = exp(-1.0 / taut);
00190         }
00191         rr = rng->normal();
00192         return Ekin
00193                         + (1.0 - factor)
00194                                         * (Ekin_ref * (stochasticVR_sumGaussians(dof - 1) + rr * rr)
00195                                                         / dof - Ekin)
00196                         + 2.0 * rr * sqrt(Ekin * Ekin_ref / dof * (1.0 - factor) * factor);
00197 }

00234 real GammaDistributionNR3rd::drawNumber(const unsigned int ia) {
00235         real a1, a2, x, v, u;
00236         if (ia <= 0)
00237                 throw std::runtime_error(
00238                                 "Error in routine stochasticVR_gammaDeviate3rd");
00239         a1 = ia - 1.0 / 3.0;
00240         a2 = 1.0 / sqrt(9.0 * a1);
00241         do {
00242                 do {
00243                         x = rng->normal();
00244                         v = 1.0 + a2 * x;
00245                 } while (v <= 0.0);
00246                 v = v * v * v;
00247                 u = (*rng)();
00248         } while ((u > 1.0 - 0.331 * x * x * x * x)
00249                         && log(u) > 0.5 * x * x + a1 * (1.0 - v + log(v)));
