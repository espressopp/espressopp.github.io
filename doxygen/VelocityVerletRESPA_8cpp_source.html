50  void VelocityVerletRESPA::run(int nsteps)
51  {
52  int nResorts = 0;
53  System& system = getSystemRef();
54  storage::Storage& storage = *system.storage;
55  real skinHalf = 0.5 * system.getSkin();
56  dtlong = dt*multistep;
57 
58  runInit(); // signal
59 
60  // Before start make sure that particles are on the right processor
61  if (resortFlag) {
62  storage.decompose();
63  maxDist = 0.0;
64  resortFlag = false;
65  }
66 
67  updateForces(true);
68  aftCalcSlow(); // signal
69 
70  for (int i = 0; i < nsteps; i++) {
71  integrate2(true);
72  aftIntSlow(); // signal
73 
74  recalc1(); // signal
75  updateForces(false);
76  recalc2(); // signal
77 
78  for (int j = 0; j < multistep; j++) {
79  befIntP(); // signal
80 
81  maxDist += integrate1();
82  aftIntP(); // signal
83 
84  if (maxDist > skinHalf) resortFlag = true;
85  if (resortFlag) {
86  storage.decompose();
87  maxDist = 0.0;
88  resortFlag = false;
89  nResorts ++;
90  }
91 
92  updateForces(false);
93  befIntV(); // signal
94 
95  integrate2(false);
96  aftIntV(); // signal
97  }
98 
99  updateForces(true);
100  aftCalcSlow(); // signal
101 
102  integrate2(true);
103  aftIntSlow(); // signal
104  }
105  }

107  real VelocityVerletRESPA::integrate1()
108  {
109  System& system = getSystemRef();
110  CellList realCells = system.storage->getRealCells();
111 
112  // loop over particles
113  real maxSqDist = 0.0;
114  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
115  real sqDist = 0.0;
116  real dtfm = 0.5 * dt / cit->mass();
117 
118  // propagate velocities
119  cit->velocity() += dtfm * cit->force();
120 
121  // propagate positions
122  Real3D deltaP = cit->velocity();
123 
124  deltaP *= dt;
125  cit->position() += deltaP;
126  sqDist += deltaP * deltaP;
127  maxSqDist = std::max(maxSqDist, sqDist);
128  }
129 
130  inIntP(maxSqDist); // signal
131 
132  real maxAllSqDist;
133  mpi::all_reduce(*system.comm, maxSqDist, maxAllSqDist, boost::mpi::maximum<real>());
134  return sqrt(maxAllSqDist);
135  }

137  void VelocityVerletRESPA::integrate2(bool slow)
138  {
139  System& system = getSystemRef();
140  CellList realCells = system.storage->getRealCells();
141 
142  // loop over particles
143  real half_dt = 0.0;
144  if(slow) {
145  half_dt = 0.5 * dtlong;
146  }
147  else {
148  half_dt = 0.5 * dt;
149  }
150  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
151  real dtfm = half_dt / cit->mass();
152  // propagate velocities
153  cit->velocity() += dtfm * cit->force();
154  }
155 
156  step++;
157  }
