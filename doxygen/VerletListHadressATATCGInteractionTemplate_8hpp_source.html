171  addForces() {
172  LOG4ESPP_INFO(theLogger, "add forces computed by the Verlet List");
173 
174  std::set<Particle*> cgZone = verletList->getCGZone();
175  std::set<Particle*> adrZone = verletList->getAdrZone();
176 
177  for (std::set<Particle*>::iterator it=adrZone.begin();
178  it != adrZone.end(); ++it) {
179  Particle &p = **it;
180  // intitialize energy diff AA-CG
181  if (p.lambda()<0.9999999 && p.lambda()>0.0000001) {
182  energydiff[&p]=0.0;
183  }
184  // energydiff[&p]=0.0;
185  }
186 
187 
188  // Pairs not inside the AdResS Zone (CG region)
189  // COMMENT FOR IDEAL GAS
190  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
191 
192  Particle &p1 = *it->first;
193  Particle &p2 = *it->second;
194  int type1 = p1.type();
195  int type2 = p2.type();
196 
197  const PotentialCG &potentialCG = getPotentialCG(type1, type2);
198 
199  Real3D force(0.0, 0.0, 0.0);
200 
201  // CG forces
202  if(potentialCG._computeForce(force, p1, p2)) {
203  p1.force() += force;
204  p2.force() -= force;
205  }
206  }
207  // REMOVE FOR IDEAL GAS
208 
209  // Compute forces (AT and VP) of Pairs inside AdResS zone
210  for (PairList::Iterator it(verletList->getAdrPairs()); it.isValid(); ++it) {
211  real w1, w2;
212  // these are the two VP interacting
213  Particle &p1 = *it->first;
214  Particle &p2 = *it->second;
215 
216  w1 = p1.lambda();
217  w2 = p2.lambda();
218 
219  real w12 = (w1 + w2)/2.0; // H-AdResS
220 
221 
222 
223  // REMOVE FOR IDEAL GAS
224  // force between VP particles
225  int type1 = p1.type();
226  int type2 = p2.type();
227  const PotentialCG &potentialCG = getPotentialCG(type1, type2);
228  Real3D forcevp(0.0, 0.0, 0.0);
229  if (w12 != 1.0) { // calculate VP force if both VP are outside AT region (CG-HY, HY-HY)
230  if (potentialCG._computeForce(forcevp, p1, p2)) {
231  forcevp *= (1.0 - w12);
232  p1.force() += forcevp;
233  p2.force() -= forcevp;
234  }
235 
236  // H-AdResS - Drift Term part 1
237  // Compute CG energies of particles in the hybrid and store and add up in map energydiff
238  if (w12 != 0.0) { //at least one particle in hybrid region => need to do the energy calculation
239  real energyvp = potentialCG._computeEnergy(p1, p2);
240  if (w1 != 0.0) { // if particle one is in hybrid region
241  energydiff[&p1] += energyvp; // add CG energy for virtual particle 1
242  }
243  if (w2 != 0.0) { // if particle two is in hybrid region
244  energydiff[&p2] += energyvp; // add CG energy for virtual particle 2
245  }
246  }
247 
248  }
249  // REMOVE FOR IDEAL GAS
250 
251  // force between AT particles
252  if (w12 != 0.0) { // calculate AT force if both VP are outside CG region (HY-HY, HY-AT, AT-AT)
253  FixedTupleListAdress::iterator it3;
254  FixedTupleListAdress::iterator it4;
255  it3 = fixedtupleList->find(&p1);
256  it4 = fixedtupleList->find(&p2);
257 
258  if (it3 != fixedtupleList->end() && it4 != fixedtupleList->end()) {
259 
260  std::vector<Particle*> atList1;
261  std::vector<Particle*> atList2;
262  atList1 = it3->second;
263  atList2 = it4->second;
264 
265  for (std::vector<Particle*>::iterator itv = atList1.begin();
266  itv != atList1.end(); ++itv) {
267 
268  Particle &p3 = **itv;
269 
270  for (std::vector<Particle*>::iterator itv2 = atList2.begin();
271  itv2 != atList2.end(); ++itv2) {
272 
273  Particle &p4 = **itv2;
274 
275  // AT forces
276  Real3D force(0.0, 0.0, 0.0);
277  const PotentialAT1 &potentialAT1 = getPotentialAT1(p3.type(), p4.type());
278  if(potentialAT1._computeForce(force, p3, p4)) {
279  force *= w12;
280  p3.force() += force;
281  p4.force() -= force;
282  }
283  // H-AdResS - Drift Term part 2
284  // Compute AT energies of particles in the hybrid and store and subtract in map energydiff
285  if(w12!=1.0) { //at least one particle in hybrid region => need to do the energy calculation
286  const real energyat = potentialAT1._computeEnergy(p3, p4);
287  if(w1!=1.0) { // if particle one is in hybrid region
288  energydiff[&p1] -= energyat; // subtract AT energy for virtual particle 1
289  }
290  if(w2!=1.0) { // if particle two is in hybrid region
291  energydiff[&p2] -= energyat; // subtract AT energy for virtual particle 2
292  }
293  }
294  const PotentialAT2 &potentialAT2 = getPotentialAT2(p3.type(), p4.type());
295  if(potentialAT2._computeForce(force, p3, p4)) {
296  force *= w12;
297  p3.force() += force;
298  p4.force() -= force;
299  }
300  // H-AdResS - Drift Term part 2
301  // Compute AT energies of particles in the hybrid and store and subtract in map energydiff
302  if(w12!=1.0) { //at least one particle in hybrid region => need to do the energy calculation
303  const real energyat = potentialAT2._computeEnergy(p3, p4);
304  if(w1!=1.0) { // if particle one is in hybrid region
305  energydiff[&p1] -= energyat; // subtract AT energy for virtual particle 1
306  }
307  if(w2!=1.0) { // if particle two is in hybrid region
308  energydiff[&p2] -= energyat; // subtract AT energy for virtual particle 2
309  }
310  }
311 
312  }
313 
314  }
315  }
316  else { // this should not happen
317  std::stringstream ss;
318  ss << "One of the VP particles not found in tuples: " << p1.id() << "-" << p1.ghost() << ", " << p2.id() << "-" << p2.ghost() << " (" << p1.position() << ") (" << p2.position();
319  throw std::runtime_error(ss.str());
320  }
321  }
322  }
323 
324  // H-AdResS - Drift Term part 3
325  // Iterate over all particles in the hybrid region and calculate drift force
326  for (std::set<Particle*>::iterator it=adrZone.begin();
327  it != adrZone.end(); ++it) { // Iterate over all particles
328  Particle &vp = **it;
329  real w = vp.lambda();
330 
331  if(w<0.9999999 && w>0.0000001) { // only chose those in the hybrid region
332  // calculate distance to nearest adress particle or center
333  std::vector<Real3D*>::iterator it2 = verletList->getAdrPositions().begin();
334  Real3D pa = **it2; // position of adress particle
335  Real3D mindriftforce(0.0, 0.0, 0.0);
336  verletList->getSystem()->bc->getMinimumImageVector(mindriftforce, vp.position(), pa);
337  real min1sq = 0.0;
338  if(verletList->getAdrRegionType()) {
339  min1sq = mindriftforce.sqr();
340  }
341  else {
342  min1sq = mindriftforce[0]*mindriftforce[0];
343  }
344  ++it2;
345  for (; it2 != verletList->getAdrPositions().end(); ++it2) {
346  pa = **it2;
347  Real3D driftforce(0.0, 0.0, 0.0);
348  verletList->getSystem()->bc->getMinimumImageVector(driftforce, vp.position(), pa);
349  real distsq1 = 0.0;
350  if(verletList->getAdrRegionType()) {
351  distsq1 = driftforce.sqr();
352  }
353  else {
354  distsq1 = driftforce[0]*driftforce[0];
355  }
356  if (distsq1 < min1sq) {
357  min1sq = distsq1;
358  mindriftforce = driftforce;
359  }
360  }
361  min1sq = sqrt(min1sq); // distance to nearest adress particle or center
362 
363  if(verletList->getAdrRegionType()) {
364  mindriftforce = (1.0/min1sq)*mindriftforce; // normalized driftforce vector
365  // mindriftforce *= (0.5 * energydiff.find(&vp)->second); // get the energy differences which were calculated previously and put in drift force
366  mindriftforce *= (0.5 * energydiff[&vp]);
367  mindriftforce *= vp.lambdaDeriv();
368  vp.force() += mindriftforce;
369  }
370  else {
371  real mindriftforceX = (1.0/min1sq)*mindriftforce[0]; // normalized driftforce vector
372  // mindriftforceX *= (0.5 * energydiff.find(&vp)->second); // get the energy differences which were calculated previously and put in drift force
373  mindriftforceX *= (0.5 * energydiff[&vp]);
374  mindriftforceX *= vp.lambdaDeriv();
375  Real3D driftforceadd(mindriftforceX,0.0,0.0);
376  vp.force() += driftforceadd;
377  }
378  vp.drift() += 0.5 * energydiff.find(&vp)->second;
379  }
380 
381  }
382 
383  // energydiff.clear(); // clear the energy difference map
384  }

390  computeEnergy() {
391  LOG4ESPP_INFO(theLogger, "compute energy of the Verlet list pairs");
392 
393  // REMOVE FOR IDEAL GAS
394  real e = 0.0;
395  for (PairList::Iterator it(verletList->getPairs());
396  it.isValid(); ++it) {
397  Particle &p1 = *it->first;
398  Particle &p2 = *it->second;
399  int type1 = p1.type();
400  int type2 = p2.type();
401  const PotentialCG &potential = getPotentialCG(type1, type2);
402  e += potential._computeEnergy(p1, p2);
403  }
404  // REMOVE FOR IDEAL GAS
405 
406  for (PairList::Iterator it(verletList->getAdrPairs());
407  it.isValid(); ++it) {
408  Particle &p1 = *it->first;
409  Particle &p2 = *it->second;
410  real w1 = p1.lambda();
411  real w2 = p2.lambda();
412  real w12 = (w1 + w2)/2.0;
413 
414  // REMOVE FOR IDEAL GAS
415  int type1 = p1.type();
416  int type2 = p2.type();
417  const PotentialCG &potentialCG = getPotentialCG(type1, type2);
418  e += (1.0-w12)*potentialCG._computeEnergy(p1, p2);
419  // REMOVE FOR IDEAL GAS
420 
421  FixedTupleListAdress::iterator it3;
422  FixedTupleListAdress::iterator it4;
423  it3 = fixedtupleList->find(&p1);
424  it4 = fixedtupleList->find(&p2);
425 
426  if (it3 != fixedtupleList->end() && it4 != fixedtupleList->end()) {
427  std::vector<Particle*> atList1;
428  std::vector<Particle*> atList2;
429  atList1 = it3->second;
430  atList2 = it4->second;
431 
432  for (std::vector<Particle*>::iterator itv = atList1.begin();
433  itv != atList1.end(); ++itv) {
434 
435  Particle &p3 = **itv;
436  for (std::vector<Particle*>::iterator itv2 = atList2.begin();
437  itv2 != atList2.end(); ++itv2) {
438  Particle &p4 = **itv2;
439 
440  // AT energies
441  const PotentialAT1 &potentialAT1 = getPotentialAT1(p3.type(), p4.type());
442  const PotentialAT2 &potentialAT2 = getPotentialAT2(p3.type(), p4.type());
443  e += w12*(potentialAT1._computeEnergy(p3, p4) + potentialAT2._computeEnergy(p3, p4));
444 
445  }
446  }
447  }
448  }

451  boost::mpi::all_reduce(*getVerletList()->getSystem()->comm, e, esum, std::plus<real>());
