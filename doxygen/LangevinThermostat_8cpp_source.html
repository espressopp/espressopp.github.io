128  {

165  void LangevinThermostat::frictionThermo(Particle& p)
166  {
167  real massf = sqrt(p.mass());
168 
169  // get a random value for each vector component
170  Real3D ranval((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
171 
172  p.force() += pref1 * p.velocity() * p.mass() +
173  pref2 * ranval * massf;
174 
175  LOG4ESPP_TRACE(theLogger, "new force of p = " << p.force());
176  }

178  void LangevinThermostat::initialize()
179  { // calculate the prefactors
180 
181  real timestep = integrator->getTimeStep();
182

184  ", gamma = " << gamma <<
185  ", temperature = " << temperature);
186 
187  pref1 = -gamma;
