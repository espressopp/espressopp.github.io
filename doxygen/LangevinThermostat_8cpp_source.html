00174     void LangevinThermostat::frictionThermo(Particle& p)
00175     {
00176       real massf = sqrt(p.mass());
00177 
00178       // get a random value for each vector component
00179 
00180       Real3D ranval((*rng)() - 0.5, (*rng)() - 0.5, (*rng)() - 0.5);
00181 
00182       p.force() += pref1 * p.velocity() * p.mass() +
00183                    pref2 * ranval * massf;
00184 
00185       LOG4ESPP_TRACE(theLogger, "new force of p = " << p.force());
00186     }
