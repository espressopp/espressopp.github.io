107 template<typename _Potential>
108 inline void
109 FixedTripleListTypesInteractionTemplate<_Potential>::addForces() {
110  LOG4ESPP_INFO(theLogger, "add forces computed by the FixedTriple List");
111  const bc::BC &bc = *getSystemRef().bc;
112 
113  for (FixedTripleList::TripleList::Iterator it(*fixedtripleList); it.isValid(); ++it) {
114  Particle &p1 = *it->first;
115  Particle &p2 = *it->second;
116  Particle &p3 = *it->third;
117  int type1 = p1.type();
118  int type2 = p2.type();
119  int type3 = p3.type();
120  const Potential &potential = getPotential(type1, type2, type3);
121 
122  Real3D dist12, dist32;
123  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
124  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
125  Real3D force12, force32;
126  potential._computeForce(force12, force32, dist12, dist32);
127  p1.force() += force12;
128  p2.force() -= force12 + force32;
129  p3.force() += force32;
130  }
131 }

133 template<typename _Potential>
134 inline real
135 FixedTripleListTypesInteractionTemplate<_Potential>::
136 computeEnergy() {
137  LOG4ESPP_INFO(theLogger, "compute energy of the FixedTriple list pairs");
138 
139  real e = 0.0;
140  const bc::BC &bc = *getSystemRef().bc; // boundary conditions
141  for (FixedTripleList::TripleList::Iterator it(*fixedtripleList); it.isValid(); ++it) {
142  const Particle &p1 = *it->first;
143  const Particle &p2 = *it->second;
144  const Particle &p3 = *it->third;
145  int type1 = p1.type();
146  int type2 = p2.type();
147  int type3 = p3.type();
148 
149  const Potential &potential = getPotential(type1, type2, type3);
150 
151  Real3D dist12 = bc.getMinimumImageVector(p1.position(), p2.position());
152  Real3D dist32 = bc.getMinimumImageVector(p3.position(), p2.position());
153  e += potential._computeEnergy(dist12, dist32);
154  LOG4ESPP_TRACE(theLogger, "id1=" << p1.id() << " id2="
155  << p2.id() << " id3=" << p3.id() << " potential energy=" << e);
156  }
157 
158  // reduce over all CPUs
159  real esum;
160  boost::mpi::all_reduce(*mpiWorld, e, esum, std::plus<real>());
161  return esum;
162 }

190 inline real
191 FixedTripleListTypesInteractionTemplate<_Potential>::
192 computeVirial() {
193  LOG4ESPP_INFO(theLogger, "compute the virial for the FixedTriple List with types");
194 
195  real w = 0.0;
196  const bc::BC &bc = *getSystemRef().bc; // boundary conditions
197  for (FixedTripleList::TripleList::Iterator it(*fixedtripleList); it.isValid(); ++it) {
198  const Particle &p1 = *it->first;
199  const Particle &p2 = *it->second;
200  const Particle &p3 = *it->third;
201  int type1 = p1.type();
202  int type2 = p2.type();
203  int type3 = p3.type();
204  const Potential &potential = getPotential(type1, type2, type3);
205 
206  Real3D dist12, dist32;
207  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
208  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
209  Real3D force12, force32;
210  potential._computeForce(force12, force32, dist12, dist32);
211  w += dist12 * force12 + dist32 * force32;
212  }
213 
214  // reduce over all CPUs
215  real wsum;
216  boost::mpi::all_reduce(*mpiWorld, w, wsum, std::plus<real>());
217  return wsum;
