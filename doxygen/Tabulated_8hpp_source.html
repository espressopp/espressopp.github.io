101  bool _computeForceRaw(Real3D& force, const Real3D& dist, real distSqr) const {
102  real ffactor;
103  if (interpolationType!=0){
104  real distrt = sqrt(distSqr);
105  ffactor = table->getForce(distrt);
106  ffactor /= distrt;
107  }
108  else {
109  //throw std::runtime_error("Tabulated potential table not available.");
110  return false;
111  }
112  force = dist * ffactor;
