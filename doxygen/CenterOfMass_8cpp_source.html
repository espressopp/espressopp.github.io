36  Real3D CenterOfMass::computeVector() const {
37 
38  System& system = getSystemRef();
39 
40  // compute the center-of-mass of the system (r_CM = \sum m_i r_i / \sum m_i)
41  real xcom = 0.0;
42  real ycom = 0.0;
43  real zcom = 0.0;
44  real mass = 0.0;
45  real xcom_sum = 0.0;
46  real ycom_sum = 0.0;
47  real zcom_sum = 0.0;
48  real mass_sum = 0.0;
49 
50  CellList realCells = system.storage->getRealCells();
51  for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
52  const Particle& p = *cit;
53  xcom += p.mass() * p.position()[0];
54  ycom += p.mass() * p.position()[1];
55  zcom += p.mass() * p.position()[2];
56  mass += p.mass();
57  }
58 
59  // it was reduce, but we need it for all cpus
60  boost::mpi::all_reduce(*mpiWorld, xcom, xcom_sum, std::plus<real>());
61  boost::mpi::all_reduce(*mpiWorld, ycom, ycom_sum, std::plus<real>());
62  boost::mpi::all_reduce(*mpiWorld, zcom, zcom_sum, std::plus<real>());
63  boost::mpi::all_reduce(*mpiWorld, mass, mass_sum, std::plus<real>());
64 
65  //Real3D force(0.0, 0.0, 0.0);
66  return Real3D(xcom_sum / mass_sum, ycom_sum / mass_sum, zcom_sum / mass_sum);
67  }
