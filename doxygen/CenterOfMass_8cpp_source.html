00036     Real3D CenterOfMass::computeVector() const {
00037 
00038       System& system = getSystemRef();
00039   
00040       // compute the center-of-mass of the system (r_CM = \sum m_i r_i / \sum m_i)
00041       real xcom = 0.0;
00042       real ycom = 0.0;
00043       real zcom = 0.0;
00044       real mass = 0.0;
00045       real xcom_sum = 0.0;
00046       real ycom_sum = 0.0;
00047       real zcom_sum = 0.0;
00048       real mass_sum = 0.0;
00049 
00050       CellList realCells = system.storage->getRealCells();
00051       for (CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00052         const Particle& p = *cit;
00053         xcom += p.mass() * p.position()[0];
00054         ycom += p.mass() * p.position()[1];
00055         zcom += p.mass() * p.position()[2];
00056         mass += p.mass();
00057       }
00058       
00059       // it was reduce, but we need it for all cpus
00060       boost::mpi::all_reduce(*mpiWorld, xcom, xcom_sum, std::plus<real>());
00061       boost::mpi::all_reduce(*mpiWorld, ycom, ycom_sum, std::plus<real>());
00062       boost::mpi::all_reduce(*mpiWorld, zcom, zcom_sum, std::plus<real>());
00063       boost::mpi::all_reduce(*mpiWorld, mass, mass_sum, std::plus<real>());
00064 
00065       //Real3D force(0.0, 0.0, 0.0);
00066       return Real3D(xcom_sum / mass_sum, ycom_sum / mass_sum, zcom_sum / mass_sum);
00067     }
