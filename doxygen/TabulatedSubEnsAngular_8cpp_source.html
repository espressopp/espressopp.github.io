95  void TabulatedSubEnsAngular::computeColVarWeights(
96  const Real3D& dist12, const Real3D& dist32, const bc::BC& bc){
97  // Sanity Check
98  if (weights.getDimension() == 0)
99  throw std::runtime_error("TabulatedSubEnsAngular requires at least one interaction.");
100  // Compute the weights for each force field
101  // given the reference and instantaneous values of ColVars
102  setColVar(dist12, dist32, bc);
103  // Compute weights up to next to last FF
104  real maxWeight = 0.;
105  int maxWeightI = 0;
106  // Check first whether we're stuck in a surface
107  bool stuck = false;
108  for (int i=0; i<numInteractions; ++i) {
109  if (weights[i] > maxWeight) {
110  maxWeight = weights[i];
111  maxWeightI = i;
112  }
113  }
114  // Lock the surface until it's sampled >= 98% of the target prob.
115  if (weightCounts > 0 &&
116  maxWeightI < numInteractions-1 &&
117  maxWeight > 0. &&
118  weightSum[maxWeightI]/weightCounts < 0.98*targetProb[maxWeightI])
119  stuck = true;
120  if (!stuck) {
121  maxWeight = 0.;
122  maxWeightI = numInteractions-1;
123  for (int i=0; i<numInteractions-1; ++i) {
124  weights[i] = 1.;
125  real norm_d_i = 0.;
126  real norm_l_i = 0.;
127  for (int j=0; j<colVar.getDimension(); ++j) {
128  int k = 0;
129  // Choose between angle, bond, and dihed
130  if (j <= 0+colVarAngleListSize) k = 0;
131  else if (j<1+colVarAngleListSize+colVarBondListSize) k = 1;
132  else k = 2;
133  real diff = colVar[j] - colVarRef[i][k];
134  if (k == 2) {
135  if (diff>M_PI) diff -= 2.0*M_PI;
136  if (diff<(-1.0*M_PI)) diff += 2.0*M_PI;
137  }
138  norm_d_i += pow(diff / colVarSd[k], 2);
139  norm_l_i += pow(colVarRef[i][3+k], 2);
140  }
141  if (norm_d_i > norm_l_i)
142  weights[i] = exp(- (sqrt(norm_d_i) - sqrt(norm_l_i)) / alpha);
143  if (weights[i] > maxWeight) {
144  maxWeight = weights[i];
145  maxWeightI = i;
146  }
147  }
148  for (int i=0; i<numInteractions-1; ++i) {
149  if (i != maxWeightI)
150  weights[i] = 0.;
151  else {
152  if (weightCounts > 0 &&
153  weights[i] > 0.01 &&
154  weightSum[i]/weightCounts < 0.98*targetProb[i]) {
155  weights[i] = 1.;
156  maxWeight = 1.;
157  }
158  }
159  }
160  if (maxWeightI == numInteractions-1)
161  maxWeight = 1.;
162  weights[numInteractions-1] = 1. - maxWeight;
163  }
164 
165  // Update weightSum
166  for (int i=0; i<numInteractions; ++i)
167  weightSum[i] += weights[i];
168  weightCounts += 1;
169  }

172  void TabulatedSubEnsAngular::setColVar(const Real3D& dist12,
173  const Real3D& dist32, const bc::BC& bc) {
174  colVar.setDimension(1);
175  real dist12_sqr = dist12 * dist12;
176  real dist32_sqr = dist32 * dist32;
177  real dist1232 = sqrt(dist12_sqr) * sqrt(dist32_sqr);
178  real cos_theta = dist12 * dist32 / dist1232;
179  colVar[0] = acos(cos_theta);
180  int i=1;
181  if (colVarAngleList != nullptr) {
182  colVarAngleListSize = colVarAngleList->size();
183  // Now all angles in colVarAngleList
184  for (FixedTripleList::TripleList::Iterator it(*colVarAngleList); it.isValid(); ++it) {
185  colVar.setDimension(i+1);
186  Particle &p1 = *it->first;
187  Particle &p2 = *it->second;
188  Particle &p3 = *it->third;
189  Real3D dist12, dist32;
190  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
191  bc.getMinimumImageVectorBox(dist32, p3.position(), p2.position());
192  real dist12_sqr = dist12 * dist12;
193  real dist32_sqr = dist32 * dist32;
194  real dist1232 = sqrt(dist12_sqr) * sqrt(dist32_sqr);
195  real cos_theta = dist12 * dist32 / dist1232;
196  colVar[i] = acos(cos_theta);
197  i+=1;
198  }
199  }
200  if (colVarBondList != nullptr) {
201  colVarBondListSize = colVarBondList->size();
202  // Now all bonds in colVarBondList
203  for (FixedPairList::PairList::Iterator it(*colVarBondList); it.isValid(); ++it) {
204  colVar.setDimension(i+1);
205  Particle &p1 = *it->first;
206  Particle &p2 = *it->second;
207  Real3D dist12;
208  bc.getMinimumImageVectorBox(dist12, p1.position(), p2.position());
209  colVar[i] = sqrt(dist12 * dist12);
210  i+=1;
211  }
212  }
213  if (colVarDihedList != nullptr) {
214  colVarDihedListSize = colVarDihedList->size();
215  // Now all dihedrals in colVarDihedList
216  for (FixedQuadrupleList::QuadrupleList::Iterator it(*colVarDihedList); it.isValid(); ++it) {
217  colVar.setDimension(i+1);
218  Particle &p1 = *it->first;
219  Particle &p2 = *it->second;
220  Particle &p3 = *it->third;
221  Particle &p4 = *it->fourth;
222  Real3D r21, r32, r43;
223  bc.getMinimumImageVectorBox(r21, p2.position(), p1.position());
224  bc.getMinimumImageVectorBox(r32, p3.position(), p2.position());
225  bc.getMinimumImageVectorBox(r43, p4.position(), p3.position());
226  colVar[i] = DihedralPotential::computePhi(r21, r32, r43);
227  i+=1;
228  }
229  }
230  }
