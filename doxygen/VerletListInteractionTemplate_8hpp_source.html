242  // local pressure tensor for layer, plane is defined by z coordinate
243  template < typename _Potential > inline void
244  VerletListInteractionTemplate < _Potential >::
245  computeVirialTensor(Tensor& w, real z) {
246  LOG4ESPP_DEBUG(_Potential::theLogger, "loop over verlet list pairs and sum up virial tensor over one z-layer");
247 
248  System& system = verletList->getSystemRef();
249  Real3D Li = system.bc->getBoxL();
250 
251  real rc_cutoff = verletList->getVerletCutoff();
252 
253  // boundaries should be taken into account
254  bool ghost_layer = false;
255  real zghost = -100.0;
256  if(z<rc_cutoff){
257  zghost = z + Li[2];
258  ghost_layer = true;
259  }
260  else if(z>=Li[2]-rc_cutoff){
261  zghost = z - Li[2];
262  ghost_layer = true;
263  }
264 
265  Tensor wlocal(0.0);
266  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
267  Particle &p1 = *it->first;
268  Particle &p2 = *it->second;
269  Real3D p1pos = p1.position();
270  Real3D p2pos = p2.position();
271 
272 
273  if( (p1pos[2]>z && p2pos[2]<z) ||
274  (p1pos[2]<z && p2pos[2]>z) ||
275  (ghost_layer &&
276  ((p1pos[2]>zghost && p2pos[2]<zghost) ||
277  (p1pos[2]<zghost && p2pos[2]>zghost))
278  )
279  ){
280  int type1 = p1.type();
281  int type2 = p2.type();
282  const Potential &potential = getPotential(type1, type2);
283 
284  Real3D force(0.0, 0.0, 0.0);
285  if(potential._computeForce(force, p1, p2)) {
286  Real3D r21 = p1pos - p2pos;
287  wlocal += Tensor(r21, force) / fabs(r21[2]);
288  }
289  }
290  }
291 
292  // reduce over all CPUs
293  Tensor wsum(0.0);
294  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, 6, (double*)&wsum, std::plus<double>());
295  w += wsum;
296  }

298  // it will calculate the pressure in 'n' layers along Z axis
299  // the first layer has coordinate 0.0 the last - (Lz - Lz/n)
300  template < typename _Potential > inline void
301  VerletListInteractionTemplate < _Potential >::
302  computeVirialTensor(Tensor *w, int n) {
303  LOG4ESPP_DEBUG(_Potential::theLogger, "loop over verlet list pairs and sum up virial tensor in bins along z-direction");
304 
305  System& system = verletList->getSystemRef();
306  Real3D Li = system.bc->getBoxL();
307 
308  real z_dist = Li[2] / float(n); // distance between two layers
309  Tensor *wlocal = new Tensor[n];
310  for(int i=0; i<n; i++) wlocal[i] = Tensor(0.0);
311  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
312  Particle &p1 = *it->first;
313  Particle &p2 = *it->second;
314  int type1 = p1.type();
315  int type2 = p2.type();
316  Real3D p1pos = p1.position();
317  Real3D p2pos = p2.position();
318 
319  const Potential &potential = getPotential(type1, type2);
320 
321  Real3D force(0.0, 0.0, 0.0);
322  Tensor ww;
323  if(potential._computeForce(force, p1, p2)) {
324  Real3D r21 = p1pos - p2pos;
325  ww = Tensor(r21, force) / fabs(r21[2]);
326 
327  int position1 = (int)( p1pos[2]/z_dist );
328  int position2 = (int)( p2pos[2]/z_dist );
329 
330  int maxpos = std::max(position1, position2);
331  int minpos = std::min(position1, position2);
332 
333  // boundaries should be taken into account
334  bool boundaries1 = false;
335  bool boundaries2 = false;
336  if(minpos < 0){
337  minpos += n;
338  boundaries1 =true;
339  }
340  if(maxpos >=n){
341  maxpos -= n;
342  boundaries2 =true;
343  }
344 
345  if(boundaries1 || boundaries2){
346  for(int i = 0; i<=maxpos; i++){
347  wlocal[i] += ww;
348  }
349  for(int i = minpos+1; i<n; i++){
350  wlocal[i] += ww;
351  }
352  }
353  else{
354  for(int i = minpos+1; i<=maxpos; i++){
355  wlocal[i] += ww;
356  }
357  }
358  }
359  }
360 
361  // reduce over all CPUs
362  Tensor *wsum = new Tensor[n];
363  boost::mpi::all_reduce(*mpiWorld, (double*)&wlocal, n, (double*)&wsum, std::plus<double>());
364 
365  for(int j=0; j<n; j++){
366  w[j] += wsum[j];
367  }
368 
369  delete [] wsum;
370  delete [] wlocal;
371  }
