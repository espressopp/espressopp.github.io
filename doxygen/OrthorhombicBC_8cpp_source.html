132  void
133  OrthorhombicBC::
134  foldCoordinate(Real3D& pos, Int3D& imageBox, int dir) const {
135  int tmp = static_cast<int>(floor(pos[dir]*invBoxL[dir]));
136 
137  imageBox[dir] += tmp;

139 
140  if(pos[dir] < 0 || pos[dir] >= boxL[dir]) {
