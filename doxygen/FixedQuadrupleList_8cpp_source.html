00096       // Particle does not exist here, return false

00100       if (!p2) {
00101         std::stringstream msg;

00106         std::stringstream msg;

00111         std::stringstream msg;
00112         msg << "quadruple particle p4 " << pid4 << " does not exists here and cannot be added";
00113         err.setException( msg.str() );
00114       }
00115     }
00116     err.checkException();
00117     
00118     if(returnVal){
00119       // add the quadruple locally
00120       this->add(p1, p2, p3, p4);
00121       // ADD THE GLOBAL QUADRUPLET
00122       // see whether the particle already has quadruples
00123       std::pair<GlobalQuadruples::const_iterator,
00124                 GlobalQuadruples::const_iterator> equalRange
00125         = globalQuadruples.equal_range(pid1);

00138         globalQuadruples.insert(equalRange.first,

00161     
00162     std::vector< longint > toSend;
00163     // loop over the particle list
00164     for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
00165       longint pid = pit->id();
00166       
00167       // LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find quadruples");

00172       //printf ("me = %d: send particle with pid %d, has %d global quadruples\n", 
00173                 //mpiWorld->rank(), pid, n);
00174 
00175       if (n > 0) {
00176             std::pair<GlobalQuadruples::const_iterator, GlobalQuadruples::const_iterator> equalRange = globalQuadruples.equal_range(pid);
00177             // first write the pid of this particle
00178             // then the number of partners (n)
00179             // and then the pids of the partners
00180             toSend.reserve(toSend.size()+3*n+1);
00181             toSend.push_back(pid);
00182             toSend.push_back(n);
00183             for (GlobalQuadruples::const_iterator it = equalRange.first; it != equalRange.second; ++it) {
00184               toSend.push_back(it->second.first);
00185               toSend.push_back(it->second.second);
00186               toSend.push_back(it->second.third);
00187           //printf ("send global quadruple: pid %d and partner %d\n", pid, it->second.first);
00188           //printf ("send global quadruple: pid %d and partner %d\n", pid, it->second.second);
00189           //printf ("send global quadruple: pid %d and partner %d\n", pid, it->second.third);
00190         }
00191             // delete all of these quadruples from the global list
00192             globalQuadruples.erase(equalRange.first, equalRange.second);
00193       }
00194     }
00195     // send the list
00196     buf.write(toSend);
