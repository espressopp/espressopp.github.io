94  Real3D retF[4];

96  Real3D rijjk = r21.cross(r32); // [r21 x r32]
97  Real3D rjkkn = r32.cross(r43); // [r32 x r43]
98 
99  real rijjk_sqr = rijjk.sqr();
100  real rjkkn_sqr = rjkkn.sqr();
101 
102  real rijjk_abs = sqrt(rijjk_sqr);
103  real rjkkn_abs = sqrt(rjkkn_sqr);
104 
105  real inv_rijjk = 1.0 / rijjk_abs;
106  real inv_rjkkn = 1.0 / rjkkn_abs;
107 
108  // cosine between planes
109  real cos_phi = (rijjk * rjkkn) * (inv_rijjk * inv_rjkkn);
110  if (cos_phi > 1.0) cos_phi = 1.0;
111  else if (cos_phi < -1.0) cos_phi = -1.0;
112 
113  real coef1 = (-2.0) * K * (cos_phi - cos_phi0);
114 
115  real A1 = inv_rijjk * inv_rjkkn;
116  real A2 = inv_rijjk * inv_rijjk;
117  real A3 = inv_rjkkn * inv_rjkkn;
118 
119  Real3D p3232 ( prod(r32,r32) );
120  Real3D p3243 ( prod(r32,r43) );
121  Real3D p2132 ( prod(r21,r32) );
122  Real3D p2143 ( prod(r21,r43) );
123  Real3D p2121 ( prod(r21,r21) );
124  Real3D p4343 ( prod(r43,r43) );
125 
126  // we have 4 particles 1,2,3,4
127  for(int l=0; l<4; l++){
128  Real3D B1, B2, B3;
129 
130  for(int i=0; i<3; i++){
131  B1[i]= r21[i] * ( p3232[i] * (d(l,2)-d(l,3)) + p3243[i] * (d(l,2)-d(l,1)) ) +
132  r32[i] * ( p2132[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,0)) ) +
133  r43[i] * ( p2132[i] * (d(l,2)-d(l,1)) + p3232[i] * (d(l,0)-d(l,1)) ) +
134  2.0 * r32[i] * p2143[i] * (d(l,1)-d(l,2));
135 
136  B2[i]= 2.0 * r21[i] * ( p3232[i] * (d(l,1)-d(l,0)) + p2132[i] * (d(l,1)-d(l,2)) ) +
137  2.0 * r32[i] * ( p2121[i] * (d(l,2)-d(l,1)) + p2132[i] * (d(l,0)-d(l,1)));
138 
139  B3[i]= 2.0 * r43[i] * ( p3232[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,2)) ) +
140  2.0 * r32[i] * ( p4343[i] * (d(l,2)-d(l,1)) + p3243[i] * (d(l,2)-d(l,3)));
141  }
142 
143  retF[l] = coef1 * ( A1*B1 - 0.5*cos_phi*(A2*B2 + A3*B3) );
144  }
145 
146  force1 = retF[0];
147  force2 = retF[1];
148  force3 = retF[2];
149  force4 = retF[3];
150  }
