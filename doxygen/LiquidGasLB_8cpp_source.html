378  void LiquidGasLB::collideStream () {
379  for (int i = 0; i < getNi().getItem(0); i++) {
380  for (int j = 0; j < getNi().getItem(1); j++) {
381  for (int k = 0; k < getNi().getItem(2); k++) {
382  /* collision phase */
383  lbfluid[i][j][k].calcLocalMoments ();
384  lbfluid[i][j][k].calcEqMoments (getExtForceFlag());
385  lbfluid[i][j][k].relaxMoments (numVels);
386  if (getLBTempFlag() == 1) {
387  lbfluid[i][j][k].thermalFluct (numVels);
388  }
389  if (getExtForceFlag() == 1) {
390  lbfluid[i][j][k].applyForces (numVels);
391  }
392  lbfluid[i][j][k].btranMomToPop (numVels);
393 
394  /* streaming phase */
395  if (getNi().getItem(0) > 1 && getNi().getItem(1) > 1 && getNi().getItem(2) > 1) {
396  streaming (i,j,k);
397  }
398  }
399  }
400  }
401 
402  /* swap pointers for two lattices */
403  for (int i = 0; i < getNi().getItem(0); i++) {
404  for (int j = 0; j < getNi().getItem(1); j++) {
405  for (int k = 0; k < getNi().getItem(2); k++) {
406  for (int l = 0; l < numVels; l++) {
407  real tmp;
408  tmp = lbfluid[i][j][k].getF_i(l);
409  lbfluid[i][j][k].setF_i(l, ghostlat[i][j][k].getPop_i(l));
410  ghostlat[i][j][k].setPop_i(l, tmp);
411  }
412 
413  /* some sanity checks */
414  if (getStepNum() % 50 == 0 || getStepNum() == 1) {
415  computeDensity (i, j, k, getNumVels(), getStepNum());
416  } else {
417  }
418  }
419  }
420  }
421  }

423  /* STREAMING ALONG THE VELOCITY VECTORS */
424  void LiquidGasLB::streaming(int _i, int _j, int _k) {
425  int _numVels;
426  int _ip, _im, _jp, _jm, _kp, _km;
427  int dir = 0;
428 
429  _numVels = getNumVels();
430 
431  /* periodic boundaries */
432  // assign iterations
433  _ip = _i + 1; _im = _i - 1;
434  _jp = _j + 1; _jm = _j - 1;
435  _kp = _k + 1; _km = _k - 1;
436 
437  // handle iterations if the site is on the "left" border of the domain
438  if (_i == 0) _im = getNi().getItem(0) - 1;
439  if (_j == 0) _jm = getNi().getItem(1) - 1;
440  if (_k == 0) _km = getNi().getItem(2) - 1;
441 
442  // handle iterations if the site is on the "right" border of the domain
443  if (_i == getNi().getItem(0) - 1) _ip = 0;
444  if (_j == getNi().getItem(1) - 1) _jp = 0;
445  if (_k == getNi().getItem(2) - 1) _kp = 0;
446 
447  /* streaming itself */
448  // do not move the staying populations
449  ghostlat[_i][_j][_k].setPop_i(0,lbfluid[_i][_j][_k].getF_i(0));
450 
451  // move populations to the nearest neighbors
452  ghostlat[_ip][_j][_k].setPop_i(1,lbfluid[_i][_j][_k].getF_i(1));
453  ghostlat[_im][_j][_k].setPop_i(2,lbfluid[_i][_j][_k].getF_i(2));
454  ghostlat[_i][_jp][_k].setPop_i(3,lbfluid[_i][_j][_k].getF_i(3));
455  ghostlat[_i][_jm][_k].setPop_i(4,lbfluid[_i][_j][_k].getF_i(4));
456  ghostlat[_i][_j][_kp].setPop_i(5,lbfluid[_i][_j][_k].getF_i(5));
457  ghostlat[_i][_j][_km].setPop_i(6,lbfluid[_i][_j][_k].getF_i(6));

475  void LiquidGasLB::computeDensity (int _i, int _j, int _k, int _numVels, int _step) {
476  real denLoc = 0.;
477  real jzLoc = 0.;
478 
479  for (int l = 0; l < _numVels; l++) {
480  denLoc += lbfluid[_i][_j][_k].getF_i(l);
481  jzLoc += lbfluid[_i][_j][_k].getF_i(l) * getCi(l).getItem(2);
482  }
483 
484  /* check velocity fluctuations at the lattice sites */
485  if (getStepNum() >= 500) {
486  int _nBins = getNBins(); // number of histogram bins
487  real _velRange = 0.4; // range of velocities fluctuations around eq.value
488  real _deltaV = _velRange / _nBins; // the histogram step
489  real _velZ; // z-comp of the velocity shifted into positive side
490  _velZ = jzLoc / denLoc + 0.5 * _velRange;
491  distr[(int)(_velZ / _deltaV)] += 1.;
492 
493  if (_step == 1000 && (_i + _j + _k) == (getNi().getItem(0) + getNi().getItem(1) + getNi().getItem(2) - 3)) {
494  real histSum = 0.;
495  for (int i = 0; i < _nBins; i++) histSum += distr[i];
496  for (int i = 0; i < _nBins; i++) distr[i] /= (histSum * _deltaV);
497  FILE * rngFile;
498  rngFile = fopen ("vel_dist.dat","a");
499  for (int i = 0; i < _nBins; i++) {
500  fprintf (rngFile, "%8.4f %8.4f \n", (i + 0.5) * _deltaV - 0.5 * _velRange, distr[i]);
501  // "+ 0.5" shifts the value of the distribution into the middle of the bin.
502  // Otherwise, the distribution values are plotted at where the bin starts!
503  }
504  fclose (rngFile);
505  } else {
506  }
