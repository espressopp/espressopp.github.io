40  template <class T> void init_pb(Py_buffer *pb, int ndim, int *shape)
41  {
42  T dum=0;
43  // Setting basic variable of the Py_buffer
44  pb->suboffsets = NULL;
45  pb->internal = NULL;
46  pb->obj = NULL;
47  pb->readonly = 1;
48  pb->ndim = ndim;
49  // The format is computed as a function of the template type T
50  pb->format = (char*)malloc(2*sizeof(const char));
51  strcpy(pb->format, get_format(dum));
52  // Allocation and setting of the shape, stride and len
53  pb->shape = (Py_ssize_t *)malloc(pb->ndim*sizeof(Py_ssize_t));
54  pb->strides = NULL;
55  pb->itemsize = sizeof(T);
56  int old_len = pb->len;
57  pb->len = pb->itemsize;
58  for (int i=0; i<pb->ndim; i++) {
59  pb->shape[i] = shape[i];
60  pb->len *= pb->shape[i];
61  }
62  pb->buf = malloc(pb->len);
63  }

119  void DumpH5MD::update() {
120  System& system = getSystemRef();
121  int shape[2];
122 
123  clear_buffers();
124 
125  NLocal = system.storage->getNRealParticles();
126 
127  shape[0] = NLocal;
128  shape[1] = 3;
129 
130  if (store_position) {
131  init_pb<real>(&position, 2, shape);
132  init_pb<longint>(&image, 2, shape);
133  }
134  init_pb<longint>(&id, 1, shape);
135  if (store_mass) init_pb<real>(&mass, 1, shape);
136  if (store_species) init_pb<int>(&species, 1, shape);
137  if (store_state) init_pb<int>(&state, 1, shape);
138  if (store_velocity) init_pb<real>(&velocity, 2, shape);
139  if (store_force) init_pb<real>(&force, 2, shape);
140  if (store_charge) init_pb<real>(&charge, 1, shape);
141  if (store_lambda) init_pb<real>(&lambda, 1, shape);
142  if (store_res_id) init_pb<int>(&res_id, 1, shape);
143  cleared = false;
144 
145  CellList realCells = system.storage->getRealCells();
146 
147  int i = 0;
148  if (is_adress_) {
149  shared_ptr<FixedTupleListAdress> ftpl = system.storage->getFixedTuples();
150  FixedTupleListAdress::iterator it3;
151  for (iterator::CellListIterator cit(realCells); !cit.isDone(); ++cit) {
152  it3 = ftpl->find(&(*cit));
153  if (it3 != ftpl->end()) {
154  std::vector<Particle*> atList = it3->second;
155  for (std::vector<Particle*>::iterator itv = atList.begin();
156  itv != atList.end(); ++itv) {
157  Particle &at = **itv;
158  if (store_position) {
159  Real3D &tmpPos = at.position();
160  Int3D &tmpImage = at.image();
161  ((real *) position.buf)[3*i] = tmpPos[0];
162  ((real *) position.buf)[3*i+1] = tmpPos[1];
163  ((real *) position.buf)[3*i+2] = tmpPos[2];
164  // Store image
165  ((longint *) image.buf)[3*i] = tmpImage[0];
166  ((longint *) image.buf)[3*i+1] = tmpImage[1];
167  ((longint *) image.buf)[3*i+2] = tmpImage[2];
168  }
169 
170  ((longint *) id.buf)[i] = at.id();
171  if (store_mass) ((real *) mass.buf)[i] = at.mass();
172  if (store_species) ((int *) species.buf)[i] = at.type();
173  if (store_state) ((int *) state.buf)[i] = at.state();
174  if (store_velocity) {
175  Real3D &tmpVel = at.velocity();
176  ((real *) velocity.buf)[3*i] = tmpVel[0];
177  ((real *) velocity.buf)[3*i+1] = tmpVel[1];
178  ((real *) velocity.buf)[3*i+2] = tmpVel[2];
179  }
180  if (store_force) {
181  Real3D &tmpF = at.force();
182  ((real *) force.buf)[3*i] = tmpF[0];
183  ((real *) force.buf)[3*i+1] = tmpF[1];
184  ((real *) force.buf)[3*i+2] = tmpF[2];
185  }
186  if (store_charge) ((real*) charge.buf)[i] = at.q();
187  if (store_lambda) ((real*) lambda.buf)[i] = at.lambda();
188  if (store_res_id) ((int*) res_id.buf)[i] = at.res_id();
189  i++;
190  }
191  }
192  }
193  } else {
194  for(iterator::CellListIterator cit(realCells); !cit.isDone(); ++cit) {
195  if (store_position) {
196  Real3D &tmpPos = cit->position();
197  Int3D &tmpImage = cit->image();
198  ((real *) position.buf)[3*i] = tmpPos[0];
199  ((real *) position.buf)[3*i+1] = tmpPos[1];
200  ((real *) position.buf)[3*i+2] = tmpPos[2];
201  // Store image
202  ((longint *) image.buf)[3*i] = tmpImage[0];
203  ((longint *) image.buf)[3*i+1] = tmpImage[1];
204  ((longint *) image.buf)[3*i+2] = tmpImage[2];
205  }
206  ((longint *) id.buf)[i] = cit->getId();
207  if (store_mass) ((real *) mass.buf)[i] = cit->getMass();
208  if (store_species) ((int *) species.buf)[i] = cit->getType();
209  if (store_state) ((int *) state.buf)[i] = cit->getState();
210  if (store_velocity) {
211  Real3D &tmpVel = cit->velocity();
212  ((real *) velocity.buf)[3*i] = tmpVel[0];
213  ((real *) velocity.buf)[3*i+1] = tmpVel[1];
214  ((real *) velocity.buf)[3*i+2] = tmpVel[2];
215  }
216  if (store_force) {
217  Real3D &tmpF = cit->force();
218  ((real *) force.buf)[3*i] = tmpF[0];
219  ((real *) force.buf)[3*i+1] = tmpF[1];
220  ((real *) force.buf)[3*i+2] = tmpF[2];
221  }
222  if (store_charge) ((real*) charge.buf)[i] = cit->q();
223  if (store_lambda) ((real*) lambda.buf)[i] = cit->lambda();
224  if (store_res_id) ((int*) res_id.buf)[i] = cit->res_id();
225  i++;
226  }
227  }
228  }
