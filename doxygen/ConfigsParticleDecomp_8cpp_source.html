69  void ConfigsParticleDecomp::gather() {
70  System& system = getSystemRef();
71  esutil::Error err(system.comm);
72 
73  int nprocs = system.comm->size();
74  int myrank = system.comm->rank();
75 
76  int localN = system.storage->getNRealParticles();
77 
78  int curNumP = 0;
79  boost::mpi::all_reduce(*system.comm, localN, curNumP, std::plus<int>());
80  if(myrank==0){
81  // check whether the number of particles is the same during the gathering
82  if( curNumP != num_of_part ){
83  stringstream msg;
84  msg<<" ConfigsParticleDecomp gathers the configurations of the same system\n"
85  " with the same number of particles. If you need to store the systems\n"
86  " with different number of particles you should use something else."
87  " E.g `Configurations`";
88  err.setException( msg.str() );
89  }
90  }
91 
92  ConfigurationPtr config = make_shared<Configuration> ();
93  for (int rank_i=0; rank_i<nprocs; rank_i++) {
94  map< size_t, Real3D > conf;
95  if (rank_i == myrank) {
96  CellList realCells = system.storage->getRealCells();
97  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
98  int id = cit->id();
99  Real3D property = Real3D(0,0,0);
100  if(key=="position")
101  property = cit->position();
102  else if(key=="velocity")
103  property = cit->velocity();
104  else if(key=="unfolded"){
105  Real3D& pos = cit->position();
106  Int3D& img = cit->image();
107  Real3D Li = system.bc->getBoxL();
108  for (int i = 0; i < 3; ++i) property[i] = pos[i] + img[i] * Li[i];
109  }
110  else{
111  stringstream msg;
112  msg<<"Error. Key "<<key<<" is unknown. Use position, unfolded or"
113  " velocity.";
114  err.setException( msg.str() );
115  }
116 
117  conf[id] = property;
118  }
119  }
120 
121  boost::mpi::broadcast(*system.comm, conf, rank_i);
122 
123  for (map<size_t,Real3D>::iterator itr=conf.begin(); itr != conf.end(); ++itr) {
124  size_t id = itr->first;
125  Real3D p = itr->second;
126  if(idToCpu[id]==myrank) config->set(id, p[0], p[1], p[2]);
127  }
128  }
129 
130  pushConfig(config);
131  }

133  void ConfigsParticleDecomp::gatherFromFile(string filename) {
134  System& system = getSystemRef();
135  esutil::Error err(system.comm);
136 
137  int nprocs = system.comm->size();
138  int myrank = system.comm->rank();
139 
140  int localN = system.storage->getNRealParticles();
141 
142  ConfigurationPtr config = make_shared<Configuration> ();
143  map< size_t, Real3D > conf;
144 
145  if (myrank==0) {
146  int id, type;
147  real xpos, ypos, zpos;
148  string line;
149  ifstream file(filename.c_str());
150  if (file.is_open()) {
151  // skip first 2 lines
152  getline(file, line);
153  getline(file, line);
154  int count = 0;
155  while (getline(file, line)) {
156  stringstream sl(line);
157  sl >> id;
158  sl >> type;
159  sl >> xpos;
160  sl >> ypos;
161  sl >> zpos;
162  // cout << id << ":" << x << "," << y << "," << z << endl;
163  conf[id] = Real3D(xpos, ypos, zpos);
164  count++;
165  }
166  file.close();
167  cout << "read " << count << " particles from file " << filename << endl;
168  if (count != num_of_part) {
169  stringstream msg;
170  msg << "Number of read particles does not match the number of particles of the system (which is " << num_of_part << ")";
171  err.setException( msg.str() );
172  }
173  } else {
174  stringstream msg;
175  msg << "Unable to open file " << filename;
176  err.setException( msg.str() );
177  }
178  }
179 
180  boost::mpi::broadcast(*system.comm, conf, 0);
181 
182  for (map<size_t,Real3D>::iterator itr=conf.begin(); itr != conf.end(); ++itr) {
183  size_t id = itr->first;
184  Real3D p = itr->second;
185  if(idToCpu[id]==myrank) config->set(id, p[0], p[1], p[2]);
186  }
187  pushConfig(config);
188  }
