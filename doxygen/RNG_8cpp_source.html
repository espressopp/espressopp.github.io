86  void RNG::saveState(long long step) {
87  std::ostringstream oss;
88  oss << *boostRNG;
89  if ( mpiWorld->rank()==0 ) {
90  std::vector<std::string> oss_str_vec;
91  boost::mpi::gather(*mpiWorld, oss.str(), oss_str_vec, 0);
92  std::string outfn = "rng." + std::to_string(step);
93  std::ofstream ofs(outfn, std::ios::out | std::ios::binary);
94  if (ofs.is_open()) {
95  for(unsigned int i = 0;i<oss_str_vec.size();++i) {
96  ofs << oss_str_vec[i] << std::endl;
97  }
98  }
99  ofs.close();
100  } else {
101  boost::mpi::gather(*mpiWorld, oss.str(), 0);
102  }
103  }

105  void RNG::loadState(const char* fname) {
106  std::ifstream ifs;
107  std::string rng_str;
108  std::vector<std::string> rng_str_vec;
109  unsigned int rng_size;
110  if ( mpiWorld->rank()==0 ) {
111  ifs.open(fname, std::ios::in | std::ios::binary);
112  std::string token;
113  while(std::getline(ifs, token)) {
114  rng_str_vec.push_back(token);
115  token.clear();
116  }
117  rng_size = rng_str_vec.size();
118  if (ifs.is_open()) ifs.close();
119  }
120  boost::mpi::broadcast(*mpiWorld, rng_size, 0);
121  if (rng_size != static_cast<unsigned int>(mpiWorld->size())) {
122  std::cerr << "# Warning: The number of processes does not match the number of saved random number generators." << std::endl << " The old states are not restored.\n";
123  return;
124  }
125  boost::mpi::scatter(*mpiWorld, rng_str_vec, rng_str, 0);
126  std::istringstream iss(rng_str);
127  iss >> *boostRNG;
128  }
