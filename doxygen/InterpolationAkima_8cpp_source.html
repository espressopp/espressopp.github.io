100  // read file, store values in r, e, f, and return number of read lines (called by readRaw())
101  int InterpolationAkima::readFile(const char* file, bool dummy) {
102 
103  char line[MAXLINE];
104  real r, e, f;
105  FILE *fp = fopen(file, "r");
106  if (fp == NULL) {
107  LOG4ESPP_ERROR(theLogger, "could not open file " << file);
108  return 0;
109  }
110 
111  int N = 0;
112 
113  while (1) {
114  if (fgets(line, MAXLINE, fp) == NULL) break;
115  int k = sscanf(line, "%lg %lg %lg", &r, &e, &f);
116  if (k < 3) continue; // do not accept this line
117  if (!dummy) {
118  radius[N] = r;
119  energy[N] = e;
120  force[N] = f;
121  }
122  N++; // increment counter for table entries
123  }
124  fclose(fp);
125 
126  if (dummy) {
127  LOG4ESPP_INFO(theLogger, "found " << N << " table entries file " << file);
128  } else {
129  LOG4ESPP_INFO(theLogger, "read " << N << " table entries in file " << file);
130  }
131 
132  return N;
133  }// readFile

136  // Akima spline method adapted from The VOTCA project (http://www.votca.org)
137  void InterpolationAkima::spline(const real* x, const real* y, int N,
138  real* p0, real* p1, real* p2, real* p3) {
139 
140  real m1, m2, m3, m4, m5;
141  real temp, g0, g1, g2, x1, x2, y1, y2, x4, x5, y4, y5;
142  real* t = new real[N];
143 
144  // Akima method:
145  // estimation of two more points on each side using a degree two polyomial.
146  // Resulting slopes t[0), t[1), t[N-2), t[N-1) are directly calculated
147 
148  // left side: t[0), t[1)
149  temp = (x[1]-x[0])/(x[2]-x[0]);
150  temp = temp*temp;
151  g0 = y[0];
152  g1 = ( (y[1]-y[0]) - temp*(y[2]-y[0]) ) / ( (x[1]-x[0]) - temp*(x[2]-x[0]) );
153  g2 = ( (y[2]-y[0]) - g1*(x[2]-x[0]) ) / ( (x[2]-x[0])*(x[2]-x[0]) );
154  x1 = x[0] - (x[2]-x[0]);
155  x2 = x[1] - (x[2]-x[0]);
156  y1 = g0 + g1*(x1-x[0]) + g2*(x1-x[0])*(x1-x[0]);
157  y2 = g0 + g1*(x2-x[0]) + g2*(x2-x[0])*(x2-x[0]);
158  m1 = (y2-y1)/(x2-x1);
159  m2 = (y[0]-y2)/(x[0]-x2);
160  m3 = (y[1]-y[0])/(x[1]-x[0]);
161  m4 = (y[2]-y[1])/(x[2]-x[1]);
162  t[0] = getSlope(m1,m2,m3,m4);
163  m5 = (y[3]-y[2])/(x[3]-x[2]);
164  t[1] = getSlope(m2,m3,m4,m5);
165 
166  // right side: t[N-2), t[N-1)
167  temp = (x[N-2]-x[N-1])/(x[N-3]-x[N-1]);
168  temp = temp*temp;
169  g0 = y[N-1];
170  g1 = ( (y[N-2]-y[N-1]) - temp*(y[N-3]-y[N-1]) ) / ( (x[N-2]-x[N-1]) - temp*(x[N-3]-x[N-1]) );
171  g2 = ( (y[N-3]-y[N-1]) - g1*(x[N-3]-x[N-1]) ) / ( (x[N-3]-x[N-1])*(x[N-3]-x[N-1]) );
172  x4 = x[N-2] + (x[N-1]-x[N-3]);
173  x5 = x[N-1] + (x[N-1]-x[N-3]);
174  y4 = g0 + g1*(x4-x[N-1]) + g2*(x4-x[N-1])*(x4-x[N-1]);
175  y5 = g0 + g1*(x5-x[N-1]) + g2*(x5-x[N-1])*(x5-x[N-1]);
176  m1 = (y[N-3]-y[N-4])/(x[N-3]-x[N-4]);
177  m2 = (y[N-2]-y[N-3])/(x[N-2]-x[N-3]);
178  m3 = (y[N-1]-y[N-2])/(x[N-1]-x[N-2]);
179  m4 = (y4-y[N-1])/(x4-x[N-1]);
180  m5 = (y5-y4)/(x5-x4);
181  t[N-2] = getSlope(m1,m2,m3,m4);
182  t[N-1] = getSlope(m2,m3,m4,m5);
183 
184  // calculate t's for all inner points [2,N-3]
185  for (int i = 2; i < N-2; i++) {
186  m1 = (y[i-1]-y[i-2])/(x[i-1]-x[i-2]);
187  m2 = (y[i]-y[i-1])/(x[i]-x[i-1]);
188  m3 = (y[i+1]-y[i])/(x[i+1]-x[i]);
189  m4 = (y[i+2]-y[i+1])/(x[i+2]-x[i+1]);
190  t[i] = getSlope(m1,m2,m3,m4);
191  }
192 
193  // calculate p0,p1,p2,p3 for all intervals 0..(N-2), where interval
194  // [x[i),x[i+1)] shall have number i (this means that the last interval
195  // has number N-2)
196  for (int i = 0; i < N-1; i++) {
197  p0[i] = y[i];
198  p1[i] = t[i];
199  p2[i] = ( 3.0*(y[i+1]-y[i])/(x[i+1]-x[i]) - 2.0*t[i] - t[i+1] ) / (x[i+1]-x[i]);
200  p3[i] = ( t[i] + t[i+1] - 2.0*(y[i+1]-y[i])/(x[i+1]-x[i]) ) / ( (x[i+1]-x[i])*(x[i+1]-x[i]) );
201  }
202 
203  delete [] t;
204  }// spline
