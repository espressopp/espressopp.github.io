00136     // Akima spline method adapted from The VOTCA project (http://www.votca.org)
00137     void InterpolationAkima::spline(const real* x, const real* y, int N,
00138                                     real* p0, real* p1, real* p2, real* p3) {
00139         
00140         real m1, m2, m3, m4, m5;
00141         real temp, g0, g1, g2, x1, x2, y1, y2, x4, x5, y4, y5;
00142         real* t = new real[N];
00143         
00144         // Akima method:
00145         //  estimation of two more points on each side using a degree two polyomial.
00146         // Resulting slopes t[0), t[1), t[N-2), t[N-1) are directly calculated
00147 
00148         // left side: t[0), t[1)
00149         temp = (x[1]-x[0])/(x[2]-x[0]);
00150         temp = temp*temp;
00151         g0 = y[0];
00152         g1 = ( (y[1]-y[0]) - temp*(y[2]-y[0]) ) / ( (x[1]-x[0]) - temp*(x[2]-x[0]) );
00153         g2 = ( (y[2]-y[0]) - g1*(x[2]-x[0]) ) / ( (x[2]-x[0])*(x[2]-x[0]) );
00154         x1 = x[0] - (x[2]-x[0]);
00155         x2 = x[1] - (x[2]-x[0]);
00156         y1 = g0 + g1*(x1-x[0]) + g2*(x1-x[0])*(x1-x[0]);
00157         y2 = g0 + g1*(x2-x[0]) + g2*(x2-x[0])*(x2-x[0]);
00158         m1 = (y2-y1)/(x2-x1);
00159         m2 = (y[0]-y2)/(x[0]-x2);
00160         m3 = (y[1]-y[0])/(x[1]-x[0]);
00161         m4 = (y[2]-y[1])/(x[2]-x[1]);
00162         t[0] = getSlope(m1,m2,m3,m4);
00163         m5 = (y[3]-y[2])/(x[3]-x[2]);
00164         t[1] = getSlope(m2,m3,m4,m5);
00165 
00166         // right side: t[N-2), t[N-1)
00167         temp = (x[N-2]-x[N-1])/(x[N-3]-x[N-1]);
00168         temp = temp*temp;
00169         g0 = y[N-1];
00170         g1 = ( (y[N-2]-y[N-1]) - temp*(y[N-3]-y[N-1]) ) / ( (x[N-2]-x[N-1]) - temp*(x[N-3]-x[N-1]) );
00171         g2 = ( (y[N-3]-y[N-1]) - g1*(x[N-3]-x[N-1]) ) / ( (x[N-3]-x[N-1])*(x[N-3]-x[N-1]) );
00172         x4 = x[N-2] + (x[N-1]-x[N-3]);
00173         x5 = x[N-1] + (x[N-1]-x[N-3]);
00174         y4 = g0 + g1*(x4-x[N-1]) + g2*(x4-x[N-1])*(x4-x[N-1]);
00175         y5 = g0 + g1*(x5-x[N-1]) + g2*(x5-x[N-1])*(x5-x[N-1]);
00176         m1 = (y[N-3]-y[N-4])/(x[N-3]-x[N-4]);
00177         m2 = (y[N-2]-y[N-3])/(x[N-2]-x[N-3]);
00178         m3 = (y[N-1]-y[N-2])/(x[N-1]-x[N-2]);
00179         m4 = (y4-y[N-1])/(x4-x[N-1]);
00180         m5 = (y5-y4)/(x5-x4);
00181         t[N-2] = getSlope(m1,m2,m3,m4);
00182         t[N-1] = getSlope(m2,m3,m4,m5);
00183         
00184         // calculate t's for all inner points [2,N-3]
00185         for (int i = 2; i < N-2; i++) {
00186             m1 = (y[i-1]-y[i-2])/(x[i-1]-x[i-2]);
00187             m2 = (y[i]-y[i-1])/(x[i]-x[i-1]);
00188             m3 = (y[i+1]-y[i])/(x[i+1]-x[i]);
00189             m4 = (y[i+2]-y[i+1])/(x[i+2]-x[i+1]);
00190             t[i] = getSlope(m1,m2,m3,m4);
00191         }
00192         
00193         // calculate p0,p1,p2,p3 for all intervals 0..(N-2), where interval
00194         // [x[i),x[i+1)] shall have number i (this means that the last interval
00195         // has number N-2)
00196         for (int i = 0; i < N-1; i++) {
00197             p0[i] = y[i];
00198             p1[i] = t[i];
00199             p2[i] = ( 3.0*(y[i+1]-y[i])/(x[i+1]-x[i]) - 2.0*t[i] - t[i+1] ) / (x[i+1]-x[i]);
00200             p3[i] = ( t[i] + t[i+1] - 2.0*(y[i+1]-y[i])/(x[i+1]-x[i]) ) / ( (x[i+1]-x[i])*(x[i+1]-x[i]) );
00201         }
00202      
00203         delete [] t;
00204     }// spline
