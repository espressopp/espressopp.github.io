39  // cut is a cutoff (without skin)
40  VerletListTriple::VerletListTriple(shared_ptr<System> system, real _cut, bool rebuildVL):SystemAccess(system){
41  LOG4ESPP_INFO(theLogger, "construct VerletListTriple, cut = " << _cut);
42 
43  if (!system->storage) {
44  throw std::runtime_error("system has no storage");
45  }
46 
47  cut = _cut;

56  boost::bind(&VerletListTriple::rebuild, this));

76  void VerletListTriple::rebuild(){
77  cutVerlet = cut + getSystem() -> getSkin();
78  cutsq = cutVerlet * cutVerlet;
79 
80  vlTriples.clear();
81 
82  // add particles to adress zone
83  CellList cl = getSystem()->storage->getRealCells();
84  LOG4ESPP_DEBUG(theLogger, "local cell list size = " << cl.size());
85  for (CellListAllTriplesIterator it(cl); it.isValid(); ++it) {
86  checkTriple(*it->first, *it->second, *it->third);
87  }
88 
89  builds++;
90  LOG4ESPP_DEBUG(theLogger, "rebuilt VerletList (count=" << builds << "), cutsq = " << cutsq
91  << " local size = " << vlTriples.size());
92  }

97  void VerletListTriple::checkTriple(Particle& pt1, Particle& pt2, Particle& pt3){
98  // check if central particle is in the exclusion list
99  if (exList.count(pt2.id()) > 0) return;
100 
101  Real3D d1 = pt1.position() - pt2.position();
102  Real3D d2 = pt2.position() - pt3.position();
103 
104  real distsq1 = d1.sqr();
105  real distsq2 = d2.sqr();
106 
107  LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
108  << " @ " << pt1.position()
109  << " - p2: " << pt2.id() << " @ " << pt2.position()
110  << " - p3: " << pt3.id() << " @ " << pt3.position()
111  << " -> distsq1 = " << distsq1
112  << " -> distsq2 = " << distsq2);
113 
114  if (distsq1>cutsq || distsq2>cutsq) return;
115 
116  vlTriples.add(pt1, pt2, pt3); // add triple to Verlet List
117  }
