00076   void VerletListTriple::rebuild(){
00077     cutVerlet = cut + getSystem() -> getSkin();
00078     cutsq = cutVerlet * cutVerlet;
00079     
00080     vlTriples.clear();
00081 
00082     // add particles to adress zone
00083     CellList cl = getSystem()->storage->getRealCells();
00084     LOG4ESPP_DEBUG(theLogger, "local cell list size = " << cl.size());
00085     for (CellListAllTriplesIterator it(cl); it.isValid(); ++it) {
00086       checkTriple(*it->first, *it->second, *it->third);
00087     }
00088 
00089     builds++;
00090     LOG4ESPP_DEBUG(theLogger, "rebuilt VerletList (count=" << builds << "), cutsq = " << cutsq
00091                  << " local size = " << vlTriples.size());
00092   }

00097   void VerletListTriple::checkTriple(Particle& pt1, Particle& pt2, Particle& pt3){
00098     // check if central particle is in the exclusion list
00099     if (exList.count(pt2.id()) > 0) return;
00100     
00101     Real3D d1 = pt1.position() - pt2.position();
00102     Real3D d2 = pt2.position() - pt3.position();
00103     
00104     real distsq1 = d1.sqr();
00105     real distsq2 = d2.sqr();
00106 
00107     LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
00108                    << " @ " << pt1.position() 
00109                    << " - p2: " << pt2.id() << " @ " << pt2.position()
00110                    << " - p3: " << pt3.id() << " @ " << pt3.position()
00111                    << " -> distsq1 = " << distsq1
00112                    << " -> distsq2 = " << distsq2);
00113 
00114     if (distsq1>cutsq || distsq2>cutsq) return;
00115     
00116     vlTriples.add(pt1, pt2, pt3); // add triple to Verlet List
00117   }
