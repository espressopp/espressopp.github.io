58  bool FixedPairDistList::
59  add(longint pid1, longint pid2) {
60  bool returnVal = true;
61  System& system = storage->getSystemRef();
62  esutil::Error err(system.comm);
63 
64  if (pid1 > pid2)
65  std::swap(pid1, pid2);
66 
67  // ADD THE LOCAL PAIR
68  Particle *p1 = storage->lookupRealParticle(pid1);
69  Particle *p2 = storage->lookupLocalParticle(pid2);
70  if (!p1){
71  // Particle does not exist here, return false
72  returnVal = false;
73  }
74  else{
75  if (!p2) {
76  std::stringstream msg;
77  msg << "bond particle p2 " << pid2 << " does not exists here and cannot be added";
78  //std::runtime_error(err.str());
79  err.setException( msg.str() );
80  }
81  }
82  err.checkException();
83 
84  if(returnVal){
85  // add the pair locally
86  this->add(p1, p2);
87 
88  Real3D pos1 = p1->position();
89  Real3D pos2 = p2->position();
90 
91  Real3D r21(0.0,0.0,0.0);
92  system.bc->getMinimumImageVectorBox(r21, pos1, pos2);
93 
94  // add the particle pair to the globalPairs list
95  pairsDist.insert(std::make_pair(pid1, std::make_pair(pid2, r21.abs()) ));
96  }
97 
98  LOG4ESPP_INFO(theLogger, "added fixed pair to global pair list");
99  return returnVal;
100  }

148  void FixedPairDistList::
149  beforeSendParticles(ParticleList& pl,
150  OutBuffer& buf) {
151  std::vector< longint > toSendInt;
152  std::vector< real > toSendReal;
153  // loop over the particle list
154  for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
155  longint pid = pit->id();
156 
157  LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
158 
159  // find all pairs that involve this particle
160 
161  int n = pairsDist.count(pid);
162 
163  if (n > 0) {
164  std::pair<PairsDist::const_iterator,
165  PairsDist::const_iterator> equalRange
166  = pairsDist.equal_range(pid);
167 
168  // first write the pid of the first particle
169  // then the number of partners
170  // and then the pids of the partners
171  toSendInt.reserve(toSendInt.size()+n+1);
172  toSendReal.reserve(toSendReal.size()+n);
173  toSendInt.push_back(pid);
174  toSendInt.push_back(n);
175  for (PairsDist::const_iterator it = equalRange.first;
176  it != equalRange.second; ++it) {
177  toSendInt.push_back(it->second.first);
178  toSendReal.push_back(it->second.second);
179  }
180 
181  // delete all of these pairs from the global list
182  //globalPairs.erase(equalRange.first->first, equalRange.second->first);
183  pairsDist.erase(pid);
184  // std::cout << "erasing particle " << pid << " from here" << std::endl;
185  }
186  }
187  // send the list
188  buf.write(toSendInt);
189  buf.write(toSendReal);
190  LOG4ESPP_INFO(theLogger, "prepared fixed pair list before send particles");
191  }

193  void FixedPairDistList::
194  afterRecvParticles(ParticleList &pl,
195  InBuffer& buf) {
196  std::vector< longint > receivedInt;
197  std::vector< real > receivedReal;
198  int n;
199  longint pid1, pid2;
200  real distVal;
201  PairsDist::iterator it = pairsDist.begin();
202  // receive the bond list
203  buf.read(receivedInt);
204  buf.read(receivedReal);
205  int size = receivedInt.size(); int i = 0; int j = 0;
206  while (i < size) {
207  // unpack the list
208  pid1 = receivedInt[i++];
209  n = receivedInt[i++];
210  for (; n > 0; --n) {
211  pid2 = receivedInt[i++];
212  distVal = receivedReal[j++];
213  it = pairsDist.insert(it, std::make_pair(pid1, std::make_pair(pid2, distVal)));
214  }
215  }
216  if (i != size) {
217  LOG4ESPP_ERROR(theLogger,
218  "ATTETNTION: recv particles might have read garbage\n");
219  }
220  LOG4ESPP_INFO(theLogger, "received fixed pair list after receive particles");
221  }

223  void FixedPairDistList::
224  onParticlesChanged() {
225  LOG4ESPP_INFO(theLogger, "rebuild local bond list from global\n");
226 
227  System& system = storage->getSystemRef();
228  esutil::Error err(system.comm);
229 
230  this->clear();
231  longint lastpid1 = -1;
232  Particle *p1;
233  Particle *p2;
234  for (PairsDist::const_iterator it = pairsDist.begin();
235  it != pairsDist.end(); ++it) {
236  if (it->first != lastpid1) {
237  p1 = storage->lookupRealParticle(it->first);
238  if (p1 == NULL) {
239  std::stringstream msg;
240  msg << "bond particle p1 " << it->first << " does not exists here";
241  //std::runtime_error(err.str());
242  err.setException( msg.str() );
243  }
244  lastpid1 = it->first;
245  }
246  p2 = storage->lookupLocalParticle(it->second.first);
247  if (p2 == NULL) {
248  std::stringstream msg;
249  msg << "bond particle p2 " << it->second.first << " does not exists here";
250  //std::runtime_error(err.str());
251  err.setException( msg.str() );
252  }
253  this->add(p1, p2);
254  }
255  err.checkException();
256  LOG4ESPP_INFO(theLogger, "regenerated local fixed pair list from global list");
257  }
