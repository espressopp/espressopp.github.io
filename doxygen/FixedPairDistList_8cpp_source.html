00151     std::vector< longint > toSendInt;
00152     std::vector< real > toSendReal;
00153     // loop over the particle list
00154     for (ParticleList::Iterator pit(pl); pit.isValid(); ++pit) {
00155       longint pid = pit->id();

00157       LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
00158 
00159       // find all pairs that involve this particle
00160       
00161       int n = pairsDist.count(pid);
00162 
00163       if (n > 0) {
00164         std::pair<PairsDist::const_iterator,
00165           PairsDist::const_iterator> equalRange
00166           = pairsDist.equal_range(pid);
00167 
00168         // first write the pid of the first particle
00169         // then the number of partners
00170         // and then the pids of the partners
00171         toSendInt.reserve(toSendInt.size()+n+1);
00172         toSendReal.reserve(toSendReal.size()+n);
00173         toSendInt.push_back(pid);
00174         toSendInt.push_back(n);
00175         for (PairsDist::const_iterator it = equalRange.first;
00176              it != equalRange.second; ++it) {
00177           toSendInt.push_back(it->second.first);
00178           toSendReal.push_back(it->second.second);
00179         }
00180 
00181         // delete all of these pairs from the global list
00182         //globalPairs.erase(equalRange.first->first, equalRange.second->first);
00183         pairsDist.erase(pid);
00184         // std::cout << "erasing particle " << pid << " from here" << std::endl;
00185       }
00186     }
00187     // send the list
00188     buf.write(toSendInt);
00189     buf.write(toSendReal);
