00178     void AssociationReaction::react() {
00179       if (integrator->getStep() % interval != 0) return;
00180 
00181       System& system = getSystemRef();
00182 
00183       LOG4ESPP_INFO(theLogger, "Perform AssociationReaction");
00184 
00185       dt = integrator->getTimeStep();
00186 
00187       Alist.clear();
00188       // loop over VL pairs
00189       for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
00190         Particle &p1 = *it->first;
00191         Particle &p2 = *it->second;
00192         // If criteria for reaction match, add the indices to Alist
00193         reactPair(p1, p2);
00194       }
00195       sendMultiMap(Alist);
00196       // Here, reduce number of partners to each A to 1
00197       // Also, keep only non-ghost A
00198       uniqueA(Alist);
00199       sendMultiMap(Alist);
00200       // Here, reduce number of partners to each B to 1
00201       // Also, keep only non-ghost B
00202       uniqueB(Alist, Blist);
00203       sendMultiMap(Blist);
00204       // Use Blist to apply the reaction.
00205       applyAR();

00211     void AssociationReaction::reactPair(Particle& p1, Particle& p2) {
00212       Real3D r = p1.position() - p2.position();
00213       real dist2 = r.sqr();
00214       bool found;
00215       found=false;
00216       if ((dist2 < cutoff_sqr) && ((*rng)() < rate*dt*interval)){
00217         if ((p1.type()==typeA) && (p2.type()==typeB) && (p1.state() >= stateAMin) && (p2.state()==0)) {
00218           Alist.insert(std::make_pair(p1.id(), p2.id()));
00219         }
00220         else if ((p2.type()==typeA) && (p1.type()==typeB) && (p2.state() >= stateAMin) && (p1.state()==0)) {
00221           Alist.insert(std::make_pair(p2.id(), p1.id()));

00265           tmp = it->first;
00266           outBuffer.write(tmp);
00267           tmp = it->second;
00268           outBuffer.write(tmp);
00269         }
00270     
00271         // lr loop: left right
00272         for (int lr = 0; lr < 2; ++lr) {
00273           int dir         = 2 * coord + lr;

00276           // Avoids double communication for size 2 directions.

00286             sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
00287

00289             if (nodeGrid.getNodePosition(coord) % 2 == 0) {

00298                 inBuffer0.recv(sender, AR_COMM_TAG);
00299               } else {
00300                 inBuffer1.recv(sender, AR_COMM_TAG);
00301               }

00311           int dir         = 2 * coord + lr;

00315           } else {

00351       longint idx1, idx2;

00354 
00355       idxSet.clear();
00356       idxList.clear();
00357       // Create idxList, containing (idx1, idx2) pairs
00358       // Create idxSet, containing each idx1 only once

00360            it!=mm.end(); it++) {
00361         idx1 = it->first;
00362         idx2 = it->second;
00363         idxList.insert(std::make_pair(idx1, idx2));
00364         idxSet.insert(idx1);
00365       }
00366 
00367       uniqueList.clear();
00368       // For each active idx1, pick a partner
00369       if (idxSet.size()>0) {
00370         for (boost::unordered_set<longint>::iterator it=idxSet.begin(); it!=idxSet.end(); it++) {
00371           idx1 = *it;
00372           Particle* p = system.storage->lookupLocalParticle(idx1);
00373           if (p==NULL) continue;
00374           if (p->ghost()) continue;

00376           if (size>0) {

00378             std::pair<boost::unordered_multimap<longint, longint>::iterator,boost::unordered_multimap<longint, longint>::iterator> candidates;
00379             candidates = idxList.equal_range(idx1);
