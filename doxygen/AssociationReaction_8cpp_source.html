178  void AssociationReaction::react() {
179  if (integrator->getStep() % interval != 0) return;
180 
181  System& system = getSystemRef();
182 
183  LOG4ESPP_INFO(theLogger, "Perform AssociationReaction");
184 
185  dt = integrator->getTimeStep();
186 
187  Alist.clear();
188  // loop over VL pairs
189  for (PairList::Iterator it(verletList->getPairs()); it.isValid(); ++it) {
190  Particle &p1 = *it->first;
191  Particle &p2 = *it->second;
192  // If criteria for reaction match, add the indices to Alist
193  reactPair(p1, p2);
194  }
195  sendMultiMap(Alist);
196  // Here, reduce number of partners to each A to 1
197  // Also, keep only non-ghost A
198  uniqueA(Alist);
199  sendMultiMap(Alist);
200  // Here, reduce number of partners to each B to 1
201  // Also, keep only non-ghost B
202  uniqueB(Alist, Blist);
203  sendMultiMap(Blist);
204  // Use Blist to apply the reaction.
205  applyAR();
206  }

211  void AssociationReaction::reactPair(Particle& p1, Particle& p2) {
212  Real3D r = p1.position() - p2.position();
213  real dist2 = r.sqr();
214  bool found;
215  found=false;
216  if ((dist2 < cutoff_sqr) && ((*rng)() < rate*dt*interval)){
217  if ((p1.type()==typeA) && (p2.type()==typeB) && (p1.state() >= stateAMin) && (p2.state()==0)) {
218  Alist.insert(std::make_pair(p1.id(), p2.id()));
219  }
220  else if ((p2.type()==typeA) && (p1.type()==typeB) && (p2.state() >= stateAMin) && (p1.state()==0)) {
221  Alist.insert(std::make_pair(p2.id(), p1.id()));
222  }
223  }
224  }

236  void AssociationReaction::sendMultiMap(boost::unordered_multimap<longint, longint> &mm) {
237 
238  LOG4ESPP_INFO(theLogger, "Entering sendMultiMap");
239 
240  InBuffer inBuffer0(*getSystem()->comm);
241  InBuffer inBuffer1(*getSystem()->comm);
242  OutBuffer outBuffer(*getSystem()->comm);
243  System& system = getSystemRef();
244  const NodeGrid& nodeGrid = domdec->getNodeGrid();
245 
246  /* direction loop: x, y, z.
247  Here we could in principle build in a one sided ghost
248  communication, simply by taking the lr loop only over one
249  value. */
250  for (int coord = 0; coord < 3; ++coord) {
251  /* inverted processing order for ghost force communication,
252  since the corner ghosts have to be collected via several
253  nodes. We now add back the corner ghost forces first again
254  to ghost forces, which only eventually go back to the real
255  particle.
256  */
257 
258  real curCoordBoxL = system.bc->getBoxL()[coord];
259 
260  outBuffer.reset();
261  // fill outBuffer from mm
262  int tmp = mm.size();
263  outBuffer.write(tmp);
264  for (boost::unordered_multimap<longint, longint>::iterator it=mm.begin(); it!=mm.end(); it++) {
265  tmp = it->first;
266  outBuffer.write(tmp);
267  tmp = it->second;
268  outBuffer.write(tmp);
269  }
270 
271  // lr loop: left right
272  for (int lr = 0; lr < 2; ++lr) {
273  int dir = 2 * coord + lr;
274  int oppositeDir = 2 * coord + (1 - lr);
275  int dirSize = nodeGrid.getGridSize(coord);
276  // Avoids double communication for size 2 directions.
277  if ( (dirSize==2) && (lr==1) ) continue;
278 
279  if (dirSize == 1) {
280  LOG4ESPP_DEBUG(theLogger, "no communication");
281  }
282  else {
283  // prepare send and receive buffers
284  longint receiver, sender;
285  receiver = nodeGrid.getNodeNeighborIndex(dir);
286  sender = nodeGrid.getNodeNeighborIndex(oppositeDir);
287 
288  // exchange particles, odd-even rule
289  if (nodeGrid.getNodePosition(coord) % 2 == 0) {
290  outBuffer.send(receiver, AR_COMM_TAG);
291  if (lr==0) {
292  inBuffer0.recv(sender, AR_COMM_TAG);
293  } else {
294  inBuffer1.recv(sender, AR_COMM_TAG);
295  }
296  } else {
297  if (lr==0) {
298  inBuffer0.recv(sender, AR_COMM_TAG);
299  } else {
300  inBuffer1.recv(sender, AR_COMM_TAG);
301  }
302  outBuffer.send(receiver, AR_COMM_TAG);
303  }
304  }
305  }
306  LOG4ESPP_DEBUG(theLogger, "Entering unpack");
307  // unpack received data
308  // add content of inBuffer to mm
309  int lengthA, Aidx, Bidx;
310  for (int lr = 0; lr < 2; ++lr) {
311  int dir = 2 * coord + lr;
312  int oppositeDir = 2 * coord + (1 - lr);
313  int dirSize = nodeGrid.getGridSize(coord);
314  if (dirSize == 1) {
315  } else {
316  // Avoids double communication for size 2 directions.
317  if ( (dirSize==2) && (lr==1) ) continue;
318  if (lr==0) {
319  inBuffer0.read(lengthA);
320  } else {
321  inBuffer1.read(lengthA);
322  }
323  for (longint i=0; i<lengthA; i++) {
324  if (lr==0) {
325  inBuffer0.read(Aidx);
326  inBuffer0.read(Bidx);
327  } else {
328  inBuffer1.read(Aidx);
329  inBuffer1.read(Bidx);
330  }
331  mm.insert(std::make_pair(Aidx, Bidx));
332  }
333  }
334  }
335  LOG4ESPP_DEBUG(theLogger, "Leaving unpack");
336  }
337 
338  LOG4ESPP_INFO(theLogger, "Leaving sendMultiMap");
339 
340  }

346  void AssociationReaction::uniqueA(boost::unordered_multimap<longint, longint> &mm) {
347 
348  // Collect indices
349  boost::unordered_set<longint> idxSet;
350  boost::unordered_multimap<longint, longint> idxList;
351  longint idx1, idx2;
352  System& system = getSystemRef();
353  boost::unordered_multimap<longint, longint> uniqueList;
354 
355  idxSet.clear();
356  idxList.clear();
357  // Create idxList, containing (idx1, idx2) pairs
358  // Create idxSet, containing each idx1 only once
359  for (boost::unordered_multimap<longint, longint>::iterator it=mm.begin();
360  it!=mm.end(); it++) {
361  idx1 = it->first;
362  idx2 = it->second;
363  idxList.insert(std::make_pair(idx1, idx2));
364  idxSet.insert(idx1);
365  }
366 
367  uniqueList.clear();
368  // For each active idx1, pick a partner
369  if (idxSet.size()>0) {
370  for (boost::unordered_set<longint>::iterator it=idxSet.begin(); it!=idxSet.end(); it++) {
371  idx1 = *it;
372  Particle* p = system.storage->lookupLocalParticle(idx1);
373  if (p==NULL) continue;
374  if (p->ghost()) continue;
375  int size = idxList.count(idx1);
376  if (size>0) {
377  int pick = (*rng)(size);
378  std::pair<boost::unordered_multimap<longint, longint>::iterator,boost::unordered_multimap<longint, longint>::iterator> candidates;
379  candidates = idxList.equal_range(idx1);
380  int i=0;
381  for (boost::unordered_multimap<longint, longint>::iterator jt=candidates.first; jt!=candidates.second; jt++, i++) {
382  if (i==pick) {
383  uniqueList.insert(std::make_pair(jt->first, jt->second));
384  break;
385  }
386  }
387  }
388  }
389  }
390  mm = uniqueList;
391  uniqueList.clear();
392  }

456  A = it->second;

459  Particle* pA = system.storage->lookupLocalParticle(A);
460  Particle* pB = system.storage->lookupLocalParticle(B);
461  if (pA==NULL) {
462  } else {

464  }
465  if (pB==NULL) {
466  } else {
467  pB->setState(pB->getState()+deltaB);
468  }
469  // Add a bond
470  fpl->add(A, B);
