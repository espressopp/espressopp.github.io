168  Real3D prod(Real3D a, Real3D b) const {
169  Real3D res(0.0, 0.0, 0.0);
170  for(int i=0; i<3; i++){
171  for(int j=0; j<3; j++){
172  res[i]+=(1.0-d(i, j))*a[j]*b[j];
173  }
174  }
175  return res;
176  }

178  void _computeForceRaw(Real3D& force1,
179  Real3D& force2,
180  Real3D& force3,
181  Real3D& force4,
182  const Real3D& r21,
183  const Real3D& r32,
184  const Real3D& r43) const {
185  Real3D retF[4];
186 
187  Real3D rijjk = r21.cross(r32); // [r21 x r32]
188  Real3D rjkkn = r32.cross(r43); // [r32 x r43]
189 
190  real rijjk_sqr = rijjk.sqr();
191  real rjkkn_sqr = rjkkn.sqr();
192 
193  real rijjk_abs = sqrt(rijjk_sqr);
194  real rjkkn_abs = sqrt(rjkkn_sqr);
195 
196  real inv_rijjk = 1.0 / rijjk_abs;
197  real inv_rjkkn = 1.0 / rjkkn_abs;
198 
199  // cosine between planes
200  real cos_phi = (rijjk * rjkkn) * (inv_rijjk * inv_rjkkn);
201  real _phi = acos(cos_phi);
202  if (cos_phi > 1.0) {
203  cos_phi = 1.0;
204  _phi = 1e-10; //not 0.0, because 1.0/sin(_phi) would cause a singularity
205  } else if (cos_phi < -1.0) {
206  cos_phi = -1.0;
207  _phi = M_PI-1e-10;
208  }
209 
210  //get sign of phi
211  //positive if (rij x rjk) x (rjk x rkn) is in the same direction as rjk, negative otherwise (see DLPOLY manual)
212  Real3D rcross = rijjk.cross(rjkkn); //(rij x rjk) x (rjk x rkn)
213  real signcheck = rcross * r32;
214  if (signcheck < 0.0) _phi *= -1.0;
215 
216  // read table
217  real a = 0.;
218  for (int i=0; i<numInteractions; ++i)
219  a += weights[i] * tables[i]->getForce(_phi);
220 
221  real coef1 = -(1.0/sin(_phi)) * a;
222 
223  real A1 = inv_rijjk * inv_rjkkn;
224  real A2 = inv_rijjk * inv_rijjk;
225  real A3 = inv_rjkkn * inv_rjkkn;
226 
227  Real3D p3232 ( prod(r32,r32) );
228  Real3D p3243 ( prod(r32,r43) );
229  Real3D p2132 ( prod(r21,r32) );
230  Real3D p2143 ( prod(r21,r43) );
231  Real3D p2121 ( prod(r21,r21) );
232  Real3D p4343 ( prod(r43,r43) );
233 
234  // we have 4 particles 1,2,3,4
235  for(int l=0; l<4; l++){
236  Real3D B1, B2, B3;
237 
238  for(int i=0; i<3; i++){
239  B1[i]= r21[i] * ( p3232[i] * (d(l,2)-d(l,3)) + p3243[i] * (d(l,2)-d(l,1)) ) +
240  r32[i] * ( p2132[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,0)) ) +
241  r43[i] * ( p2132[i] * (d(l,2)-d(l,1)) + p3232[i] * (d(l,0)-d(l,1)) ) +
242  2.0 * r32[i] * p2143[i] * (d(l,1)-d(l,2));
243 
244  B2[i]= 2.0 * r21[i] * ( p3232[i] * (d(l,1)-d(l,0)) + p2132[i] * (d(l,1)-d(l,2)) ) +
245  2.0 * r32[i] * ( p2121[i] * (d(l,2)-d(l,1)) + p2132[i] * (d(l,0)-d(l,1)));
246 
247  B3[i]= 2.0 * r43[i] * ( p3232[i] * (d(l,3)-d(l,2)) + p3243[i] * (d(l,1)-d(l,2)) ) +
248  2.0 * r32[i] * ( p4343[i] * (d(l,2)-d(l,1)) + p3243[i] * (d(l,2)-d(l,3)));
249  }
250 
251  retF[l] = coef1 * ( A1*B1 - 0.5*cos_phi*(A2*B2 + A3*B3) );
252  }
253 
254  force1 = retF[0];
255  force2 = retF[1];
256  force3 = retF[2];
257  force4 = retF[3];
258 
259  }
