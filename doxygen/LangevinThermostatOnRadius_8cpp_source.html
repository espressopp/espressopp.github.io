124  void LangevinThermostatOnRadius::frictionThermo(Particle& p) {
125 
126  // get a random value for a radius
127 
128  real ranval = (*rng)() - 0.5;
129 
130  p.fradius() += pref1 * p.vradius() * dampingmass +
131  pref2 * ranval * massf;
