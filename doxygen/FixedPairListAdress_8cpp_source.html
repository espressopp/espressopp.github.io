63  bool FixedPairListAdress::add(longint pid1, longint pid2) {
64 
65  if (pid1 > pid2)
66  std::swap(pid1, pid2);
67 
68  bool returnVal = true;
69  System& system = storage->getSystemRef();
70  esutil::Error err(system.comm);
71 
72  // ADD THE LOCAL PAIR
73  Particle *p1 = storage->lookupAdrATParticle(pid1);
74  Particle *p2 = storage->lookupAdrATParticle(pid2);
75  if (!p1){
76  // Particle does not exist here, return false
77  returnVal = false;
78  }
79  else{
80  if (!p2) {
81  std::stringstream msg;
82  msg << "Atomistic bond particle p2 (id=" << pid2 << ") does not exists here "
83  << "and cannot be added. "
84  << " The pair " << pid1 << " - " << pid2 << " could not be created.";
85  err.setException( msg.str() );
86  }
87  }
88  err.checkException();
89 
90  if(returnVal){
91  // add the pair locally
92  this->add(p1, p2);
93  // ADD THE GLOBAL PAIR
94  // see whether the particle already has pairs
95  std::pair<GlobalPairs::const_iterator,
96  GlobalPairs::const_iterator> equalRange
97  = globalPairs.equal_range(pid1);
98  if (equalRange.first == globalPairs.end()) {
99  // if it hasn't, insert the new pair
100  globalPairs.insert(std::make_pair(pid1, pid2));
101  }
102  else {
103  // otherwise test whether the pair already exists
104  for (GlobalPairs::const_iterator it = equalRange.first; it != equalRange.second; ++it) {
105  if (it->second == pid2) {
106  // TODO: Pair already exists, generate error!
107  ;
108  }
109  }
110  // if not, insert the new pair
111  globalPairs.insert(equalRange.first, std::make_pair(pid1, pid2));
112  }
113  LOG4ESPP_INFO(theLogger, "added fixed pair to global pair list");
114  }
115  return returnVal;
116  }

119  void FixedPairListAdress::beforeSendATParticles(std::vector<longint>& atpl,
120  OutBuffer& buf) {
121 
122  //std::cout << "beforeSendATParticles() fixed pl (size " << atpl.size() << ")\n";
123 
124  std::vector< longint > toSend;
125 
126  // loop over the VP particle list
127  for (std::vector<longint>::iterator it = atpl.begin();
128  it != atpl.end(); ++it) {
129  longint pid = *it;
130 
131  LOG4ESPP_DEBUG(theLogger, "send particle with pid " << pid << ", find pairs");
132 
133  // find all pairs that involve this particle
134 
135  int n = globalPairs.count(pid);
136 
137  if (n > 0) {
138  std::pair<GlobalPairs::const_iterator,
139  GlobalPairs::const_iterator> equalRange
140  = globalPairs.equal_range(pid);
141 
142  // first write the pid of the first particle
143  // then the number of partners
144  // and then the pids of the partners
145  toSend.reserve(toSend.size()+n+1);
146  toSend.push_back(pid);
147  toSend.push_back(n);
148  for (GlobalPairs::const_iterator it = equalRange.first;
149  it != equalRange.second; ++it) {
150  toSend.push_back(it->second);
151  LOG4ESPP_DEBUG(theLogger, "send global bond: pid "
152  << pid << " and partner " << it->second);
153  }
154 
155  // delete all of these pairs from the global list
156  globalPairs.erase(pid);
157  }
158  }
159 
160  // send the list
161  buf.write(toSend);
162  LOG4ESPP_INFO(theLogger, "prepared fixed pair list before send particles");
163  }

172  void FixedPairListAdress::onParticlesChanged() {
173 
174  LOG4ESPP_INFO(theLogger, "rebuild local bond list from global\n");
175 
176  System& system = storage->getSystemRef();
177  esutil::Error err(system.comm);
178 
179  this->clear();
180  longint lastpid1 = -1;
181  Particle *p1;
182  Particle *p2;
183 
184  for (GlobalPairs::const_iterator it = globalPairs.begin();
185  it != globalPairs.end(); ++it) {
186 
187  if (it->first != lastpid1) {
188  p1 = storage->lookupAdrATParticle(it->first);
189  if (p1 == NULL) {
190  std::stringstream msg;
191  msg << "FixedPairListAdress ";
192  msg << "Atomistic bond particle p1 (id=" << it->first << ") does not exists here.";
193  err.setException( msg.str() );
194  }
195  lastpid1 = it->first;
196  }
197 
198  p2 = storage->lookupAdrATParticle(it->second);
199  if (p2 == NULL) {
200  std::stringstream msg;
201  msg << "FixedPairListAdress ";
202  msg << "Atomistic bond particle p2 (id=" << it->second << ") does not exists here.";
203  err.setException( msg.str() );
204  }

206  //std::cout << " adding (" << p1->getId() << ", " << p2->getId() << ")\n";
207  this->add(p1, p2);
208  }
209  err.checkException();
210 
211  LOG4ESPP_INFO(theLogger, "Regenerated local fixed pair list from global list");
212  }
