00141 
00142             // first write the pid of the first particle
00143             // then the number of partners
00144             // and then the pids of the partners
00145             toSend.reserve(toSend.size()+n+1);
00146             toSend.push_back(pid);
00147             toSend.push_back(n);
00148             for (GlobalPairs::const_iterator it = equalRange.first;
00149                  it != equalRange.second; ++it) {
00150               toSend.push_back(it->second);
00151               LOG4ESPP_DEBUG(theLogger, "send global bond: pid "
00152                            << pid << " and partner " << it->second);
00153             }
00154 
00155             // delete all of these pairs from the global list
00156             globalPairs.erase(pid);
00157           }
00158         }
00159 
00160         // send the list
