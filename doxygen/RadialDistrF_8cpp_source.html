00046     python::list RadialDistrF::computeArray(int rdfN) const {

00048       System& system = getSystemRef();
00049       esutil::Error err(system.comm);
00050       Real3D Li = system.bc->getBoxL();
00051       Real3D Li_half = Li / 2.;
00052       
00053       int nprocs = system.comm->size();
00054       int myrank = system.comm->rank();
00055       
00056       real histogram [rdfN];
00057       for(int i=0;i<rdfN;i++) histogram[i]=0;
00058           
00059       real dr = Li_half[1] / (real)rdfN; // If you work with nonuniform Lx, Ly, Lz, you
00060                                          // should use for Li_half[XXX] the shortest side length   
00061       
00062       int num_part = 0;
00063       ConfigurationPtr config = make_shared<Configuration> ();

00065         map< size_t, Real3D > conf;

00067           CellList realCells = system.storage->getRealCells();
00068           for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00069             int id = cit->id();
00070             conf[id] = cit->position();
00071           }
00072         }
00073 
00074         boost::mpi::broadcast(*system.comm, conf, rank_i);
00075 
00076         // for simplicity we will number the particles from 0
00077         for (map<size_t,Real3D>::iterator itr=conf.begin(); itr != conf.end(); ++itr) {
00078           //size_t id = itr->first;
00079           Real3D p = itr->second;
00080           config->set(num_part, p[0], p[1], p[2]);
00081           num_part ++;
00082         }
00083       }
00084       // now all CPUs have all particle coords and num_part is the total number of particles
00085       
00086       // use all cpus
00087       // TODO it could be a problem if   n_nodes > num_part
00088       int numi = num_part / nprocs + 1;
00089       int mini = myrank * numi;
00090       int maxi = mini + numi;
00091       
00092       if(maxi>num_part) maxi = num_part;
00093       
00094       int perc=0;
00095       real denom = 100.0 / (real)(maxi-mini);
00096       for(int i = mini; i<maxi; i++){
00097         Real3D coordP1 = config->getCoordinates(i);
00098         for(int j = i+1; j<num_part; j++){
00099           Real3D coordP2 = config->getCoordinates(j);
00100           Real3D distVector = coordP1 - coordP2;
00101           
00102           // minimize the distance in simulation box
00103           for(int ii=0; ii<3; ii++){
00104             if( distVector[ii] < -Li_half[ii] ) distVector[ii] += Li[ii];
00105             if( distVector[ii] >  Li_half[ii] ) distVector[ii] -= Li[ii];
00106           }
00107           
00108           int bin = (int)( distVector.abs() / dr );
00109           if( bin < rdfN){
00110             histogram[bin] += 1.0;
00111           }
00112           
00113         }
00114         /*
00115          * additional calculations slow down routine but from the other hand
00116          * it helps to monitor progress
00117          */
00118         if(print_progress && system.comm->rank()==0){
00119           perc = (int)((i-mini)*denom);
00120           if(perc%5==0){
00121             cout<<"calculation progress (radial distr. func.): "<< perc << " %\r"<<flush;
00122           }
00123         }
00124         
00125       }
00126       if(system.comm->rank()==0)
00127         cout<<"calculation progress (radial distr. func.): 100 %" <<endl;
00128 
00129       real totHistogram[rdfN];
00130       boost::mpi::all_reduce(*system.comm, histogram, rdfN, totHistogram, plus<real>());
00131       
00132       // normalizing
00133       int nconfigs = 1; //config - 1
00134       //avg_part = part_total / float(nconfigs)
00135       real rho = (real)num_part / (Li[0]*Li[1]*Li[2]);
00136       real factor = 2.0 * M_PIl * dr * rho * (real)nconfigs * (real)num_part;
00137       
00138       for(int i=0; i < rdfN; i++){
00139         real radius = (i + 0.5) * dr;
00140         totHistogram[i] /= factor * (radius*radius + dr*dr / 12.0);
00141       }
00142       
00143       python::list pyli;
00144       for(int i=0; i < rdfN; i++){
00145         pyli.append( totHistogram[i] );
00146       }
00147       return pyli;
