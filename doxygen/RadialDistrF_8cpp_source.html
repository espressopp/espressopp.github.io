45  // rdfN is a level of discretisation of rdf (how many elements it contains)
46  python::list RadialDistrF::computeArray(int rdfN) const {
47 
48  System& system = getSystemRef();
49  esutil::Error err(system.comm);
50  Real3D Li = system.bc->getBoxL();
51  Real3D Li_half = Li / 2.;
52 
53  int nprocs = system.comm->size();
54  int myrank = system.comm->rank();
55 
56  real histogram [rdfN];
57  for(int i=0;i<rdfN;i++) histogram[i]=0;
58 
59  real dr = Li_half[1] / (real)rdfN; // If you work with nonuniform Lx, Ly, Lz, you
60  // should use for Li_half[XXX] the shortest side length
61 
62  int num_part = 0;
63  ConfigurationPtr config = make_shared<Configuration> ();
64  for (int rank_i=0; rank_i<nprocs; rank_i++) {
65  map< size_t, Real3D > conf;
66  if (rank_i == myrank) {
67  CellList realCells = system.storage->getRealCells();
68  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
69  int id = cit->id();
70  conf[id] = cit->position();
71  }
72  }
73 
74  boost::mpi::broadcast(*system.comm, conf, rank_i);
75 
76  // for simplicity we will number the particles from 0
77  for (map<size_t,Real3D>::iterator itr=conf.begin(); itr != conf.end(); ++itr) {
78  //size_t id = itr->first;
79  Real3D p = itr->second;
80  config->set(num_part, p[0], p[1], p[2]);
81  num_part ++;
82  }
83  }
84  // now all CPUs have all particle coords and num_part is the total number of particles
85 
86  // use all cpus
87  // TODO it could be a problem if n_nodes > num_part
88  int numi = num_part / nprocs + 1;
89  int mini = myrank * numi;
90  int maxi = mini + numi;
91 
92  if(maxi>num_part) maxi = num_part;
93 
94  int perc=0;
95  real denom = 100.0 / (real)(maxi-mini);
96  for(int i = mini; i<maxi; i++){
97  Real3D coordP1 = config->getCoordinates(i);
98  for(int j = i+1; j<num_part; j++){
99  Real3D coordP2 = config->getCoordinates(j);
100  Real3D distVector = coordP1 - coordP2;
101 
102  // minimize the distance in simulation box
103  for(int ii=0; ii<3; ii++){
104  if( distVector[ii] < -Li_half[ii] ) distVector[ii] += Li[ii];
105  if( distVector[ii] > Li_half[ii] ) distVector[ii] -= Li[ii];
106  }
107 
108  int bin = (int)( distVector.abs() / dr );
109  if( bin < rdfN){
110  histogram[bin] += 1.0;
111  }
112 
113  }
114  /*
115  * additional calculations slow down routine but from the other hand
116  * it helps to monitor progress
117  */
118  if(print_progress && system.comm->rank()==0){
119  perc = (int)((i-mini)*denom);
120  if(perc%5==0){
121  cout<<"calculation progress (radial distr. func.): "<< perc << " %\r"<<flush;
122  }
123  }
124 
125  }
126  if(system.comm->rank()==0)
127  cout<<"calculation progress (radial distr. func.): 100 %" <<endl;
128 
129  real totHistogram[rdfN];
130  boost::mpi::all_reduce(*system.comm, histogram, rdfN, totHistogram, plus<real>());
131 
132  // normalizing
133  int nconfigs = 1; //config - 1
134  //avg_part = part_total / float(nconfigs)
135  real rho = (real)num_part / (Li[0]*Li[1]*Li[2]);
136  real factor = 2.0 * M_PIl * dr * rho * (real)nconfigs * (real)num_part;
137 
138  for(int i=0; i < rdfN; i++){
139  real radius = (i + 0.5) * dr;
140  totHistogram[i] /= factor * (radius*radius + dr*dr / 12.0);
141  }
142 
143  python::list pyli;
144  for(int i=0; i < rdfN; i++){
145  pyli.append( totHistogram[i] );
146  }
147  return pyli;
148  }
