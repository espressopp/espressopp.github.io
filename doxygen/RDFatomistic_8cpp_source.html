78  for (std::vector<Particle*>::iterator it3 = atList.begin();
79  it3 != atList.end(); ++it3) {
80  Particle &at = **it3;
81  int id = at.id();
82  data tmp = { at.position(), at.type(), vp.id(), vp.lambda() };
83  conf[id] = tmp;
84  }
85  }

102  }
103  }

106  // now all CPUs have all particle coords and num_part is the total number of particles

108  // use all cpus

110  int numi = num_part / nprocs + 1;
111  int mini = myrank * numi;
112  int maxi = mini + numi;
113 
114  if(maxi>num_part) maxi = num_part;

121  real resolutionP1 = config[i].resolution;
122 
123

236  it3 != atList.end(); ++it3) {
237  Particle &at = **it3;

240  int id = at.id();
241  dataPathIntegral tmp = { at.position(), at.type(), vp.id(), at.pib() };

244  }
245  }
246 
247  else { // If not, use CG particle itself for calculation.

266 
267  int numi = num_part / nprocs + 1;
268  int mini = myrank * numi;
269  int maxi = mini + numi;
270 
271  if(maxi>num_part) maxi = num_part;
272 
273  int perc=0, perc1=0;
274  for(int i = mini; i<maxi; i++) {
275  Real3D coordP1 = config[i].pos;
276  int typeP1 = config[i].type;
277  int molP1 = config[i].molecule;
278  int pibP1 = config[i].pib;
279 
280  if( (coordP1[0] < Xmax) && (coordP1[0] > Xmin) ) {
281  for(int j = i+1; j<num_part; j++) {
282  Real3D coordP2 = config[j].pos;
283  int typeP2 = config[j].type;
284  int molP2 = config[j].molecule;
285  int pibP2 = config[j].pib;
286 
287  if ( (molP1 != molP2) && (pibP1 == pibP2) && (coordP2[0] < Xmax) && (coordP2[0] > Xmin) ) {
288 
289  if( ( (typeP1 == target1) && (typeP2 == target2) ) || ( (typeP1 == target2) && (typeP2 == target1) ) ) {
290 
291  Real3D distVector = coordP1 - coordP2;
292 
293  // minimize the distance in simulation box
294  for(int ii=0; ii<3; ii++) {
295  if( distVector[ii] < -Li_half[ii] ) distVector[ii] += Li[ii];
296  if( distVector[ii] > Li_half[ii] ) distVector[ii] -= Li[ii];
297  }
298 
299  real absdist = distVector.abs();
300 
301  int bin = (int)( absdist / dr );
302  if( bin < rdfN) {
303 
304  // Normalization (see R. Potestio et al., Phys. Rev. Lett. 111, 060601 (2013), Supp. Info.)
305  real Xplus = Xmax - coordP1[0];
306  real Xminus = coordP1[0] - Xmin;
307 
308  real h_norm = 0.0;
309 
310  if (Xplus <= absdist) {
311  h_norm += absdist - Xplus;
312  }
313  if (Xminus <= absdist) {
314  h_norm += absdist - Xminus;
315  }
316 
317  real v_norm = 1.0/(2.0*M_PIl*dr*absdist*(2.0*absdist - h_norm));
318 
319  histogram[bin] += v_norm;
320  }
321  num_pairs += 1;
322 
323  }
324  }
325  }
326  }
327 
328  }

331  boost::mpi::all_reduce(*system.comm, histogram, rdfN, totHistogram, plus<real>());
332 
333  int tot_num_pairs = 0;
334  boost::mpi::all_reduce(*system.comm, num_pairs, tot_num_pairs, plus<int>());
335 
336  // Normalization (see R. Potestio et al., Phys. Rev. Lett. 111, 060601 (2013), Supp. Info.)
337  real subvol = 2.0*Li[1]*Li[2]*span;
338  if(span > 0.5*Li[0]) {
339  subvol = Li[1]*Li[2]*Li[0];
340  }
341  real rho = (real)1.0 / (Li[0]*Li[1]*Li[2]);
342  for(int i=0; i < rdfN; i++) {
343  totHistogram[i] /= ( (real)tot_num_pairs / subvol );
344  }
345  // Normalization
346 
347  python::list pyli;
348  for(int i=0; i < rdfN; i++) {
349  pyli.append( totHistogram[i] );
350  }
351  return pyli;
352  }
