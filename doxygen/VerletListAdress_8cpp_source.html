75  void VerletListAdress::rebuild()
76  {
77  vlPairs.clear();
78  adrZone.clear(); // particles in adress zone
79  cgZone.clear(); // particles in CG zone
80  adrPairs.clear(); // pairs in adress zone
81  const bc::BC& bc = *getSystemRef().bc;
82 
83  // get local cells
84  CellList localcells = getSystem()->storage->getLocalCells();
85 
86  // if adrCenter is not set, the center of adress zone moves along with some particles
87  if (!adrCenterSet) {
88  // loop over all VP particles (reals and ghosts) on node
89  for (CellListIterator it(localcells); it.isValid(); ++it) {
90 
91  // loop over positions
92  for (std::vector<Real3D*>::iterator it2 = adrPositions.begin(); it2 != adrPositions.end(); ++it2){
93  Real3D dist;
94  real distsq;
95  bc.getMinimumImageVectorBox(dist, it->getPos(), **it2);
96 
97  if (getAdrRegionType()){ // spherical adress region
98  distsq=dist.sqr();
99  }
100  else { // slab-type adress region
101  distsq=dist[0]*dist[0];
102  }
103  if (distsq <= adrsq) {
104  adrZone.insert(&(*it));
105  break; // do not need to loop further
106  }
107  }
108  // if not near enough to any adrPositions, put in cgZone
109  if (adrZone.count(&(*it)) == 0) {
110  cgZone.insert(&(*it));
111  }
112  }
113  }
114  // center of adress zone is fixed
115  else {
116  for (CellListIterator it(localcells); it.isValid(); ++it) {
117  Real3D dist;
118  real distsq;
119  bc.getMinimumImageVectorBox(dist, it->getPos(), adrCenter);
120  //Real3D dist = it->getPos() - adrCenter; // CHANGE FOR X SPLIT VS SPHERE
121  //real dist = it->getPos()[0] - adrCenter[0]; // CHANGE FOR X SPLIT VS SPHERE
122  //real distsq = dist.sqr(); // CHANGE FOR X SPLIT VS SPHERE
123  //real distsq = dist*dist; // CHANGE FOR X SPLIT VS SPHERE
124  if (getAdrRegionType()){ // spherical adress region
125  distsq=dist.sqr();
126  }
127  else { // slab-type adress region
128  distsq=dist[0]*dist[0];
129  }
130  if (distsq <= adrsq) {
131  adrZone.insert(&(*it));
132  }
133  else {
134  cgZone.insert(&(*it));
135  }
136  }
137 
138  }
139 
140  // add particles to adress pairs and VL
141  CellList cl = getSystem()->storage->getRealCells();
142  int count=0;
143 
144  for (CellListAllPairsIterator it(cl); it.isValid(); ++it) {
145  checkPair(*it->first, *it->second);
146  count+=1;
147  }
148 
149  LOG4ESPP_INFO(theLogger, "rebuilt VerletList, cutsq = " << cutsq
150  << " local size = " << vlPairs.size());
151  builds++;
152  }

156  void VerletListAdress::checkPair(Particle& pt1, Particle& pt2)
157  {
158 
159  Real3D d = pt1.position() - pt2.position();
160  real distsq = d.sqr();
161 
162  LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
163  << " @ " << pt1.position()
164  << " - p2: " << pt2.id() << " @ " << pt2.position()
165  << " -> distsq = " << distsq);
166 
167  // see if it's in the exclusion list (both directions, CG particles only)
168  if (exList.count(std::make_pair(pt1.id(), pt2.id())) == 1) return;
169  if (exList.count(std::make_pair(pt2.id(), pt1.id())) == 1) return;
170  // see if it's in the adress zone
171  if (adrZone.count(&pt1) == 1 || adrZone.count(&pt2) == 1) {
172  if (distsq > adrcutsq) return;
173  adrPairs.add(pt1, pt2); // add to adress pairs
174  }
175  else {
176  if (distsq > cutsq) return;
177  vlPairs.add(pt1, pt2); // add pair to Verlet List
178  }
179  }
