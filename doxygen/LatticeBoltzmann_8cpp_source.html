375  /* (RE)INITIALIZATION OF THERMAL FLUCTUATIONS */
376  void LatticeBoltzmann::initFluctuations () {
377  using std::setprecision;
378  using std::fixed;
379  using std::setw;
380 
381  /* set amplitudes of local fluctuations */
382  real _lbTemp;
383  real mu, a3;
384 
385  int _myRank = getSystem()->comm->rank();
386 
387  if (_myRank == 0) {
388  std::cout << "Conversion coefficients from MD to LB:\n";
389  std::cout << "Mass: " << convMassMDtoLB() << "\n";
390  std::cout << "Length: " << convLenMDtoLB() << "\n";
391  std::cout << "Time: " << convTimeMDtoLB() << "\n";
392  std::cout << "MD steps between LB-update: " << getNSteps() << "\n";
393  std::cout << "-------------------------------------\n";
394  }
395 
396  _lbTemp = getLBTemp() * convMassMDtoLB() * pow(convLenMDtoLB() / convTimeMDtoLB(), 2.);
397  a3 = getA() * getA() * getA(); // a^3
398  mu = _lbTemp / (getCs2() * a3); // thermal mass density
399 
400  if (_lbTemp <= ROUND_ERROR_PREC) {
401  // account for fluctuations being turned off
402  setDoFluct(false);
403  if (_myRank == 0) std::cout << "Atermal LB-fluid. No fluctuations!" << std::endl;
404  } else {
405  // account for fluctuations being turned on!
406  setDoFluct(true);
407 
408  if (_myRank == 0) {
409  std::cout << setprecision(8);
410  std::cout << fixed; // some output tricks
411  std::cout << "The fluctuations have been introduced into the system:\n";
412  std::cout << "lbTemp = " << getLBTemp() << " in LJ-units or " << _lbTemp << " in lattice units" << "\n";
413  }
414 
415  // calculate phi coefficients
416  setPhi(0, 0.);
417  setPhi(1, 0.); setPhi(2, 0.); setPhi(3, 0.);
418  setPhi(4, sqrt(mu / getInvB(4) * (1. - getGamma(0) * getGamma(0))));
419  for (int l = 5; l < 10; l++) {
420  setPhi(l, sqrt(mu / getInvB(l) * (1. - getGamma(1) * getGamma(1))));
421  }
422  for (int l = 10; l < getNumVels(); l++) {
423  setPhi(l, sqrt(mu / getInvB(l)));
424  }
425 
426  // set phi on every lattice site
427  Int3D _Ni = getMyNi();
428  for (int i = 0; i < _Ni[0]; i++) {
429  for (int j = 0; j < _Ni[1]; j++) {
430  for (int k = 0; k < _Ni[2]; k++) {
431  for (int l = 0; l < getNumVels(); l++) {
432  (*lbfluid)[i][j][k].setPhiLoc(l,getPhi(l));
433  }
434  }
435  }
436  }
437 
438  if (_myRank == 0) {
439  std::cout << "The amplitudes phi_i of the fluctuations have been redefined.\n";
440  std::cout << "-------------------------------------\n";
441  }
442  }
443  }

480  /* REAL MD-PARTICLES DECOMPOSITION IF THEY MOVED TO THE GHOST REGION */
483  void LatticeBoltzmann::makeDecompose() {
484  int _offset = getHaloSkin();
485  real _a = getA();
486  Int3D _myNi = getMyNi();
487  Real3D _myLeft = getMyLeft();
488 
489  System& system = getSystemRef();
490  CellList realCells = system.storage->getRealCells();
491 
492  bool _myDecompose = false;
493  bool decompose = false;
494  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
495  if (cit->position()[0] < _myLeft[0] ||
496  cit->position()[1] < _myLeft[1] ||
497  cit->position()[2] < _myLeft[2] ||
498  cit->position()[0] + 2 * _offset > _myLeft[0] + _myNi[0]*_a ||
499  cit->position()[1] + 2 * _offset > _myLeft[1] + _myNi[1]*_a ||
500  cit->position()[2] + 2 * _offset > _myLeft[2] + _myNi[2]*_a ) {
501  _myDecompose = true;
502  }
503  }
504 
505  boost::mpi::all_reduce(*system.comm,
506  _myDecompose, decompose, std::plus<bool>());
507 
508  if (decompose) {
509  system.storage->decompose();
510  }
511 
512  }

516  /* COLLIDE-STREAM STEP */
517  void LatticeBoltzmann::collideStream () {
518  int _offset = getHaloSkin();
519  bool _extForce = doExtForce();
520  bool _fluct = doFluct();
521  bool _coupling = doCoupling();
522  Int3D _myNi = getMyNi();
523 
524  // copy forces from halo region to the real one //
525  if (_coupling) {
526  copyForcesFromHalo();
527  }
528 
529  // collision-streaming //
530  real timer = colstream.getElapsedTime();
531  for (int i = _offset; i < _myNi[0]-_offset; i++) {

646  void LatticeBoltzmann::calcViscForce (Particle& p) {
647  int _offset = getHaloSkin();
648  real _a = getA();
649  real _invA = 1. / _a;
650  real _fricCoeff = getFricCoeff();
651  Real3D Li = getSystem()->bc->getBoxL();
652 
653  // account for particle's positions with respect to CPU's left border
654  Real3D _pos = p.position() - getMyLeft();
655  Real3D _posLB = ( _pos + (double)_offset ) * _invA;
656 
657  Int3D bin = Int3D( floor(_posLB[0]), floor(_posLB[1]), floor(_posLB[2]));
658 
659  // weight factors, dimensionless
660  std::vector<real> delta = std::vector<real>(6, 0.);
661  delta[0] = _posLB[0] - bin[0];
662  delta[1] = _posLB[1] - bin[1];
663  delta[2] = _posLB[2] - bin[2];
664  delta[3] = _a - delta[0];
665  delta[4] = _a - delta[1];
666  delta[5] = _a - delta[2];
667 
668  real _convTimeMDtoLB = convTimeMDtoLB();
669  real _convLenMDtoLB = convLenMDtoLB();
670  real _convMassMDtoLB = convMassMDtoLB();
671  real _convCoeff = _convTimeMDtoLB / _convLenMDtoLB;
672 
673  Real3D interpVel = Real3D (0.);
674  // loop over neighboring LB nodes
675  int _ip, _jp, _kp;
676  for (int _i = 0; _i < 2; _i++) {
677  for (int _j = 0; _j < 2; _j++) {
678  for (int _k = 0; _k < 2; _k++) {
679  // assign iterations
680  _ip = bin[0] + _i; _jp = bin[1] + _j; _kp = bin[2] + _k;
681 
682  // force acting onto the fluid node at the moment (midpoint scheme)
683  Real3D _f = (*lbfor)[_ip][_jp][_kp].getExtForceLoc()
684  + (*lbfor)[_ip][_jp][_kp].getCouplForceLoc();
685  Real3D _jLoc = Real3D((*lbmom)[_ip][_jp][_kp].getMom_i(1)+_f[0],
686  (*lbmom)[_ip][_jp][_kp].getMom_i(2)+_f[1],
687  (*lbmom)[_ip][_jp][_kp].getMom_i(3)+_f[2] );
688  real _invDenLoc = 1. / (*lbmom)[_ip][_jp][_kp].getMom_i(0);
689 
690  Real3D _u = _jLoc * _invDenLoc * _convCoeff;
691  interpVel += _u * delta[3 * _i] * delta[3 * _j + 1] * delta[3 * _k + 2];
692  }
693  }
694  }
695 
696  // add visc force to the buffered rand force acting onto particle p.id()
697  addFOnPart(p.id(), -_fricCoeff * (p.velocity() - interpVel));
698 
699  // apply buffered force to the MD-particle p.id()
700  p.force() += getFOnPart(p.id());
701 
702  // convert coupl force (LJ units) to mom change on a lattice (LB units)
703  Real3D deltaJLoc = Real3D(0.);
704  deltaJLoc -= getFOnPart(p.id()) * _convMassMDtoLB / (_convCoeff * _convTimeMDtoLB);
705 
706  // loop over neighboring LB nodes
707  for (int _i = 0; _i < 2; _i++) {
708  for (int _j = 0; _j < 2; _j++) {
709  for (int _k = 0; _k < 2; _k++) {
710  // PBC on the right (left is safe)
711  _ip = bin[0] + _i; _jp = bin[1] + _j; _kp = bin[2] + _k;
712 
713  // converting momentum into coupling force with weights delta[i]
714  Real3D _fLoc = deltaJLoc;
715  _fLoc *= delta[3*_i]; _fLoc *= delta[3*_j+1]; _fLoc *= delta[3*_k+2];
716 
717  // add coupling force to the correspondent lattice cite
718  (*lbfor)[_ip][_jp][_kp].addCouplForceLoc(_fLoc);
719  }
720  }
721  }
722  }
