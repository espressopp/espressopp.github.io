785  void LatticeBoltzmann::calcViscForce (Particle& p) {
786  int _offset = getHaloSkin();
787  real _a = getA();
788  real _invA = 1. / _a;
789  real _fricCoeff = getFricCoeff();
790  Real3D Li = getSystem()->bc->getBoxL();
791 
792  // account for particle's positions with respect to CPU's left border
793  Real3D _pos = p.position() - getMyLeft();
794 
795  Real3D _posLB = _pos;
796  _posLB += (double)_offset;
797  _posLB *= _invA;
798 
799  Int3D bin;
800  bin[0] = floor (_posLB[0]); bin[1] = floor (_posLB[1]); bin[2] = floor (_posLB[2]);
801 
802  // weight factors, dimensionless
803  std::vector<real> delta = std::vector<real>(6, 0.);
804  delta[0] = _posLB[0] - bin[0];
805  delta[1] = _posLB[1] - bin[1];
806  delta[2] = _posLB[2] - bin[2];
807  delta[3] = _a - delta[0];
808  delta[4] = _a - delta[1];
809  delta[5] = _a - delta[2];
810 
811  real _convTimeMDtoLB = convTimeMDtoLB();
812  real _convLenMDtoLB = convLenMDtoLB();
813  real _convMassMDtoLB = convMassMDtoLB();
814  int _numVels = getNumVels();
815  real _invDenLoc;
816  Real3D _jLoc;
817  Real3D _u;
818 
819  Real3D interpVel = Real3D (0.);
820  real _convCoeff = _convTimeMDtoLB / _convLenMDtoLB;
821  // loop over neighboring LB nodes
822  int _ip, _jp, _kp;
823  for (int _i = 0; _i < 2; _i++) {
824  for (int _j = 0; _j < 2; _j++) {
825  for (int _k = 0; _k < 2; _k++) {
826  // assign iterations
827  _ip = bin[0] + _i; _jp = bin[1] + _j; _kp = bin[2] + _k;
828 
829  _invDenLoc = 1. / (*lbmom)[_ip][_jp][_kp].getMom_i(0);
830  _jLoc[0] = (*lbmom)[_ip][_jp][_kp].getMom_i(1);
831  _jLoc[1] = (*lbmom)[_ip][_jp][_kp].getMom_i(2);
832  _jLoc[2] = (*lbmom)[_ip][_jp][_kp].getMom_i(3);
833 
834  _u = _jLoc;
835  _u *= _invDenLoc;
836  _u *= _convCoeff;
837 
838  interpVel += _u * delta[3 * _i] * delta[3 * _j + 1] * delta[3 * _k + 2];
839  }
840  }
841  }
842 
843  // add viscous force to the buffered random force acting onto particle p.id()
844  addFOnPart(p.id(), -_fricCoeff * (p.velocity() - interpVel));
845 
846  // apply buffered force to the MD-particle p.id()
847  p.force() += getFOnPart(p.id());
848 
849  // convert coupling force (LJ units) to the momentum change on a lattice (LB units)
850  Real3D deltaJLoc = Real3D(0.);
851  deltaJLoc -= getFOnPart(p.id()) * _convMassMDtoLB / (_convCoeff * _convTimeMDtoLB);
852 
853  // loop over neighboring LB nodes
854  for (int _i = 0; _i < 2; _i++) {
855  for (int _j = 0; _j < 2; _j++) {
856  for (int _k = 0; _k < 2; _k++) {
857  // periodic boundaries on the right side
858  _ip = bin[0] + _i; _jp = bin[1] + _j; _kp = bin[2] + _k;
859 
860  // converting momentum into coupling force with weights delta[i]
861  Real3D _fLoc = deltaJLoc;
862  _fLoc *= delta[3*_i]; _fLoc *= delta[3*_j+1]; _fLoc *= delta[3*_k+2];
863 
864  // add coupling force to the correspondent lattice cite
865  (*lbfor)[_ip][_jp][_kp].addCouplForceLoc(_fLoc);
866  }
867  }
868  }
869  }
