143  real _computeEnergy(const Real3D& r12, const Real3D& r32) const {
144  // 2 is central particle
145  real d12 = r12.abs();
146  real d32 = r32.abs();
147 
148  // probably it should be moved to AngularPotential.hpp
149  // should check does it good for cosine etc.
150  if (d12 >= rc1 || d32 >= rc2 )
151  return 0.0;
152  else{
153  real inv_d12_a = 1.0 / (d12 - sigmarc1);
154  real inv_d32_a = 1.0 / (d32 - sigmarc2);
155 
156  real cosTeta123 = (r12 * r32) / (d12 * d32);
157  real difCos = cosTeta123 - cosTeta0;
158  real difCos2 = difCos * difCos;
159 
160  real expProduct = exp(sigmaGamma1 * inv_d12_a + sigmaGamma2 * inv_d32_a);
161 
162  real energy3 = epsilonLambda * expProduct * difCos2;
163 
164  return energy3;
165  }

168  bool _computeForceRaw(Real3D& force12, Real3D& force32,
169  const Real3D& r12, const Real3D& r32) const{
170 
171  real d12 = r12.abs();
172  real d32 = r32.abs();
173 
174  if (d12 >= rc1 || d32 >= rc2 ){
175  force12 = 0.0;
176  force32 = 0.0;
177  return false;
178  }
179  else{
180  real inv_d12 = 1.0 / d12;
181  real inv_d32 = 1.0 / d32;
182 
183  Real3D e12 = r12 * inv_d12;
184  Real3D e32 = r32 * inv_d32;
185 
186  real inv_d12_a = 1.0 / (d12 - sigmarc1);
187  real inv_d32_a = 1.0 / (d32 - sigmarc2);
188 
189  real inv_d12_a2 = inv_d12_a*inv_d12_a;
190  real inv_d32_a2 = inv_d32_a*inv_d32_a;
191 
192  real cosTeta123 = (r12 * r32) * ( inv_d12 * inv_d32 );
193  real difCos = cosTeta123 - cosTeta0;
194  real difCos2 = difCos * difCos;
195 
196  real expProduct = epsilonLambda *
197  exp( sigmaGamma1*inv_d12_a+sigmaGamma2*inv_d32_a);
198 
199  real energy3 = expProduct * difCos2;
200 
201  real factor1 = sigmaGamma1 * energy3;
202  real factor2 = sigmaGamma2 * energy3;
203 
204  real expTerm = 2.0 * expProduct * difCos;
205 
206  force12 = factor1 * e12 * inv_d12_a2 -
207  expTerm * (e32 - e12 * cosTeta123) * inv_d12;
208 
209  force32 = factor2 * e32 * inv_d32_a2 -
210  expTerm * (e12 - e32 * cosTeta123) * inv_d32;
211  return true;
212  }
