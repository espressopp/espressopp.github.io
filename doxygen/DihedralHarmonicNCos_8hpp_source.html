197  /*
198  * Compute raw force for the tabulated potential, depend only on the phi value
199  * \f$F(\phi) = -grad V = -grad [ K*(1+cos(n*\phi - \phi_0)) ] = K*n*sin(\phi_0 - n*\phi) \f$
200  * where n is a multiplicity and \f$phi_0\f$ is the reference angle
201  *
202  * @param[in] phi Degree in radians
203  *
204  * @return The value of the force
205  */
206  real _computeForceRaw(real phi) const {
207  real sin_phi = sin(phi);
208  if (fabs(sin_phi) < 1e-9) {
209  if (sin_phi>0.0) sin_phi = 1e-9;
210  else sin_phi = -1e-9;
211  }
212  return -1.0*(1.0/sin(phi))*K * multiplicity * sin(phi0 - multiplicity*phi);
213  }
