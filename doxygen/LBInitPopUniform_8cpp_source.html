34  void LBInitPopUniform::createDenVel (real _rho0, Real3D _u0) {
35  if (mpiWorld->rank()==0) {
36  printf("Creating an initial configuration with uniform density %f and \nvelocity %f, %f, %f\n",
37  _rho0, _u0.getItem(0), _u0.getItem(1), _u0.getItem(2));
38  printf("-------------------------------------\n");
39  } else {
40  // do nothing
41  }
42 
43  real invCs2 = 1. / latticeboltzmann->getCs2();
44 
45  real invCs4 = invCs2*invCs2;
46  real scalp, value;
47  int _offset = latticeboltzmann->getHaloSkin();
48  Int3D _Ni = latticeboltzmann->getMyNi();
49  int _numVels = latticeboltzmann->getNumVels(); // number of velocities in the model
50 
51  // set initial velocity of the populations from Maxwell's distribution
52  for (int i = 0; i < _Ni.getItem(0); i++) {
53  real trace = _u0 * _u0 * invCs2;
54  for (int j = 0; j < _Ni.getItem(1); j++) {
55  for (int k = 0; k < _Ni.getItem(2); k++) {
56  for (int l = 0; l < _numVels; l++) {
57  scalp = _u0 * latticeboltzmann->getCi(l);
58  value = 0.5 * latticeboltzmann->getEqWeight(l) * _rho0 * (2. + 2. * scalp * invCs2 + scalp * scalp * invCs4 - trace);
59  latticeboltzmann->setPops( Int3D(i,j,k), l, value );
60  latticeboltzmann->setGhostFluid( Int3D(i,j,k), l, 0.0 );
61  }
62  /* fill in den and j values for real and halo regions */
63  latticeboltzmann->setLBMom(Int3D(i,j,k), 0, _rho0);
64  latticeboltzmann->setLBMom(Int3D(i,j,k), 1, _u0[0] * _rho0);
65  latticeboltzmann->setLBMom(Int3D(i,j,k), 2, _u0[1] * _rho0);
66  latticeboltzmann->setLBMom(Int3D(i,j,k), 3, _u0[2] * _rho0);
67  }
68  }
69  }
70  }
