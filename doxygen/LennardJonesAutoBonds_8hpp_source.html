179  inline bool _computeForce(Real3D& force, const Particle &p1, const Particle &p2) const {
180  Real3D dist = p1.position() - p2.position();
181  if (p1.ghost()) {
182  printf("WARNING (this should never happen !): didn't add crosslink, because particle1 is ghosts");
183  return _computeForce(force, dist);
184  }
185  if (dist.sqr() <= cutoffSqr) {
186  FixedPairList::GlobalPairs* globalPairs = bondlist->getGlobalPairs();
187  if (globalPairs->count(p1.id()) + globalPairs->count(p2.id()) < max_crosslinks) {
188  globalPairs->insert(globalPairs->begin(), std::make_pair(p1.id(), p2.id()));
189  bondlist->add(p1.id(), p2.id());
190  };
191  }
192  return _computeForce(force, dist);
193  }

204  real _computeEnergySqrRaw(real distSqr) const {
205  real frac2 = sigma*sigma / distSqr;
206  real frac6 = frac2 * frac2 * frac2;
207  real energy = 4.0 * epsilon * (frac6 * frac6 - frac6);
208  return energy;
209  }
210 
211  bool _computeForceRaw(Real3D& force,
212  const Real3D& dist,
213  real distSqr) const {
214 
215  real frac2 = 1.0 / distSqr;
216  real frac6 = frac2 * frac2 * frac2;
217  real ffactor = frac6 * (ff1 * frac6 - ff2) * frac2;
218  force = dist * ffactor;
219  return true;
220  }
