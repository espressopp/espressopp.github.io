00168     void InterpolationCubic::spline(const real *x, const real *y, int n,
00169                                     real yp1, real ypn, real *y2) {
00170       real *u = new real[n];
00171      
00172       if (yp1 > 0.99e30) {
00173         y2[0] = 0.0;
00174         u[0]  = 0.0;
00175       }
00176       else {
00177         y2[0] = -0.5;
00178         u[0]  = (3.0/(x[1]-x[0])) * ((y[1]-y[0]) / (x[1]-x[0]) - yp1);
00179       }
00180      
00181       for (int i = 1; i < n-1; i++) {
00182      
00183         real sig = (x[i] - x[i-1]) / (x[i+1]-x[i-1]);
00184         real p   = sig * y2[i-1] + 2.0;
00185      
00186         y2[i] = (sig-1.0) / p;
00187         u[i]  = (y[i+1] - y[i]) / (x[i+1] - x[i]) -
00188                 (y[i] - y[i-1]) / (x[i] - x[i-1]);
00189         u[i]  = (6.0*u[i] / (x[i+1]-x[i-1]) - sig*u[i-1]) / p;
00190       }
00191      
00192       real qn, un;
00193      
00194       if (ypn > 0.99e30) {
00195         qn = 0.0;
00196         un = 0.0;
00197       } else {
00198         qn = 0.5;
00199         un = (3.0 / (x[n-1]-x[n-2])) * (ypn - (y[n-1]-y[n-2]) / (x[n-1]-x[n-2]));
00200       }
00201      
00202       y2[n-1] = (un - qn * u[n-2]) / (qn * y2[n-2] + 1.0);
00203         
00204       for (int k = n-2; k >= 0; k--) {
00205         y2[k] = y2[k] * y2[k+1] + u[k];
00206       }
00207      
00208       delete [] u;
00209     }// spline
