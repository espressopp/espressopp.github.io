52 void DumpXTCAdress::dump() {
53  shared_ptr<System> system = getSystem();
54  analysis::ConfigurationsExtAdress conf(system, ftpl);
55  conf.setUnfolded(unfolded);
56  conf.gather();
57 
58  if (system->comm->rank() == 0) {
59  analysis::ConfigurationExtPtr conf_real = conf.back();
60 
61  int num_of_particles = conf_real->getSize();
62 
63  if (this->open("a")) {
64  analysis::ConfigurationExtIterator cei = conf_real-> getIterator();
65  matrix box;
66  rvec *coord = new rvec[num_of_particles];
67  RealND props;
68  props.setDimension(cei.currentProperties().getDimension());
69 
70  // TODO(anyone): this works only for orthorombic BC
71  Real3D bl = system->bc->getBoxL();
72 
73  for (int i = 0; i < dim; i++) {
74  box[i][0] = 0.;
75  box[i][1] = 0.;
76  box[i][2] = 0.;
77 
78  box[i][i] = bl[i];
79  }
80 
81  for (int i = 0; i < num_of_particles; i++) {
82  props = cei.nextProperties();
83  coord[i][0] = props[0]*length_factor; // We only write coordinates to .xtc
84  coord[i][1] = props[1]*length_factor;
85  coord[i][2] = props[2]*length_factor;
86  }
87 
88  int step = integrator->getStep();
89  float time = integrator->getTimeStep()*step;
90 
91  write_xtc(fio, num_of_particles, step, time, box, coord, xtcprec);
92 
93  delete [] coord;
94 
95  this->close();
96  } else {
97  std::cout << "Unable to open file: "<< file_name << std::endl;
98  }
99  }
100 }
