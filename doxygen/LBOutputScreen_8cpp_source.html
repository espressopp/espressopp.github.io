33  void LBOutputScreen::writeOutput() {
34  /* It is a not straightforward way. The idea is that the Output onto
35  the Screen takes place when the step is already finished, i.e. at
36  (t+ dt). However, the LB couples to the integrator signal befIntV,
37  i.e. at (t+1/2dt). Therefore, to check momentum conservation for LB+MD,
38  it is necessary to go forward in time by 1/2dt in MD-part. This is
39  done by employing the same forces as for integrate1() but withont
40  actually altering velocities of MD-particles. */
41 
42  bool _restart = latticeboltzmann->doRestart();
43  real _timestep = latticeboltzmann->getCopyTimestep();
44 
45  Real3D _myCMVel = Real3D(0.); // cm velocity (current CPU)
46  Real3D _totCMVel = Real3D(0.); // cm velocity (all CPUs)
47 
48  System& system = getSystemRef();
49  CellList realCells = system.storage->getRealCells();
50 
51  // loop over all particles in the curr CPU //
52  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
53  Real3D& vel = cit->velocity();
54  Real3D& force = cit->force();
55  _myCMVel += vel + 0.5 * _timestep * force;
56  }
57 
58  if (_restart) {
59  // collect cm vels and share the result between CPUs
60  boost::mpi::all_reduce(*system.comm,
61  _myCMVel, _totCMVel, std::plus<Real3D>());
62  findLBMom(1);
63 
64  // correct for difference between LB and MD momenta
65  Real3D corrForCMVel = getLBMom() + _totCMVel;
66  Real3D corrPerPart = corrForCMVel / latticeboltzmann->getTotNPart();
67  latticeboltzmann->galileanTransf(corrPerPart);
68 
69  // set flag after restart
70  latticeboltzmann->setDoRestart(false);
71  } else {
72  // collect cm vels from CPUs
73  boost::mpi::reduce(*system.comm,
74  _myCMVel, _totCMVel, std::plus<Real3D>(), 0);
75  findLBMom(0);
76  }
77 
78  // output MD momentum and statistics of the run //
79  if (system.comm->rank() == 0) {
80  Int3D _Ni = latticeboltzmann->getNi();
81  long int latSize = _Ni[0] * _Ni[1] * _Ni[2];
82  long int _step = latticeboltzmann->getStepNum();
83  real _invNSteps = 1. / latticeboltzmann->getNSteps();
84  long int madeLBSteps = (long int)((_step-getOldStepNum()) * _invNSteps);
85 
86  // output MD momentum //
87  setMDMom(_totCMVel); // need this for testsuite
88  printf("MD mom at (+1/2dt): %17.12f %17.12f %17.12f \n",
89  _totCMVel[0], _totCMVel[1], _totCMVel[2]);
90 
91  if (_step != 0) {
92  // calculate time performance
93  setLBTimerNew( timeLBtoMD.getElapsedTime() );
94  real timelb = getLBTimerNew() - getLBTimerOld();
95  setLBTimerOld( getLBTimerNew() );
96 
97  // output LB statistics
98  printf ("%ld LB steps on %ld nodes done in %f sec, relative MLUPS: %f \n",
99  madeLBSteps, latSize, timelb,
100  madeLBSteps * latSize * 1e-6 / timelb);
101  printf ("Speed in LJ units: %f tau / sec \n",
102  (_step - getOldStepNum()) * _timestep / timelb);
103 
104  } else {
105  // timers //
106  timeLBtoMD.reset();
107  setLBTimerOld(0.);
108  }
109 
110  setOldStepNum(_step);
111 
112  printf("------------------------------------");
113  printf("-------------------------------------\n");
114  }
115  }

117  void LBOutputScreen::findLBMom (int _mode) {
118  Real3D result = Real3D(0.);
119  Real3D _myMom = Real3D(0.);
120  Int3D _Ni = latticeboltzmann->getMyNi();
121  int _offset = latticeboltzmann->getHaloSkin();
122 
123  for (int i = _offset; i < _Ni[0] - _offset; i++) {
124  for (int j = _offset; j < _Ni[1] - _offset; j++) {
125  for (int k = _offset; k < _Ni[2] - _offset; k++) {
126  _myMom += Real3D( latticeboltzmann->getLBMom(Int3D(i,j,k), 1),
127  latticeboltzmann->getLBMom(Int3D(i,j,k), 2),
128  latticeboltzmann->getLBMom(Int3D(i,j,k), 3) );
129  }
130  }
131  }
132  _myMom *= (latticeboltzmann->convTimeMDtoLB() / latticeboltzmann->getA());
133 
134  System& system = getSystemRef();
135 
136  if (_mode == 0) {
137  // collect momenta from CPUs
138  boost::mpi::reduce(*system.comm,
139  _myMom, result, std::plus<Real3D>(), 0);
140  } else {
141  // collect momenta and share the result between CPUs
142  boost::mpi::all_reduce(*system.comm,
143  _myMom, result, std::plus<Real3D>());
144  setLBMom(result);
145  }
146 
147  // output LB momentum //
148  if (system.comm->rank() == 0) {
149  setLBMom(result); // need this for testsuite
150  printf ("after step %d:\n", latticeboltzmann->getStepNum());
151  printf ("LB mom in LJ units: %17.12f %17.12f %17.12f \n",
152  result[0], result[1], result[2]);
153  }
154  }
