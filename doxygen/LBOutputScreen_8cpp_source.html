133  void LBOutputScreen::findLBMom () {
134  real _fi = 0.;
135  Real3D result = Real3D(0.,0.,0.);
136  Real3D _myU = Real3D(0.,0.,0.);
137  Real3D _ci = Real3D(0.,0.,0.);
138  Int3D _Ni = latticeboltzmann->getMyNi();
139  int _offset = latticeboltzmann->getHaloSkin();
140  int _numVels = latticeboltzmann->getNumVels();
141 
142  for (int i = _offset; i < _Ni[0] - _offset; i++) {
143  for (int j = _offset; j < _Ni[1] - _offset; j++) {
144  for (int k = _offset; k < _Ni[2] - _offset; k++) {
145  Real3D _jLoc = Real3D(0.,0.,0.);
146 
147  // calculation of density and momentum flux on the lattice site
148  for (int l = 0; l < _numVels; l++) {
149  _fi = latticeboltzmann->getLBFluid(Int3D(i,j,k),l);
150  _ci = latticeboltzmann->getCi(l);
151  _jLoc += _fi * _ci;
152  }
153 
154 /* _jLoc[0] = latticeboltzmann->getLBMom(Int3D(i,j,k),1);
155  _jLoc[1] = latticeboltzmann->getLBMom(Int3D(i,j,k),2);
156  _jLoc[2] = latticeboltzmann->getLBMom(Int3D(i,j,k),3);
157 */
158  _myU += _jLoc;
159  }
160  }
161  }
162  _myU *= (latticeboltzmann->convTimeMDtoLB());
163  _myU /= (latticeboltzmann->getA());

165  boost::mpi::reduce(*getSystem()->comm, _myU, result, std::plus<Real3D>(), 0);

168  int _step = latticeboltzmann->getStepNum();

171  printf("LB-fluid mom after streaming (LJ units) is %18.14f %18.14f %18.14f \n",
