233  real VelocityVerlet::integrate1()
234  {
235  System& system = getSystemRef();
236  CellList realCells = system.storage->getRealCells();
237 
238  // loop over all particles of the local cells
239  int count = 0;
240  real maxSqDist = 0.0; // maximal square distance a particle moves
241  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
242  real sqDist = 0.0;
243  LOG4ESPP_INFO(theLogger, "updating first half step of velocities and full step of positions")
244  LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() <<
245  ", pos = " << cit->position() <<
246  ", v = " << cit->velocity() <<
247  ", f = " << cit->force());
248 
249  /* more precise for DEBUG:
250  printf("Particle %d, pos = %16.12f %16.12f %16.12f, v = %16.12f, %16.12f %16.12f, f = %16.12f %16.12f %16.12f\n",
251  cit->p.id, cit->r.p[0], cit->r.p[1], cit->r.p[2],
252  cit->m.v[0], cit->m.v[1], cit->m.v[2],
253  cit->f.f[0], cit->f.f[1], cit->f.f[2]);
254  */
255 
256  real dtfm = 0.5 * dt / cit->mass();
257 
258  // Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * f(t)
259  cit->velocity() += dtfm * cit->force();
260 
261  // Propagate positions (only NVT): p(t + dt) = p(t) + dt * v(t+0.5*dt)
262  Real3D deltaP = cit->velocity();
263 
264  deltaP *= dt;
265  cit->position() += deltaP;
266  sqDist += deltaP * deltaP;
267 
268  count++;
269 
270  maxSqDist = std::max(maxSqDist, sqDist);
271  }
272 
273  // signal
274  inIntP(maxSqDist);
275 
276  real maxAllSqDist;
277  mpi::all_reduce(*system.comm, maxSqDist, maxAllSqDist, boost::mpi::maximum<real>());
278 
279  LOG4ESPP_INFO(theLogger, "moved " << count << " particles in integrate1" <<
280  ", max move local = " << sqrt(maxSqDist) <<
281  ", global = " << sqrt(maxAllSqDist));
282 
283  return sqrt(maxAllSqDist);
284  }
