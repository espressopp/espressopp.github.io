00233     real VelocityVerlet::integrate1()
00234     {
00235       System& system = getSystemRef();
00236       CellList realCells = system.storage->getRealCells();
00237 
00238       // loop over all particles of the local cells
00239       int count = 0;
00240       real maxSqDist = 0.0; // maximal square distance a particle moves
00241       for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
00242         real sqDist = 0.0;
00243         LOG4ESPP_INFO(theLogger, "updating first half step of velocities and full step of positions")
00244         LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() << 
00245                 ", pos = " << cit->position() <<
00246                 ", v = " << cit->velocity() << 
00247                 ", f = " << cit->force());
00248 
00249         /* more precise for DEBUG:
00250         printf("Particle %d, pos = %16.12f %16.12f %16.12f, v = %16.12f, %16.12f %16.12f, f = %16.12f %16.12f %16.12f\n",
00251             cit->p.id, cit->r.p[0], cit->r.p[1], cit->r.p[2],
00252                 cit->m.v[0], cit->m.v[1], cit->m.v[2],
00253             cit->f.f[0], cit->f.f[1], cit->f.f[2]);
00254         */
00255 
00256         real dtfm = 0.5 * dt / cit->mass();
00257 
00258         // Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * f(t) 
00259         cit->velocity() += dtfm * cit->force();
00260 
00261         // Propagate positions (only NVT): p(t + dt) = p(t) + dt * v(t+0.5*dt) 
00262         Real3D deltaP = cit->velocity();
00263         
00264         deltaP *= dt;
00265         cit->position() += deltaP;
00266         sqDist += deltaP * deltaP;
00267 
00268         count++;
00269 
00270         maxSqDist = std::max(maxSqDist, sqDist);
00271       }
00272       
00273       // signal
00274       inIntP(maxSqDist);
00275 
00276       real maxAllSqDist;
00277       mpi::all_reduce(*system.comm, maxSqDist, maxAllSqDist, boost::mpi::maximum<real>());
00278 
00279       LOG4ESPP_INFO(theLogger, "moved " << count << " particles in integrate1" <<
00280                     ", max move local = " << sqrt(maxSqDist) <<
00281                     ", global = " << sqrt(maxAllSqDist));
00282       
00283       return sqrt(maxAllSqDist);
00284     }
