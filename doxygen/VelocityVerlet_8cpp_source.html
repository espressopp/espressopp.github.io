63  void VelocityVerlet::run(int nsteps)
64  {
65  VT_TRACER("run");
66  int nResorts = 0;
67  real time;
68  timeIntegrate.reset();
69  resetTimers();
70  System& system = getSystemRef();
71  storage::Storage& storage = *system.storage;
72  real skinHalf = 0.5 * system.getSkin();
73 
74  // signal
75  runInit();
76 
77  // Before start make sure that particles are on the right processor
78  if (resortFlag) {
79  VT_TRACER("resort");
80  // time = timeIntegrate.getElapsedTime();
81  LOG4ESPP_INFO(theLogger, "resort particles");
82  storage.decompose();
83  maxDist = 0.0;
84  resortFlag = false;
85  // timeResort += timeIntegrate.getElapsedTime();
86  }
87 
88  bool recalcForces = true; // TODO: more intelligent
89 
90  if (recalcForces) {
91  LOG4ESPP_INFO(theLogger, "recalc forces before starting main integration loop");
92 
93  // signal
94  recalc1();
95 
96  updateForces();
97  if (LOG4ESPP_DEBUG_ON(theLogger)) {
98  // printForces(false); // forces are reduced to real particles
99  }
100 
101  // signal
102  recalc2();
103  }
104 
105  LOG4ESPP_INFO(theLogger, "starting main integration loop (nsteps=" << nsteps << ")");
106 
107  for (int i = 0; i < nsteps; i++) {
108  LOG4ESPP_INFO(theLogger, "Next step " << i << " of " << nsteps << " starts");
109 
110  //saveOldPos(); // save particle positions needed for constraints
111 
112  // signal
113  befIntP();
114 
115  time = timeIntegrate.getElapsedTime();
116  LOG4ESPP_INFO(theLogger, "updating positions and velocities")
117  maxDist += integrate1();
118  timeInt1 += timeIntegrate.getElapsedTime() - time;
119 
120  /*
121  real cellsize = 1.4411685442;
122  if (maxDist > 1.4411685442){
123  cout<<"WARNING!!!!!! huge jump: "<<maxDist<<endl;
124  exit(1);
125  }*/
126 
127  // signal
128  aftIntP();
129 
130  LOG4ESPP_INFO(theLogger, "maxDist = " << maxDist << ", skin/2 = " << skinHalf);
131 
132  if (maxDist > skinHalf) resortFlag = true;
133 
134  if (resortFlag) {
135  VT_TRACER("resort1");
136  time = timeIntegrate.getElapsedTime();
137  LOG4ESPP_INFO(theLogger, "step " << i << ": resort particles");
138  storage.decompose();
139  maxDist = 0.0;
140  resortFlag = false;
141  nResorts ++;
142  timeResort += timeIntegrate.getElapsedTime() - time;
143  }
144 
145  LOG4ESPP_INFO(theLogger, "updating forces")
146  updateForces();
147 
148  // signal
149  befIntV();
150 
151  time = timeIntegrate.getElapsedTime();
152  integrate2();
153  timeInt2 += timeIntegrate.getElapsedTime() - time;
154 
155  // signal
156  aftIntV();
157  }
158 
159  timeRun = timeIntegrate.getElapsedTime();
160  timeLost = timeRun - (timeForceComp[0] + timeForceComp[1] + timeForceComp[2] +
161  timeComm1 + timeComm2 + timeInt1 + timeInt2 + timeResort);
162 
163  LOG4ESPP_INFO(theLogger, "finished run");
164  }

236  real VelocityVerlet::integrate1()
237  {
238  System& system = getSystemRef();
239  CellList realCells = system.storage->getRealCells();
240 
241  // loop over all particles of the local cells
242  int count = 0;
243  real maxSqDist = 0.0; // maximal square distance a particle moves
244  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
245  real sqDist = 0.0;
246  LOG4ESPP_INFO(theLogger, "updating first half step of velocities and full step of positions")
247  LOG4ESPP_DEBUG(theLogger, "Particle " << cit->id() <<
248  ", pos = " << cit->position() <<
249  ", v = " << cit->velocity() <<
250  ", f = " << cit->force());
251 
252  /* more precise for DEBUG:
253  printf("Particle %d, pos = %16.12f %16.12f %16.12f, v = %16.12f, %16.12f %16.12f, f = %16.12f %16.12f %16.12f\n",
254  cit->p.id, cit->r.p[0], cit->r.p[1], cit->r.p[2],
255  cit->m.v[0], cit->m.v[1], cit->m.v[2],
256  cit->f.f[0], cit->f.f[1], cit->f.f[2]);
257  */
258 
259  real dtfm = 0.5 * dt / cit->mass();
260 
261  // Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * f(t)
262  cit->velocity() += dtfm * cit->force();
263 
264  // Propagate positions (only NVT): p(t + dt) = p(t) + dt * v(t+0.5*dt)
265  Real3D deltaP = cit->velocity();
266 
267  deltaP *= dt;
268  cit->position() += deltaP;
269  sqDist += deltaP * deltaP;
270 
271  count++;
272 
273  maxSqDist = std::max(maxSqDist, sqDist);
274  }
275 
276  // signal
277  inIntP(maxSqDist);
278 
279  real maxAllSqDist;
280  mpi::all_reduce(*system.comm, maxSqDist, maxAllSqDist, boost::mpi::maximum<real>());
281 
282  LOG4ESPP_INFO(theLogger, "moved " << count << " particles in integrate1" <<
283  ", max move local = " << sqrt(maxSqDist) <<
284  ", global = " << sqrt(maxAllSqDist));
285 
286  return sqrt(maxAllSqDist);
287  }

289  void VelocityVerlet::integrate2()
290  {
291  LOG4ESPP_INFO(theLogger, "updating second half step of velocities")
292  System& system = getSystemRef();
293  CellList realCells = system.storage->getRealCells();
294 
295  // loop over all particles of the local cells
296  real half_dt = 0.5 * dt;
297  for(CellListIterator cit(realCells); !cit.isDone(); ++cit) {
298  real dtfm = half_dt / cit->mass();
299  /* Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * f(t) */
300  cit->velocity() += dtfm * cit->force();
301  }
302 
303  step++;
