152  template< bool VEC_MODE_AOS, bool PACK_NEIGHBORS >
153  void VerletList::rebuild_p_nc_pack_stencil()
154  {
155  {
156  const auto& cellNborList = vec->getNeighborList();
157  const auto& particleArray = vec->getParticleArray();
158  const size_t* __restrict cellRange = &(particleArray.cellRange()[0]);
159  const size_t* __restrict sizes = &(particleArray.sizes()[0]);
160 
161  const auto* __restrict position = &(particleArray.position[0]);
162  const auto* __restrict pa_p_x = &(particleArray.p_x[0]);
163  const auto* __restrict pa_p_y = &(particleArray.p_y[0]);
164  const auto* __restrict pa_p_z = &(particleArray.p_z[0]);
165  const auto* __restrict pa_p_type = &(particleArray.type[0]);
166 
167  // number of cells with neighbors
168  const size_t numRealCells = cellNborList.numCells();
169  const size_t numCells = particleArray.sizes().size();
170 
171  neighborList.clist.resize(numRealCells);
172  neighborList.crange.resize(numRealCells);
173 
175  size_t plist_reserve = 0;
176  for(size_t irow=0; irow<numRealCells; irow++)
177  {
178  plist_reserve += sizes[cellNborList.cellId(irow)];
179  }
180  neighborList.plist.reserve(plist_reserve);
181  neighborList.prange.reserve(plist_reserve);
182 
183  size_t max_nneighbors = cellNborList.maxNumNeighbors();
184  size_t max_cell_size = 0, nplist_reserve;
185  for(size_t icell=0; icell<numCells; icell++)
186  max_cell_size = std::max(max_cell_size, sizes[icell]);
187  size_t max_pairs_per_cell = max_cell_size*max_cell_size*(max_nneighbors+1);
188  nplist_reserve = neighborList.nplist.size(); // re-use previous allocation
189 
192 
193  std::vector<int> c_range;
194 
195  if(PACK_NEIGHBORS)
196  {
197  c_range.reserve(numRealCells+1);
198  size_t vec_max_cell_size = ((max_cell_size+ESPP_VECTOR_WIDTH-1)/ESPP_VECTOR_WIDTH)*ESPP_VECTOR_WIDTH;
199  size_t c_reserve = numRealCells*max_nneighbors*vec_max_cell_size;
200  if(c_reserve > c_j.size())
201  {
202  c_j.resize(c_reserve);
203  c_x.resize(c_reserve);
204  c_y.resize(c_reserve);
205  c_z.resize(c_reserve);
206  }
207  }
208 
209  auto* __restrict c_x_ptr = &(c_x[0]);
210  auto* __restrict c_y_ptr = &(c_y[0]);
211  auto* __restrict c_z_ptr = &(c_z[0]);
212  auto* __restrict c_j_ptr = &(c_j[0]);
213  int c_j_max = 0;
214 
215  if(PACK_NEIGHBORS)
216  {
217  c_range.push_back(c_j_max);
218  for(size_t irow=0; irow<numRealCells; irow++)
219  {
220  const size_t cell_nnbrs = cellNborList.numNeighbors(irow);
221  for(size_t inbr=0; inbr<cell_nnbrs; inbr++)
222  {
223  size_t cell_id = cellNborList.at(irow,inbr);
224  size_t cell_start = cellRange[cell_id];
225  size_t cell_size = sizes[cell_id];
226  size_t cell_end = cell_start + cell_size;
227  int* __restrict c_j_ctr = c_j_ptr + c_j_max;
228  int ll = 0;
229 
230  #pragma vector always
231  #pragma vector aligned
232  #pragma ivdep
233  for(size_t ll=0; ll<cell_size; ll++)
234  {
235  c_j_ctr[ll] = cell_start+ll;
236  }
237  c_j_max += cell_size;
238  }
239 
240  // padding
241  {
242  int last_nbr = cellNborList.at(irow,cell_nnbrs-1);
243  int padding = cellRange[last_nbr]+sizes[last_nbr];
244  int pad_end = ((c_j_max+ESPP_VECTOR_WIDTH-1)/ESPP_VECTOR_WIDTH)*ESPP_VECTOR_WIDTH;
245  int num_padding = pad_end - c_j_max;
246  int* __restrict c_j_ctr = c_j_ptr + c_j_max;
247 
248  #pragma vector always
249  #pragma vector aligned
250  #pragma ivdep
251  for(size_t ll=0; ll<num_padding; ll++)
252  {
253  c_j_ctr[ll] = padding;
254  }
255  c_j_max += num_padding;
256  }
257  c_range.push_back(c_j_max);
258  }
259  if(c_j_max>c_j.size()) throw std::runtime_error("rebuild_p_nc_pack_stencil: Reserve size exceeded.");
260 
262  #pragma vector always
263  #pragma vector aligned
264  #pragma ivdep
265  for(int ii=0; ii<c_j_max; ii++)
266  {
267  int p = c_j_ptr[ii];
268  if(VEC_MODE_AOS)
269  {
270  c_x_ptr[ii] = position[p].x;
271  c_y_ptr[ii] = position[p].y;
272  c_z_ptr[ii] = position[p].z;
273  }
274  else
275  {
276  c_x_ptr[ii] = pa_p_x[p];
277  c_y_ptr[ii] = pa_p_y[p];
278  c_z_ptr[ii] = pa_p_z[p];
279  }
280  }
281  }
283 
284  size_t prev_num_pairs = num_pairs;
285  for(size_t irow=0; irow < numRealCells; irow++)
286  {
289  if(num_pairs + max_pairs_per_cell > (neighborList.nplist.size()))
290  {
291  Int3D cellgrid = getSystem()->storage->getInt3DCellGrid();
292  const size_t threshold = std::max(cellgrid[0],std::max(cellgrid[1],cellgrid[2]));
293  size_t cells_left = numRealCells-irow, add_size;
294  if(cells_left <= threshold)
295  // over-estimate with few cells left
296  add_size = max_pairs_per_cell*(numRealCells-irow);
297  else
298  // under-estimate with many cells left
299  add_size = max_pairs_per_cell*(numRealCells-irow) / threshold;
300  size_t old_size = neighborList.nplist.size();
301  size_t new_size = ESPP_FIT_TO_VECTOR_WIDTH(old_size+add_size);
302  neighborList.nplist.resize(new_size);
303  LOG4ESPP_INFO(theLogger,"Resizing neighbor list. num_pairs="<<num_pairs
304  <<" old_size="<<old_size<<" new_size="<<new_size);
305  }
307 
308  size_t cell_id = cellNborList.cellId(irow);
309  size_t cell_nnbrs = cellNborList.numNeighbors(irow);
310  size_t cell_start = cellRange[cell_id];
311  size_t cell_data_end = cellRange[cell_id+1];
312  size_t cell_size = sizes[cell_id];
313  size_t cell_end = cell_start + cell_size;
314 
315  for(size_t p=cell_start; p < cell_end; p++)
316  {
317  real p_x, p_y, p_z;
318  if(VEC_MODE_AOS)
319  {
320  p_x = position[p].x;
321  p_y = position[p].y;
322  p_z = position[p].z;
323  }
324  else
325  {
326  p_x = pa_p_x[p];
327  p_y = pa_p_y[p];
328  p_z = pa_p_z[p];
329  }
330 
331  // self-loop
332  {
333  size_t ncell_id = cell_id;
334 
335  size_t ncell_start = cellRange[ncell_id];
336  size_t ncell_data_end = cellRange[ncell_id+1];
337  size_t ncell_end = ncell_start + sizes[ncell_id];
338  int* __restrict npptr = &(neighborList.nplist[num_pairs]);
339 
340  {
341  int ll=0;
342 
343  #pragma vector always
344  #pragma vector aligned
345  #pragma ivdep
346  for(size_t np = ncell_start; np<ncell_data_end; np++)
347  {
348  {
349  real dist_x, dist_y, dist_z;
350  if(VEC_MODE_AOS)
351  {
352  dist_x = p_x - position[np].x;
353  dist_y = p_y - position[np].y;
354  dist_z = p_z - position[np].z;
355  }
356  else
357  {
358  dist_x = p_x - pa_p_x[np];
359  dist_y = p_y - pa_p_y[np];
360  dist_z = p_z - pa_p_z[np];
361  }
362 
363  const real distSqr = dist_x*dist_x + dist_y*dist_y + dist_z*dist_z;
364 
365  if(p < np && distSqr <= cutsq) {
366  npptr[ll++] = np;
367  }
368  }
369  }
370  num_pairs += ll;
371  }
372  }
373 
374  // neighbor-loop
375  size_t last_ncell = cell_id, prev_num_pairs_2 = num_pairs;
376 
377  if(PACK_NEIGHBORS)
378  {
379  const int c_start = c_range[irow];
380  const int c_end = c_range[irow+1];
381  int* __restrict npptr = &(neighborList.nplist[num_pairs]);
382  int ll=0;
383 
384  #pragma vector always
385  #pragma vector aligned
386  #pragma ivdep
387  for(int ii=c_start; ii<c_end; ii++)
388  {
389  const real dist_x = p_x - c_x_ptr[ii];
390  const real dist_y = p_y - c_y_ptr[ii];
391  const real dist_z = p_z - c_z_ptr[ii];
392 
393  const real distSqr = dist_x*dist_x + dist_y*dist_y + dist_z*dist_z;
394 
395  int include_np = 1;
396  if(distSqr > cutsq) include_np = 0;
397  if(include_np) npptr[ll++] = c_j_ptr[ii];
398  }
399  num_pairs += ll;
400  }
401  else
402  for(size_t inbr=0; inbr<cell_nnbrs; inbr++)
403  {
404  size_t ncell_id = cellNborList.at(irow,inbr);
405 
406  size_t ncell_start = cellRange[ncell_id];
407  size_t ncell_data_end = cellRange[ncell_id+1];
408  size_t ncell_end = ncell_start + sizes[ncell_id];
409  int* __restrict npptr = &(neighborList.nplist[num_pairs]);
410 
411  {
412  int ll=0;
413 
414  #pragma vector always
415  #pragma vector aligned
416  #pragma ivdep
417  for(size_t np = ncell_start; np<ncell_data_end; np++)
418  {
419  {
420  real dist_x, dist_y, dist_z;
421  if(VEC_MODE_AOS)
422  {
423  dist_x = p_x - position[np].x;
424  dist_y = p_y - position[np].y;
425  dist_z = p_z - position[np].z;
426  }
427  else
428  {
429  dist_x = p_x - pa_p_x[np];
430  dist_y = p_y - pa_p_y[np];
431  dist_z = p_z - pa_p_z[np];
432  }
433 
434  const real distSqr = dist_x*dist_x + dist_y*dist_y + dist_z*dist_z;
435 
436  if(distSqr <= cutsq) {
437  npptr[ll++] = np;
438  }
439  }
440  }
441 
442  num_pairs += ll;
443  }
444 
445  // track the last cell to have a neighbor of p
446  if(num_pairs - prev_num_pairs_2)
447  last_ncell = ncell_id;
448  prev_num_pairs_2 = num_pairs;
449  }
450 
451  size_t new_pairs = (num_pairs - prev_num_pairs);
452 
453  if(new_pairs)
454  {
455  // pad remaining part of list with stray neighbor particle
456  size_t num_rem = num_pairs % ESPP_VECTOR_WIDTH;
457  size_t num_pad = (num_rem > 0) * (ESPP_VECTOR_WIDTH - num_rem);
458  size_t padding = cellRange[last_ncell]+sizes[last_ncell];
459  for(size_t pad = 0; pad < num_pad; pad++)
460  neighborList.nplist[num_pairs++] = padding;
461  neighborList.plist.push_back(p);
462  neighborList.prange.push_back(num_pairs);
463  }
464  prev_num_pairs = num_pairs;
465  }
466 
467  if(VEC_MODE_AOS)
468  for(size_t p=cell_start; p < cell_end; p++)
469  max_type = std::max(max_type, position[p].t);
470  else
471  for(size_t p=cell_start; p < cell_end; p++)
472  max_type = std::max(max_type, pa_p_type[p]);
473  }
474 
475  if(num_pairs>nplist_reserve) {
476  LOG4ESPP_WARN(theLogger,"Reserve size exceeded. "
477  "Expected "<< nplist_reserve<<". Got "<<num_pairs);
478  }
479  }
480 
481  LOG4ESPP_DEBUG(theLogger, "rebuilt VerletList (count=" << builds << "), cutsq = " << cutsq
482  << " local size = " << vlPairs.size());
483  }

487  void VerletList::checkPair(Particle& pt1, Particle& pt2)
488  {
489 
490  Real3D d = pt1.position() - pt2.position();
491  real distsq = d.sqr();
492 
493  LOG4ESPP_TRACE(theLogger, "p1: " << pt1.id()
494  << " @ " << pt1.position()
495  << " - p2: " << pt2.id() << " @ " << pt2.position()
496  << " -> distsq = " << distsq);
497 
498  if (distsq > cutsq) return;
499 
500  // see if it's in the exclusion list (both directions)
501  if (exList.count(std::make_pair(pt1.id(), pt2.id())) == 1) return;
502  if (exList.count(std::make_pair(pt2.id(), pt1.id())) == 1) return;
503 
504  max_type = std::max(max_type, std::max(pt1.type(), pt2.type()));
505  vlPairs.add(pt1, pt2); // add pair to Verlet List
506  }
