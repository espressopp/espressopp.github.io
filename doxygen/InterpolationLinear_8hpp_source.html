82  inline real InterpolationLinear::splineInterpolation(real r,
83  const real* a, const real* b) const {
84  int index;
85  index = static_cast<int>((r - inner) * invdelta);
86  if (index < 0) {
87  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
88  << inner << " - " << inner + (N - 1) * delta);
89  index = 0;
90  }
91  if (index >= N) {
92  LOG4ESPP_ERROR(theLogger, "distance " << r << " out of range "
93  << inner << " - " << inner + (N - 1) * delta);
94  index = N-1;
95  }
96 
97  return a[index]*r + b[index];
98  }
