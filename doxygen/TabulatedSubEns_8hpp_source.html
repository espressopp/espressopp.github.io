160  real _computeEnergySqrRaw(real distSqr) const {
161  real e = 0.;
162  for (int i=0; i<numInteractions; ++i)
163  e += weights[i] * tables[i]->getEnergy(sqrt(distSqr));
164  return e;
165  }
166 
167  bool _computeForceRaw(Real3D& force, const Real3D& dist, real distSqr) const {
168  real ffactor = 0;
169  real distrt = sqrt(distSqr);
170  for (int i=0; i<numInteractions; ++i)
171  ffactor += weights[i] * tables[i]->getForce(distrt) / distrt;
172  force = dist * ffactor;
173  return true;
174  }
