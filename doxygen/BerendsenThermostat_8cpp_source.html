00083     void BerendsenThermostat::thermostat(){
00084       LOG4ESPP_DEBUG(theLogger, "equilibrating the temperature");
00085 
00086       System& system = getSystemRef();
00087       esutil::Error err(system.comm);
00088       
00089       static Temperature Tcurrent(getSystem());
00090       
00091       real T = Tcurrent.computeRaw();  // calculating the current temperature in system
00092       
00093       real lambda2 = 1 + pref * (T0/T - 1);
00094       
00095       if(lambda2<0.0){
00096         std::stringstream msg;
00097         msg << "Scaling coefficient is <0 (Berendsen thermostat). lambda^2="<<lambda2;
00098         err.setException( msg.str() );
00099         err.checkException();
00100       }
00101 
00102       real lambda = sqrt( lambda2 );  // current scaling parameter
00103 
00104       CellList cells = system.storage->getRealCells();
00105       for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
00106         scaleVelocity(*cit, lambda);
00107       }
00108     }
