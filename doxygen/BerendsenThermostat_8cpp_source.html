83  void BerendsenThermostat::thermostat(){
84  LOG4ESPP_DEBUG(theLogger, "equilibrating the temperature");
85 
86  System& system = getSystemRef();
87  esutil::Error err(system.comm);
88 
89  static Temperature Tcurrent(getSystem());
90 
91  real T = Tcurrent.compute_real(); // calculating the current temperature in system
92 
93  real lambda2 = 1 + pref * (T0/T - 1);
94 
95  if(lambda2<0.0){
96  std::stringstream msg;
97  msg << "Scaling coefficient is <0 (Berendsen thermostat). lambda^2="<<lambda2;
98  err.setException( msg.str() );
99  err.checkException();
100  }
101 
102  real lambda = sqrt( lambda2 ); // current scaling parameter
103 
104  CellList cells = system.storage->getRealCells();
105  for(CellListIterator cit(cells); !cit.isDone(); ++cit) {
106  scaleVelocity(*cit, lambda);
107  }
108  }
