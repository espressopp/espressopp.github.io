57  public:
58  FixedLocalTupleComListInteractionTemplate
59  (shared_ptr < System > _system,
60  shared_ptr < FixedLocalTupleList > _fixedtupleList,
61  shared_ptr < Potential > _potential)
62  : SystemAccess(_system), fixedtupleList(_fixedtupleList), potential(_potential) {
63  if (! potential) {
64  LOG4ESPP_ERROR(theLogger, "NULL potential");
65  }
66  System& system = getSystemRef();
67  esutil::Error err(system.comm);
68 
69  long unsigned int localN = system.storage -> getNRealParticles();
70  boost::mpi::all_reduce(*system.comm, localN, num_of_part, std::plus<int>());
71 
72  N_Constrain = 0;
73 
74  if (!fixedtupleList->empty()) {
75 
76  FixedLocalTupleList::TupleList::Iterator it(*fixedtupleList);
77  std::vector<Particle*> pList = it->second;
78 
79  N_Constrain = pList.size() + 1;
80 
81  // Check the length of individula particle list in FixedLocalTuple
82  for (FixedLocalTupleList::TupleList::Iterator it(*fixedtupleList); it.isValid(); ++it) {
83  std::vector<Particle*> pList = it->second;
84 
85  if (pList.size() + 1 != N_Constrain) {
86  std::stringstream msg;
87  msg << "ERROR: Tuple Length is not constant\n";
88  err.setException(msg.str());
89  }
90  }
91  }
92 
93  std::vector<int> nList;
94  boost::mpi::all_gather(*system.comm, N_Constrain, nList);
95  int check_N = 0;
96  for (int i = 0; i < nList.size(); i++) {
97  if (nList[i] != 0) {
98  check_N = nList[i];
99  break;
100  }
101  }
102  for (int i = 0; i < nList.size(); i++) {
103  if (nList[i] != 0) {
104  if (nList[i] != check_N) {
105  std::stringstream msg;
106  msg << "ERROR: Tuple Length is not constant\n";
107  err.setException(msg.str());
108  }
109  }
110  }
111  N_Constrain = check_N;
112 
113  num_of_subchain = num_of_part/N_Constrain;
114 
115  int* pid_list;
116  longint pidK = 0;
117  pid_list = new int[num_of_subchain];
118 
119  // Check the particle id in FixedLocalTuple
120  for (FixedLocalTupleList::TupleList::Iterator it(*fixedtupleList); it.isValid(); ++it) {
121  Particle *p = it->first;
122  pid_list[pidK] = p->id();
123 
124  for(long unsigned int i = 0; i < pidK; i++) {
125  if (pid_list[i]/N_Constrain == pid_list[pidK]/N_Constrain) {
126  std::stringstream msg;
127  msg << "ERROR: Particle ID is redundant\n";
128  err.setException(msg.str());
129  }
130  }
131  pidK++;
132  }
133 
134  delete [] pid_list;
135  pid_list = NULL;
136 
137  Real3D* totCOM = new Real3D[num_of_subchain];
138  Real3D* COM = new Real3D[num_of_subchain];
139 
140  computeCOM();
141 
142  for(long unsigned int i = 0; i < num_of_subchain; i++) {
143  totCOM[i] = 0.;
144 
145  if (com_temporal.count(i)) {
146  COM[i] = com_temporal[i];
147  } else {
148  COM[i] = 0.;
149  }
150  }
151 
152  boost::mpi::all_reduce(*system.comm, COM, num_of_subchain, totCOM, std::plus<Real3D>() );
153 
154  for (long unsigned int i = 0; i < num_of_subchain; i++) {
155  com_origin[i] = totCOM[i];
156  }
157 
158  delete [] COM;
159  COM = NULL;
160  delete [] totCOM;
161  totCOM = NULL;
162  }

209  void computeCOM() {
210 
211  com_temporal.clear();
212  total_mass.clear();
213 
214  const bc::BC& bc = *getSystemRef().bc; // boundary conditions
215 
216  for (FixedLocalTupleList::TupleList::Iterator it(*fixedtupleList); it.isValid(); ++it) {
217  Particle *p = it->first;
218  Real3D pos_i = p->position();
219  real mass = p->mass();
220  Real3D tmp_com = mass*pos_i;
221  real tmp_mass = mass;
222  std::vector<Particle*> pList = it->second;
223 
224  for (long unsigned int j = 0; j < N_Constrain - 1; j++) {
225  p = pList[j];
226  Real3D pos_j = p->position();
227  Real3D dist;
228  bc.getMinimumImageVectorBox(dist, pos_j, pos_i);
229  pos_j = pos_i + dist;
230  mass = p->mass();
231  tmp_com += mass*pos_j;
232  tmp_mass += mass;
233  pos_i = pos_j;
234  }
235  com_temporal[(p->id() - 1)/N_Constrain] = tmp_com/tmp_mass;
236  total_mass[(p->id() - 1)/N_Constrain] = tmp_mass;
237  }
238  }
